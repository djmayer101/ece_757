00043 AccessTraceForAddress&
00044 lookupTraceForAddress(const Address& addr, AddressMap& record_map)
00045 {
00046     // we create a static default object here that is used to insert
00047     // since the insertion will create a copy of the object in the
00048     // process.  Perhaps this is optimizing early, but it doesn't seem
00049     // like it could hurt.
00050     static const AccessTraceForAddress dflt;
00051 
00052     pair<AddressMap::iterator, bool> r =
00053         record_map.insert(make_pair(addr, dflt));
00054     AddressMap::iterator i = r.first;
00055     AccessTraceForAddress &access_trace = i->second;
00056     if (r.second) {
00057         // there was nothing there and the insert succeed, so we need
00058         // to actually set the address.
00059         access_trace.setAddress(addr);
00060     }
00061 
00062     return access_trace;
00063 }

00066 printSorted(ostream& out, int num_of_sequencers, const AddressMap &record_map,
00067             string description)
00068 {
00069     const int records_printed = 100;
00070 
00071     uint64 misses = 0;
00072     std::vector<const AccessTraceForAddress *> sorted;
00073 
00074     AddressMap::const_iterator i = record_map.begin();
00075     AddressMap::const_iterator end = record_map.end();
00076     for (; i != end; ++i) {
00077         const AccessTraceForAddress* record = &i->second;
00078         misses += record->getTotal();
00079         sorted.push_back(record);
00080     }
00081     sort(sorted.begin(), sorted.end(), AccessTraceForAddress::less_equal);
00082 
00083     out << "Total_entries_" << description << ": " << record_map.size()
00084         << endl;
00085     if (g_system_ptr->getProfiler()->getAllInstructions())
00086         out << "Total_Instructions_" << description << ": " << misses << endl;
00087     else
00088         out << "Total_data_misses_" << description << ": " << misses << endl;
00089 
00090     out << "total | load store atomic | user supervisor | sharing | touched-by"
00091         << endl;
00092 
00093     Histogram remaining_records(1, 100);
00094     Histogram all_records(1, 100);
00095     Histogram remaining_records_log(-1);
00096     Histogram all_records_log(-1);
00097 
00098     // Allows us to track how many lines where touched by n processors
00099     std::vector<int64> m_touched_vec;
00100     std::vector<int64> m_touched_weighted_vec;
00101     m_touched_vec.resize(num_of_sequencers+1);
00102     m_touched_weighted_vec.resize(num_of_sequencers+1);
00103     for (int j = 0; j < m_touched_vec.size(); j++) {
00104         m_touched_vec[j] = 0;
00105         m_touched_weighted_vec[j] = 0;
00106     }
00107 
00108     int counter = 0;
00109     int max = sorted.size();
00110     while (counter < max && counter < records_printed) {
00111         const AccessTraceForAddress* record = sorted[counter];
00112         double percent = 100.0 * (record->getTotal() / double(misses));
00113         out << description << " | " << percent << " % " << *record << endl;
00114         all_records.add(record->getTotal());
00115         all_records_log.add(record->getTotal());
00116         counter++;
00117         m_touched_vec[record->getTouchedBy()]++;
00118         m_touched_weighted_vec[record->getTouchedBy()] += record->getTotal();
00119     }
00120 
00121     while (counter < max) {
00122         const AccessTraceForAddress* record = sorted[counter];
00123         all_records.add(record->getTotal());
00124         remaining_records.add(record->getTotal());
00125         all_records_log.add(record->getTotal());
00126         remaining_records_log.add(record->getTotal());
00127         m_touched_vec[record->getTouchedBy()]++;
00128         m_touched_weighted_vec[record->getTouchedBy()] += record->getTotal();
00129     }
00130     out << endl;
00131     out << "all_records_" << description << ": "
00132         << all_records << endl
00133         << "all_records_log_" << description << ": "
00134         << all_records_log << endl
00135         << "remaining_records_" << description << ": "
00136         << remaining_records << endl
00137         << "remaining_records_log_" << description << ": "
00138         << remaining_records_log << endl
00139         << "touched_by_" << description << ": "
00140         << m_touched_vec << endl
00141         << "touched_by_weighted_" << description << ": "
00142         << m_touched_weighted_vec << endl
00143         << endl;

00281 void
00282 AddressProfiler::addTraceSample(Address data_addr, Address pc_addr,
00283                                 RubyRequestType type,
00284                                 RubyAccessMode access_mode, NodeID id,
00285                                 bool sharing_miss)
00286 {
00287     if (m_all_instructions) {
00288         if (sharing_miss) {
00289             m_sharing_miss_counter++;
00290         }
00291 
00292         // record data address trace info
00293         data_addr.makeLineAddress();
00294         lookupTraceForAddress(data_addr, m_dataAccessTrace).
00295             update(type, access_mode, id, sharing_miss);
00296 
00297         // record macro data address trace info
00298 
00299         // 6 for datablock, 4 to make it 16x more coarse
00300         Address macro_addr(data_addr.maskLowOrderBits(10));
00301         lookupTraceForAddress(macro_addr, m_macroBlockAccessTrace).
00302             update(type, access_mode, id, sharing_miss);
00303 
00304         // record program counter address trace info
00305         lookupTraceForAddress(pc_addr, m_programCounterAccessTrace).
00306             update(type, access_mode, id, sharing_miss);
00307     }
00308 
00309     if (m_all_instructions) {
00310         // This code is used if the address profiler is an
00311         // all-instructions profiler record program counter address
00312         // trace info
00313         lookupTraceForAddress(pc_addr, m_programCounterAccessTrace).
00314             update(type, access_mode, id, sharing_miss);
00315     }
