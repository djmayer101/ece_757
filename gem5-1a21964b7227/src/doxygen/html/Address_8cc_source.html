00060 Index
00061 Address::memoryModuleIndex() const
00062 {
00063     Index index =
00064         bitSelect(RubySystem::getBlockSizeBits() +
00065                   RubySystem::getMemorySizeBits(), ADDRESS_WIDTH);
00066     assert (index >= 0);
00067     return index;
00068 
00069     // Index indexHighPortion =
00070     //     address.bitSelect(MEMORY_SIZE_BITS - 1,
00071     //                       PAGE_SIZE_BITS + NUMBER_OF_MEMORY_MODULE_BITS);
00072     // Index indexLowPortion =
00073     //     address.bitSelect(DATA_BLOCK_BITS, PAGE_SIZE_BITS - 1);
00074     //
00075     // Index index = indexLowPortion |
00076     //     (indexHighPortion << (PAGE_SIZE_BITS - DATA_BLOCK_BITS));
00077 
00078     /*
00079       Round-robin mapping of addresses, at page size granularity
00080 
00081 ADDRESS_WIDTH    MEMORY_SIZE_BITS        PAGE_SIZE_BITS  DATA_BLOCK_BITS
00082   |                    |                       |               |
00083  \ /                  \ /                     \ /             \ /       0
00084   -----------------------------------------------------------------------
00085   |       unused        |xxxxxxxxxxxxxxx|       |xxxxxxxxxxxxxxx|       |
00086   |                     |xxxxxxxxxxxxxxx|       |xxxxxxxxxxxxxxx|       |
00087   -----------------------------------------------------------------------
00088                         indexHighPortion         indexLowPortion
00089                                         <------->
00090                                NUMBER_OF_MEMORY_MODULE_BITS
00091     */
00092 }
