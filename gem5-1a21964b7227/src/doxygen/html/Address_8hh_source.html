00131 inline physical_address_t
00132 Address::bitSelect(int small, int big) const
00133 {
00134     physical_address_t mask;
00135     assert((unsigned)big >= (unsigned)small);
00136 
00137     if (big >= ADDRESS_WIDTH - 1) {
00138         return (m_address >> small);
00139     } else {
00140         mask = ~((physical_address_t)~0 << (big + 1));

00147 // removes bits inclusive
00148 inline physical_address_t
00149 Address::bitRemove(int small, int big) const
00150 {
00151     physical_address_t mask;
00152     assert((unsigned)big >= (unsigned)small);
00153 
00154     if (small >= ADDRESS_WIDTH - 1) {
00155         return m_address;
00156     } else if (big >= ADDRESS_WIDTH - 1) {
00157         mask = (physical_address_t)~0 >> small;
00158         return (m_address & mask);
00159     } else if (small == 0) {
00160         mask = (physical_address_t)~0 << big;
00161         return (m_address & mask);
00162     } else {
00163         mask = ~((physical_address_t)~0 << small);
00164         physical_address_t lower_bits = m_address & mask;
00165         mask = (physical_address_t)~0 << (big + 1);
00166         physical_address_t higher_bits = m_address & mask;
00167 
00168         // Shift the valid high bits over the removed section
00169         higher_bits = higher_bits >> (big - small + 1);
00170         return (higher_bits | lower_bits);
00171     }
00172 }
