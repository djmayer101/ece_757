00038 BaseGarnetNetwork::BaseGarnetNetwork(const Params *p)
00039     : Network(p), m_ruby_start(0)
00040 {
00041     m_ni_flit_size = p->ni_flit_size;
00042     m_vcs_per_vnet = p->vcs_per_vnet;
00043     m_enable_fault_model = p->enable_fault_model;
00044     if (m_enable_fault_model)
00045         fault_model = p->fault_model;
00046 
00047     // Currently Garnet only supports uniform bandwidth for all
00048     // links and network interfaces.
00049     for (std::vector<BasicExtLink*>::const_iterator i = 
00050              m_topology_ptr->params()->ext_links.begin();
00051          i != m_topology_ptr->params()->ext_links.end(); ++i) {
00052         BasicExtLink* ext_link = (*i);
00053         if (ext_link->params()->bandwidth_factor != m_ni_flit_size) {
00054             fatal("Garnet only supports uniform bw across all links and NIs\n");
00055         }
00056     }
00057     for (std::vector<BasicIntLink*>::const_iterator i = 
00058              m_topology_ptr->params()->int_links.begin();
00059          i != m_topology_ptr->params()->int_links.end(); ++i) {
00060         BasicIntLink* int_link = (*i);
00061         if (int_link->params()->bandwidth_factor != m_ni_flit_size) {
00062             fatal("Garnet only supports uniform bw across all links and NIs\n");
00063         }
00064     }
00065 
00066     // Allocate to and from queues
00067 
00068     // Queues that are getting messages from protocol
00069     m_toNetQueues.resize(m_nodes);
00070 
00071     // Queues that are feeding the protocol
00072     m_fromNetQueues.resize(m_nodes);
00073 
00074     m_in_use.resize(m_virtual_networks);
00075     m_ordered.resize(m_virtual_networks);
00076     m_flits_received.resize(m_virtual_networks);
00077     m_flits_injected.resize(m_virtual_networks);
00078     m_network_latency.resize(m_virtual_networks);
00079     m_queueing_latency.resize(m_virtual_networks);
00080     for (int i = 0; i < m_virtual_networks; i++) {
00081         m_in_use[i] = false;
00082         m_ordered[i] = false;
00083         m_flits_received[i] = 0;
00084         m_flits_injected[i] = 0;
00085         m_network_latency[i] = 0.0;
00086         m_queueing_latency[i] = 0.0;
00087     }
00088 
00089     for (int node = 0; node < m_nodes; node++) {
00090         // Setting number of virtual message buffers per Network Queue
00091         m_toNetQueues[node].resize(m_virtual_networks);
00092         m_fromNetQueues[node].resize(m_virtual_networks);
00093 
00094         // Instantiating the Message Buffers that
00095         // interact with the coherence protocol
00096         for (int j = 0; j < m_virtual_networks; j++) {
00097             m_toNetQueues[node][j] = new MessageBuffer();
00098             m_fromNetQueues[node][j] = new MessageBuffer();
00099         }
00100     }
00101 }

00149 void
00150 BaseGarnetNetwork::printPerformanceStats(ostream& out) const
00151 {
00152     int total_flits_injected = 0;
00153     int total_flits_received = 0;
00154     int total_network_latency = 0.0;
00155     int total_queueing_latency = 0.0;
00156 
00157     for (int i = 0; i < m_virtual_networks; i++) {
00158         if (!m_in_use[i])
00159             continue;
00160 
00161         out << "[Vnet " << i << "]: flits injected = "
00162             << m_flits_injected[i] << endl;
00163         out << "[Vnet " << i << "]: flits received = "
00164             << m_flits_received[i] << endl;
00165         out << "[Vnet " << i << "]: average network latency = "
00166             << ((double) m_network_latency[i] / (double) m_flits_received[i])
00167             << endl;
00168         out << "[Vnet " << i << "]: average queueing (at source NI) latency = "
00169             << ((double) m_queueing_latency[i] / (double) m_flits_received[i])
00170             << endl;
00171 
00172         out << endl;
00173         total_flits_injected += m_flits_injected[i];
00174         total_flits_received += m_flits_received[i];
00175         total_network_latency += m_network_latency[i];
00176         total_queueing_latency += m_queueing_latency[i];
00177     }
00178     out << "Total flits injected = " << total_flits_injected << endl;
00179     out << "Total flits received = " << total_flits_received << endl;
00180     out << "Average network latency = "
00181         << ((double) total_network_latency/ (double) total_flits_received) << endl;
00182     out << "Average queueing (at source NI) latency = "
00183         << ((double) total_queueing_latency/ (double) total_flits_received) << endl;
00184     out << "Average latency = "
00185         << ((double)  (total_queueing_latency + total_network_latency) /
00186             (double) total_flits_received)<< endl;
00187     out << "-------------" << endl;
00188     out << endl;
