00084     else
00085         assert(false);
00086 
00087     m_cache.resize(m_cache_num_sets);
00088     for (int i = 0; i < m_cache_num_sets; i++) {
00089         m_cache[i].resize(m_cache_assoc);
00090         for (int j = 0; j < m_cache_assoc; j++) {
00091             m_cache[i][j] = NULL;
00092         }
00093     }

00347 {

00350                                                   * (uint64)m_cache_assoc;
00351 
00352     for (int i = 0; i < m_cache_num_sets; i++) {
00353         for (int j = 0; j < m_cache_assoc; j++) {

00356                 RubyRequestType request_type = RubyRequestType_NULL;
00357                 if (perm == AccessPermission_Read_Only) {
00358                     if (m_is_instruction_only_cache) {
00359                         request_type = RubyRequestType_IFETCH;
00360                     } else {
00361                         request_type = RubyRequestType_LD;
00362                     }
00363                 } else if (perm == AccessPermission_Read_Write) {
00364                     request_type = RubyRequestType_ST;
00365                 }
00366 
00367                 if (request_type != RubyRequestType_NULL) {
00368                     tr->addRecord(cntrl, m_cache[i][j]->m_Address.getAddress(),
00369                                   0, request_type,
00370                                   m_replacementPolicy_ptr->getLastAccess(i, j),
00371                                   m_cache[i][j]->getDataBlk());
00372                     warmedUpBlocks++;
00373                 }
00374             }
00375         }
00376     }
00377 
00378     DPRINTF(RubyCacheTrace, "%s: %lli blocks of %lli total blocks"
00379             "recorded %.2f%% \n", name().c_str(), warmedUpBlocks,

00381             (float(warmedUpBlocks)/float(totalBlocks))*100.0);
