00148 CacheRecorder::aggregateRecords(uint8_t** buf, uint64 total_size)
00149 {
00150     std::sort(m_records.begin(), m_records.end(), compareTraceRecords);
00151 
00152     int size = m_records.size();
00153     uint64 current_size = 0;
00154     int record_size = sizeof(TraceRecord) + RubySystem::getBlockSizeBytes();
00155 
00156     for (int i = 0; i < size; ++i) {
00157         // Determine if we need to expand the buffer size
00158         if (current_size + record_size > total_size) {
00159             uint8_t* new_buf = new (nothrow) uint8_t[total_size * 2];
00160             if (new_buf == NULL) {
00161                 fatal("Unable to allocate buffer of size %s\n",
00162                       total_size * 2);
00163             }
00164             total_size = total_size * 2;
00165             uint8_t* old_buf = *buf;
00166             memcpy(new_buf, old_buf, current_size);
00167             *buf = new_buf;
00168             delete [] old_buf;
00169         }
00170 
00171         // Copy the current record into the buffer
00172         memcpy(&((*buf)[current_size]), m_records[i], record_size);
00173         current_size += record_size;
00174 
00175         free(m_records[i]);
00176         m_records[i] = NULL;
00177     }
