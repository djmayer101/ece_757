00080 void
00081 CheckTable::addCheck(const Address& address)
00082 {
00083     if (floorLog2(CHECK_SIZE) != 0) {
00084         if (address.bitSelect(0, CHECK_SIZE_BITS - 1) != 0) {
00085             panic("Check not aligned");
00086         }
00087     }
00088 
00089     for (int i = 0; i < CHECK_SIZE; i++) {
00090         if (m_lookup_map.count(Address(address.getAddress()+i))) {
00091             // A mapping for this byte already existed, discard the
00092             // entire check
00093             return;
00094         }
00095     }
00096 
00097     Check* check_ptr = new Check(address, Address(100 + m_check_vector.size()),
00098                                  m_num_writers, m_num_readers, m_tester_ptr);
00099     for (int i = 0; i < CHECK_SIZE; i++) {
00100         // Insert it once per byte
00101         m_lookup_map[Address(address.getAddress() + i)] = check_ptr;
00102     }
00103     m_check_vector.push_back(check_ptr);
00104 }
