00092     // 1 in 8 chance this will be an exclusive prefetch
00093     if ((random() & 0x7) != 0) {

00097         if (m_tester_ptr->isInstReadableCpuPort(index)) {
00098             flags.set(Request::INST_FETCH);

00101         cmd = MemCmd::WriteReq;
00102         flags.set(Request::PF_EXCLUSIVE);
00103     }
00104 
00105     // Prefetches are assumed to be 0 sized
00106     Request *req = new Request(m_address.getAddress(), 0, flags,
00107             m_tester_ptr->masterId(), curTick(), m_pc.getAddress());
00108     req->setThreadContext(index, 0);
00109 
00110     PacketPtr pkt = new Packet(req, cmd);
00111 
00112     // push the subblock onto the sender state.  The sequencer will
00113     // update the subblock on the return
00114     pkt->senderState = new SenderState(m_address, req->getSize());
00115 
00116     if (port->sendTimingReq(pkt)) {
00117         DPRINTF(RubyTest, "successfully initiated prefetch.\n");

00281 
00282     if (m_status == TesterStatus_Action_Pending) {
00283         DPRINTF(RubyTest, "Action callback write value: %d, currently %d\n",
00284                 (m_value + m_store_count), data->getByte(0));

00289             m_status = TesterStatus_Ready;

00291             m_status = TesterStatus_Idle;

00293         DPRINTF(RubyTest, "Action callback return data now %d\n",
00294                 data->getByte(0));
00295     } else if (m_status == TesterStatus_Check_Pending) {
00296         DPRINTF(RubyTest, "Check callback\n");
00297         // Perform load/check
00298         for (int byte_number=0; byte_number<CHECK_SIZE; byte_number++) {
00299             if (uint8_t(m_value + byte_number) != data->getByte(byte_number)) {
00300                 panic("Action/check failure: proc: %d address: %s data: %s "
00301                       "byte_number: %d m_value+byte_number: %d byte: %d %s"
00302                       "Time: %d\n",
00303                       proc, address, data, byte_number,
00304                       (int)m_value + byte_number,
00305                       (int)data->getByte(byte_number), *this, curTime);
00306             }
00307         }
00308         DPRINTF(RubyTest, "Action/check success\n");
00309         debugPrint();
00310 
00311         // successful check complete, increment complete
00312         m_tester_ptr->incrementCheckCompletions();
00313 
00314         m_status = TesterStatus_Idle;

00316 
00317     } else {
