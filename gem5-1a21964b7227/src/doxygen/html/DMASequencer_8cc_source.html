00050 RequestStatus
00051 DMASequencer::makeRequest(PacketPtr pkt)
00052 {
00053     if (m_is_busy) {
00054         return RequestStatus_BufferFull;
00055     }
00056 
00057     uint64_t paddr = pkt->getAddr();
00058     uint8_t* data =  pkt->getPtr<uint8_t>(true);
00059     int len = pkt->getSize();
00060     bool write = pkt->isWrite();
00061 
00062     assert(!m_is_busy);  // only support one outstanding DMA request
00063     m_is_busy = true;
00064 
00065     active_request.start_paddr = paddr;
00066     active_request.write = write;
00067     active_request.data = data;
00068     active_request.len = len;
00069     active_request.bytes_completed = 0;
00070     active_request.bytes_issued = 0;
00071     active_request.pkt = pkt;
00072 
00073     SequencerMsg *msg = new SequencerMsg(clockEdge());
00074     msg->getPhysicalAddress() = Address(paddr);
00075     msg->getLineAddress() = line_address(msg->getPhysicalAddress());
00076     msg->getType() = write ? SequencerRequestType_ST : SequencerRequestType_LD;
00077     int offset = paddr & m_data_block_mask;
00078 
00079     msg->getLen() = (offset + len) <= RubySystem::getBlockSizeBytes() ?
00080         len : RubySystem::getBlockSizeBytes() - offset;
00081 
00082     if (write && (data != NULL)) {
00083         if (active_request.data != NULL) {
00084             msg->getDataBlk().setData(data, offset, msg->getLen());
00085         }
00086     }
00087 
00088     assert(m_mandatory_q_ptr != NULL);
00089     m_mandatory_q_ptr->enqueue(msg);
00090     active_request.bytes_issued += msg->getLen();
00091 
00092     return RequestStatus_Issued;
00093 }

00095 void
00096 DMASequencer::issueNext()
00097 {
00098     assert(m_is_busy == true);
00099     active_request.bytes_completed = active_request.bytes_issued;
00100     if (active_request.len == active_request.bytes_completed) {
00101         //
00102         // Must unset the busy flag before calling back the dma port because
00103         // the callback may cause a previously nacked request to be reissued
00104         //
00105         DPRINTF(RubyDma, "DMA request completed\n");
00106         m_is_busy = false;
00107         ruby_hit_callback(active_request.pkt);
00108         return;
00109     }
00110 
00111     SequencerMsg *msg = new SequencerMsg(clockEdge());
00112     msg->getPhysicalAddress() = Address(active_request.start_paddr +
00113                                        active_request.bytes_completed);
00114 
00115     assert((msg->getPhysicalAddress().getAddress() & m_data_block_mask) == 0);
00116     msg->getLineAddress() = line_address(msg->getPhysicalAddress());
00117 
00118     msg->getType() = (active_request.write ? SequencerRequestType_ST :
00119                      SequencerRequestType_LD);
00120 
00121     msg->getLen() =
00122         (active_request.len -
00123          active_request.bytes_completed < RubySystem::getBlockSizeBytes() ?
00124          active_request.len - active_request.bytes_completed :
00125          RubySystem::getBlockSizeBytes());
00126 
00127     if (active_request.write) {
00128         msg->getDataBlk().
00129             setData(&active_request.data[active_request.bytes_completed],
00130                     0, msg->getLen());
00131         msg->getType() = SequencerRequestType_ST;
00132     } else {
00133         msg->getType() = SequencerRequestType_LD;
00134     }
00135 
00136     assert(m_mandatory_q_ptr != NULL);
00137     m_mandatory_q_ptr->enqueue(msg);
00138     active_request.bytes_issued += msg->getLen();
00139     DPRINTF(RubyDma, 
00140             "DMA request bytes issued %d, bytes completed %d, total len %d\n",
00141             active_request.bytes_issued, active_request.bytes_completed,
00142             active_request.len);
00143 }

00145 void
00146 DMASequencer::dataCallback(const DataBlock & dblk)
00147 {
00148     assert(m_is_busy == true);
00149     int len = active_request.bytes_issued - active_request.bytes_completed;
00150     int offset = 0;
00151     if (active_request.bytes_completed == 0)
00152         offset = active_request.start_paddr & m_data_block_mask;
00153     assert(active_request.write == false);
00154     if (active_request.data != NULL) {
00155         memcpy(&active_request.data[active_request.bytes_completed],
00156                dblk.getData(offset, len), len);
00157     }
00158     issueNext();
00159 }
