00071 void DecoderUnit::init()
00072 {
00073     if (m_dec_width >= 4)
00074     { // 2-level decoder
00075         m_num_in_1st = (m_dec_width == 4)? 2:3;
00076         m_num_out_0th = 1 << (m_num_in_1st - 1);
00077         m_num_in_2nd = (uint32_t)ceil((double)m_dec_width/(double)m_num_in_1st);
00078         m_num_out_1st = 1 << (m_dec_width - m_num_in_1st);
00079     }
00080     else if (m_dec_width >= 2)
00081     { // 1-level decoder
00082         m_num_in_1st = m_dec_width;
00083         m_num_out_0th = 1 << (m_num_in_1st - 1);
00084         m_num_in_2nd = m_num_out_1st = 0;
00085     }
00086     else
00087     {
00088         m_num_in_1st = m_num_out_0th = m_num_in_2nd = m_num_out_1st = 0;
00089     }
00090 
00091     // compute energy constants
00092     double e_factor = m_tech_param_ptr->get_vdd() * m_tech_param_ptr->get_vdd();
00093     if (m_dec_width >= 4)
00094     {
00095         m_e_chg_l1 = calc_chgl1_cap() * e_factor;
00096         m_e_chg_output = calc_select_cap() * e_factor;
00097     }
00098     else if (m_dec_width >= 2)
00099     {
00100         m_e_chg_l1 = calc_chgl1_cap() * e_factor;
00101         m_e_chg_output = 0;

00104     {
00105         m_e_chg_l1 = m_e_chg_output = 0;
