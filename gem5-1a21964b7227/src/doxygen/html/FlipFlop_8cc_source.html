00106 double FlipFlop::calc_node_cap(uint32_t num_fanin_, uint32_t num_fanout_)
00107 {
00108     double total_cap = 0;
00109 
00110     //FIXME: all need actual size
00111     //part 1: drain cap of NOR gate
00112     double WdecNORn = m_tech_param_ptr->get_WdecNORn();
00113     double WdecNORp = m_tech_param_ptr->get_WdecNORp();
00114     total_cap += num_fanin_*m_tech_param_ptr->calc_draincap(WdecNORn, TechParameter::NCH, 1) + m_tech_param_ptr->calc_draincap(WdecNORp, TechParameter::PCH, num_fanin_);
00115 
00116     //part 2: gate cap of NOR gate
