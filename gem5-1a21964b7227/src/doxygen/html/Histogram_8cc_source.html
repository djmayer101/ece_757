00087 Histogram::add(int64 value)
00088 {
00089     assert(value >= 0);
00090     m_max = max(m_max, value);
00091     m_count++;
00092 
00093     m_sumSamples += value;
00094     m_sumSquaredSamples += (value*value);
00095 
00096     uint32_t index;
00097 
00098     if (m_binsize == -1) {
00099         // This is a log base 2 histogram
00100         if (value == 0) {
00101             index = 0;
00102         } else {
00103             index = floorLog2(value) + 1;
00104             if (index >= m_data.size()) {
00105                 index = m_data.size() - 1;
00106             }
00107         }
00108     } else {
00109         // This is a linear histogram
00110         uint32_t t_bins = m_data.size();
00111 
00112         while (m_max >= (t_bins * m_binsize)) doubleBinSize();
00113         index = value/m_binsize;
00114     }
00115 
00116     assert(index < m_data.size());
00117     m_data[index]++;
00118     m_largest_bin = max(m_largest_bin, index);

00122 Histogram::add(Histogram& hist)
00123 {
00124     uint32_t t_bins = m_data.size();
00125 
00126     if (hist.getBins() != t_bins) {
00127         fatal("Histograms with different number of bins cannot be combined!");
00128     }
00129 
00130     m_max = max(m_max, hist.getMax());
00131     m_count += hist.size();
00132     m_sumSamples += hist.getTotal();
00133     m_sumSquaredSamples += hist.getSquaredTotal();
00134 
00135     // Both histograms are log base 2.
00136     if (hist.getBinSize() == -1 && m_binsize == -1) {
00137         for (int j = 0; j < hist.getData(0); j++) {
00138             add(0);
00139         }
00140 
00141         for (uint32_t i = 1; i < t_bins; i++) {
00142             for (int j = 0; j < hist.getData(i); j++) {
00143                 add(1<<(i-1));  // account for the + 1 index
00144             }
00145         }
00146     } else if (hist.getBinSize() >= 1 && m_binsize >= 1) {
00147         // Both the histogram are linear.
00148         // We are assuming that the two histograms have the same
00149         // minimum value that they can store.
00150 
00151         while (m_binsize > hist.getBinSize()) hist.doubleBinSize();
00152         while (hist.getBinSize() > m_binsize) doubleBinSize();
00153 
00154         assert(m_binsize == hist.getBinSize());
00155 
00156         for (uint32_t i = 0; i < t_bins; i++) {
00157             m_data[i] += hist.getData(i);
00158 
00159             if (m_data[i] > 0) m_largest_bin = i;
00160         }
00161     } else {
00162         fatal("Don't know how to combine log and linear histograms!");
00163     }

00166 // Computation of standard deviation of samples a1, a2, ... aN
00167 // variance = [SUM {ai^2} - (SUM {ai})^2/N]/(N-1)
00168 // std deviation equals square root of variance
00169 double
00170 Histogram::getStandardDeviation() const
00171 {
00172     if (m_count <= 1)
00173         return 0.0;
00174 
00175     double variance =
00176         (double)(m_sumSquaredSamples - m_sumSamples * m_sumSamples / m_count)
00177         / (m_count - 1);
00178     return sqrt(variance);
00179 }
