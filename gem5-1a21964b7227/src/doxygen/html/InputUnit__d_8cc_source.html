00066 void
00067 InputUnit_d::wakeup()
00068 {
00069     flit_d *t_flit;
00070     if (m_in_link->isReady(m_router->curCycle())) {
00071 
00072         t_flit = m_in_link->consumeLink();
00073         int vc = t_flit->get_vc();
00074 
00075         if ((t_flit->get_type() == HEAD_) ||
00076            (t_flit->get_type() == HEAD_TAIL_)) {
00077 
00078             assert(m_vcs[vc]->get_state() == IDLE_);
00079             // Do the route computation for this vc
00080             m_router->route_req(t_flit, this, vc);
00081 
00082             m_vcs[vc]->set_enqueue_time(m_router->curCycle());
00083         } else {
00084             t_flit->advance_stage(SA_, m_router->curCycle());
00085             m_router->swarb_req();
00086         }
00087         // write flit into input buffer
00088         m_vcs[vc]->insertFlit(t_flit);
00089 
00090         int vnet = vc/m_vc_per_vnet;
00091         // number of writes same as reads
00092         // any flit that is written will be read only once
00093         m_num_buffer_writes[vnet]++;
00094         m_num_buffer_reads[vnet]++;
00095     }
00096 }
