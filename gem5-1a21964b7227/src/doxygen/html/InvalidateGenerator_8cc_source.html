00052 bool
00053 InvalidateGenerator::initiate()
00054 {
00055     MasterPort* port;
00056     Request::Flags flags;
00057     PacketPtr pkt;
00058     Packet::Command cmd;
00059 
00060     // For simplicity, requests are assumed to be 1 byte-sized
00061     Request *req = new Request(m_address, 1, flags, masterId);
00062 
00063     //
00064     // Based on the current state, issue a load or a store
00065     //
00066     if (m_status == InvalidateGeneratorStatus_Load_Waiting) {
00067         DPRINTF(DirectedTest, "initiating read\n");
00068         cmd = MemCmd::ReadReq;
00069         port = m_directed_tester->getCpuPort(m_active_read_node);
00070         pkt = new Packet(req, cmd);
00071     } else if (m_status == InvalidateGeneratorStatus_Inv_Waiting) {
00072         DPRINTF(DirectedTest, "initiating invalidating write\n");
00073         cmd = MemCmd::WriteReq;
00074         port = m_directed_tester->getCpuPort(m_active_inv_node);
00075         pkt = new Packet(req, cmd);
00076     } else {
00077         panic("initiate was unexpectedly called\n");
00078     }
00079     uint8_t* dummyData = new uint8_t;
00080     *dummyData = 0;
00081     pkt->dataDynamic(dummyData);
00082 
00083     if (port->sendTimingReq(pkt)) {
00084         DPRINTF(DirectedTest, "initiating request - successful\n");
00085         if (m_status == InvalidateGeneratorStatus_Load_Waiting) {
00086             m_status = InvalidateGeneratorStatus_Load_Pending;
00087         } else {
00088             m_status = InvalidateGeneratorStatus_Inv_Pending;
00089         }
00090         return true;
00091     } else {
00092         // If the packet did not issue, must delete
00093         // Note: No need to delete the data, the packet destructor
00094         // will delete it
00095         delete pkt->req;
00096         delete pkt;
00097 
00098         DPRINTF(DirectedTest, "failed to issue request - sequencer not ready\n");
00099         return false;
00100     }
00101 }

00103 void 
00104 InvalidateGenerator::performCallback(uint32_t proc, Addr address)
00105 {
00106     assert(m_address == address);  
00107 
00108     if (m_status == InvalidateGeneratorStatus_Load_Pending) {
00109         assert(m_active_read_node == proc);
00110         m_active_read_node++;
00111         //
00112         // Once all cpus have the block in S state, issue the invalidate
00113         //
00114         if (m_active_read_node == m_num_cpus) {
00115             m_status = InvalidateGeneratorStatus_Inv_Waiting;
00116             m_active_read_node = 0;
00117         } else {
00118             m_status = InvalidateGeneratorStatus_Load_Waiting;
00119         }
00120     } else if (m_status == InvalidateGeneratorStatus_Inv_Pending) {
00121         assert(m_active_inv_node == proc);
00122         m_active_inv_node++;
00123         if (m_active_inv_node == m_num_cpus) {
00124             m_address += m_addr_increment_size;
00125             m_active_inv_node = 0;
00126         }
00127         //
00128         // Invalidate completed, send that info to the tester and restart
00129         // the cycle
00130         //
00131         m_directed_tester->incrementCycleCompletions();
00132         m_status = InvalidateGeneratorStatus_Load_Waiting;
00133     } 
00134     
00135 }
