00051 void
00052 MemCntrlProfiler::printStats(ostream& out) const
00053 {
00054     if (!m_memReq && !m_memRefresh) {
00055         out << "Memory Controller: " << m_description
00056             << " no stats recorded." << endl
00057             << endl
00058             << endl;
00059         return;
00060     }
00061 
00062     // if there's a memory controller at all
00063     uint64 total_stalls = m_memInputQ + m_memBankQ + m_memWaitCycles;
00064     double stallsPerReq = total_stalls * 1.0 / m_memReq;
00065     out << "Memory controller: " << m_description << ":" << endl;
00066 
00067     // does not include refreshes
00068     out << "  memory_total_requests: " << m_memReq << endl;
00069     out << "  memory_reads: " << m_memRead << endl;
00070     out << "  memory_writes: " << m_memWrite << endl;
00071     out << "  memory_refreshes: " << m_memRefresh << endl;
00072     out << "  memory_total_request_delays: " << total_stalls << endl;
00073     out << "  memory_delays_per_request: " << stallsPerReq << endl;
00074     out << "  memory_delays_in_input_queue: " << m_memInputQ << endl;
00075     out << "  memory_delays_behind_head_of_bank_queue: "
00076         << m_memBankQ << endl;
00077     out << "  memory_delays_stalled_at_head_of_bank_queue: "
00078         << m_memWaitCycles << endl;
00079 
00080     // Note: The following "memory stalls" entries are a breakdown of
00081     // the cycles which already showed up in m_memWaitCycles.  The
00082     // order is significant; it is the priority of attributing the
00083     // cycles.  For example, bank_busy is before arbitration because
00084     // if the bank was busy, we didn't even check arbitration.
00085     // Note: "not old enough" means that since we grouped waiting
00086     // heads-of-queues into batches to avoid starvation, a request in
00087     // a newer batch didn't try to arbitrate yet because there are
00088     // older requests waiting.
00089     out << "  memory_stalls_for_bank_busy: " << m_memBankBusy << endl;
00090     out << "  memory_stalls_for_random_busy: " << m_memRandBusy << endl;
00091     out << "  memory_stalls_for_anti_starvation: " << m_memNotOld << endl;
00092     out << "  memory_stalls_for_arbitration: " << m_memArbWait << endl;
00093     out << "  memory_stalls_for_bus: " << m_memBusBusy << endl;
00094     out << "  memory_stalls_for_tfaw: " << m_memTfawBusy << endl;
00095     out << "  memory_stalls_for_read_write_turnaround: "
00096         << m_memReadWriteBusy << endl;
00097     out << "  memory_stalls_for_read_read_turnaround: "
00098         << m_memDataBusBusy << endl;
00099     out << "  accesses_per_bank: ";
00100 
00101     for (int bank = 0; bank < m_memBankCount.size(); bank++) {
00102         out << m_memBankCount[bank] << "  ";
00103     }
00104     out << endl;
00105     out << endl;
00106 }
