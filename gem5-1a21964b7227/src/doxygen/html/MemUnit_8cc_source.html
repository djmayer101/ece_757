00090     const TechParameter* tp = m_tech_param_ptr;

00092     double total_cap = 0;
00093     // part 1: drain capacitance of pass transistors
00094     total_cap += tp->calc_draincap(Wmemcellr, TechParameter::NCH, 1)*num_read_port*num_data_end/2.0;
00095     total_cap += tp->calc_draincap(Wmemcellw, TechParameter::NCH, 1)*num_write_port;
00096 
00097     // has coefficient (1/2 * 2)
00098     // part 2: drain capacitance of memory cell
00099     total_cap += tp->calc_draincap(Wmemcella, TechParameter::NCH, 1) + tp->calc_draincap(Wmemcella*Wmemcellbscale, TechParameter::PCH, 1
00100             );
00101 
00102     // has coefficient (1/2 * 2)
00103     // part 3: gate capacitance of memory cell
00104     total_cap += tp->calc_gatecap(Wmemcella, 1) + tp->calc_gatecap(Wmemcella*Wmemcellbscale, 1);
