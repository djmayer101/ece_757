00079 bool
00080 MessageBuffer::areNSlotsAvailable(int n)
00081 {
00082 
00083     // fast path when message buffers have infinite size
00084     if (m_max_size == -1) {
00085         return true;
00086     }
00087 
00088     // determine my correct size for the current cycle
00089     // pop operations shouldn't effect the network's visible size
00090     // until next cycle, but enqueue operations effect the visible
00091     // size immediately
00092     int current_size = max(m_size_at_cycle_start, m_size);
00093     if (m_time_last_time_pop < m_receiver_ptr->curCycle()) {
00094         // no pops this cycle - m_size is correct
00095         current_size = m_size;
00096     } else {
00097         if (m_time_last_time_enqueue < m_receiver_ptr->curCycle()) {
00098             // no enqueues this cycle - m_size_at_cycle_start is correct
00099             current_size = m_size_at_cycle_start;
00100         } else {
00101             // both pops and enqueues occured this cycle - add new
00102             // enqueued msgs to m_size_at_cycle_start
00103             current_size = m_size_at_cycle_start+m_msgs_this_cycle;
00104         }
00105     }
00106 
00107     // now compare the new size with our max size
00108     if (current_size + n <= m_max_size) {
00109         return true;
00110     } else {
00111         DPRINTF(RubyQueue, "n: %d, current_size: %d, m_size: %d, "
00112                 "m_max_size: %d\n",
00113                 n, current_size, m_size, m_max_size);
00114         m_not_avail_count++;
00115         return false;
00116     }
00117 }

00184             if (m_last_arrival_time < current_time) {
00185                 m_last_arrival_time = current_time;

00199                   *this, m_name, current_time * m_receiver_ptr->clockPeriod(),
00200                   delta * m_receiver_ptr->clockPeriod(),

00202                   m_last_arrival_time * m_receiver_ptr->clockPeriod());
