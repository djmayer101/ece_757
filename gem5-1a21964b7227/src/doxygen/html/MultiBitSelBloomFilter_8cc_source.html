00168 int
00169 MultiBitSelBloomFilter::get_index(const Address& addr, int i)
00170 {
00171     // m_skip_bits is used to perform BitSelect after skipping some
00172     // bits. Used to simulate BitSel hashing on larger than cache-line
00173     // granularities
00174     uint64 x = (addr.getLineAddress()) >> m_skip_bits;
00175     int y = hash_bitsel(x, i, m_num_hashes, 30, m_filter_size_bits);
00176     //36-bit addresses, 6-bit cache lines
00177 
00178     if (isParallel) {
00179         return (y % m_par_filter_size) + i*m_par_filter_size;
00180     } else {
00181         return y % m_filter_size;
00182     }
00183 }

00186 MultiBitSelBloomFilter::hash_bitsel(uint64 value, int index, int jump,
00187                                     int maxBits, int numBits)
00188 {
00189     uint64 mask = 1;
00190     int result = 0;
00191     int bit, i;
00192 
00193     for (i = 0; i < numBits; i++) {
00194         bit = (index + jump*i) % maxBits;
00195         if (value & (mask << bit)) result += mask << i;
