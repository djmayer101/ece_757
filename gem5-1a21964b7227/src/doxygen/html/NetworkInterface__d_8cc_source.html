00126 NetworkInterface_d::flitisizeMessage(MsgPtr msg_ptr, int vnet)
00127 {
00128     NetworkMessage *net_msg_ptr = safe_cast<NetworkMessage *>(msg_ptr.get());
00129     NetDest net_msg_dest = net_msg_ptr->getInternalDestination();
00130 
00131     // gets all the destinations associated with this message.
00132     vector<NodeID> dest_nodes = net_msg_dest.getAllDest();
00133 
00134     // Number of flits is dependent on the link bandwidth available.
00135     // This is expressed in terms of bytes/cycle or the flit size
00136     int num_flits = (int) ceil((double) m_net_ptr->MessageSizeType_to_int(
00137         net_msg_ptr->getMessageSize())/m_net_ptr->getNiFlitSize());
00138 
00139     // loop to convert all multicast messages into unicast messages
00140     for (int ctr = 0; ctr < dest_nodes.size(); ctr++) {
00141 
00142         // this will return a free output virtual channel
00143         int vc = calculateVC(vnet);
00144 
00145         if (vc == -1) {
00146             return false ;
00147         }
00148         MsgPtr new_msg_ptr = msg_ptr->clone();
00149         NodeID destID = dest_nodes[ctr];
00150 
00151         NetworkMessage *new_net_msg_ptr =
00152             safe_cast<NetworkMessage *>(new_msg_ptr.get());
00153         if (dest_nodes.size() > 1) {
00154             NetDest personal_dest;
00155             for (int m = 0; m < (int) MachineType_NUM; m++) {
00156                 if ((destID >= MachineType_base_number((MachineType) m)) &&
00157                     destID < MachineType_base_number((MachineType) (m+1))) {
00158                     // calculating the NetDest associated with this destID
00159                     personal_dest.clear();
00160                     personal_dest.add((MachineID) {(MachineType) m, (destID -
00161                         MachineType_base_number((MachineType) m))});
00162                     new_net_msg_ptr->getInternalDestination() = personal_dest;
00163                     break;
00164                 }
00165             }
00166             net_msg_dest.removeNetDest(personal_dest);
00167             // removing the destination from the original message to reflect
00168             // that a message with this particular destination has been
00169             // flitisized and an output vc is acquired
00170             net_msg_ptr->getInternalDestination().removeNetDest(personal_dest);
00171         }
00172 
00173         for (int i = 0; i < num_flits; i++) {
00174             m_net_ptr->increment_injected_flits(vnet);
00175             flit_d *fl = new flit_d(i, vc, vnet, num_flits, new_msg_ptr,
00176                 m_net_ptr->curCycle());
00177 
00178             fl->set_delay(m_net_ptr->curCycle() -
00179                           m_net_ptr->ticksToCycles(msg_ptr->getTime()));
00180             m_ni_buffers[vc]->insert(fl);
00181         }
00182 
00183         m_ni_enqueue_time[vc] = m_net_ptr->curCycle();
00184         m_out_vc_state[vc]->setState(ACTIVE_, m_net_ptr->curCycle());
00185     }
00186     return true ;

00190 int
00191 NetworkInterface_d::calculateVC(int vnet)
00192 {
00193         for (int i = 0; i < m_vc_per_vnet; i++) {
00194                 int delta = m_vc_allocator[vnet];
00195                 m_vc_allocator[vnet]++;
00196                 if(m_vc_allocator[vnet] == m_vc_per_vnet)
00197                         m_vc_allocator[vnet] = 0;
00198 
00199                 if (m_out_vc_state[(vnet*m_vc_per_vnet) + delta]->isInState(
00200                     IDLE_, m_net_ptr->curCycle())) {
00201                         return ((vnet*m_vc_per_vnet) + delta);
00202                 }
00203         }
00204         return -1;

00207 /*
00208  * The NI wakeup checks whether there are any ready messages in the protocol
00209  * buffer. If yes, it picks that up, flitisizes it into a number of flits and
00210  * puts it into an output buffer and schedules the output link. On a wakeup
00211  * it also checks whether there are flits in the input link. If yes, it picks
00212  * them up and if the flit is a tail, the NI inserts the corresponding message
00213  * into the protocol buffer. It also checks for credits being sent by the
00214  * downstream router.

00217 void
00218 NetworkInterface_d::wakeup()
00219 {
00220     DPRINTF(RubyNetwork, "m_id: %d woke up at time: %lld",
00221             m_id, m_net_ptr->curCycle());
00222 
00223     MsgPtr msg_ptr;
00224 
00225     // Checking for messages coming from the protocol
00226     // can pick up a message/cycle for each virtual net
00227     for (int vnet = 0; vnet < m_virtual_networks; vnet++) {
00228         while (inNode_ptr[vnet]->isReady()) { // Is there a message waiting
00229             msg_ptr = inNode_ptr[vnet]->peekMsgPtr();
00230             if (flitisizeMessage(msg_ptr, vnet)) {
00231                 inNode_ptr[vnet]->pop();
00232             } else {
00233                 break;
00234             }
00235         }
00236     }
00237 
00238     scheduleOutputLink();
00239     checkReschedule();
00240 
00241     /*********** Picking messages destined for this NI **********/
00242 
00243     if (inNetLink->isReady(m_net_ptr->curCycle())) {
00244         flit_d *t_flit = inNetLink->consumeLink();
00245         bool free_signal = false;
00246         if (t_flit->get_type() == TAIL_ || t_flit->get_type() == HEAD_TAIL_) {
00247             free_signal = true;
00248 
00249             outNode_ptr[t_flit->get_vnet()]->enqueue(
00250                 t_flit->get_msg_ptr(), Cycles(1));
00251         }
00252         // Simply send a credit back since we are not buffering
00253         // this flit in the NI
00254         flit_d *credit_flit = new flit_d(t_flit->get_vc(), free_signal,
00255                                          m_net_ptr->curCycle());
00256         creditQueue->insert(credit_flit);
00257         m_ni_credit_link->scheduleEvent(Cycles(1));
00258 
00259         int vnet = t_flit->get_vnet();
00260         m_net_ptr->increment_received_flits(vnet);
00261         Cycles network_delay = m_net_ptr->curCycle() -
00262                                t_flit->get_enqueue_time();
00263         Cycles queueing_delay = t_flit->get_delay();
00264 
00265         m_net_ptr->increment_network_latency(network_delay, vnet);
00266         m_net_ptr->increment_queueing_latency(queueing_delay, vnet);
00267         delete t_flit;
00268     }
00269 
00270     /****************** Checking for credit link *******/
00271 
00272     if (m_credit_link->isReady(m_net_ptr->curCycle())) {
00273         flit_d *t_flit = m_credit_link->consumeLink();
00274         m_out_vc_state[t_flit->get_vc()]->increment_credit();
00275         if (t_flit->is_free_signal()) {
00276             m_out_vc_state[t_flit->get_vc()]->setState(IDLE_,
00277                 m_net_ptr->curCycle());
00278         }
00279         delete t_flit;
00280     }

00289 void
00290 NetworkInterface_d::scheduleOutputLink()
00291 {
00292     int vc = m_vc_round_robin;
00293     m_vc_round_robin++;
00294     if (m_vc_round_robin == m_num_vcs)
00295         m_vc_round_robin = 0;
00296 
00297     for (int i = 0; i < m_num_vcs; i++) {
00298         vc++;
00299         if (vc == m_num_vcs)
00300             vc = 0;
00301 
00302         // model buffer backpressure
00303         if (m_ni_buffers[vc]->isReady(m_net_ptr->curCycle()) &&
00304             m_out_vc_state[vc]->has_credits()) {
00305 
00306             bool is_candidate_vc = true;
00307             int t_vnet = get_vnet(vc);
00308             int vc_base = t_vnet * m_vc_per_vnet;
00309 
00310             if (m_net_ptr->isVNetOrdered(t_vnet)) {
00311                 for (int vc_offset = 0; vc_offset < m_vc_per_vnet;
00312                      vc_offset++) {
00313                     int t_vc = vc_base + vc_offset;
00314                     if (m_ni_buffers[t_vc]->isReady(m_net_ptr->curCycle())) {
00315                         if (m_ni_enqueue_time[t_vc] < m_ni_enqueue_time[vc]) {
00316                             is_candidate_vc = false;
00317                             break;
00318                         }
00319                     }
00320                 }
00321             }
00322             if (!is_candidate_vc)
00323                 continue;
00324 
00325             m_out_vc_state[vc]->decrement_credit();
00326             // Just removing the flit
00327             flit_d *t_flit = m_ni_buffers[vc]->getTopFlit();
00328             t_flit->set_time(m_net_ptr->curCycle() + Cycles(1));
00329             outSrcQueue->insert(t_flit);
00330             // schedule the out link
00331             outNetLink->scheduleEvent(Cycles(1));
00332 
00333             if (t_flit->get_type() == TAIL_ ||
00334                t_flit->get_type() == HEAD_TAIL_) {
00335                 m_ni_enqueue_time[vc] = INFINITE_;
00336             }
00337             return;
00338         }
00339     }
00340 }
