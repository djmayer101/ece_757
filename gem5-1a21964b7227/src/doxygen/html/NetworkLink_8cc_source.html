00134 void
00135 NetworkLink::wakeup()
00136 {
00137     if (!link_srcQueue->isReady(curCycle()))
00138         return;
00139 
00140     flit *t_flit = link_srcQueue->getTopFlit();
00141     t_flit->set_time(curCycle() + m_latency);
00142     linkBuffer->insert(t_flit);
00143     link_consumer->scheduleEvent(m_latency);
00144     m_link_utilized++;
00145     m_vc_load[t_flit->get_vc()]++;
00146 }
