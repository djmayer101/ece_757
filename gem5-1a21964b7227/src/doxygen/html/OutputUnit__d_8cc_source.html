00069     if (m_credit_link->isReady(m_router->curCycle())) {
00070         flit_d *t_flit = m_credit_link->consumeLink();
00071         int out_vc = t_flit->get_vc();
00072         m_outvc_state[out_vc]->increment_credit();
00073         m_router->update_incredit(m_outvc_state[out_vc]->get_inport(),
00074                                   m_outvc_state[out_vc]->get_invc(),
00075                                   m_outvc_state[out_vc]->get_credit_count());
00076 
00077         if (t_flit->is_free_signal())
00078             set_vc_state(IDLE_, out_vc, m_router->curCycle());
00079 
00080         delete t_flit;
00081     }
00082 }
