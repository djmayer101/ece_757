00060 double RRArbiter::calc_dynamic_energy(double num_req_, bool is_max_) const
00061 {
00062     if (num_req_ > m_req_width)
00063     {
00064         cerr << "WARNING: (num_req_ > m_req_width). Set num_req_ = m_req_width" << endl;
00065         num_req_ = m_req_width;
00066     }
00067 
00068     double num_grant;
00069     if (num_req_ >= 1) num_grant = 1;
00070     else if (num_req_) num_grant = 1.0 / ceil(1.0/num_req_);
00071     else num_grant = 0;
00072 
00073     double e_atomic;
00074     double e_arb = 0;
00075 
00076     e_atomic = m_e_chg_req*num_req_;
00077     e_arb += e_atomic;
00078 
00079     e_atomic = m_e_chg_grant*num_grant;
00080     e_arb += e_atomic;
00081 
00082     // assume carry signal propagates half length in average case */
00083     // carry does not propagate in maximum case, i.e. all carrys go down */
00084     e_atomic = m_e_chg_carry*m_req_width*(is_max_? 1:0.5)*num_grant;
00085     e_arb += e_atomic;
00086 
00087     e_atomic = m_e_chg_carry_in*(m_req_width*(is_max_? 1:0.5)-1)*num_grant;
00088     e_arb += e_atomic;
00089 
00090     // priority register
00091     e_atomic = m_ff_ptr->get_e_switch()*2*num_grant;
00092     e_arb += e_atomic;
00093 
00094     e_atomic = m_ff_ptr->get_e_keep_0()*(m_req_width-2*num_grant);
00095     e_arb += e_atomic;
00096 
00097     e_atomic = m_ff_ptr->get_e_clock()*m_req_width;
00098     e_arb += e_atomic;
00099 
00100     return e_arb;
00101 }

00105     double e_factor = m_tech_param_ptr->get_EnergyFactor();
00106 
00107     m_e_chg_req = calc_req_cap()/2*e_factor;

00120 // switch cap of request signal (round robin arbiter)
00121 double RRArbiter::calc_req_cap()
00122 {
00123     double total_cap = 0;
00124 
00125     // part 1: gate cap of 2 NOR gates
00126     // FIXME: need actual size
00127     double WdecNORn = m_tech_param_ptr->get_WdecNORn();
00128     double WdecNORp = m_tech_param_ptr->get_WdecNORp();
00129     total_cap += 2*m_tech_param_ptr->calc_gatecap(WdecNORn+WdecNORp, 0);
00130 
00131     // part 2: inverter
00132     // FIXME: need actual size
00133     double Wdecinvn = m_tech_param_ptr->get_Wdecinvn();
00134     double Wdecinvp = m_tech_param_ptr->get_Wdecinvp();
00135     total_cap += m_tech_param_ptr->calc_draincap(Wdecinvn, TechParameter::NCH, 1)
00136         + m_tech_param_ptr->calc_draincap(Wdecinvp, TechParameter::PCH, 1)
00137         + m_tech_param_ptr->calc_gatecap(Wdecinvn+Wdecinvp, 0);
00138 
00139     // part 3: wire cap
00140     double Cmetal = m_tech_param_ptr->get_Cmetal();
00141     total_cap += m_len_in_wire*Cmetal;
00142 
00143     return total_cap;
