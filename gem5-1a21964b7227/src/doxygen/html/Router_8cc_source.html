00263 Router::routeCompute(flit *m_flit, int inport)
00264 {
00265     int invc = m_flit->get_vc();
00266     int outport = m_in_vc_state[inport][invc]->get_outport();
00267     int outvc = m_in_vc_state[inport][invc]->get_outvc();
00268 
00269     assert(m_net_ptr->getNumPipeStages() >= 1);
00270 
00271     // Subtract 1 as 1 cycle will be consumed in scheduling the output link
00272     m_flit->set_time(curCycle() + Cycles((m_net_ptr->getNumPipeStages() - 1)));
00273     m_flit->set_vc(outvc);
00274     m_router_buffers[outport][outvc]->insert(m_flit);
00275 
00276     if (m_net_ptr->getNumPipeStages() > 1)
00277         scheduleEvent(Cycles(m_net_ptr->getNumPipeStages() - 1));
00278 
00279     if ((m_flit->get_type() == HEAD_) || (m_flit->get_type() == HEAD_TAIL_)) {
00280         NetworkMessage *nm =
00281             safe_cast<NetworkMessage*>(m_flit->get_msg_ptr().get());
00282         NetDest destination = nm->getInternalDestination();
00283 
00284         if (m_net_ptr->getNumPipeStages() > 1) {
00285             m_out_vc_state[outport][outvc]->setState(VC_AB_, curCycle() +
00286                                                      Cycles(1));
00287             m_out_link[outport]->request_vc_link(outvc, destination,
00288                                                  curCycle() + Cycles(1));
00289         } else {
00290             m_out_vc_state[outport][outvc]->setState(VC_AB_, curCycle());
00291             m_out_link[outport]->request_vc_link(outvc, destination,
00292                 curCycle());
00293         }
00294     }
00295 
00296     if ((m_flit->get_type() == TAIL_) || (m_flit->get_type() == HEAD_TAIL_)) {
00297         m_in_vc_state[inport][invc]->setState(IDLE_, curCycle() + Cycles(1));

00303 Router::wakeup()
00304 {
00305     flit *t_flit;
00306 
00307     // This is for round-robin scheduling of incoming ports
00308     int incoming_port = m_round_robin_start;
00309     m_round_robin_start++;
00310     if (m_round_robin_start >= m_in_link.size()) {
00311         m_round_robin_start = 0;
00312     }
00313 
00314     for (int port = 0; port < m_in_link.size(); port++) {
00315         // Round robin scheduling
00316         incoming_port++;
00317         if (incoming_port >= m_in_link.size())
00318             incoming_port = 0;
00319 
00320         // checking the incoming link
00321         if (m_in_link[incoming_port]->isReady()) {
00322             DPRINTF(RubyNetwork, "m_id: %d, Time: %lld\n", m_id, curCycle());
00323             t_flit = m_in_link[incoming_port]->peekLink();
00324             routeCompute(t_flit, incoming_port);
00325             m_in_link[incoming_port]->consumeLink();
00326         }
00327     }
00328     scheduleOutputLinks();
00329     checkReschedule(); // This is for flits lying in the router buffers
00330     vc_arbitrate();
00331     check_arbiter_reschedule();

00350

00352             vc_tolookat++;

00354                 vc_tolookat = 0;

00360                    curCycle()) &&
