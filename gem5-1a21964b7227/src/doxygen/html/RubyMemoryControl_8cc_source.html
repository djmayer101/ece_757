00030 /*
00031  * Description:  This module simulates a basic DDR-style memory controller
00032  * (and can easily be extended to do FB-DIMM as well).
00033  *
00034  * This module models a single channel, connected to any number of
00035  * DIMMs with any number of ranks of DRAMs each.  If you want multiple
00036  * address/data channels, you need to instantiate multiple copies of
00037  * this module.
00038  *
00039  * Each memory request is placed in a queue associated with a specific
00040  * memory bank.  This queue is of finite size; if the queue is full
00041  * the request will back up in an (infinite) common queue and will
00042  * effectively throttle the whole system.  This sort of behavior is
00043  * intended to be closer to real system behavior than if we had an
00044  * infinite queue on each bank.  If you want the latter, just make
00045  * the bank queues unreasonably large.
00046  *
00047  * The head item on a bank queue is issued when all of the
00048  * following are true:
00049  *   the bank is available
00050  *   the address path to the DIMM is available
00051  *   the data path to or from the DIMM is available
00052  *
00053  * Note that we are not concerned about fixed offsets in time.  The bank
00054  * will not be used at the same moment as the address path, but since
00055  * there is no queue in the DIMM or the DRAM it will be used at a constant
00056  * number of cycles later, so it is treated as if it is used at the same
00057  * time.
00058  *
00059  * We are assuming closed bank policy; that is, we automatically close
00060  * each bank after a single read or write.  Adding an option for open
00061  * bank policy is for future work.
00062  *
00063  * We are assuming "posted CAS"; that is, we send the READ or WRITE
00064  * immediately after the ACTIVATE.  This makes scheduling the address
00065  * bus trivial; we always schedule a fixed set of cycles.  For DDR-400,
00066  * this is a set of two cycles; for some configurations such as
00067  * DDR-800 the parameter tRRD forces this to be set to three cycles.
00068  *
00069  * We assume a four-bit-time transfer on the data wires.  This is
00070  * the minimum burst length for DDR-2.  This would correspond
00071  * to (for example) a memory where each DIMM is 72 bits wide
00072  * and DIMMs are ganged in pairs to deliver 64 bytes at a shot.
00073  * This gives us the same occupancy on the data wires as on the
00074  * address wires (for the two-address-cycle case).
00075  *
00076  * The only non-trivial scheduling problem is the data wires.
00077  * A write will use the wires earlier in the operation than a read
00078  * will; typically one cycle earlier as seen at the DRAM, but earlier
00079  * by a worst-case round-trip wire delay when seen at the memory controller.
00080  * So, while reads from one rank can be scheduled back-to-back
00081  * every two cycles, and writes (to any rank) scheduled every two cycles,
00082  * when a read is followed by a write we need to insert a bubble.
00083  * Furthermore, consecutive reads from two different ranks may need
00084  * to insert a bubble due to skew between when one DRAM stops driving the
00085  * wires and when the other one starts.  (These bubbles are parameters.)
00086  *
00087  * This means that when some number of reads and writes are at the
00088  * heads of their queues, reads could starve writes, and/or reads
00089  * to the same rank could starve out other requests, since the others
00090  * would never see the data bus ready.
00091  * For this reason, we have implemented an anti-starvation feature.
00092  * A group of requests is marked "old", and a counter is incremented
00093  * each cycle as long as any request from that batch has not issued.
00094  * if the counter reaches twice the bank busy time, we hold off any
00095  * newer requests until all of the "old" requests have issued.
00096  *
00097  * We also model tFAW.  This is an obscure DRAM parameter that says
00098  * that no more than four activate requests can happen within a window
00099  * of a certain size.  For most configurations this does not come into play,
00100  * or has very little effect, but it could be used to throttle the power
00101  * consumption of the DRAM.  In this implementation (unlike in a DRAM
00102  * data sheet) TFAW is measured in memory bus cycles; i.e. if TFAW = 16
00103  * then no more than four activates may happen within any 16 cycle window.
00104  * Refreshes are included in the activates.
00105  *
00106  */

00587         if (m_bankBusyCounter[bank] > 0) m_bankBusyCounter[bank]--;

00617         anyOld |= m_oldRequest[bank];

00621             if (!m_bankQueues[bank].empty()) m_oldRequest[bank] = 1;
00622         }

00624     }

00627     if (m_mem_random_arbitrate) {

00629     }
00630 
00631     // For each channel, scan round-robin, and pick an old, ready
00632     // request and issue it.  Treat a refresh request as if it were at
00633     // the head of its bank queue.  After we issue something, keep
00634     // scanning the queues just to gather statistics about how many
00635     // are waiting.  If in mem_fixed_delay mode, we can issue more
00636     // than one request per cycle.
00637     int queueHeads = 0;
00638     int banksIssued = 0;
00639     for (int i = 0; i < m_total_banks; i++) {
00640         m_roundRobin++;
00641         if (m_roundRobin >= m_total_banks) m_roundRobin = 0;
00642         issueRefresh(m_roundRobin);
00643         int qs = m_bankQueues[m_roundRobin].size();
00644         if (qs > 1) {
00645             m_profiler_ptr->profileMemBankQ(qs-1);
00646         }
00647         if (qs > 0) {
00648             // we're not idle if anything is queued
00649             m_idleCount = IDLECOUNT_MAX_VALUE;
00650             queueHeads++;
00651             if (queueReady(m_roundRobin)) {
00652                 issueRequest(m_roundRobin);
00653                 banksIssued++;
00654                 if (m_mem_fixed_delay) {
00655                     m_profiler_ptr->profileMemWaitCycles(m_mem_fixed_delay);
00656                 }
00657             }
00658         }
00659     }
00660 
00661     // memWaitCycles is a redundant catch-all for the specific
00662     // counters in queueReady
00663     m_profiler_ptr->profileMemWaitCycles(queueHeads - banksIssued);
00664 
00665     // Check input queue and move anything to bank queues if not full.
00666     // Since this is done here at the end of the cycle, there will
00667     // always be at least one cycle of latency in the bank queue.  We
00668     // deliberately move at most one request per cycle (to simulate
00669     // typical hardware).  Note that if one bank queue fills up, other
00670     // requests can get stuck behind it here.
00671     if (!m_input_queue.empty()) {
00672         // we're not idle if anything is pending
00673         m_idleCount = IDLECOUNT_MAX_VALUE;
00674         MemoryNode req = m_input_queue.front();
00675         int bank = getBank(req.m_addr);
00676         if (m_bankQueues[bank].size() < m_bank_queue_size) {
00677             m_input_queue.pop_front();
00678             m_bankQueues[bank].push_back(req);
00679         }
00680         m_profiler_ptr->profileMemInputQ(m_input_queue.size());
00681     }
