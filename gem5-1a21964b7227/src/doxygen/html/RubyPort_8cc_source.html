00294 void
00295 RubyPort::ruby_hit_callback(PacketPtr pkt)
00296 {
00297     // Retrieve the request port from the sender State
00298     RubyPort::SenderState *senderState =
00299         safe_cast<RubyPort::SenderState *>(pkt->senderState);
00300     M5Port *port = senderState->port;
00301     assert(port != NULL);
00302 
00303     // pop the sender state from the packet
00304     pkt->senderState = senderState->predecessor;
00305     delete senderState;
00306 
00307     port->hitCallback(pkt);
00308 
00309     //
00310     // If we had to stall the M5Ports, wake them up because the sequencer
00311     // likely has free resources now.
00312     //
00313     if (waitingOnSequencer) {
00314         //
00315         // Record the current list of ports to retry on a temporary list before
00316         // calling sendRetry on those ports.  sendRetry will cause an 
00317         // immediate retry, which may result in the ports being put back on the
00318         // list. Therefore we want to clear the retryList before calling
00319         // sendRetry.
00320         //
00321         std::list<M5Port*> curRetryList(retryList);
00322 
00323         retryList.clear();
00324         waitingOnSequencer = false;
00325         
00326         for (std::list<M5Port*>::iterator i = curRetryList.begin();
00327              i != curRetryList.end(); ++i) {
00328             DPRINTF(RubyPort,
00329                     "Sequencer may now be free.  SendRetry to port %s\n",
00330                     (*i)->name());
00331             (*i)->onRetryList(false);
00332             (*i)->sendRetry();
00333         }
00334     }
00335 
00336     testDrainComplete();
00337 }

00415 void
00416 RubyPort::M5Port::hitCallback(PacketPtr pkt)
00417 {
00418     bool needsResponse = pkt->needsResponse();
00419 
00420     //
00421     // Unless specified at configuraiton, all responses except failed SC 
00422     // and Flush operations access M5 physical memory.
00423     //
00424     bool accessPhysMem = access_phys_mem;
00425 
00426     if (pkt->isLLSC()) {
00427         if (pkt->isWrite()) {
00428             if (pkt->req->getExtraData() != 0) {
00429                 //
00430                 // Successful SC packets convert to normal writes
00431                 //
00432                 pkt->convertScToWrite();
00433             } else {
00434                 //
00435                 // Failed SC packets don't access physical memory and thus
00436                 // the RubyPort itself must convert it to a response.
00437                 //
00438                 accessPhysMem = false;
00439             }
00440         } else {
00441             //
00442             // All LL packets convert to normal loads so that M5 PhysMem does
00443             // not lock the blocks.
00444             //
00445             pkt->convertLlToRead();
00446         }
00447     }
00448 
00449     //
00450     // Flush requests don't access physical memory
00451     //
00452     if (pkt->isFlush()) {
00453         accessPhysMem = false;
00454     }
00455 
00456     DPRINTF(RubyPort, "Hit callback needs response %d\n", needsResponse);
00457 
00458     if (accessPhysMem) {
00459         ruby_port->system->getPhysMem().access(pkt);
00460     } else if (needsResponse) {
00461         pkt->makeResponse();
00462     }
00463 
00464     // turn packet around to go back to requester if response expected
00465     if (needsResponse) {
00466         DPRINTF(RubyPort, "Sending packet back over port\n");
00467         // send next cycle
00468         schedTimingResp(pkt, curTick() + g_system_ptr->clockPeriod());
00469     } else {
00470         delete pkt;
00471     }
00472     DPRINTF(RubyPort, "Hit callback done!\n");
00473 }
