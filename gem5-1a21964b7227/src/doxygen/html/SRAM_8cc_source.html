00107 double SRAM::calc_e_read(
00108         bool is_max_
00109         ) const
00110 {
00111     double e_atomic;
00112     double e_read = 0;
00113 
00114     // decoder
00115     if (m_rowdec_unit_ptr != NULL)
00116     {
00117         e_atomic = m_rowdec_unit_ptr->get_e_chg_addr()*m_rowdec_unit_ptr->get_dec_width()*(is_max_? 1:0.5);
00118         e_atomic += m_rowdec_unit_ptr->get_e_chg_output();
00119         // assume all 1st-level decoders change output
00120         e_atomic += m_rowdec_unit_ptr->get_e_chg_l1()*m_rowdec_unit_ptr->get_num_in_2nd();
00121         e_read += e_atomic;
00122     }
00123 
00124     //wordline
00125     e_atomic = m_wl_unit_ptr->get_e_read();
00126     e_read += e_atomic;
00127 
00128     //bitline pre
00129     e_atomic = m_bl_pre_unit_ptr->get_e_charge_gate()*m_line_width;
00130     if (m_num_data_end == 2)
00131     {
00132         e_atomic += m_bl_pre_unit_ptr->get_e_charge_drain()*m_line_width;
00133     }
00134     else
00135     {
00136         e_atomic += m_bl_pre_unit_ptr->get_e_charge_drain()*m_line_width*(is_max_? 1:0.5);
00137     }
00138     e_read += e_atomic;
00139 
00140     //bitline
00141     if (m_num_data_end == 2)
00142     {
00143         e_atomic = m_bl_unit_ptr->get_e_col_read()*m_line_width;
00144     }
00145     else
00146     {
00147         e_atomic = m_bl_unit_ptr->get_e_col_read()*m_line_width*(is_max_? 1:0.5);
00148     }
00149     e_read += e_atomic;
00150 
00151     if (m_num_data_end == 2)
00152     {
00153         e_atomic = m_amp_unit_ptr->get_e_access()*m_line_width;
00154         e_read += e_atomic;
00155     }
00156 
00157     if (m_outdrv_unit_ptr != NULL)
00158     {
00159         e_atomic = m_outdrv_unit_ptr->get_e_select();
00160 
00161         e_atomic += m_outdrv_unit_ptr->get_e_chg_data()*m_line_width*(is_max_? 1:0.5);
00162 
00163         //assume 1 and 0 are uniformly distributed
00164         if ((m_outdrv_unit_ptr->get_e_out_1() >= m_outdrv_unit_ptr->get_e_out_0()) || (!is_max_))
00165         {
00166             e_atomic += m_outdrv_unit_ptr->get_e_out_1()*m_line_width*(is_max_? 1:0.5);
00167         }
00168         if ((m_outdrv_unit_ptr->get_e_out_1() < m_outdrv_unit_ptr->get_e_out_0()) || (!is_max_))
00169         {
00170             e_atomic += m_outdrv_unit_ptr->get_e_out_0()*m_line_width*(is_max_? 1:0.5);
00171         }
00172 
00173         e_read += e_atomic;
00174     }
