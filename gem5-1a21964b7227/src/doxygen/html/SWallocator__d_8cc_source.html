00088 void
00089 SWallocator_d::arbitrate_inports()
00090 {
00091     // First do round robin arbitration on a set of input vc requests
00092     for (int inport = 0; inport < m_num_inports; inport++) {
00093         int invc = m_round_robin_inport[inport];
00094 
00095         // Select next round robin vc candidate within valid vnet
00096         int next_round_robin_invc = invc;
00097         do {
00098             next_round_robin_invc++;
00099 
00100             if (next_round_robin_invc >= m_num_vcs)
00101                 next_round_robin_invc = 0;
00102         } while (!((m_router->get_net_ptr())->validVirtualNetwork(
00103                     get_vnet(next_round_robin_invc))));
00104 
00105         m_round_robin_inport[inport] = next_round_robin_invc;
00106 
00107         for (int invc_iter = 0; invc_iter < m_num_vcs; invc_iter++) {
00108             invc++;
00109             if (invc >= m_num_vcs)
00110                 invc = 0;
00111 
00112             if (!((m_router->get_net_ptr())->validVirtualNetwork(
00113                 get_vnet(invc))))
00114                 continue;
00115 
00116             if (m_input_unit[inport]->need_stage(invc, ACTIVE_, SA_,
00117                                                  m_router->curCycle()) &&
00118                 m_input_unit[inport]->has_credits(invc)) {
00119 
00120                 if (is_candidate_inport(inport, invc)) {
00121                     int outport = m_input_unit[inport]->get_route(invc);
00122                     m_local_arbiter_activity++;
00123                     m_port_req[outport][inport] = true;
00124                     m_vc_winners[outport][inport]= invc;
00125                     break; // got one vc winner for this port
00126                 }
00127             }
00128         }
00129     }

00132 bool
00133 SWallocator_d::is_candidate_inport(int inport, int invc)
00134 {
00135     int outport = m_input_unit[inport]->get_route(invc);
00136     int t_enqueue_time = m_input_unit[inport]->get_enqueue_time(invc);
00137     int t_vnet = get_vnet(invc);
00138     int vc_base = t_vnet*m_vc_per_vnet;
00139     if ((m_router->get_net_ptr())->isVNetOrdered(t_vnet)) {
00140         for (int vc_offset = 0; vc_offset < m_vc_per_vnet; vc_offset++) {
00141             int temp_vc = vc_base + vc_offset;
00142             if (m_input_unit[inport]->need_stage(temp_vc, ACTIVE_, SA_,
00143                                                  m_router->curCycle()) &&
00144                (m_input_unit[inport]->get_route(temp_vc) == outport) &&
00145                (m_input_unit[inport]->get_enqueue_time(temp_vc) <
00146                     t_enqueue_time)) {
00147                 return false;
00148                 break;
00149             }
00150         }
00151     }
00152     return true;
00153 }

00156 void
00157 SWallocator_d::arbitrate_outports()
00158 {
00159     // Now there are a set of input vc requests for output vcs.
00160     // Again do round robin arbitration on these requests
00161     for (int outport = 0; outport < m_num_outports; outport++) {
00162         int inport = m_round_robin_outport[outport];
00163         m_round_robin_outport[outport]++;
00164 
00165         if (m_round_robin_outport[outport] >= m_num_outports)
00166             m_round_robin_outport[outport] = 0;
00167 
00168         for (int inport_iter = 0; inport_iter < m_num_inports; inport_iter++) {
00169             inport++;
00170             if (inport >= m_num_inports)
00171                 inport = 0;
00172 
00173             // inport has a request this cycle for outport:
00174             if (m_port_req[outport][inport]) {
00175                 m_port_req[outport][inport] = false;
00176                 int invc = m_vc_winners[outport][inport];
00177                 int outvc = m_input_unit[inport]->get_outvc(invc);
00178 
00179                 // remove flit from Input Unit
00180                 flit_d *t_flit = m_input_unit[inport]->getTopFlit(invc);
00181                 t_flit->advance_stage(ST_, m_router->curCycle());
00182                 t_flit->set_vc(outvc);
00183                 t_flit->set_outport(outport);
00184                 t_flit->set_time(m_router->curCycle() + Cycles(1));
00185 
00186                 m_output_unit[outport]->decrement_credit(outvc);
00187                 m_router->update_sw_winner(inport, t_flit);
00188                 m_global_arbiter_activity++;
00189 
00190                 if ((t_flit->get_type() == TAIL_) ||
00191                     t_flit->get_type() == HEAD_TAIL_) {
00192 
00193                     // Send a credit back
00194                     // along with the information that this VC is now idle
00195                     m_input_unit[inport]->increment_credit(invc, true,
00196                         m_router->curCycle());
00197 
00198                     // This Input VC should now be empty
00199                     assert(m_input_unit[inport]->isReady(invc,
00200                         m_router->curCycle()) == false);
00201 
00202                     m_input_unit[inport]->set_vc_state(IDLE_, invc,
00203                         m_router->curCycle());
00204                     m_input_unit[inport]->set_enqueue_time(invc,
00205                         Cycles(INFINITE_));
00206                 } else {
00207                     // Send a credit back
00208                     // but do not indicate that the VC is idle
00209                     m_input_unit[inport]->increment_credit(invc, false,
00210                         m_router->curCycle());
00211                 }
00212                 break; // got a in request for this outport
00213             }
00214         }
00215     }
