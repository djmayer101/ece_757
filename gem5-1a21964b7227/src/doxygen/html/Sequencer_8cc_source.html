00323 bool
00324 Sequencer::handleLlsc(const Address& address, SequencerRequest* request)
00325 {
00326     //
00327     // The success flag indicates whether the LLSC operation was successful.
00328     // LL ops will always succeed, but SC may fail if the cache line is no
00329     // longer locked.
00330     //
00331     bool success = true;
00332     if (request->m_type == RubyRequestType_Store_Conditional) {
00333         if (!m_dataCache_ptr->isLocked(address, m_version)) {
00334             //
00335             // For failed SC requests, indicate the failure to the cpu by
00336             // setting the extra data to zero.
00337             //
00338             request->pkt->req->setExtraData(0);
00339             success = false;
00340         } else {
00341             //
00342             // For successful SC requests, indicate the success to the cpu by
00343             // setting the extra data to one.  
00344             //
00345             request->pkt->req->setExtraData(1);
00346         }
00347         //
00348         // Independent of success, all SC operations must clear the lock
00349         //
00350         m_dataCache_ptr->clearLocked(address);
00351     } else if (request->m_type == RubyRequestType_Load_Linked) {
00352         //
00353         // Note: To fully follow Alpha LLSC semantics, should the LL clear any
00354         // previously locked cache lines?
00355         //
00356         m_dataCache_ptr->setLocked(address, m_version);
00357     } else if ((m_dataCache_ptr->isTagPresent(address)) &&
00358                (m_dataCache_ptr->isLocked(address, m_version))) {
00359         //
00360         // Normal writes should clear the locked address
00361         //
00362         m_dataCache_ptr->clearLocked(address);
00363     }
00364     return success;
00365 }

00585         //
00586         // Alpha LL/SC instructions need to be handled carefully by the cache
00587         // coherence protocol to ensure they follow the proper semantics. In
00588         // particular, by identifying the operations as atomic, the protocol
00589         // should understand that migratory sharing optimizations should not
00590         // be performed (i.e. a load between the LL and SC should not steal

00597             DPRINTF(RubySequencer, "Issuing LL\n");
00598             assert(pkt->isRead());
00599             primary_type = RubyRequestType_Load_Linked;
00600         }
00601         secondary_type = RubyRequestType_ATOMIC;
00602     } else if (pkt->req->isLocked()) {
00603         //
00604         // x86 locked instructions are translated to store cache coherence
00605         // requests because these requests should always be treated as read
00606         // exclusive operations and should leverage any migratory sharing
00607         // optimization built into the protocol.
00608         //
00609         if (pkt->isWrite()) {
00610             DPRINTF(RubySequencer, "Issuing Locked RMW Write\n");
00611             primary_type = RubyRequestType_Locked_RMW_Write;
00612         } else {
00613             DPRINTF(RubySequencer, "Issuing Locked RMW Read\n");
00614             assert(pkt->isRead());
00615             primary_type = RubyRequestType_Locked_RMW_Read;
00616         }
00617         secondary_type = RubyRequestType_ST;
00618     } else {
00619         if (pkt->isRead()) {
00620             if (pkt->req->isInstFetch()) {
00621                 primary_type = secondary_type = RubyRequestType_IFETCH;
00622             } else {
00623 #if THE_ISA == X86_ISA
00624                 uint32_t flags = pkt->req->getFlags();
00625                 bool storeCheck = flags &
00626                         (TheISA::StoreCheck << TheISA::FlagShift);
00627 #else
00628                 bool storeCheck = false;
00629 #endif // X86_ISA
00630                 if (storeCheck) {
00631                     primary_type = RubyRequestType_RMW_Read;
00632                     secondary_type = RubyRequestType_ST;
00633                 } else {
00634                     primary_type = secondary_type = RubyRequestType_LD;
00635                 }
00636             }
00637         } else if (pkt->isWrite()) {
00638             //
00639             // Note: M5 packets do not differentiate ST from RMW_Write
00640             //
00641             primary_type = secondary_type = RubyRequestType_ST;
00642         } else if (pkt->isFlush()) {
00643           primary_type = secondary_type = RubyRequestType_FLUSH;
00644         } else {
00645             panic("Unsupported ruby packet type\n");
00646         }
00647     }
00648 
00649     RequestStatus status = insertRequest(pkt, primary_type);
00650     if (status != RequestStatus_Ready)
00651         return status;
00652 
00653     issueRequest(pkt, secondary_type);
00654 
00655     // TODO: issue hardware prefetches here
00656     return RequestStatus_Issued;
00657 }

00659 void
00660 Sequencer::issueRequest(PacketPtr pkt, RubyRequestType secondary_type)
00661 {
00662     assert(pkt != NULL);
00663     int proc_id = -1;
00664     if (pkt->req->hasContextId()) {
00665         proc_id = pkt->req->contextId();
00666     }
00667 
00668     // If valid, copy the pc to the ruby request
00669     Addr pc = 0;
00670     if (pkt->req->hasPC()) {
00671         pc = pkt->req->getPC();
00672     }
00673 
00674     RubyRequest *msg = new RubyRequest(clockEdge(), pkt->getAddr(),
00675                                        pkt->getPtr<uint8_t>(true),
00676                                        pkt->getSize(), pc, secondary_type,
00677                                        RubyAccessMode_Supervisor, pkt,
00678                                        PrefetchBit_No, proc_id);
00679 
00680     DPRINTFR(ProtocolTrace, "%15s %3s %10s%20s %6s>%-6s %s %s\n",
00681             curTick(), m_version, "Seq", "Begin", "", "",
00682             msg->getPhysicalAddress(),
00683             RubyRequestType_to_string(secondary_type));
00684 
00685     Cycles latency(0);  // initialzed to an null value
00686 
00687     if (secondary_type == RubyRequestType_IFETCH)
00688         latency = m_instCache_ptr->getLatency();
00689     else
00690         latency = m_dataCache_ptr->getLatency();
00691 
00692     // Send the message to the cache controller
00693     assert(latency > 0);
00694 
00695     assert(m_mandatory_q_ptr != NULL);
00696     m_mandatory_q_ptr->enqueue(msg, latency);
00697 }
