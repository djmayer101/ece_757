00049 bool
00050 SeriesRequestGenerator::initiate()
00051 {
00052     DPRINTF(DirectedTest, "initiating request\n");
00053     assert(m_status == SeriesRequestGeneratorStatus_Thinking);
00054 
00055     MasterPort* port = m_directed_tester->getCpuPort(m_active_node);
00056 
00057     Request::Flags flags;
00058 
00059     // For simplicity, requests are assumed to be 1 byte-sized
00060     Request *req = new Request(m_address, 1, flags, masterId);
00061 
00062     Packet::Command cmd;
00063     bool do_write = ((random() % 100) < m_percent_writes);
00064     if (do_write) {
00065         cmd = MemCmd::WriteReq;
00066     } else {
00067         cmd = MemCmd::ReadReq;
00068     }
00069 
00070     PacketPtr pkt = new Packet(req, cmd);
00071     uint8_t* dummyData = new uint8_t;
00072     *dummyData = 0;
00073     pkt->dataDynamic(dummyData);
00074 
00075     if (port->sendTimingReq(pkt)) {
00076         DPRINTF(DirectedTest, "initiating request - successful\n");
00077         m_status = SeriesRequestGeneratorStatus_Request_Pending;
00078         return true;
00079     } else {
00080         // If the packet did not issue, must delete
00081         // Note: No need to delete the data, the packet destructor
00082         // will delete it
00083         delete pkt->req;
00084         delete pkt;
00085 
00086         DPRINTF(DirectedTest, "failed to initiate request - sequencer not ready\n");
00087         return false;
00088     }
00089 }

00091 void 
00092 SeriesRequestGenerator::performCallback(uint32_t proc, Addr address)
00093 {
00094     assert(m_active_node == proc);
00095     assert(m_address == address);  
00096     assert(m_status == SeriesRequestGeneratorStatus_Request_Pending);
00097 
00098     m_status = SeriesRequestGeneratorStatus_Thinking;
00099     m_active_node++;
00100     if (m_active_node == m_num_cpus) {
00101         //
00102         // Cycle of requests completed, increment cycle completions and restart

00105         m_directed_tester->incrementCycleCompletions();
