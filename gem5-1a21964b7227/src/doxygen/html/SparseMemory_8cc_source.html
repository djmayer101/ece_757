00043         - RubySystem::getBlockSizeBits();;

00045     m_number_of_levels = number_of_levels;
00046

00048     // Create the array that describes the bits per level

00051     even_level_bits = m_total_number_of_bits / m_number_of_levels;

00063     m_adds_per_level = new uint64_t[m_number_of_levels];

00065     for (int level = 0; level < m_number_of_levels; level++) {
00066         m_adds_per_level[level] = 0;

00081 // Delete the lowest table first, the tables above
00082 void
00083 SparseMemory::recursivelyRemoveTables(SparseMapType* curTable, int curLevel)
00084 {
00085     SparseMapType::iterator iter;
00086 
00087     for (iter = curTable->begin(); iter != curTable->end(); iter++) {
00088         SparseMemEntry entry = (*iter).second;
00089 
00090         if (curLevel != (m_number_of_levels - 1)) {
00091             // If the not at the last level, analyze those lower level
00092             // tables first, then delete those next tables
00093             SparseMapType* nextTable = (SparseMapType*)(entry);
00094             recursivelyRemoveTables(nextTable, (curLevel + 1));
00095             delete nextTable;
00096         } else {
00097             // If at the last level, delete the directory entry
00098             delete (AbstractEntry*)(entry);
00099         }
00100         entry = NULL;
00101     }
00102 
00103     // Once all entries have been deleted, erase the entries
00104     curTable->erase(curTable->begin(), curTable->end());

00214                                       CurNextInfo& curInfo)
00215 {
00216     Address curAddress;
00217     CurNextInfo nextInfo;

00219 
00220     // create the appropriate address for this level
00221     // Note: that set Address is inclusive of the specified range,
00222     // thus the high bit is one less than the total number of bits
00223     // used to create the address.

00225                                             curInfo.highBit - 1));
00226 
00227     DPRINTF(RubyCache, "address: %s, curInfo.level: %d, curInfo.lowBit: %d, "
00228             "curInfo.highBit - 1: %d, curAddress: %s\n",
00229             address, curInfo.level, curInfo.lowBit,
00230             curInfo.highBit - 1, curAddress);
00231 
00232     assert(curInfo.curTable->count(curAddress) != 0);
00233 
00234     entry = (*(curInfo.curTable))[curAddress];
00235 
00236     if (curInfo.level < (m_number_of_levels - 1)) {
00237         // set up next level's info
00238         nextInfo.curTable = (SparseMapType*)(entry);
00239         nextInfo.level = curInfo.level + 1;
00240 
00241         nextInfo.highBit = curInfo.highBit -
00242             m_number_of_bits_per_level[curInfo.level];
00243 
00244         nextInfo.lowBit = curInfo.lowBit -
00245             m_number_of_bits_per_level[curInfo.level + 1];
00246 
00247         // recursively search the table hierarchy
00248         int tableSize = recursivelyRemoveLevels(address, nextInfo);
00249 
00250         // If this table below is now empty, we must delete it and
00251         // erase it from our table.
00252         if (tableSize == 0) {
00253             m_removes_per_level[curInfo.level]++;
00254             delete nextInfo.curTable;
00255             entry = NULL;
00256             curInfo.curTable->erase(curAddress);
00257         }
00258     } else {
00259         // if this is the last level, we have reached the Directory
00260         // Entry and thus we should delete it including the
00261         // SparseMemEntry container struct.
00262         delete (AbstractEntry*)(entry);
00263         entry = NULL;
00264         curInfo.curTable->erase(curAddress);
00265         m_removes_per_level[curInfo.level]++;
00266     }
00267     return curInfo.curTable->size();

00380             address_of_nodes[parity_of_level].pop();

00385                 SparseMemEntry entry = (*iter).second;

00387 
00388                 if (cur_level != (m_number_of_levels - 1)) {
00389                     // If not at the last level, put this node in the queue
00390                     unexplored_nodes[1 - parity_of_level].push(
00391                                                      (SparseMapType*)(entry));
00392                     address_of_nodes[1 - parity_of_level].push(address |
00393                                          (curAddress.getAddress() << lowBit));

00397                                                                    << lowBit);

00401                 }

00403         }
00404 
00405         // Adjust the highBit value for the next level
00406         highBit -= m_number_of_bits_per_level[cur_level];
