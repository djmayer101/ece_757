00106 StoreTrace::store(NodeID node)

00110     assert((m_last_writer == -1) || (m_last_writer == node));

00112     m_last_writer = node;

00126 void
00127 StoreTrace::downgrade(NodeID node)
00128 {
00129     if (node == m_last_writer) {
00130         Time current = curTick();
00131         assert(m_stores_this_interval != 0);
00132         assert(m_last_store != 0);
00133         assert(m_first_store != 0);
00134         assert(m_last_writer != -1);
00135 
00136         // Per line stats
00137         m_store_first_to_stolen.add(current - m_first_store);
00138         m_store_count.add(m_stores_this_interval);
00139         m_store_last_to_stolen.add(current - m_last_store);
00140         m_store_first_to_last.add(m_last_store - m_first_store);
00141         m_total_samples++;
00142 
00143         // Global stats
00144         assert(s_store_first_to_stolen_ptr != NULL);
00145         s_store_first_to_stolen_ptr->add(current - m_first_store);
00146         s_store_count_ptr->add(m_stores_this_interval);
00147         s_store_last_to_stolen_ptr->add(current - m_last_store);
00148         s_store_first_to_last_ptr->add(m_last_store - m_first_store);
00149         s_total_samples++;
00150 
00151         // Initilize for next go round
00152         m_stores_this_interval = 0;
00153         m_last_store = 0;
00154         m_first_store = 0;
00155         m_last_writer = -1;
