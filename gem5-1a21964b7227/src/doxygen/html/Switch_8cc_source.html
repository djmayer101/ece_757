00077 void
00078 Switch::addOutPort(const vector<MessageBuffer*>& out,
00079     const NetDest& routing_table_entry, Cycles link_latency, int bw_multiplier)

00081     // Create a throttle
00082     Throttle* throttle_ptr = new Throttle(m_id, m_throttles.size(),
00083             link_latency, bw_multiplier, m_network_ptr->getEndpointBandwidth(),
00084             this);
00085     m_throttles.push_back(throttle_ptr);
00086 
00087     // Create one buffer per vnet (these are intermediaryQueues)
00088     vector<MessageBuffer*> intermediateBuffers;
00089     for (int i = 0; i < out.size(); i++) {
00090         out[i]->setSender(this);
00091 
00092         MessageBuffer* buffer_ptr = new MessageBuffer;
00093         // Make these queues ordered
00094         buffer_ptr->setOrdering(true);
00095         if (m_network_ptr->getBufferSize() > 0) {
00096             buffer_ptr->resize(m_network_ptr->getBufferSize());
00097         }
00098 
00099         intermediateBuffers.push_back(buffer_ptr);
00100         m_buffers_to_free.push_back(buffer_ptr);
00101 
00102         buffer_ptr->setSender(this);
00103         buffer_ptr->setReceiver(this);
00104     }
00105 
00106     // Hook the queues to the PerfectSwitch
00107     m_perfect_switch_ptr->addOutPort(intermediateBuffers, routing_table_entry);
00108 
00109     // Hook the queues to the Throttle
00110     throttle_ptr->addLinks(intermediateBuffers, out);

00174

00176         Throttle* throttle_ptr = m_throttles[link];

00179                 << "_link_" << link << ": "
00180                 << throttle_ptr->getUtilization() << " bw: "
00181                 << throttle_ptr->getLinkBandwidth()
00182                 << " base_latency: " << throttle_ptr->getLatency() << endl;
00183         }
00184     }
00185     out << endl;
00186 
00187     // Traffic breakdown
00188     for (int link = 0; link < m_throttles.size(); link++) {
00189         Throttle* throttle_ptr = m_throttles[link];
00190         if (!throttle_ptr)
00191             continue;
00192 
00193         const vector<vector<int> >& message_counts =
