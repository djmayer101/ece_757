00064 void
00065 Switch_d::wakeup()
00066 {
00067     DPRINTF(RubyNetwork, "Switch woke up at time: %lld\n",
00068             m_router->curCycle());
00069 
00070     for (int inport = 0; inport < m_num_inports; inport++) {
00071         if (!m_switch_buffer[inport]->isReady(m_router->curCycle()))
00072             continue;
00073         flit_d *t_flit = m_switch_buffer[inport]->peekTopFlit();
00074         if (t_flit->is_stage(ST_, m_router->curCycle())) {
00075             int outport = t_flit->get_outport();
00076             t_flit->advance_stage(LT_, m_router->curCycle());
00077             t_flit->set_time(m_router->curCycle() + Cycles(1));
00078 
00079             // This will take care of waking up the Network Link
00080             m_output_unit[outport]->insert_flit(t_flit);
00081             m_switch_buffer[inport]->getTopFlit();
00082             m_crossbar_activity++;
00083         }
00084     }
00085     check_for_wakeup();
00086 }
