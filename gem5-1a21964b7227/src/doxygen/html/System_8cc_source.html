00424 {
00425     Address address(pkt->getAddr());

00427     line_address.makeLineAddress();
00428 
00429     AccessPermission access_perm = AccessPermission_NotPresent;
00430     int num_controllers = m_abs_cntrl_vec.size();
00431 
00432     DPRINTF(RubySystem, "Functional Read request for %s\n",address);
00433 
00434     unsigned int num_ro = 0;
00435     unsigned int num_rw = 0;
00436     unsigned int num_busy = 0;
00437     unsigned int num_backing_store = 0;
00438     unsigned int num_invalid = 0;
00439 
00440     // In this loop we count the number of controllers that have the given
00441     // address in read only, read write and busy states.
00442     for (unsigned int i = 0; i < num_controllers; ++i) {
00443         access_perm = m_abs_cntrl_vec[i]-> getAccessPermission(line_address);
00444         if (access_perm == AccessPermission_Read_Only)
00445             num_ro++;
00446         else if (access_perm == AccessPermission_Read_Write)
00447             num_rw++;
00448         else if (access_perm == AccessPermission_Busy)
00449             num_busy++;
00450         else if (access_perm == AccessPermission_Backing_Store)
00451             // See RubySlicc_Exports.sm for details, but Backing_Store is meant
00452             // to represent blocks in memory *for Broadcast/Snooping protocols*,
00453             // where memory has no idea whether it has an exclusive copy of data
00454             // or not.
00455             num_backing_store++;
00456         else if (access_perm == AccessPermission_Invalid ||
00457                  access_perm == AccessPermission_NotPresent)
00458             num_invalid++;
00459     }
00460     assert(num_rw <= 1);
00461 
00462     uint8_t *data = pkt->getPtr<uint8_t>(true);
00463     unsigned int size_in_bytes = pkt->getSize();
00464     unsigned startByte = address.getAddress() - line_address.getAddress();
00465 
00466     // This if case is meant to capture what happens in a Broadcast/Snoop
00467     // protocol where the block does not exist in the cache hierarchy. You
00468     // only want to read from the Backing_Store memory if there is no copy in
00469     // the cache hierarchy, otherwise you want to try to read the RO or RW
00470     // copies existing in the cache hierarchy (covered by the else statement).
00471     // The reason is because the Backing_Store memory could easily be stale, if
00472     // there are copies floating around the cache hierarchy, so you want to read
00473     // it only if it's not in the cache hierarchy at all.
00474     if (num_invalid == (num_controllers - 1) &&
00475             num_backing_store == 1) {
00476         DPRINTF(RubySystem, "only copy in Backing_Store memory, read from it\n");
00477         for (unsigned int i = 0; i < num_controllers; ++i) {
00478             access_perm = m_abs_cntrl_vec[i]->getAccessPermission(line_address);
00479             if (access_perm == AccessPermission_Backing_Store) {
00480                 DataBlock& block = m_abs_cntrl_vec[i]->
00481                     getDataBlock(line_address);
00482 
00483                 DPRINTF(RubySystem, "reading from %s block %s\n",
00484                         m_abs_cntrl_vec[i]->name(), block);
00485                 for (unsigned j = 0; j < size_in_bytes; ++j) {
00486                     data[j] = block.getByte(j + startByte);
00487                 }
00488                 return true;
00489             }
00490         }
00491     } else if (num_ro > 0 || num_rw == 1) {
00492         // In Broadcast/Snoop protocols, this covers if you know the block
00493         // exists somewhere in the caching hierarchy, then you want to read any
00494         // valid RO or RW block.  In directory protocols, same thing, you want
00495         // to read any valid readable copy of the block.
00496         DPRINTF(RubySystem, "num_busy = %d, num_ro = %d, num_rw = %d\n",
00497                 num_busy, num_ro, num_rw);
00498         // In this loop, we try to figure which controller has a read only or
00499         // a read write copy of the given address. Any valid copy would suffice
00500         // for a functional read.
00501         for (unsigned int i = 0;i < num_controllers;++i) {
00502             access_perm = m_abs_cntrl_vec[i]->getAccessPermission(line_address);
00503             if (access_perm == AccessPermission_Read_Only ||

00511                     data[j] = block.getByte(j + startByte);
00512                 }
00513                 return true;
00514             }
00515         }
00516     }
00517
