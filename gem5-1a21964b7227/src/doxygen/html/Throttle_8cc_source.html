00128 void
00129 Throttle::wakeup()
00130 {
00131     // Limits the number of message sent to a limited number of bytes/cycle.
00132     assert(getLinkBandwidth() > 0);
00133     int bw_remaining = getLinkBandwidth();
00134 
00135     // Give the highest numbered link priority most of the time
00136     m_wakeups_wo_switch++;
00137     int highest_prio_vnet = m_vnets-1;
00138     int lowest_prio_vnet = 0;
00139     int counter = 1;
00140     bool schedule_wakeup = false;
00141 
00142     // invert priorities to avoid starvation seen in the component network
00143     if (m_wakeups_wo_switch > PRIORITY_SWITCH_LIMIT) {
00144         m_wakeups_wo_switch = 0;
00145         highest_prio_vnet = 0;
00146         lowest_prio_vnet = m_vnets-1;
00147         counter = -1;
00148     }
00149 
00150     for (int vnet = highest_prio_vnet;
00151          (vnet * counter) >= (counter * lowest_prio_vnet);
00152          vnet -= counter) {
00153 
00154         assert(m_out[vnet] != NULL);
00155         assert(m_in[vnet] != NULL);
00156         assert(m_units_remaining[vnet] >= 0);
00157 
00158         while (bw_remaining > 0 &&
00159             (m_in[vnet]->isReady() || m_units_remaining[vnet] > 0) &&
00160             m_out[vnet]->areNSlotsAvailable(1)) {
00161 
00162             // See if we are done transferring the previous message on
00163             // this virtual network
00164             if (m_units_remaining[vnet] == 0 && m_in[vnet]->isReady()) {
00165                 // Find the size of the message we are moving
00166                 MsgPtr msg_ptr = m_in[vnet]->peekMsgPtr();
00167                 NetworkMessage* net_msg_ptr =
00168                     safe_cast<NetworkMessage*>(msg_ptr.get());
00169                 m_units_remaining[vnet] +=
00170                     network_message_to_size(net_msg_ptr);
00171 
00172                 DPRINTF(RubyNetwork, "throttle: %d my bw %d bw spent "
00173                         "enqueueing net msg %d time: %lld.\n",
00174                         m_node, getLinkBandwidth(), m_units_remaining[vnet],
00175                         g_system_ptr->curCycle());
00176 
00177                 // Move the message
00178                 m_out[vnet]->enqueue(m_in[vnet]->peekMsgPtr(), m_link_latency);
00179                 m_in[vnet]->pop();
00180 
00181                 // Count the message
00182                 m_message_counters[net_msg_ptr->getMessageSize()][vnet]++;
00183 
00184                 DPRINTF(RubyNetwork, "%s\n", *m_out[vnet]);
00185             }
00186 
00187             // Calculate the amount of bandwidth we spent on this message
00188             int diff = m_units_remaining[vnet] - bw_remaining;
00189             m_units_remaining[vnet] = max(0, diff);
00190             bw_remaining = max(0, -diff);
00191         }
00192 
00193         if (bw_remaining > 0 &&
00194             (m_in[vnet]->isReady() || m_units_remaining[vnet] > 0) &&
00195             !m_out[vnet]->areNSlotsAvailable(1)) {
00196             DPRINTF(RubyNetwork, "vnet: %d", vnet);
00197             // schedule me to wakeup again because I'm waiting for my
00198             // output queue to become available
00199             schedule_wakeup = true;
00200         }
00201     }
00202 
00203     // We should only wake up when we use the bandwidth
00204     // This is only mostly true
00205     // assert(bw_remaining != getLinkBandwidth());
00206 
00207     // Record that we used some or all of the link bandwidth this cycle
00208     double ratio = 1.0 - (double(bw_remaining) / double(getLinkBandwidth()));
00209 
00210     // If ratio = 0, we used no bandwidth, if ratio = 1, we used all
00211     linkUtilized(ratio);
00212 
00213     if (bw_remaining > 0 && !schedule_wakeup) {
00214         // We have extra bandwidth and our output buffer was
00215         // available, so we must not have anything else to do until
00216         // another message arrives.
00217         DPRINTF(RubyNetwork, "%s not scheduled again\n", *this);
00218     } else {
00219         DPRINTF(RubyNetwork, "%s scheduled again\n", *this);
00220 
00221         // We are out of bandwidth for this cycle, so wakeup next
00222         // cycle and continue
00223         scheduleEvent(Cycles(1));
00224     }
00225 }
