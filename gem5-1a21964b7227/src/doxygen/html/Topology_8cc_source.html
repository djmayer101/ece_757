00070     // Total nodes/controllers in network
00071     // Must make sure this is called after the State Machine constructors
00072     m_nodes = MachineType_base_number(MachineType_NUM);
00073     assert(m_nodes > 1);
00074 
00075     if (m_nodes != params()->ext_links.size() &&
00076         m_nodes != params()->ext_links.size()) {
00077         fatal("m_nodes (%d) != ext_links vector length (%d)\n",
00078               m_nodes, params()->ext_links.size());
00079     }
00080 
00081     // analyze both the internal and external links, create data structures
00082     // Note that the python created links are bi-directional, but that the
00083     // topology and networks utilize uni-directional links.  Thus each 
00084     // BasicLink is converted to two calls to add link, on for each direction
00085     for (vector<BasicExtLink*>::const_iterator i = params()->ext_links.begin();
00086          i != params()->ext_links.end(); ++i) {
00087         BasicExtLink *ext_link = (*i);
00088         AbstractController *abs_cntrl = ext_link->params()->ext_node;
00089         BasicRouter *router = ext_link->params()->int_node;
00090 
00091         // Store the ExtLink pointers for later
00092         m_ext_link_vector.push_back(ext_link);
00093 
00094         int ext_idx1 = abs_cntrl->params()->cntrl_id;
00095         int ext_idx2 = ext_idx1 + m_nodes;
00096         int int_idx = router->params()->router_id + 2*m_nodes;
00097 
00098         // create the internal uni-directional links in both directions
00099         //   the first direction is marked: In
00100         addLink(ext_idx1, int_idx, ext_link, LinkDirection_In);
00101         //   the first direction is marked: Out
00102         addLink(int_idx, ext_idx2, ext_link, LinkDirection_Out);
00103     }
00104 
00105     for (vector<BasicIntLink*>::const_iterator i = params()->int_links.begin();
00106          i != params()->int_links.end(); ++i) {
00107         BasicIntLink *int_link = (*i);
00108         BasicRouter *router_a = int_link->params()->node_a;
00109         BasicRouter *router_b = int_link->params()->node_b;
00110 
00111         // Store the IntLink pointers for later
00112         m_int_link_vector.push_back(int_link);
00113 
00114         int a = router_a->params()->router_id + 2*m_nodes;
00115         int b = router_b->params()->router_id + 2*m_nodes;
00116 
00117         // create the internal uni-directional links in both directions
00118         //   the first direction is marked: In
00119         addLink(a, b, int_link, LinkDirection_In);
00120         //   the second direction is marked: Out
00121         addLink(b, a, int_link, LinkDirection_Out);

00142 void
00143 Topology::createLinks(Network *net, bool isReconfiguration)
00144 {
00145     // Find maximum switchID
00146     SwitchID max_switch_id = 0;
00147     for (LinkMap::const_iterator i = m_link_map.begin();
00148          i != m_link_map.end(); ++i) {
00149         std::pair<int, int> src_dest = (*i).first;
00150         max_switch_id = max(max_switch_id, src_dest.first);
00151         max_switch_id = max(max_switch_id, src_dest.second);        
00152     }
00153 
00154     // Initialize weight, latency, and inter switched vectors
00155     Matrix topology_weights;
00156     int num_switches = max_switch_id+1;
00157     topology_weights.resize(num_switches);
00158     m_component_latencies.resize(num_switches);
00159     m_component_inter_switches.resize(num_switches);
00160 
00161     for (int i = 0; i < topology_weights.size(); i++) {
00162         topology_weights[i].resize(num_switches);
00163         m_component_latencies[i].resize(num_switches);
00164         m_component_inter_switches[i].resize(num_switches);
00165 
00166         for (int j = 0; j < topology_weights[i].size(); j++) {
00167             topology_weights[i][j] = INFINITE_LATENCY;
00168 
00169             // initialize to invalid values
00170             m_component_latencies[i][j] = -1;
00171 
00172             // initially assume direct connections / no intermediate
00173             // switches between components
00174             m_component_inter_switches[i][j] = 0;
00175         }
00176     }
00177 
00178     // Set identity weights to zero
00179     for (int i = 0; i < topology_weights.size(); i++) {
00180         topology_weights[i][i] = 0;
00181     }
00182 
00183     // Fill in the topology weights and bandwidth multipliers
00184     for (LinkMap::const_iterator i = m_link_map.begin();
00185          i != m_link_map.end(); ++i) {
00186         std::pair<int, int> src_dest = (*i).first;
00187         BasicLink* link = (*i).second.link;
00188         int src = src_dest.first;
00189         int dst = src_dest.second;
00190         m_component_latencies[src][dst] = link->m_latency;
00191         topology_weights[src][dst] = link->m_weight;
00192     }
00193         
00194     // Walk topology and hookup the links
00195     Matrix dist = shortest_path(topology_weights, m_component_latencies,
00196         m_component_inter_switches);
00197     for (int i = 0; i < topology_weights.size(); i++) {
00198         for (int j = 0; j < topology_weights[i].size(); j++) {
00199             int weight = topology_weights[i][j];
00200             if (weight > 0 && weight != INFINITE_LATENCY) {
00201                 NetDest destination_set = shortest_path_to_node(i, j,
00202                                                      topology_weights, dist);
00203                 makeLink(net, i, j, destination_set, isReconfiguration);
00204             }
00205         }
00206     }
00207 }

00230     // Make sure we're not trying to connect two end-point nodes

00232     assert(src >= 2 * m_nodes || dest >= 2 * m_nodes);
00233 
00234     std::pair<int, int> src_dest;
00235     LinkEntry link_entry;

00237     if (src < m_nodes) {
00238         src_dest.first = src;
00239         src_dest.second = dest;
00240         link_entry = m_link_map[src_dest];
00241         net->makeInLink(src, dest - (2 * m_nodes), link_entry.link,
00242                         link_entry.direction,

00245     } else if (dest < 2*m_nodes) {
00246         assert(dest >= m_nodes);
00247         NodeID node = dest - m_nodes;
00248         src_dest.first = src;
00249         src_dest.second = dest;

00252                          link_entry.direction, 
00253                          routing_table_entry,

00266 // The following all-pairs shortest path algorithm is based on the
00267 // discussion from Cormen et al., Chapter 26.1.
00268 void
00269 extend_shortest_path(Matrix& current_dist, Matrix& latencies,
00270     Matrix& inter_switches)
00271 {
00272     bool change = true;
00273     int nodes = current_dist.size();
00274 
00275     while (change) {
00276         change = false;
00277         for (int i = 0; i < nodes; i++) {
00278             for (int j = 0; j < nodes; j++) {
00279                 int minimum = current_dist[i][j];
00280                 int previous_minimum = minimum;
00281                 int intermediate_switch = -1;
00282                 for (int k = 0; k < nodes; k++) {
00283                     minimum = min(minimum,
00284                         current_dist[i][k] + current_dist[k][j]);
00285                     if (previous_minimum != minimum) {
00286                         intermediate_switch = k;
00287                         inter_switches[i][j] =
00288                             inter_switches[i][k] +
00289                             inter_switches[k][j] + 1;
00290                     }
00291                     previous_minimum = minimum;
00292                 }
00293                 if (current_dist[i][j] != minimum) {
00294                     change = true;
00295                     current_dist[i][j] = minimum;
00296                     assert(intermediate_switch >= 0);
00297                     assert(intermediate_switch < latencies[i].size());
00298                     latencies[i][j] = latencies[i][intermediate_switch] +
00299                         latencies[intermediate_switch][j];
00300                 }
00301             }
00302         }
00303     }
00304 }

00321 NetDest
00322 shortest_path_to_node(SwitchID src, SwitchID next, const Matrix& weights,
00323     const Matrix& dist)
00324 {
00325     NetDest result;
00326     int d = 0;
00327     int machines;
00328     int max_machines;
00329 
00330     machines = MachineType_NUM;
00331     max_machines = MachineType_base_number(MachineType_NUM);
00332 
00333     for (int m = 0; m < machines; m++) {
00334         for (int i = 0; i < MachineType_base_count((MachineType)m); i++) {
00335             // we use "d+max_machines" below since the "destination"
00336             // switches for the machines are numbered
00337             // [MachineType_base_number(MachineType_NUM)...
00338             //  2*MachineType_base_number(MachineType_NUM)-1] for the
00339             // component network
00340             if (link_is_shortest_path_to_node(src, next, d + max_machines,
00341                     weights, dist)) {
00342                 MachineID mach = {(MachineType)m, i};
00343                 result.add(mach);
00344             }
00345             d++;
00346         }
00347     }
00348 
00349     DPRINTF(RubyNetwork, "Returning shortest path\n"
00350             "(src-(2*max_machines)): %d, (next-(2*max_machines)): %d, "
00351             "src: %d, next: %d, result: %s\n",
00352             (src-(2*max_machines)), (next-(2*max_machines)),
00353             src, next, result);
00354 
00355     return result;
00356 }
