00198 double VCAllocator::get_static_power() const
00199 {
00200     double p_va = 0;
00201     switch(m_va_model)
00202     {
00203         case ONE_STAGE_ARB:
00204             p_va = m_global_arb_ptr->get_static_power()*m_num_out_port*m_num_vclass*m_num_vchannel;
00205             break;
00206         case TWO_STAGE_ARB:
00207             p_va += m_local_arb_ptr->get_static_power()*m_num_in_port*m_num_vclass*m_num_vchannel;
00208             p_va += m_global_arb_ptr->get_static_power()*m_num_out_port*m_num_vclass*m_num_vchannel;
00209             break;   
00210         case VC_SELECT:
00211             p_va = m_vc_select_ptr->get_static_power()*m_num_out_port*m_num_vclass;
00212             break;
00213         default:
00214             cerr << "ERROR: Invalid VA model" << endl;
00215             exit(1);
00216     }
00217     return p_va;
