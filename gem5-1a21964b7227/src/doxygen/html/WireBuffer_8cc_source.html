00072 void
00073 WireBuffer::enqueue(MsgPtr message, Cycles latency)
00074 {
00075     m_msg_counter++;
00076     Cycles current_time = g_system_ptr->curCycle();
00077     Cycles arrival_time = current_time + latency;
00078     assert(arrival_time > current_time);
00079 
00080     MessageBufferNode thisNode(arrival_time, m_msg_counter, message);
00081     m_message_queue.push_back(thisNode);
00082     if (m_consumer_ptr != NULL) {
00083         m_consumer_ptr->scheduleEventAbsolute(arrival_time);
00084     } else {
00085         panic("No Consumer for WireBuffer! %s\n", *this);
00086     }
00087 }

00115 void
00116 WireBuffer::recycle()
00117 {
00118     // Because you don't want anything reordered, make sure the recycle latency
00119     // is just 1 cycle. As a result, you really want to use this only in
00120     // Wire-like situations because you don't want to deadlock as a result of
00121     // being stuck behind something if you're not actually supposed to.
00122     assert(isReady());
00123     MessageBufferNode node = m_message_queue.front();
00124     pop_heap(m_message_queue.begin(), m_message_queue.end(),
00125         greater<MessageBufferNode>());
00126 
00127     node.m_time = g_system_ptr->curCycle() + Cycles(1);
00128     m_message_queue.back() = node;
00129     push_heap(m_message_queue.begin(), m_message_queue.end(),
00130         greater<MessageBufferNode>());
00131     m_consumer_ptr->scheduleEventAbsolute(node.m_time);
00132 }
