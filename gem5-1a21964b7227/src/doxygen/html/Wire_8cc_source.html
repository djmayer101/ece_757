00083     {
00084         case MIN_DELAY:
00085             {
00086                 if (m_is_shielding)
00087                 {
00088                     double r = m_res_unit_len*len_;
00089                     double c_g = 2*m_gnd_cap_unit_len*len_;
00090                     double c_c = 2*m_couple_cap_unit_len*len_;

00092                             (0.7*BufferDriveResistance*BufferInputCapacitance)); //k is the number of buffers to be inserted
00093                     *h_ = sqrt(((0.7*BufferDriveResistance*c_g)+
00094                                 (1.4*1.5*BufferDriveResistance*c_c))/(0.7*r*BufferInputCapacitance)); //the size of the buffers to be inserted
00095                     break;
00096                 }
00097                 else

00103                             (0.7*BufferDriveResistance*BufferInputCapacitance));

00105                                 (1.4*2.2*BufferDriveResistance*c_c))/(0.7*r*BufferInputCapacitance));

00223 // The following function computes the wire resistance considering
00224 // width-spacing combination and a width-dependent resistivity model
00225 double Wire::calc_res_unit_len()
00226 {
00227     double r = -1.0;
00228     double rho;
00229     // r, rho is in ohm.m
00230 
00231     double WireMinWidth = m_tech_param_ptr->get_WireMinWidth();
00232     double WireBarrierThickness = m_tech_param_ptr->get_WireBarrierThickness();
00233     double WireMetalThickness = m_tech_param_ptr->get_WireMetalThickness();
00234 
00235     switch(m_width_spacing_model)
00236     {
00237         case SWIDTH_SSPACE:
00238         case SWIDTH_DSPACE:
00239             rho = 2.202e-8 + (1.030e-15  / (WireMinWidth - 2*WireBarrierThickness));
00240             r = ((rho) / ((WireMinWidth - 2*WireBarrierThickness) *
00241                         (WireMetalThickness - WireBarrierThickness)));
00242             break;
00243         case DWIDTH_SSPACE:
00244         case DWIDTH_DSPACE:
00245             rho = 2.202e-8 + (1.030e-15  / (2*WireMinWidth - 2*WireBarrierThickness));
00246             r = ((rho) / ((2*WireMinWidth - 2*WireBarrierThickness) *
00247                         (WireMetalThickness - WireBarrierThickness)));
00248             break;
00249         default:
00250             warn("Orion: Width spacing model not found: %s\n", m_width_spacing_model);
00251             r = 1.0;
00252     }
00253     return r;
00254 }
