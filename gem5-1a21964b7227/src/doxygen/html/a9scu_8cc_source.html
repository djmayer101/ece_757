00053 Tick
00054 A9SCU::read(PacketPtr pkt)
00055 {
00056     assert(pkt->getAddr() >= pioAddr && pkt->getAddr() < pioAddr + pioSize);
00057     assert(pkt->getSize() == 4);
00058     Addr daddr = pkt->getAddr() - pioAddr;
00059     pkt->allocate();
00060 
00061     switch(daddr) {
00062       case Control:
00063         pkt->set(1); // SCU already enabled
00064         break;
00065       case Config:
00066         assert(sys->numContexts() <= 4);
00067         int smp_bits, core_cnt;
00068         smp_bits = power(2,sys->numContexts()) - 1;
00069         core_cnt = sys->numContexts() - 1;
00070         pkt->set(smp_bits << 4 | core_cnt);
00071         break;
00072       default:
00073         // Only configuration register is implemented
00074         panic("Tried to read SCU at offset %#x\n", daddr);
00075         break;
00076     }
00077     pkt->makeAtomicResponse();
00078     return pioDelay;
00079 
00080 }
