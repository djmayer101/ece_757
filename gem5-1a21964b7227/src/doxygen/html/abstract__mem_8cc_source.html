00184 // Add load-locked to tracking list.  Should only be called if the
00185 // operation is a load and the LLSC flag is set.
00186 void
00187 AbstractMemory::trackLoadLocked(PacketPtr pkt)
00188 {
00189     Request *req = pkt->req;
00190     Addr paddr = LockedAddr::mask(req->getPaddr());
00191 
00192     // first we check if we already have a locked addr for this
00193     // xc.  Since each xc only gets one, we just update the
00194     // existing record with the new address.
00195     list<LockedAddr>::iterator i;
00196 
00197     for (i = lockedAddrList.begin(); i != lockedAddrList.end(); ++i) {
00198         if (i->matchesContext(req)) {
00199             DPRINTF(LLSC, "Modifying lock record: context %d addr %#x\n",
00200                     req->contextId(), paddr);
00201             i->addr = paddr;
00202             return;
00203         }
00204     }
00205 
00206     // no record for this xc: need to allocate a new one
00207     DPRINTF(LLSC, "Adding lock record: context %d addr %#x\n",
00208             req->contextId(), paddr);
00209     lockedAddrList.push_front(LockedAddr(req));
00210 }

00213 // Called on *writes* only... both regular stores and
00214 // store-conditional operations.  Check for conventional stores which
00215 // conflict with locked addresses, and for success/failure of store
00216 // conditionals.
00217 bool
00218 AbstractMemory::checkLockedAddrList(PacketPtr pkt)
00219 {
00220     Request *req = pkt->req;
00221     Addr paddr = LockedAddr::mask(req->getPaddr());
00222     bool isLLSC = pkt->isLLSC();
00223 
00224     // Initialize return value.  Non-conditional stores always
00225     // succeed.  Assume conditional stores will fail until proven
00226     // otherwise.
00227     bool allowStore = !isLLSC;
00228 
00229     // Iterate over list.  Note that there could be multiple matching records,
00230     // as more than one context could have done a load locked to this location.
00231     // Only remove records when we succeed in finding a record for (xc, addr);
00232     // then, remove all records with this address.  Failed store-conditionals do
00233     // not blow unrelated reservations.
00234     list<LockedAddr>::iterator i = lockedAddrList.begin();
00235 
00236     if (isLLSC) {
00237         while (i != lockedAddrList.end()) {
00238             if (i->addr == paddr && i->matchesContext(req)) {
00239                 // it's a store conditional, and as far as the memory system can
00240                 // tell, the requesting context's lock is still valid.
00241                 DPRINTF(LLSC, "StCond success: context %d addr %#x\n",
00242                         req->contextId(), paddr);
00243                 allowStore = true;
00244                 break;
00245             }
00246             // If we didn't find a match, keep searching!  Someone else may well
00247             // have a reservation on this line here but we may find ours in just
00248             // a little while.
00249             i++;
00250         }
00251         req->setExtraData(allowStore ? 1 : 0);
00252     }
00253     // LLSCs that succeeded AND non-LLSC stores both fall into here:
00254     if (allowStore) {
00255         // We write address paddr.  However, there may be several entries with a
00256         // reservation on this address (for other contextIds) and they must all
00257         // be removed.
00258         i = lockedAddrList.begin();
00259         while (i != lockedAddrList.end()) {
00260             if (i->addr == paddr) {
00261                 DPRINTF(LLSC, "Erasing lock record: context %d addr %#x\n",
00262                         i->contextId, paddr);
00263                 i = lockedAddrList.erase(i);
00264             } else {
00265                 i++;
00266             }
00267         }
00268     }
00269 
00270     return allowStore;
00271 }

00303 void
00304 AbstractMemory::access(PacketPtr pkt)
00305 {
00306     assert(AddrRange(pkt->getAddr(),
00307                      pkt->getAddr() + pkt->getSize() - 1).isSubset(range));
00308 
00309     if (pkt->memInhibitAsserted()) {
00310         DPRINTF(MemoryAccess, "mem inhibited on 0x%x: not responding\n",
00311                 pkt->getAddr());
00312         return;
00313     }
00314 
00315     uint8_t *hostAddr = pmemAddr + pkt->getAddr() - range.start();
00316 
00317     if (pkt->cmd == MemCmd::SwapReq) {
00318         TheISA::IntReg overwrite_val;
00319         bool overwrite_mem;
00320         uint64_t condition_val64;
00321         uint32_t condition_val32;
00322 
00323         if (!pmemAddr)
00324             panic("Swap only works if there is real memory (i.e. null=False)");
00325         assert(sizeof(TheISA::IntReg) >= pkt->getSize());
00326 
00327         overwrite_mem = true;
00328         // keep a copy of our possible write value, and copy what is at the
00329         // memory address into the packet
00330         std::memcpy(&overwrite_val, pkt->getPtr<uint8_t>(), pkt->getSize());
00331         std::memcpy(pkt->getPtr<uint8_t>(), hostAddr, pkt->getSize());
00332 
00333         if (pkt->req->isCondSwap()) {
00334             if (pkt->getSize() == sizeof(uint64_t)) {
00335                 condition_val64 = pkt->req->getExtraData();
00336                 overwrite_mem = !std::memcmp(&condition_val64, hostAddr,
00337                                              sizeof(uint64_t));
00338             } else if (pkt->getSize() == sizeof(uint32_t)) {
00339                 condition_val32 = (uint32_t)pkt->req->getExtraData();
00340                 overwrite_mem = !std::memcmp(&condition_val32, hostAddr,
00341                                              sizeof(uint32_t));
00342             } else
00343                 panic("Invalid size for conditional read/write\n");
00344         }
00345 
00346         if (overwrite_mem)
00347             std::memcpy(hostAddr, &overwrite_val, pkt->getSize());
00348 
00349         assert(!pkt->req->isInstFetch());
00350         TRACE_PACKET("Read/Write");
00351         numOther[pkt->req->masterId()]++;
00352     } else if (pkt->isRead()) {
00353         assert(!pkt->isWrite());
00354         if (pkt->isLLSC()) {
00355             trackLoadLocked(pkt);
00356         }
00357         if (pmemAddr)
00358             memcpy(pkt->getPtr<uint8_t>(), hostAddr, pkt->getSize());
00359         TRACE_PACKET(pkt->req->isInstFetch() ? "IFetch" : "Read");
00360         numReads[pkt->req->masterId()]++;
00361         bytesRead[pkt->req->masterId()] += pkt->getSize();
00362         if (pkt->req->isInstFetch())
00363             bytesInstRead[pkt->req->masterId()] += pkt->getSize();
00364     } else if (pkt->isWrite()) {
00365         if (writeOK(pkt)) {
00366             if (pmemAddr)
00367                 memcpy(hostAddr, pkt->getPtr<uint8_t>(), pkt->getSize());
00368             assert(!pkt->req->isInstFetch());
00369             TRACE_PACKET("Write");
00370             numWrites[pkt->req->masterId()]++;
00371             bytesWritten[pkt->req->masterId()] += pkt->getSize();
00372         }
00373     } else if (pkt->isInvalidate()) {
00374         // no need to do anything
00375     } else {
00376         panic("unimplemented");
00377     }
00378 
00379     if (pkt->needsResponse()) {
00380         pkt->makeResponse();
00381     }
00382 }

00413               pkt->cmdString());
