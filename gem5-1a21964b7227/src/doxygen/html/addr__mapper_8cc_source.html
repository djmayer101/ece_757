00120 bool
00121 AddrMapper::recvTimingReq(PacketPtr pkt)
00122 {
00123     Addr orig_addr = pkt->getAddr();
00124     bool needsResponse = pkt->needsResponse();
00125     bool memInhibitAsserted = pkt->memInhibitAsserted();
00126 
00127     if (needsResponse && !memInhibitAsserted) {
00128         pkt->pushSenderState(new AddrMapperSenderState(orig_addr));
00129     }
00130 
00131     pkt->setAddr(remapAddr(orig_addr));
00132 
00133     // Attempt to send the packet (always succeeds for inhibited
00134     // packets)
00135     bool successful = masterPort.sendTimingReq(pkt);
00136 
00137     // If not successful, restore the sender state
00138     if (!successful && needsResponse) {
00139         delete pkt->popSenderState();
00140     }
00141 
00142     return successful;
00143 }

00145 bool
00146 AddrMapper::recvTimingResp(PacketPtr pkt)
00147 {
00148     AddrMapperSenderState* receivedState =
00149         dynamic_cast<AddrMapperSenderState*>(pkt->senderState);
00150 
00151     // Restore initial sender state
00152     if (receivedState == NULL)
00153         panic("AddrMapper %s got a response without sender state\n",
00154               name());
00155 
00156     Addr remapped_addr = pkt->getAddr();
00157 
00158     // Restore the state and address
00159     pkt->senderState = receivedState->predecessor;
00160     pkt->setAddr(receivedState->origAddr);
00161 
00162     // Attempt to send the packet
00163     bool successful = slavePort.sendTimingResp(pkt);
00164 
00165     // If packet successfully sent, delete the sender state, otherwise
00166     // restore state
00167     if (successful) {
00168         delete receivedState;
00169     } else {
00170         // Don't delete anything and let the packet look like we did
00171         // not touch it
00172         pkt->senderState = receivedState;
00173         pkt->setAddr(remapped_addr);
00174     }
00175     return successful;
00176 }

00250 Addr
00251 RangeAddrMapper::remapAddr(Addr addr) const
00252 {
00253     for (int i = 0; i < originalRanges.size(); ++i) {
00254         if (originalRanges[i].contains(addr)) {
00255             Addr offset = addr - originalRanges[i].start();
00256             return offset + remappedRanges[i].start();
00257         }
00258     }
00259 
00260     return addr;
00261 }
