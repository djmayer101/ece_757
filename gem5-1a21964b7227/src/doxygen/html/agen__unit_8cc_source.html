00052 AGENUnit::execute(int slot_num)
00053 {
00054     ResourceRequest* agen_req = reqs[slot_num];
00055     DynInstPtr inst = reqs[slot_num]->inst;
00056 #if TRACING_ON
00057     ThreadID tid = inst->readTid();
00058 #endif
00059     InstSeqNum seq_num = inst->seqNum;
00060 
00061     if (inst->fault != NoFault) {
00062         DPRINTF(InOrderAGEN,
00063                 "[tid:%i]: [sn:%i]: Detected %s fault @ %x. Forwarding to "
00064                 "next stage.\n", tid, inst->seqNum, inst->fault->name(),
00065                 inst->pcState());
00066         agen_req->done();
00067         return;
00068     }
00069 
00070     switch (agen_req->cmd)
00071     {
00072       case GenerateAddr:
00073         {
00074             // Load/Store Instruction
00075             if (inst->isMemRef()) {
00076                 DPRINTF(InOrderAGEN,
00077                         "[tid:%i] Generating Address for [sn:%i] (%s).\n",
00078                         tid, seq_num, inst->staticInst->getName());
00079 
00080                 inst->fault = inst->calcEA();
00081                 inst->setMemAddr(inst->getEA());
00082 
00083                 DPRINTF(InOrderAGEN,
00084                     "[tid:%i] [sn:%i] Effective address calculated as: %#x\n",
00085                     tid, seq_num, inst->getEA());
00086 
00087                 if (inst->fault == NoFault) {
00088                     agen_req->done();
00089                 } else {
00090                     fatal("%s encountered while calculating address [sn:%i] %s",
00091                           inst->fault->name(), seq_num, inst->instName());
00092                 }
00093 
00094                 agens++;
00095             } else {
00096                 DPRINTF(InOrderAGEN,
00097                         "[tid:] Ignoring non-memory instruction [sn:%i]\n",
00098                         tid, seq_num);
00099                 agen_req->done();
00100             }
00101         }
00102         break;
00103 
00104       default:
00105         fatal("Unrecognized command to %s", resName);
00106     }
