00067 template <class XC>
00068 inline void
00069 handleLockedSnoop(XC *xc, PacketPtr pkt, Addr cacheBlockMask)
00070 {
00071     // If we see a snoop come into the CPU and we currently have an LLSC
00072     // operation pending we need to clear the lock flag if it is to the same
00073     // cache line.
00074 
00075     if (!xc->readMiscReg(MISCREG_LOCKFLAG))
00076         return;
00077 
00078     Addr locked_addr = xc->readMiscReg(MISCREG_LOCKADDR) & cacheBlockMask;
00079     Addr snoop_addr = pkt->getAddr();
00080 
00081     assert((cacheBlockMask & snoop_addr) == snoop_addr);
00082 
00083     if (locked_addr == snoop_addr)
00084         xc->setMiscReg(MISCREG_LOCKFLAG, false);
00085 }

00088 template <class XC>
00089 inline void
00090 handleLockedRead(XC *xc, Request *req)
00091 {
00092     xc->setMiscReg(MISCREG_LOCKADDR, req->getPaddr() & ~0xf);
00093     xc->setMiscReg(MISCREG_LOCKFLAG, true);
00094 }

00097 template <class XC>
00098 inline bool
00099 handleLockedWrite(XC *xc, Request *req)
00100 {
00101     if (req->isUncacheable()) {
00102         // Funky Turbolaser mailbox access...don't update
00103         // result register (see stq_c in decoder.isa)
00104         req->setExtraData(2);
00105     } else {
00106         // standard store conditional
00107         bool lock_flag = xc->readMiscReg(MISCREG_LOCKFLAG);
00108         Addr lock_addr = xc->readMiscReg(MISCREG_LOCKADDR);
00109         if (!lock_flag || (req->getPaddr() & ~0xf) != lock_addr) {
00110             // Lock flag not set or addr mismatch in CPU;
00111             // don't even bother sending to memory system
00112             req->setExtraData(0);
00113             xc->setMiscReg(MISCREG_LOCKFLAG, false);
00114             // the rest of this code is not architectural;
00115             // it's just a debugging aid to help detect
00116             // livelock by warning on long sequences of failed
00117             // store conditionals
00118             int stCondFailures = xc->readStCondFailures();
00119             stCondFailures++;
00120             xc->setStCondFailures(stCondFailures);
00121             if (stCondFailures % 100000 == 0) {
00122                 warn("context %d: %d consecutive "
00123                      "store conditional failures\n",
00124                      xc->contextId(), stCondFailures);
00125             }
00126 
00127             // store conditional failed already, so don't issue it to mem
00128             return false;
00129         }
00130     }
00131 
00132     return true;
00133 }
