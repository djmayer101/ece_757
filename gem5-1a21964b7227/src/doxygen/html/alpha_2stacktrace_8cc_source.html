00137 {
00138     tc = _tc;
00139 
00140     System *sys = tc->getSystemPtr();
00141 
00142     bool usermode =
00143         (tc->readMiscRegNoEffect(IPR_DTB_CM) & 0x18) != 0;
00144 
00145     Addr pc = tc->pcState().pc();
00146     bool kernel = sys->kernelStart <= pc && pc <= sys->kernelEnd;
00147 
00148     if (usermode) {
00149         stack.push_back(user);
00150         return;
00151     }
00152 
00153     if (!kernel) {
00154         stack.push_back(console);
00155         return;
00156     }
00157 
00158     SymbolTable *symtab = sys->kernelSymtab;
00159     Addr ksp = tc->readIntReg(StackPointerReg);
00160     Addr bottom = ksp & ~0x3fff;
00161 
00162     if (is_call) {
00163         Addr addr;
00164         if (!symtab->findNearestAddr(pc, addr))
00165             panic("could not find address %#x", pc);
00166 
00167         stack.push_back(addr);
00168         pc = tc->pcState().pc();
00169     }
00170 
00171     while (ksp > bottom) {
00172         Addr addr;
00173         if (!symtab->findNearestAddr(pc, addr))
00174             panic("could not find symbol for pc=%#x", pc);
00175         assert(pc >= addr && "symbol botch: callpc < func");
00176 
00177         stack.push_back(addr);
00178 
00179         if (isEntry(addr))
00180             return;
00181 
00182         Addr ra;
00183         int size;
00184         if (decodePrologue(ksp, pc, addr, size, ra)) {
00185             if (!ra)
00186                 return;
00187 
00188             if (size <= 0) {
00189                 stack.push_back(unknown);
00190                 return;
00191             }
00192 
00193             pc = ra;
00194             ksp += size;
00195         } else {
00196             stack.push_back(unknown);
00197             return;
00198         }
00199 
00200         kernel = sys->kernelStart <= pc && pc <= sys->kernelEnd;
00201         if (!kernel)
00202             return;

00235 bool
00236 StackTrace::decodeStack(MachInst inst, int &disp)
00237 {
00238     // lda $sp, -disp($sp)
00239     //
00240     // Opcode<31:26> == 0x08
00241     // RA<25:21> == 30
00242     // RB<20:16> == 30
00243     // Disp<15:0>
00244     const MachInst mem_mask = 0xffff0000;
00245     const MachInst lda_pattern = 0x23de0000;
00246     const MachInst lda_disp_mask = 0x0000ffff;
00247 
00248     // subq $sp, disp, $sp
00249     // addq $sp, disp, $sp
00250     //
00251     // Opcode<31:26> == 0x10
00252     // RA<25:21> == 30
00253     // Lit<20:13>
00254     // One<12> = 1
00255     // Func<11:5> == 0x20 (addq)
00256     // Func<11:5> == 0x29 (subq)
00257     // RC<4:0> == 30
00258     const MachInst intop_mask = 0xffe01fff;
00259     const MachInst addq_pattern = 0x43c0141e;
00260     const MachInst subq_pattern = 0x43c0153e;
00261     const MachInst intop_disp_mask = 0x001fe000;
00262     const int intop_disp_shift = 13;
00263 
00264     if ((inst & mem_mask) == lda_pattern)
00265         disp = -sext<16>(inst & lda_disp_mask);
00266     else if ((inst & intop_mask) == addq_pattern)
00267         disp = -int((inst & intop_disp_mask) >> intop_disp_shift);
00268     else if ((inst & intop_mask) == subq_pattern)
00269         disp = int((inst & intop_disp_mask) >> intop_disp_shift);
00270     else
00271         return false;
00272 
00273     return true;
00274 }

00303  * which registers are stored where, and how large the stack frame is.

00305 bool
00306 StackTrace::decodePrologue(Addr sp, Addr callpc, Addr func, int &size,
00307                            Addr &ra)
00308 {
00309     size = 0;
00310     ra = 0;
00311 
00312     for (Addr pc = func; pc < callpc; pc += sizeof(MachInst)) {
00313         MachInst inst;
00314         CopyOut(tc, (uint8_t *)&inst, pc, sizeof(MachInst));
00315 
00316         int reg, disp;
00317         if (decodeStack(inst, disp)) {
00318             if (size) {
00319                 // panic("decoding frame size again");
00320                 return true;
00321             }
00322             size += disp;
00323         } else if (decodeSave(inst, reg, disp)) {
00324             if (!ra && reg == ReturnAddressReg) {
00325                 CopyOut(tc, (uint8_t *)&ra, sp + disp, sizeof(Addr));
00326                 if (!ra) {
00327                     // panic("no return address value pc=%#x\n", pc);
00328                     return false;
00329                 }
00330             }
00331         }
00332     }
00333 
00334     return true;
