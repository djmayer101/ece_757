00073 readId(PacketPtr pkt, uint64_t amba_id, Addr pio_addr)
00074 {
00075     Addr daddr = pkt->getAddr() - pio_addr;
00076     if (daddr < AMBA_PER_ID0 || daddr > AMBA_CEL_ID3)
00077         return false;
00078 
00079     pkt->allocate();
00080 
00081     int byte = (daddr - AMBA_PER_ID0) << 1;
00082     // Too noisy right now
00083     DPRINTF(AMBA, "Returning %#x for offset %#x(%d)\n", (amba_id >> byte) & 0xFF,
00084             pkt->getAddr() - pio_addr, byte);
00085     assert(pkt->getSize() == 4);
00086     pkt->set<uint32_t>((amba_id >> byte) & 0xFF);
00087     return true;
00088 }
