00108 void
00109 AlphaFault::invoke(ThreadContext *tc, StaticInstPtr inst)
00110 {
00111     FaultBase::invoke(tc);
00112     if (!FullSystem)
00113         return;
00114     countStat()++;
00115 
00116     PCState pc = tc->pcState();
00117 
00118     // exception restart address
00119     if (setRestartAddress() || !(pc.pc() & 0x3))
00120         tc->setMiscRegNoEffect(IPR_EXC_ADDR, pc.pc());
00121 
00122     if (skipFaultingInstruction()) {
00123         // traps...  skip faulting instruction.
00124         tc->setMiscRegNoEffect(IPR_EXC_ADDR,
00125                    tc->readMiscRegNoEffect(IPR_EXC_ADDR) + 4);
00126     }
00127 
00128     pc.set(tc->readMiscRegNoEffect(IPR_PAL_BASE) + vect());
00129     tc->pcState(pc);
00130 }

00141 void
00142 DtbFault::invoke(ThreadContext *tc, StaticInstPtr inst)
00143 {
00144     if (FullSystem) {
00145         // Set fault address and flags.  Even though we're modeling an
00146         // EV5, we use the EV6 technique of not latching fault registers
00147         // on VPTE loads (instead of locking the registers until IPR_VA is
00148         // read, like the EV5).  The EV6 approach is cleaner and seems to
00149         // work with EV5 PAL code, but not the other way around.
00150         if (!tc->misspeculating() &&
00151             reqFlags.noneSet(Request::VPTE | Request::PREFETCH)) {
00152             // set VA register with faulting address
00153             tc->setMiscRegNoEffect(IPR_VA, vaddr);
00154 
00155             // set MM_STAT register flags
00156             MachInst machInst = inst->machInst;
00157             tc->setMiscRegNoEffect(IPR_MM_STAT,
00158                 (((Opcode(machInst) & 0x3f) << 11) |
00159                  ((Ra(machInst) & 0x1f) << 6) |
00160                  (flags & 0x3f)));
00161 
00162             // set VA_FORM register with faulting formatted address
00163             tc->setMiscRegNoEffect(IPR_VA_FORM,
00164                 tc->readMiscRegNoEffect(IPR_MVPTBR) | (vaddr.vpn() << 3));
00165         }
00166     }
00167 
00168     AlphaFault::invoke(tc);
00169 }

00186 ItbPageFault::invoke(ThreadContext *tc, StaticInstPtr inst)
00187 {
00188     if (FullSystem) {
00189         ItbFault::invoke(tc);

00204 void
00205 NDtbMissFault::invoke(ThreadContext *tc, StaticInstPtr inst)
00206 {
00207     if (FullSystem) {
00208         DtbFault::invoke(tc, inst);
00209         return;
00210     }
00211 
00212     Process *p = tc->getProcessPtr();
00213     TlbEntry entry;
00214     bool success = p->pTable->lookup(vaddr, entry);
00215     if (!success) {

00217             success = p->pTable->lookup(vaddr, entry);

00219     if (!success) {
00220         panic("Tried to access unmapped address %#x.\n", (Addr)vaddr);
00221     } else {
00222         tc->getDTBPtr()->insert(vaddr.page(), entry);
