00180 LinuxAlphaSystem::setDelayLoop(ThreadContext *tc)
00181 {
00182     Addr addr = 0;
00183     if (kernelSymtab->findAddress("loops_per_jiffy", addr)) {
00184         Tick cpuFreq = tc->getCpuPtr()->frequency();
00185         assert(intrFreq);
