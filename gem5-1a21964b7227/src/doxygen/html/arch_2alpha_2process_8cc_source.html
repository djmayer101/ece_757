00047 AlphaLiveProcess::AlphaLiveProcess(LiveProcessParams *params,
00048                                    ObjectFile *objFile)
00049     : LiveProcess(params, objFile)
00050 {
00051     brk_point = objFile->dataBase() + objFile->dataSize() + objFile->bssSize();
00052     brk_point = roundUp(brk_point, VMPageSize);
00053 
00054     // Set up stack.  On Alpha, stack goes below text section.  This
00055     // code should get moved to some architecture-specific spot.
00056     stack_base = objFile->textBase() - (409600+4096);
00057 
00058     // Set up region for mmaps.  Tru64 seems to start just above 0 and
00059     // grow up from there.
00060     mmap_start = mmap_end = 0x10000;
00061 
00062     // Set pointer for next thread stack.  Reserve 8M for main stack.
00063     next_thread_stack_base = stack_base - (8 * 1024 * 1024);
00064 
00065 }

00067 void
00068 AlphaLiveProcess::argsInit(int intSize, int pageSize)
00069 {
00070     objFile->loadSections(initVirtMem);
00071 
00072     typedef AuxVector<uint64_t> auxv_t;
00073     std::vector<auxv_t>  auxv;
00074 
00075     ElfObject * elfObject = dynamic_cast<ElfObject *>(objFile);
00076     if(elfObject)
00077     {
00078         // modern glibc uses a bunch of auxiliary vectors to set up 
00079         // TLS as well as do a bunch of other stuff
00080         // these vectors go on the bottom of the stack, below argc/argv/envp
00081         // pointers but above actual arg strings
00082         // I don't have all the ones glibc looks at here, but so far it doesn't
00083         // seem to be a problem.
00084         // check out _dl_aux_init() in glibc/elf/dl-support.c for details
00085         // --Lisa
00086         auxv.push_back(auxv_t(M5_AT_PAGESZ, AlphaISA::VMPageSize));
00087         auxv.push_back(auxv_t(M5_AT_CLKTCK, 100));
00088         auxv.push_back(auxv_t(M5_AT_PHDR, elfObject->programHeaderTable()));
00089         DPRINTF(Loader, "auxv at PHDR %08p\n", elfObject->programHeaderTable());
00090         auxv.push_back(auxv_t(M5_AT_PHNUM, elfObject->programHeaderCount()));
00091         auxv.push_back(auxv_t(M5_AT_ENTRY, objFile->entryPoint()));
00092         auxv.push_back(auxv_t(M5_AT_UID, uid()));
00093         auxv.push_back(auxv_t(M5_AT_EUID, euid()));
00094         auxv.push_back(auxv_t(M5_AT_GID, gid()));
00095         auxv.push_back(auxv_t(M5_AT_EGID, egid()));
00096 
00097     }
00098 
00099     // Calculate how much space we need for arg & env & auxv arrays.
00100     int argv_array_size = intSize * (argv.size() + 1);
00101     int envp_array_size = intSize * (envp.size() + 1);
00102     int auxv_array_size = intSize * 2 * (auxv.size() + 1);
00103 
00104     int arg_data_size = 0;
00105     for (vector<string>::size_type i = 0; i < argv.size(); ++i) {
00106         arg_data_size += argv[i].size() + 1;
00107     }
00108     int env_data_size = 0;
00109     for (vector<string>::size_type i = 0; i < envp.size(); ++i) {
00110         env_data_size += envp[i].size() + 1;
00111     }
00112 
00113     int space_needed =
00114         argv_array_size + 
00115         envp_array_size + 
00116         auxv_array_size +
00117         arg_data_size + 
00118         env_data_size;
00119 
00120     if (space_needed < 32*1024)
00121         space_needed = 32*1024;
00122 
00123     // set bottom of stack
00124     stack_min = stack_base - space_needed;
00125     // align it
00126     stack_min = roundDown(stack_min, pageSize);
00127     stack_size = stack_base - stack_min;
00128     // map memory
00129     allocateMem(stack_min, roundUp(stack_size, pageSize));
00130 
00131     // map out initial stack contents
00132     Addr argv_array_base = stack_min + intSize; // room for argc
00133     Addr envp_array_base = argv_array_base + argv_array_size;
00134     Addr auxv_array_base = envp_array_base + envp_array_size;
00135     Addr arg_data_base = auxv_array_base + auxv_array_size;
00136     Addr env_data_base = arg_data_base + arg_data_size;
00137 
00138     // write contents to stack
00139     uint64_t argc = argv.size();
00140     if (intSize == 8)
00141         argc = htog((uint64_t)argc);
00142     else if (intSize == 4)
00143         argc = htog((uint32_t)argc);
00144     else
00145         panic("Unknown int size");
00146 
00147     initVirtMem.writeBlob(stack_min, (uint8_t*)&argc, intSize);
00148 
00149     copyStringArray(argv, argv_array_base, arg_data_base, initVirtMem);
00150     copyStringArray(envp, envp_array_base, env_data_base, initVirtMem);
00151 
00152     //Copy the aux stuff
00153     for (vector<auxv_t>::size_type x = 0; x < auxv.size(); x++) {
00154         initVirtMem.writeBlob(auxv_array_base + x * 2 * intSize,
00155                 (uint8_t*)&(auxv[x].a_type), intSize);
00156         initVirtMem.writeBlob(auxv_array_base + (x * 2 + 1) * intSize,
00157                 (uint8_t*)&(auxv[x].a_val), intSize);
00158     }
00159 
00160     ThreadContext *tc = system->getThreadContext(contextIds[0]);
00161 
00162     setSyscallArg(tc, 0, argc);
00163     setSyscallArg(tc, 1, argv_array_base);
00164     tc->setIntReg(StackPointerReg, stack_min);
00165 
00166     tc->pcState(objFile->entryPoint());
00167 }

00222 void
00223 AlphaLiveProcess::setSyscallReturn(ThreadContext *tc,
00224         SyscallReturn return_value)
00225 {
00226     // check for error condition.  Alpha syscall convention is to
00227     // indicate success/failure in reg a3 (r19) and put the
00228     // return value itself in the standard return value reg (v0).
00229     if (return_value.successful()) {
00230         // no error
00231         tc->setIntReg(SyscallSuccessReg, 0);
00232         tc->setIntReg(ReturnValueReg, return_value.value());
00233     } else {
00234         // got an error, return details
00235         tc->setIntReg(SyscallSuccessReg, (IntReg)-1);
00236         tc->setIntReg(ReturnValueReg, -return_value.value());
00237     }
00238 }
