00153 /*
00154  * Determine if the mapping at va..(va+len) is valid.
00155  */
00156 bool
00157 RemoteGDB::acc(Addr va, size_t len)
00158 {
00159     if (!FullSystem)
00160         panic("acc function needs to be rewritten for SE mode\n");
00161 
00162     Addr last_va;
00163 
00164     va = TruncPage(va);
00165     last_va = RoundPage(va + len);
00166 
00167     do  {
00168         if (IsK0Seg(va)) {
00169             if (va < (K0SegBase + system->memSize())) {
00170                 DPRINTF(GDBAcc, "acc:   Mapping is valid  K0SEG <= "
00171                         "%#x < K0SEG + size\n", va);
00172                 return true;
00173             } else {
00174                 DPRINTF(GDBAcc, "acc:   Mapping invalid %#x "
00175                         "> K0SEG + size\n", va);
00176                 return false;
00177             }
00178         }
00179 
00188         if (PcPAL(va) || va < 0x10000)
00189             return true;
00190 
00191         Addr ptbr = context->readMiscRegNoEffect(IPR_PALtemp20);
00192         PageTableEntry pte =
00193             kernel_pte_lookup(context->getPhysProxy(), ptbr, va);
00194         if (!pte.valid()) {
00195             DPRINTF(GDBAcc, "acc:   %#x pte is invalid\n", va);
00196             return false;
00197         }
00198         va += PageBytes;
00199     } while (va < last_va);
00200 
00201     DPRINTF(GDBAcc, "acc:   %#x mapping is valid\n", va);
00202     return true;
00203 }

00209 void
00210 RemoteGDB::getregs()
00211 {
00212     memset(gdbregs.regs, 0, gdbregs.bytes());
00213 
00214     gdbregs.regs[KGDB_REG_PC] = context->pcState().pc();
00215 
00216     // @todo: Currently this is very Alpha specific.
00217     if (PcPAL(gdbregs.regs[KGDB_REG_PC])) {
00218         for (int i = 0; i < NumIntArchRegs; ++i) {
00219             gdbregs.regs[i] = context->readIntReg(reg_redir[i]);
00220         }
00221     } else {
00222         for (int i = 0; i < NumIntArchRegs; ++i) {
00223             gdbregs.regs[i] = context->readIntReg(i);
00224         }
00225     }
00226 
00227 #ifdef KGDB_FP_REGS
00228     for (int i = 0; i < NumFloatArchRegs; ++i) {
00229         gdbregs.regs[i + KGDB_REG_F0] = context->readFloatRegBits(i);
00230     }
00231 #endif

00273 void
00274 RemoteGDB::setSingleStep()
00275 {
00276     PCState pc = context->pcState();
00277     PCState bpc;
00278     bool set_bt = false;
00279 
00280     // User was stopped at pc, e.g. the instruction at pc was not
00281     // executed.
00282     MachInst inst = read<MachInst>(pc.pc());
00283     StaticInstPtr si = context->getDecoderPtr()->decode(inst, pc.pc());
00284     if (si->hasBranchTarget(pc, context, bpc)) {
00285         // Don't bother setting a breakpoint on the taken branch if it
00286         // is the same as the next pc
00287         if (bpc.pc() != pc.npc())
00288             set_bt = true;
00289     }
00290 
00291     DPRINTF(GDBMisc, "setSingleStep bt_addr=%#x nt_addr=%#x\n",
00292             takenBkpt, notTakenBkpt);
00293 
00294     setTempBreakpoint(notTakenBkpt = pc.npc());
00295 
00296     if (set_bt)
00297         setTempBreakpoint(takenBkpt = bpc.pc());
00298 }
