00186 Addr
00187 AlphaSystem::fixFuncEventAddr(Addr addr)
00188 {
00189     // mask for just the opcode, Ra, and Rb fields (not the offset)
00190     const uint32_t inst_mask = 0xffff0000;
00191     // ldah gp,X(pv): opcode 9, Ra = 29, Rb = 27
00192     const uint32_t gp_ldah_pattern = (9 << 26) | (29 << 21) | (27 << 16);
00193     // lda  gp,Y(gp): opcode 8, Ra = 29, rb = 29
00194     const uint32_t gp_lda_pattern  = (8 << 26) | (29 << 21) | (29 << 16);
00195 
00196     uint32_t i1 = virtProxy.read<uint32_t>(addr);
00197     uint32_t i2 = virtProxy.read<uint32_t>(addr + sizeof(MachInst));
00198 
00199     if ((i1 & inst_mask) == gp_ldah_pattern &&
00200         (i2 & inst_mask) == gp_lda_pattern) {
00201         Addr new_addr = addr + 2 * sizeof(MachInst);
00202         DPRINTF(Loader, "fixFuncEventAddr: %p -> %p", addr, new_addr);
00203         return new_addr;
00204     } else {
00205         return addr;
00206     }
00207 }
