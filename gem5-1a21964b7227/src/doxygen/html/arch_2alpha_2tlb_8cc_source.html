00163 TLB::lookup(Addr vpn, uint8_t asn)
00164 {
00165     // assume not found...
00166     TlbEntry *retval = NULL;
00167 
00168     if (EntryCache[0]) {
00169         if (vpn == EntryCache[0]->tag &&
00170             (EntryCache[0]->asma || EntryCache[0]->asn == asn))
00171             retval = EntryCache[0];
00172         else if (EntryCache[1]) {
00173             if (vpn == EntryCache[1]->tag &&
00174                 (EntryCache[1]->asma || EntryCache[1]->asn == asn))
00175                 retval = EntryCache[1];
00176             else if (EntryCache[2] && vpn == EntryCache[2]->tag &&
00177                      (EntryCache[2]->asma || EntryCache[2]->asn == asn))
00178                 retval = EntryCache[2];
00179         }
00180     }
00181 
00182     if (retval == NULL) {
00183         PageTable::const_iterator i = lookupTable.find(vpn);
00184         if (i != lookupTable.end()) {
00185             while (i->first == vpn) {
00186                 int index = i->second;
00187                 TlbEntry *entry = &table[index];
00188                 assert(entry->valid);
00189                 if (vpn == entry->tag && (entry->asma || entry->asn == asn)) {
00190                     retval = updateCache(entry);
00191                     break;
00192                 }
00193 
00194                 ++i;
00195             }
00196         }
00197     }
00198 
00199     DPRINTF(TLB, "lookup %#x, asn %#x -> %s ppn %#x\n", vpn, (int)asn,
00200             retval ? "hit" : "miss", retval ? retval->ppn : 0);

00204 Fault
00205 TLB::checkCacheability(RequestPtr &req, bool itb)
00206 {
00207     // in Alpha, cacheability is controlled by upper-level bits of the
00208     // physical address
00209 
00210     /*
00211      * We support having the uncacheable bit in either bit 39 or bit
00212      * 40.  The Turbolaser platform (and EV5) support having the bit
00213      * in 39, but Tsunami (which Linux assumes uses an EV6) generates
00214      * accesses with the bit in 40.  So we must check for both, but we
00215      * have debug flags to catch a weird case where both are used,
00216      * which shouldn't happen.
00217      */
00218 
00219 
00220     if (req->getPaddr() & PAddrUncachedBit43) {
00221         // IPR memory space not implemented
00222         if (PAddrIprSpace(req->getPaddr())) {
00223             return new UnimpFault("IPR memory space not implemented!");
00224         } else {
00225             // mark request as uncacheable
00226             req->setFlags(Request::UNCACHEABLE);
00227 
00228             // Clear bits 42:35 of the physical address (10-2 in
00229             // Tsunami manual)
00230             req->setPaddr(req->getPaddr() & PAddrUncachedMask);
00231         }
00232         // We shouldn't be able to read from an uncachable address in Alpha as
00233         // we don't have a ROM and we don't want to try to fetch from a device 
00234         // register as we destroy any data that is clear-on-read. 
00235         if (req->isUncacheable() && itb) 
00236             return new UnimpFault("CPU trying to fetch from uncached I/O");
00237 
00238     }
00239     return NoFault;
00240 }

00370 Fault
00371 TLB::translateInst(RequestPtr req, ThreadContext *tc)
00372 {
00373     //If this is a pal pc, then set PHYSICAL
00374     if (FullSystem && PcPAL(req->getPC()))
00375         req->setFlags(Request::PHYSICAL);
00376 
00377     if (PcPAL(req->getPC())) {
00378         // strip off PAL PC marker (lsb is 1)
00379         req->setPaddr((req->getVaddr() & ~3) & PAddrImplMask);
00380         fetch_hits++;
00381         return NoFault;
00382     }
00383 
00384     if (req->getFlags() & Request::PHYSICAL) {
00385         req->setPaddr(req->getVaddr());
00386     } else {
00387         // verify that this is a good virtual address
00388         if (!validVirtualAddress(req->getVaddr())) {
00389             fetch_acv++;
00390             return new ItbAcvFault(req->getVaddr());
00391         }
00392 
00393 
00394         // VA<42:41> == 2, VA<39:13> maps directly to PA<39:13> for EV5
00395         // VA<47:41> == 0x7e, VA<40:13> maps directly to PA<40:13> for EV6
00396         if (VAddrSpaceEV6(req->getVaddr()) == 0x7e) {
00397             // only valid in kernel mode
00398             if (ICM_CM(tc->readMiscRegNoEffect(IPR_ICM)) !=
00399                 mode_kernel) {
00400                 fetch_acv++;
00401                 return new ItbAcvFault(req->getVaddr());
00402             }
00403 
00404             req->setPaddr(req->getVaddr() & PAddrImplMask);
00405 
00406             // sign extend the physical address properly
00407             if (req->getPaddr() & PAddrUncachedBit40)
00408                 req->setPaddr(req->getPaddr() | ULL(0xf0000000000));
00409             else
00410                 req->setPaddr(req->getPaddr() & ULL(0xffffffffff));
00411         } else {
00412             // not a physical address: need to look up pte
00413             int asn = DTB_ASN_ASN(tc->readMiscRegNoEffect(IPR_DTB_ASN));
00414             TlbEntry *entry = lookup(VAddr(req->getVaddr()).vpn(),
00415                               asn);
00416 
00417             if (!entry) {
00418                 fetch_misses++;
00419                 return new ItbPageFault(req->getVaddr());
00420             }
00421 
00422             req->setPaddr((entry->ppn << PageShift) +
00423                           (VAddr(req->getVaddr()).offset()
00424                            & ~3));
00425 
00426             // check permissions for this access
00427             if (!(entry->xre &
00428                   (1 << ICM_CM(tc->readMiscRegNoEffect(IPR_ICM))))) {
00429                 // instruction access fault
00430                 fetch_acv++;
00431                 return new ItbAcvFault(req->getVaddr());
00432             }
00433 
00434             fetch_hits++;
00435         }
00436     }
00437 
00438     // check that the physical address is ok (catch bad physical addresses)
00439     if (req->getPaddr() & ~PAddrImplMask) {
00440         return new MachineCheckFault();
00441     }
00442 
00443     return checkCacheability(req, true);
00444 
00445 }

00452

00457         DPRINTF(TLB, "Alignment Fault on %#x, size = %d\n", req->getVaddr(),

00465             (mode_type)ALT_MODE_AM(
00466                 tc->readMiscRegNoEffect(IPR_ALT_MODE))

00468     }
00469 
00470     if (req->getFlags() & Request::PHYSICAL) {
00471         req->setPaddr(req->getVaddr());
00472     } else {

00475             if (write) { write_acv++; } else { read_acv++; }
00476             uint64_t flags = (write ? MM_STAT_WR_MASK : 0) |
00477                 MM_STAT_BAD_VA_MASK |
00478                 MM_STAT_ACV_MASK;
00479             return new DtbPageFault(req->getVaddr(), req->getFlags(), flags);
00480         }
00481 
00482         // Check for "superpage" mapping
00483         if (VAddrSpaceEV6(req->getVaddr()) == 0x7e) {
00484             // only valid in kernel mode
00485             if (DTB_CM_CM(tc->readMiscRegNoEffect(IPR_DTB_CM)) !=
00486                 mode_kernel) {
00487                 if (write) { write_acv++; } else { read_acv++; }
00488                 uint64_t flags = ((write ? MM_STAT_WR_MASK : 0) |
00489                                   MM_STAT_ACV_MASK);
00490 
00491                 return new DtbAcvFault(req->getVaddr(), req->getFlags(),
00492                                        flags);
00493             }
00494 
00495             req->setPaddr(req->getVaddr() & PAddrImplMask);
00496 
00497             // sign extend the physical address properly
00498             if (req->getPaddr() & PAddrUncachedBit40)
00499                 req->setPaddr(req->getPaddr() | ULL(0xf0000000000));
00500             else
00501                 req->setPaddr(req->getPaddr() & ULL(0xffffffffff));
00502         } else {
00503             if (write)
00504                 write_accesses++;
00505             else
00506                 read_accesses++;
00507 
00508             int asn = DTB_ASN_ASN(tc->readMiscRegNoEffect(IPR_DTB_ASN));

00513             if (!entry) {

00522                                               flags));
00523             }
00524 
00525             req->setPaddr((entry->ppn << PageShift) +
00526                           VAddr(req->getVaddr()).offset());
00527 
00528             if (write) {

00530                     // declare the instruction access fault

00536                                             flags);

00540                     uint64_t flags = MM_STAT_WR_MASK | MM_STAT_FONW_MASK;
00541                     return new DtbPageFault(req->getVaddr(), req->getFlags(),
00542                                             flags);
00543                 }

00546                     read_acv++;

00548                         (entry->fonr ? MM_STAT_FONR_MASK : 0);
00549                     return new DtbAcvFault(req->getVaddr(), req->getFlags(),
00550                                            flags);
00551                 }
00552                 if (entry->fonr) {
00553                     read_acv++;
00554                     uint64_t flags = MM_STAT_FONR_MASK;
00555                     return new DtbPageFault(req->getVaddr(), req->getFlags(),
00556                                             flags);
00557                 }
00558             }
00559         }
00560 
00561         if (write)
00562             write_hits++;
00563         else
00564             read_hits++;

00566

00568     if (req->getPaddr() & ~PAddrImplMask) {

00571
