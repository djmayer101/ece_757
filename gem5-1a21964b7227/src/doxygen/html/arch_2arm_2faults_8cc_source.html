00100 ArmFault::invoke(ThreadContext *tc, StaticInstPtr inst)
00101 {
00102     // ARM ARM B1.6.3
00103     FaultBase::invoke(tc);
00104     if (!FullSystem)
00105         return;
00106     countStat()++;
00107 
00108     SCTLR sctlr = tc->readMiscReg(MISCREG_SCTLR);
00109     CPSR cpsr = tc->readMiscReg(MISCREG_CPSR);
00110     CPSR saved_cpsr = tc->readMiscReg(MISCREG_CPSR);
00111     saved_cpsr.nz = tc->readIntReg(INTREG_CONDCODES_NZ);
00112     saved_cpsr.c = tc->readIntReg(INTREG_CONDCODES_C);
00113     saved_cpsr.v = tc->readIntReg(INTREG_CONDCODES_V);
00114     saved_cpsr.ge = tc->readIntReg(INTREG_CONDCODES_GE);
00115 
00116     Addr curPc M5_VAR_USED = tc->pcState().pc();
00117     ITSTATE it = tc->pcState().itstate();
00118     saved_cpsr.it2 = it.top6;
00119     saved_cpsr.it1 = it.bottom2;
00120 
00121     cpsr.mode = nextMode();
00122     cpsr.it1 = cpsr.it2 = 0;
00123     cpsr.j = 0;
00124    
00125     cpsr.t = sctlr.te;
00126     cpsr.a = cpsr.a | abortDisable();
00127     cpsr.f = cpsr.f | fiqDisable();
00128     cpsr.i = 1;
00129     cpsr.e = sctlr.ee;
00130     tc->setMiscReg(MISCREG_CPSR, cpsr);
00131     // Make sure mailbox sets to one always
00132     tc->setMiscReg(MISCREG_SEV_MAILBOX, 1);
00133     tc->setIntReg(INTREG_LR, curPc +
00134             (saved_cpsr.t ? thumbPcOffset() : armPcOffset()));
00135 
00136     switch (nextMode()) {
00137       case MODE_FIQ:
00138         tc->setMiscReg(MISCREG_SPSR_FIQ, saved_cpsr);
00139         break;
00140       case MODE_IRQ:
00141         tc->setMiscReg(MISCREG_SPSR_IRQ, saved_cpsr);
00142         break;
00143       case MODE_SVC:
00144         tc->setMiscReg(MISCREG_SPSR_SVC, saved_cpsr);
00145         break;
00146       case MODE_UNDEFINED:
00147         tc->setMiscReg(MISCREG_SPSR_UND, saved_cpsr);
00148         break;
00149       case MODE_ABORT:
00150         tc->setMiscReg(MISCREG_SPSR_ABT, saved_cpsr);
00151         break;
00152       default:
00153         panic("unknown Mode\n");
00154     }
00155 
00156     Addr newPc = getVector(tc);
00157     DPRINTF(Faults, "Invoking Fault:%s cpsr:%#x PC:%#x lr:%#x newVec: %#x\n",
00158             name(), cpsr, curPc, tc->readIntReg(INTREG_LR), newPc);
00159     PCState pc(newPc);
00160     pc.thumb(cpsr.t);
00161     pc.nextThumb(pc.thumb());
