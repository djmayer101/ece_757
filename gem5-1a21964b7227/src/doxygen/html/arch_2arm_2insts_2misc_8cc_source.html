00100     }
00101     if (!foundPsr) {
00102         os << "????";

00104     }
00105     if (bits(byteMask, 3)) {
00106         if (apsr) {
00107             os << "nzcvq";

00109             os << "f";
00110         }
00111     }
00112     if (bits(byteMask, 2)) {
00113         if (apsr) {
00114             os << "g";

00116             os << "s";

00118     }
