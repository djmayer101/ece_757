00265     // Create a new unique identifier for this pid

00270             warn_once("Error out of identifiers for cache occupancy stats");

00292     std::string next_task_str = ti.curTaskName(task_descriptor);
00293 
00294     // Streamline treats pid == -1 as the kernel process.
00295     // Also pid == 0 implies idle process (except during Linux boot)
00296     int32_t mm = ti.curTaskMm(task_descriptor);
00297     bool is_kernel = (mm == 0);
00298     if (is_kernel && (pid != 0)) {
00299         pid = -1;
00300         tgid = -1;
00301         next_task_str = "kernel";
00302     }
00303 
00304     LinuxArmSystem* sys = dynamic_cast<LinuxArmSystem *>(tc->getSystemPtr());
00305     if (!sys) {
00306         panic("System is not LinuxArmSystem while getting Linux process info!");
00307     }
00308     std::map<uint32_t, uint32_t>& taskMap = sys->taskMap;
00309 
00310     // Create a new unique identifier for this pid
00311     sys->mapPid(tc, pid);
00312 
00313     // Set cpu task id, output process info, and dump stats
00314     tc->getCpuPtr()->taskId(taskMap[pid]);
00315     tc->getCpuPtr()->setPid(pid);
00316 
00317     std::ostream* taskFile = sys->taskFile;
00318 
00319     // Task file is read by cache occupancy plotting script or
00320     // Streamline conversion script.
00321     ccprintf(*taskFile,
00322              "tick=%lld %d cpu_id=%d next_pid=%d next_tgid=%d next_task=%s\n",
00323              curTick(), taskMap[pid], tc->cpuId(), (int) pid, (int) tgid,
00324              next_task_str);
00325     taskFile->flush();
