00065 void
00066 Trace::ArmNativeTrace::ThreadState::update(NativeTrace *parent)
00067 {
00068     oldState = state[current];
00069     current = (current + 1) % 2;
00070     newState = state[current];
00071 
00072     memcpy(newState, oldState, sizeof(state[0]));
00073 
00074     uint64_t diffVector;
00075     parent->read(&diffVector, sizeof(diffVector));
00076     diffVector = ArmISA::gtoh(diffVector);
00077 
00078     int changes = 0;
00079     for (int i = 0; i < STATE_NUMVALS; i++) {
00080         if (diffVector & 0x1) {
00081             changed[i] = true;
00082             changes++;
00083         } else {
00084             changed[i] = false;
00085         }
00086         diffVector >>= 1;
00087     }
00088 
00089     uint64_t values[changes];
00090     parent->read(values, sizeof(values));
00091     int pos = 0;
00092     for (int i = 0; i < STATE_NUMVALS; i++) {
00093         if (changed[i]) {
00094             newState[i] = ArmISA::gtoh(values[pos++]);
00095             changed[i] = (newState[i] != oldState[i]);
00096         }
00097     }
00098 }

00100 void
00101 Trace::ArmNativeTrace::ThreadState::update(ThreadContext *tc)
00102 {
00103     oldState = state[current];
00104     current = (current + 1) % 2;
00105     newState = state[current];
00106 
00107     // Regular int regs
00108     for (int i = 0; i < 15; i++) {
00109         newState[i] = tc->readIntReg(i);
00110         changed[i] = (oldState[i] != newState[i]);
00111     }
00112 
00113     //R15, aliased with the PC
00114     newState[STATE_PC] = tc->pcState().npc();
00115     changed[STATE_PC] = (newState[STATE_PC] != oldState[STATE_PC]);
00116 
00117     //CPSR
00118     CPSR cpsr = tc->readMiscReg(MISCREG_CPSR);
00119     cpsr.nz = tc->readIntReg(INTREG_CONDCODES_NZ);
00120     cpsr.c = tc->readIntReg(INTREG_CONDCODES_C);
00121     cpsr.v = tc->readIntReg(INTREG_CONDCODES_V);
00122     cpsr.ge = tc->readIntReg(INTREG_CONDCODES_GE);
00123 
00124     newState[STATE_CPSR] = cpsr;
00125     changed[STATE_CPSR] = (newState[STATE_CPSR] != oldState[STATE_CPSR]);
00126 
00127     for (int i = 0; i < NumFloatArchRegs; i += 2) {
00128         newState[STATE_F0 + (i >> 1)] =
00129             static_cast<uint64_t>(tc->readFloatRegBits(i + 1)) << 32 |
00130             tc->readFloatRegBits(i);
00131     }
00132     newState[STATE_FPSCR] = tc->readMiscRegNoEffect(MISCREG_FPSCR) |
00133                             tc->readIntReg(INTREG_FPCONDCODES);
00134 }

00136 void
00137 Trace::ArmNativeTrace::check(NativeTraceRecord *record)
00138 {
00139     ThreadContext *tc = record->getThread();
00140     // This area is read only on the target. It can't stop there to tell us
00141     // what's going on, so we should skip over anything there also.
00142     if (tc->nextInstAddr() > 0xffff0000)
00143         return;
00144     nState.update(this);
00145     mState.update(tc);
00146 
00147     // If a syscall just happened native trace needs another tick
00148     if ((mState.oldState[STATE_PC] == nState.oldState[STATE_PC]) &&
00149             (mState.newState[STATE_PC] - 4 == nState.newState[STATE_PC])) {
00150             DPRINTF(ExecRegDelta, "Advancing to match PCs after syscall\n");
00151             nState.update(this);
00152 
00153     }
00154 
00155     bool errorFound = false;
00156     // Regular int regs
00157     for (int i = 0; i < STATE_NUMVALS; i++) {
00158         if (nState.changed[i] || mState.changed[i]) {
00159             bool oldMatch = (mState.oldState[i] == nState.oldState[i]);
00160             bool newMatch = (mState.newState[i] == nState.newState[i]);
00161             if (oldMatch && newMatch) {
00162                 // The more things change, the more they stay the same.
00163                 continue;
00164             }
00165 
00166             errorFound = true;
00167 
00168 #ifndef NDEBUG
00169             const char *vergence = "  ";
00170             if (oldMatch && !newMatch) {
00171                 vergence = "<>";
00172             } else if (!oldMatch && newMatch) {
00173                 vergence = "><";
00174             }
00175 
00176             if (!nState.changed[i]) {
00177                 DPRINTF(ExecRegDelta, "%s [%5s] "\
00178                                       "Native:         %#010x         "\
00179                                       "M5:     %#010x => %#010x\n",
00180                                       vergence, regNames[i],
00181                                       nState.newState[i],
00182                                       mState.oldState[i], mState.newState[i]);
00183             } else if (!mState.changed[i]) {
00184                 DPRINTF(ExecRegDelta, "%s [%5s] "\
00185                                       "Native: %#010x => %#010x "\
00186                                       "M5:             %#010x        \n",
00187                                       vergence, regNames[i],
00188                                       nState.oldState[i], nState.newState[i],
00189                                       mState.newState[i]);
00190             } else {
00191                 DPRINTF(ExecRegDelta, "%s [%5s] "\
00192                                       "Native: %#010x => %#010x "\
00193                                       "M5:     %#010x => %#010x\n",
00194                                       vergence, regNames[i],
00195                                       nState.oldState[i], nState.newState[i],
00196                                       mState.oldState[i], mState.newState[i]);
00197             }
00198 #endif
00199         }
00200     }
00201     if (errorFound) {
00202         StaticInstPtr inst = record->getStaticInst();
00203         assert(inst);
00204         bool ran = true;
00205         if (inst->isMicroop()) {
00206             ran = false;
00207             inst = record->getMacroStaticInst();
00208         }
00209         assert(inst);
00210         record->traceInst(inst, ran);
00211 
00212         bool pcError = (mState.newState[STATE_PC] !=
00213                         nState.newState[STATE_PC]);
00214         if (stopOnPCError && pcError)
00215             panic("Native trace detected an error in control flow!");
00216     }
00217 }
