00077 void
00078 ArmLiveProcess::initState()
00079 {
00080     LiveProcess::initState();
00081     argsInit(MachineBytes, VMPageSize);
00082     for (int i = 0; i < contextIds.size(); i++) {
00083         ThreadContext * tc = system->getThreadContext(contextIds[i]);
00084         CPACR cpacr = tc->readMiscReg(MISCREG_CPACR);
00085         // Enable the floating point coprocessors.
00086         cpacr.cp10 = 0x3;
00087         cpacr.cp11 = 0x3;
00088         tc->setMiscReg(MISCREG_CPACR, cpacr);
00089         // Generically enable floating point support.
00090         FPEXC fpexc = tc->readMiscReg(MISCREG_FPEXC);
00091         fpexc.en = 1;
00092         tc->setMiscReg(MISCREG_FPEXC, fpexc);
00093     }
00094 }
