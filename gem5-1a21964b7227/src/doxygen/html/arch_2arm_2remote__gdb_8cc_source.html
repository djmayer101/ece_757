00169 bool
00170 RemoteGDB::acc(Addr va, size_t len)
00171 {
00172     if (FullSystem) {
00173         Addr last_va;
00174         va       = truncPage(va);
00175         last_va  = roundPage(va + len);
00176 
00177         do  {
00178             if (virtvalid(context, va)) {
00179                 return true;
00180             }
00181             va += PageBytes;
00182         } while (va < last_va);
00183 
00184         DPRINTF(GDBAcc, "acc:   %#x mapping is valid\n", va);
00185         return true;
00186     } else {
00187         TlbEntry entry;
00188         //Check to make sure the first byte is mapped into the processes address
00189         //space.
00190         if (context->getProcessPtr()->pTable->lookup(va, entry))
00191             return true;
00192         return false;

00197  * Translate the kernel debugger register format into the GDB register
00198  * format.
00199  */
00200 void
00201 RemoteGDB::getregs()
00202 {
00203     DPRINTF(GDBAcc, "getregs in remotegdb \n");
00204 
00205     memset(gdbregs.regs, 0, gdbregs.bytes());
00206 
00207     // R0-R15 supervisor mode
00208     // arm registers are 32 bits wide, gdb registers are 64 bits wide
00209     // two arm registers are packed into one gdb register (little endian)
00210     gdbregs.regs[REG_R0 + 0] = context->readIntReg(INTREG_R1) << 32 |
00211                                        context->readIntReg(INTREG_R0);
00212     gdbregs.regs[REG_R0 + 1] = context->readIntReg(INTREG_R3) << 32 |
00213                                        context->readIntReg(INTREG_R2);
00214     gdbregs.regs[REG_R0 + 2] = context->readIntReg(INTREG_R5) << 32 |
00215                                        context->readIntReg(INTREG_R4);
00216     gdbregs.regs[REG_R0 + 3] = context->readIntReg(INTREG_R7) << 32 |
00217                                        context->readIntReg(INTREG_R6);
00218     gdbregs.regs[REG_R0 + 4] = context->readIntReg(INTREG_R9) << 32 |
00219                                        context->readIntReg(INTREG_R8);
00220     gdbregs.regs[REG_R0 + 5] = context->readIntReg(INTREG_R11) << 32|
00221                                        context->readIntReg(INTREG_R10);
00222     gdbregs.regs[REG_R0 + 6] = context->readIntReg(INTREG_SP) << 32 |
00223                                        context->readIntReg(INTREG_R12);
00224     gdbregs.regs[REG_R0 + 7] = context->pcState().pc() << 32        |
00225                                        context->readIntReg(INTREG_LR);
00226 
00227     // CPSR
00228     gdbregs.regs[REG_CPSR]  = context->readMiscRegNoEffect(MISCREG_CPSR);
00229 
00230     // vfpv3/neon floating point registers (32 double or 64 float)
00231 
00232     gdbregs.regs[REG_F0] =
00233         static_cast<uint64_t>(context->readFloatRegBits(0)) << 32 |
00234         gdbregs.regs[REG_CPSR];
00235 
00236     for (int i = 1; i < (NumFloatArchRegs>>1); ++i) {
00237       gdbregs.regs[i + REG_F0] =
00238           static_cast<uint64_t>(context->readFloatRegBits(2*i)) << 32 |
00239           context->readFloatRegBits(2*i-1);
00240     }
00241 
00242   // FPSCR
00243     gdbregs.regs[REG_FPSCR] =
00244         static_cast<uint64_t>(context->readMiscRegNoEffect(MISCREG_FPSCR)) << 32 |
00245         context->readFloatRegBits(NumFloatArchRegs - 1);
00246 }

00268     context->setIntReg(INTREG_R7 , bits(gdbregs.regs[REG_R0 + 3], 63, 32));
00269     context->setIntReg(INTREG_R8 , bits(gdbregs.regs[REG_R0 + 4], 31, 0));
00270     context->setIntReg(INTREG_R9 , bits(gdbregs.regs[REG_R0 + 4], 63, 32));
00271     context->setIntReg(INTREG_R10, bits(gdbregs.regs[REG_R0 + 5], 31, 0));
00272     context->setIntReg(INTREG_R11, bits(gdbregs.regs[REG_R0 + 5], 63, 32));
00273     context->setIntReg(INTREG_R12, bits(gdbregs.regs[REG_R0 + 6], 31, 0));
00274     context->setIntReg(INTREG_SP , bits(gdbregs.regs[REG_R0 + 6], 63, 32));
00275     context->setIntReg(INTREG_LR , bits(gdbregs.regs[REG_R0 + 7], 31, 0));
00276     context->pcState(bits(gdbregs.regs[REG_R0 + 7], 63, 32));
00277 
00278     //CPSR
00279     context->setMiscRegNoEffect(MISCREG_CPSR, gdbregs.regs[REG_CPSR]);
00280 
00281     //vfpv3/neon floating point registers (32 double or 64 float)
00282     context->setFloatRegBits(0, gdbregs.regs[REG_F0]>>32);
00283 
00284     for (int i = 1; i < NumFloatArchRegs; ++i) {
00285       if(i%2){
00286         int j = (i+1)/2;
00287         context->setFloatRegBits(i, bits(gdbregs.regs[j + REG_F0], 31, 0));
00288       }
00289       else{
00290         int j = i/2;
00291         context->setFloatRegBits(i, gdbregs.regs[j + REG_F0]>>32);
00292       }
00293     }
00294 
00295     //FPSCR
00296     context->setMiscReg(MISCREG_FPSCR, gdbregs.regs[REG_FPSCR]>>32);

00312 void
00313 RemoteGDB::setSingleStep()
00314 {
00315     PCState pc = context->pcState();
00316     PCState bpc;
00317     bool set_bt = false;
00318 
00319     // User was stopped at pc, e.g. the instruction at pc was not
00320     // executed.
00321     MachInst inst = read<MachInst>(pc.pc());
00322     StaticInstPtr si = context->getDecoderPtr()->decode(inst, pc.pc());
00323     if (si->hasBranchTarget(pc, context, bpc)) {
00324         // Don't bother setting a breakpoint on the taken branch if it
00325         // is the same as the next pc
00326         if (bpc.pc() != pc.npc())
00327             set_bt = true;
00328     }
00329 
00330     DPRINTF(GDBMisc, "setSingleStep bt_addr=%#x nt_addr=%#x\n",
00331             takenBkpt, notTakenBkpt);
00332 
00333     setTempBreakpoint(notTakenBkpt = pc.npc());
00334 
00335     if (set_bt)
00336         setTempBreakpoint(takenBkpt = bpc.pc());
00337 }
