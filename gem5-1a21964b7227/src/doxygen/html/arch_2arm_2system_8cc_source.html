00070 void
00071 ArmSystem::initState()
00072 {
00073     // Moved from the constructor to here since it relies on the
00074     // address map being resolved in the interconnect
00075 
00076     // Call the initialisation of the super class
00077     System::initState();
00078 
00079     const Params* p = params();
00080 
00081     if (bootldr) {
00082         bootldr->loadSections(physProxy);
00083 
00084         uint8_t jump_to_bl[] =
00085         {
00086             0x07, 0xf0, 0xa0, 0xe1  // branch to r7
00087         };
00088         physProxy.writeBlob(0x0, jump_to_bl, sizeof(jump_to_bl));
00089 
00090         inform("Using bootloader at address %#x\n", bootldr->entryPoint());
00091 
00092         // Put the address of the boot loader into r7 so we know
00093         // where to branch to after the reset fault
00094         // All other values needed by the boot loader to know what to do
00095         if (!p->gic_cpu_addr || !p->flags_addr)
00096             fatal("gic_cpu_addr && flags_addr must be set with bootloader\n");
00097 
00098         for (int i = 0; i < threadContexts.size(); i++) {
00099             threadContexts[i]->setIntReg(3, kernelEntry & loadAddrMask);
00100             threadContexts[i]->setIntReg(4, params()->gic_cpu_addr);
00101             threadContexts[i]->setIntReg(5, params()->flags_addr);
00102             threadContexts[i]->setIntReg(7, bootldr->entryPoint());
00103         }
00104     } else {
00105         // Set the initial PC to be at start of the kernel code
00106         threadContexts[0]->pcState(kernelEntry & loadAddrMask);
00107     }
00108 }
