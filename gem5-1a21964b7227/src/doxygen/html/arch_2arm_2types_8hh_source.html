00332         advance()
00333         {
00334             Base::advance();
00335             flags = nextFlags;
00336             npc(pc() + (thumb() ? 2 : 4));
00337 
00338             if (_nextItstate) {
00339                 _itstate = _nextItstate;

00341             } else if (_itstate) {

00343                 uint8_t cond_mask = it.mask;
00344                 uint8_t thumb_cond = it.cond;
00345                 DPRINTF(Decoder, "Advancing ITSTATE from %#x,%#x.\n",

00348                 uint8_t new_bit = bits(cond_mask, 4);
00349                 cond_mask &= mask(4);
00350                 if (cond_mask == 0)
00351                     thumb_cond = 0;

00388         // Perform an interworking branch.
00389         void
00390         instIWNPC(uint32_t val)
00391         {
00392             bool thumbEE = (thumb() && jazelle());
00393 
00394             Addr newPC = val;
00395             if (thumbEE) {
00396                 if (bits(newPC, 0)) {
00397                     newPC = newPC & ~mask(1);
00398                 }  // else we have a bad interworking address; do not call
00399                    // panic() since the instruction could be executed
00400                    // speculatively
00401             } else {
00402                 if (bits(newPC, 0)) {
00403                     nextThumb(true);
00404                     newPC = newPC & ~mask(1);
00405                 } else if (!bits(newPC, 1)) {
00406                     nextThumb(false);
00407                 } else {
00408                     // This state is UNPREDICTABLE in the ARM architecture
00409                     // The easy thing to do is just mask off the bit and
00410                     // stay in the current mode, so we'll do that.
00411                     newPC &= ~mask(2);
00412                 }
00413             }
00414             npc(newPC);
00415         }
