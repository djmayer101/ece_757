00101 void
00102 MipsFaultBase::setExceptionState(ThreadContext *tc, uint8_t excCode)
00103 {
00104     // modify SRS Ctl - Save CSS, put ESS into CSS
00105     StatusReg status = tc->readMiscReg(MISCREG_STATUS);
00106     if (status.exl != 1 && status.bev != 1) {
00107         // SRS Ctl is modified only if Status_EXL and Status_BEV are not set
00108         SRSCtlReg srsCtl = tc->readMiscReg(MISCREG_SRSCTL);
00109         srsCtl.pss = srsCtl.css;
00110         srsCtl.css = srsCtl.ess;
00111         tc->setMiscRegNoEffect(MISCREG_SRSCTL, srsCtl);
00112     }
00113 
00114     // set EXL bit (don't care if it is already set!)
00115     status.exl = 1;
00116     tc->setMiscRegNoEffect(MISCREG_STATUS, status);
00117 
00118     // write EPC
00119     PCState pc = tc->pcState();
00120     DPRINTF(MipsPRA, "PC: %s\n", pc);
00121     bool delay_slot = pc.pc() + sizeof(MachInst) != pc.npc();
00122     tc->setMiscRegNoEffect(MISCREG_EPC,
00123             pc.pc() - (delay_slot ? sizeof(MachInst) : 0));
00124 
00125     // Set Cause_EXCCODE field
00126     CauseReg cause = tc->readMiscReg(MISCREG_CAUSE);
00127     cause.excCode = excCode;
00128     cause.bd = delay_slot ? 1 : 0;
00129     cause.ce = 0;
00130     tc->setMiscRegNoEffect(MISCREG_CAUSE, cause);
00131 }

00145 void
00146 ResetFault::invoke(ThreadContext *tc, StaticInstPtr inst)
00147 {
00148     if (FullSystem) {
00149         DPRINTF(MipsPRA, "%s encountered.\n", name());
00150         /* All reset activity must be invoked from here */
00151         Addr handler = vect(tc);
00152         tc->pcState(handler);
00153         DPRINTF(MipsPRA, "ResetFault::invoke : PC set to %x", handler);
00154     }
00155 
00156     // Set Coprocessor 1 (Floating Point) To Usable
00157     StatusReg status = tc->readMiscRegNoEffect(MISCREG_STATUS);
00158     status.cu.cu1 = 1;
00159     tc->setMiscReg(MISCREG_STATUS, status);
