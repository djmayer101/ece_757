00048 MipsLiveProcess::MipsLiveProcess(LiveProcessParams * params,
00049         ObjectFile *objFile)
00050     : LiveProcess(params, objFile)
00051 {
00052     // Set up stack. On MIPS, stack starts at the top of kuseg
00053     // user address space. MIPS stack grows down from here
00054     stack_base = 0x7FFFFFFF;
00055 
00056     // Set pointer for next thread stack.  Reserve 8M for main stack.
00057     next_thread_stack_base = stack_base - (8 * 1024 * 1024);
00058 
00059     // Set up break point (Top of Heap)
00060     brk_point = objFile->dataBase() + objFile->dataSize() + objFile->bssSize();
00061     brk_point = roundUp(brk_point, VMPageSize);
00062 
00063     // Set up region for mmaps.  Start it 1GB above the top of the heap.
00064     mmap_start = mmap_end = brk_point + 0x40000000L;
00065 }

00077 MipsLiveProcess::argsInit(int pageSize)
00078 {
00079     int intSize = sizeof(IntType);
00080 
00081     // load object file into target memory
00082     objFile->loadSections(initVirtMem);
00083 
00084     typedef AuxVector<IntType> auxv_t;
00085     std::vector<auxv_t> auxv;
00086 
00087     ElfObject * elfObject = dynamic_cast<ElfObject *>(objFile);
00088     if (elfObject)
00089     {
00090         // Set the system page size
00091         auxv.push_back(auxv_t(M5_AT_PAGESZ, MipsISA::VMPageSize));
00092         // Set the frequency at which time() increments
00093         auxv.push_back(auxv_t(M5_AT_CLKTCK, 100));
00094         // For statically linked executables, this is the virtual
00095         // address of the program header tables if they appear in the
00096         // executable image.
00097         auxv.push_back(auxv_t(M5_AT_PHDR, elfObject->programHeaderTable()));
00098         DPRINTF(Loader, "auxv at PHDR %08p\n", elfObject->programHeaderTable());
00099         // This is the size of a program header entry from the elf file.
00100         auxv.push_back(auxv_t(M5_AT_PHENT, elfObject->programHeaderSize()));
00101         // This is the number of program headers from the original elf file.
00102         auxv.push_back(auxv_t(M5_AT_PHNUM, elfObject->programHeaderCount()));
00103         //The entry point to the program
00104         auxv.push_back(auxv_t(M5_AT_ENTRY, objFile->entryPoint()));
00105         //Different user and group IDs
00106         auxv.push_back(auxv_t(M5_AT_UID, uid()));
00107         auxv.push_back(auxv_t(M5_AT_EUID, euid()));
00108         auxv.push_back(auxv_t(M5_AT_GID, gid()));
00109         auxv.push_back(auxv_t(M5_AT_EGID, egid()));
00110     }
00111 
00112     // Calculate how much space we need for arg & env & auxv arrays.
00113     int argv_array_size = intSize * (argv.size() + 1);
00114     int envp_array_size = intSize * (envp.size() + 1);
00115     int auxv_array_size = intSize * 2 * (auxv.size() + 1);
00116 
00117     int arg_data_size = 0;
00118     for (vector<string>::size_type i = 0; i < argv.size(); ++i) {
00119         arg_data_size += argv[i].size() + 1;
00120     }
00121     int env_data_size = 0;
00122     for (vector<string>::size_type i = 0; i < envp.size(); ++i) {
00123         env_data_size += envp[i].size() + 1;
00124     }
00125 
00126     int space_needed =
00127         argv_array_size +
00128         envp_array_size +
00129         auxv_array_size +
00130         arg_data_size +
00131         env_data_size;
00132 
00133     // set bottom of stack
00134     stack_min = stack_base - space_needed;
00135     // align it
00136     stack_min = roundDown(stack_min, pageSize);
00137     stack_size = stack_base - stack_min;
00138     // map memory
00139     allocateMem(stack_min, roundUp(stack_size, pageSize));
00140 
00141     // map out initial stack contents
00142     IntType argv_array_base = stack_min + intSize; // room for argc
00143     IntType envp_array_base = argv_array_base + argv_array_size;
00144     IntType auxv_array_base = envp_array_base + envp_array_size;
00145     IntType arg_data_base = auxv_array_base + auxv_array_size;
00146     IntType env_data_base = arg_data_base + arg_data_size;
00147 
00148     // write contents to stack
00149     IntType argc = argv.size();
00150 
00151     argc = htog((IntType)argc);
00152 
00153     initVirtMem.writeBlob(stack_min, (uint8_t*)&argc, intSize);
00154 
00155     copyStringArray(argv, argv_array_base, arg_data_base, initVirtMem);
00156 
00157     copyStringArray(envp, envp_array_base, env_data_base, initVirtMem);
00158 
00159     // Copy the aux vector
00160     for (typename vector<auxv_t>::size_type x = 0; x < auxv.size(); x++) {
00161         initVirtMem.writeBlob(auxv_array_base + x * 2 * intSize,
00162                 (uint8_t*)&(auxv[x].a_type), intSize);
00163         initVirtMem.writeBlob(auxv_array_base + (x * 2 + 1) * intSize,
00164                 (uint8_t*)&(auxv[x].a_val), intSize);
00165     }
00166 
00167     // Write out the terminating zeroed auxilliary vector
00168     for (unsigned i = 0; i < 2; i++) {
00169         const IntType zero = 0;
00170         const Addr addr = auxv_array_base + 2 * intSize * (auxv.size() + i);
00171         initVirtMem.writeBlob(addr, (uint8_t*)&zero, intSize);
00172     }
00173 
00174     ThreadContext *tc = system->getThreadContext(contextIds[0]);
00175 
00176     setSyscallArg(tc, 0, argc);
00177     setSyscallArg(tc, 1, argv_array_base);
00178     tc->setIntReg(StackPointerReg, stack_min);
00179 
00180     tc->pcState(objFile->entryPoint());
00181 }
