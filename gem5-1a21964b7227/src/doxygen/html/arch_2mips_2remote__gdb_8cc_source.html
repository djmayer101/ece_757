00170 /*
00171  * Translate the kernel debugger register format into the GDB register
00172  * format.
00173  */
00174 void
00175 RemoteGDB::getregs()
00176 {
00177     DPRINTF(GDBAcc, "getregs in remotegdb \n");
00178     memset(gdbregs.regs, 0, gdbregs.bytes());
00179 
00180     // MIPS registers are 32 bits wide, gdb registers are 64 bits wide
00181     // two MIPS registers are packed into one gdb register (little endian)
00182 
00183     // INTREG: R0~R31
00184     for (int i = 0; i < GdbIntArchRegs; i++) {
00185         gdbregs.regs[i] = pack(
00186                 context->readIntReg(i * 2),
00187                 context->readIntReg(i * 2 + 1));
00188     }
00189     // SR, LO, HI, BADVADDR, CAUSE, PC
00190     gdbregs.regs[GdbIntArchRegs + 0] = pack(
00191                 context->readMiscRegNoEffect(MISCREG_STATUS),
00192                 context->readIntReg(INTREG_LO));
00193     gdbregs.regs[GdbIntArchRegs + 1] = pack(
00194                 context->readIntReg(INTREG_HI),
00195                 context->readMiscRegNoEffect(MISCREG_BADVADDR));
00196     gdbregs.regs[GdbIntArchRegs + 2] = pack(
00197                 context->readMiscRegNoEffect(MISCREG_CAUSE),
00198                 context->pcState().pc());
00199     // FLOATREG: F0~F31
00200     for (int i = 0; i < GdbFloatArchRegs; i++) {
00201         gdbregs.regs[GdbIntRegs + i] = pack(
00202                 context->readFloatRegBits(i * 2),
00203                 context->readFloatRegBits(i * 2 + 1));
00204     }
00205     // FCR, FIR
00206     gdbregs.regs[GdbIntRegs + GdbFloatArchRegs + 0] = pack(
00207                 context->readFloatRegBits(FLOATREG_FCCR),
00208                 context->readFloatRegBits(FLOATREG_FIR));
00209 }

00214  */
00215 void
00216 RemoteGDB::setregs()
00217 {
00218     DPRINTF(GDBAcc, "setregs in remotegdb \n");
00219 
00220     // MIPS registers are 32 bits wide, gdb registers are 64 bits wide
00221     // two MIPS registers are packed into one gdb register (little endian)
00222 
00223     // INTREG: R0~R31
00224     for (int i = 0; i < GdbIntArchRegs; i++) {
00225         if (i) context->setIntReg(i * 2,
00226                 unpackLo(gdbregs.regs[i]));
00227         context->setIntReg(i * 2 + 1,
00228                 unpackHi(gdbregs.regs[i]));
00229     }
00230     // SR, LO, HI, BADVADDR, CAUSE, PC
00231     context->setMiscRegNoEffect(MISCREG_STATUS,
00232                 unpackLo(gdbregs.regs[GdbIntArchRegs + 0]));
00233     context->setIntReg(INTREG_LO,
00234                 unpackHi(gdbregs.regs[GdbIntArchRegs + 0]));
00235     context->setIntReg(INTREG_HI,
00236                 unpackLo(gdbregs.regs[GdbIntArchRegs + 1]));
00237     context->setMiscRegNoEffect(MISCREG_BADVADDR,
00238                 unpackHi(gdbregs.regs[GdbIntArchRegs + 1]));
00239     context->setMiscRegNoEffect(MISCREG_CAUSE,
00240                 unpackLo(gdbregs.regs[GdbIntArchRegs + 2]));
00241     context->pcState(
00242                 unpackHi(gdbregs.regs[GdbIntArchRegs + 2]));
00243     // FLOATREG: F0~F31
00244     for (int i = 0; i < GdbFloatArchRegs; i++) {
00245         context->setFloatRegBits(i * 2,
00246                 unpackLo(gdbregs.regs[GdbIntRegs + i]));
00247         context->setFloatRegBits(i * 2 + 1,
00248                 unpackHi(gdbregs.regs[GdbIntRegs + i]));
00249     }
00250     // FCR, FIR
00251     context->setFloatRegBits(FLOATREG_FCCR,
00252                 unpackLo(gdbregs.regs[GdbIntRegs + GdbFloatArchRegs + 0]));
00253     context->setFloatRegBits(FLOATREG_FIR,
00254                 unpackHi(gdbregs.regs[GdbIntRegs + GdbFloatArchRegs + 0]));

00270 void
00271 RemoteGDB::setSingleStep()
00272 {
00273     PCState pc = context->pcState();
00274     PCState bpc;
00275     bool set_bt = false;
00276 
00277     // User was stopped at pc, e.g. the instruction at pc was not
00278     // executed.
00279     MachInst inst = read<MachInst>(pc.pc());
00280     StaticInstPtr si = context->getDecoderPtr()->decode(inst, pc.pc());
00281     if (si->hasBranchTarget(pc, context, bpc)) {
00282         // Don't bother setting a breakpoint on the taken branch if it
00283         // is the same as the next npc
00284         if (bpc.npc() != pc.nnpc())
00285             set_bt = true;
00286     }
00287 
00288     DPRINTF(GDBMisc, "setSingleStep bt_addr=%#x nt_addr=%#x\n",
00289             takenBkpt, notTakenBkpt);
00290 
00291     notTakenBkpt = pc.nnpc();
00292     setTempBreakpoint(notTakenBkpt);
00293 
00294     if (set_bt) {
00295         takenBkpt = bpc.npc();
00296         setTempBreakpoint(takenBkpt);
00297     }
00298 }
