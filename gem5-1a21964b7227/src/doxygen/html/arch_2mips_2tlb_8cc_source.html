00091 // look up an entry in the TLB
00092 MipsISA::PTE *
00093 TLB::lookup(Addr vpn, uint8_t asn) const
00094 {
00095     // assume not found...
00096     PTE *retval = NULL;
00097     PageTable::const_iterator i = lookupTable.find(vpn);
00098     if (i != lookupTable.end()) {
00099         while (i->first == vpn) {
00100             int index = i->second;
00101             PTE *pte = &table[index];
00102 
00103             /* 1KB TLB Lookup code - from MIPS ARM Volume III - Rev. 2.50 */
00104             Addr Mask = pte->Mask;
00105             Addr InvMask = ~Mask;
00106             Addr VPN  = pte->VPN;
00107             if (((vpn & InvMask) == (VPN & InvMask)) &&
00108                     (pte->G  || (asn == pte->asid))) {
00109                 // We have a VPN + ASID Match
00110                 retval = pte;
00111                 break;
00112             }
00113             ++i;
00114         }
00115     }
00116 
00117     DPRINTF(TLB, "lookup %#x, asn %#x -> %s ppn %#x\n", vpn, (int)asn,
00118             retval ? "hit" : "miss", retval ? retval->PFN1 : 0);
00119     return retval;
00120 }

00171 void
00172 TLB::insertAt(PTE &pte, unsigned Index, int _smallPages)
00173 {
00174     smallPages = _smallPages;
00175     if (Index > size) {
00176         warn("Attempted to write at index (%d) beyond TLB size (%d)",
00177                 Index, size);
00178     } else {
00179         // Update TLB
00180         DPRINTF(TLB, "TLB[%d]: %x %x %x %x\n",
00181                 Index, pte.Mask << 11,
00182                 ((pte.VPN << 11) | pte.asid),
00183                 ((pte.PFN0 << 6) | (pte.C0 << 3) |
00184                  (pte.D0 << 2) | (pte.V0 <<1) | pte.G),
00185                 ((pte.PFN1 <<6) | (pte.C1 << 3) |
00186                  (pte.D1 << 2) | (pte.V1 <<1) | pte.G));
00187         if (table[Index].V0 == true || table[Index].V1 == true) {
00188             // Previous entry is valid
00189             PageTable::iterator i = lookupTable.find(table[Index].VPN);
00190             lookupTable.erase(i);
00191         }
00192         table[Index]=pte;
00193         // Update fast lookup table
00194         lookupTable.insert(make_pair(table[Index].VPN, Index));
00195     }
00196 }
