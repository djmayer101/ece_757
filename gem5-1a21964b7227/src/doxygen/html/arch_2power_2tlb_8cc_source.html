00087     if (i != lookupTable.end()) {
00088         while (i->first == vpn) {
00089             int index = i->second;
00090             PowerISA::PTE *pte = &table[index];
00091             Addr Mask = pte->Mask;
00092             Addr InvMask = ~Mask;
00093             Addr VPN  = pte->VPN;
00094             if (((vpn & InvMask) == (VPN & InvMask))
00095                && (pte->G  || (asn == pte->asid))) {
00096 
00097                 // We have a VPN + ASID Match
00098                 retval = pte;
00099                 break;
00100             }
00101             ++i;
00102         }
00103     }
00104
