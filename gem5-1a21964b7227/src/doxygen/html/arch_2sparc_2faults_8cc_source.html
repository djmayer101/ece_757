00295 void
00296 doREDFault(ThreadContext *tc, TrapType tt)
00297 {
00298     MiscReg TL = tc->readMiscRegNoEffect(MISCREG_TL);
00299     MiscReg TSTATE = tc->readMiscRegNoEffect(MISCREG_TSTATE);
00300     PSTATE pstate = tc->readMiscRegNoEffect(MISCREG_PSTATE);
00301     HPSTATE hpstate = tc->readMiscRegNoEffect(MISCREG_HPSTATE);
00302     MiscReg CCR = tc->readIntReg(NumIntArchRegs + 2);
00303     MiscReg ASI = tc->readMiscRegNoEffect(MISCREG_ASI);
00304     MiscReg CWP = tc->readMiscRegNoEffect(MISCREG_CWP);
00305     MiscReg CANSAVE = tc->readMiscRegNoEffect(NumIntArchRegs + 3);
00306     MiscReg GL = tc->readMiscRegNoEffect(MISCREG_GL);
00307     PCState pc = tc->pcState();
00308 
00309     TL++;
00310 
00311     Addr pcMask = pstate.am ? mask(32) : mask(64);
00312 
00313     // set TSTATE.gl to gl
00314     replaceBits(TSTATE, 42, 40, GL);
00315     // set TSTATE.ccr to ccr
00316     replaceBits(TSTATE, 39, 32, CCR);
00317     // set TSTATE.asi to asi
00318     replaceBits(TSTATE, 31, 24, ASI);
00319     // set TSTATE.pstate to pstate
00320     replaceBits(TSTATE, 20, 8, pstate);
00321     // set TSTATE.cwp to cwp
00322     replaceBits(TSTATE, 4, 0, CWP);
00323 
00324     // Write back TSTATE
00325     tc->setMiscRegNoEffect(MISCREG_TSTATE, TSTATE);
00326 
00327     // set TPC to PC
00328     tc->setMiscRegNoEffect(MISCREG_TPC, pc.pc() & pcMask);
00329     // set TNPC to NPC
00330     tc->setMiscRegNoEffect(MISCREG_TNPC, pc.npc() & pcMask);
00331 
00332     // set HTSTATE.hpstate to hpstate
00333     tc->setMiscRegNoEffect(MISCREG_HTSTATE, hpstate);
00334 
00335     // TT = trap type;
00336     tc->setMiscRegNoEffect(MISCREG_TT, tt);
00337 
00338     // Update GL
00339     tc->setMiscReg(MISCREG_GL, min<int>(GL+1, MaxGL));
00340 
00341     bool priv = pstate.priv; // just save the priv bit
00342     pstate = 0;
00343     pstate.priv = priv;
00344     pstate.pef = 1;
00345     tc->setMiscRegNoEffect(MISCREG_PSTATE, pstate);
00346 
00347     hpstate.red = 1;
00348     hpstate.hpriv = 1;
00349     hpstate.ibe = 0;
00350     hpstate.tlz = 0;
00351     tc->setMiscRegNoEffect(MISCREG_HPSTATE, hpstate);
00352 
00353     bool changedCWP = true;
00354     if (tt == 0x24)
00355         CWP++;
00356     else if (0x80 <= tt && tt <= 0xbf)
00357         CWP += (CANSAVE + 2);
00358     else if (0xc0 <= tt && tt <= 0xff)
00359         CWP--;
00360     else
00361         changedCWP = false;
00362 
00363     if (changedCWP) {
00364         CWP = (CWP + NWindows) % NWindows;
00365         tc->setMiscReg(MISCREG_CWP, CWP);
00366     }
00367 }

00374 void
00375 doNormalFault(ThreadContext *tc, TrapType tt, bool gotoHpriv)
00376 {
00377     MiscReg TL = tc->readMiscRegNoEffect(MISCREG_TL);
00378     MiscReg TSTATE = tc->readMiscRegNoEffect(MISCREG_TSTATE);
00379     PSTATE pstate = tc->readMiscRegNoEffect(MISCREG_PSTATE);
00380     HPSTATE hpstate = tc->readMiscRegNoEffect(MISCREG_HPSTATE);
00381     MiscReg CCR = tc->readIntReg(NumIntArchRegs + 2);
00382     MiscReg ASI = tc->readMiscRegNoEffect(MISCREG_ASI);
00383     MiscReg CWP = tc->readMiscRegNoEffect(MISCREG_CWP);
00384     MiscReg CANSAVE = tc->readIntReg(NumIntArchRegs + 3);
00385     MiscReg GL = tc->readMiscRegNoEffect(MISCREG_GL);
00386     PCState pc = tc->pcState();
00387 
00388     // Increment the trap level
00389     TL++;
00390     tc->setMiscRegNoEffect(MISCREG_TL, TL);
00391 
00392     Addr pcMask = pstate.am ? mask(32) : mask(64);
00393 
00394     // Save off state
00395 
00396     // set TSTATE.gl to gl
00397     replaceBits(TSTATE, 42, 40, GL);
00398     // set TSTATE.ccr to ccr
00399     replaceBits(TSTATE, 39, 32, CCR);
00400     // set TSTATE.asi to asi
00401     replaceBits(TSTATE, 31, 24, ASI);
00402     // set TSTATE.pstate to pstate
00403     replaceBits(TSTATE, 20, 8, pstate);
00404     // set TSTATE.cwp to cwp
00405     replaceBits(TSTATE, 4, 0, CWP);
00406 
00407     // Write back TSTATE
00408     tc->setMiscRegNoEffect(MISCREG_TSTATE, TSTATE);
00409 
00410     // set TPC to PC
00411     tc->setMiscRegNoEffect(MISCREG_TPC, pc.pc() & pcMask);
00412     // set TNPC to NPC
00413     tc->setMiscRegNoEffect(MISCREG_TNPC, pc.npc() & pcMask);
00414 
00415     // set HTSTATE.hpstate to hpstate
00416     tc->setMiscRegNoEffect(MISCREG_HTSTATE, hpstate);
00417 
00418     // TT = trap type;
00419     tc->setMiscRegNoEffect(MISCREG_TT, tt);
00420 
00421     // Update the global register level
00422     if (!gotoHpriv)
00423         tc->setMiscReg(MISCREG_GL, min<int>(GL + 1, MaxPGL));
00424     else
00425         tc->setMiscReg(MISCREG_GL, min<int>(GL + 1, MaxGL));
00426 
00427     // pstate.mm is unchanged
00428     pstate.pef = 1; // PSTATE.pef = whether or not an fpu is present
00429     pstate.am = 0;
00430     pstate.ie = 0;
00431     // pstate.tle is unchanged
00432     // pstate.tct = 0
00433 
00434     if (gotoHpriv) {
00435         pstate.cle = 0;
00436         // The manual says PSTATE.priv should be 0, but Legion leaves it alone
00437         hpstate.red = 0;
00438         hpstate.hpriv = 1;
00439         hpstate.ibe = 0;
00440         // hpstate.tlz is unchanged
00441         tc->setMiscRegNoEffect(MISCREG_HPSTATE, hpstate);
00442     } else { // we are going to priv
00443         pstate.priv = 1;
00444         pstate.cle = pstate.tle;
00445     }
00446     tc->setMiscRegNoEffect(MISCREG_PSTATE, pstate);
00447 
00448 
00449     bool changedCWP = true;
00450     if (tt == 0x24)
00451         CWP++;
00452     else if (0x80 <= tt && tt <= 0xbf)
00453         CWP += (CANSAVE + 2);
00454     else if (0xc0 <= tt && tt <= 0xff)
00455         CWP--;
00456     else
00457         changedCWP = false;
00458 
00459     if (changedCWP) {
00460         CWP = (CWP + NWindows) % NWindows;
00461         tc->setMiscReg(MISCREG_CWP, CWP);
00462     }
00463 }

00482 void
00483 getPrivVector(ThreadContext *tc, Addr &PC, Addr &NPC, MiscReg TT, MiscReg TL)
00484 {
00485     Addr TBA = tc->readMiscRegNoEffect(MISCREG_TBA);

00492 void
00493 SparcFaultBase::invoke(ThreadContext * tc, StaticInstPtr inst)
00494 {
00495     FaultBase::invoke(tc);
00496     if (!FullSystem)
00497         return;
00498 
00499     countStat()++;
00500 
00501     // We can refer to this to see what the trap level -was-, but something
00502     // in the middle could change it in the regfile out from under us.
00503     MiscReg tl = tc->readMiscRegNoEffect(MISCREG_TL);
00504     MiscReg tt = tc->readMiscRegNoEffect(MISCREG_TT);
00505     PSTATE pstate = tc->readMiscRegNoEffect(MISCREG_PSTATE);
00506     HPSTATE hpstate = tc->readMiscRegNoEffect(MISCREG_HPSTATE);
00507 
00508     Addr PC, NPC;
00509 
00510     PrivilegeLevel current;
00511     if (hpstate.hpriv)
00512         current = Hyperprivileged;
00513     else if (pstate.priv)
00514         current = Privileged;
00515     else
00516         current = User;
00517 
00518     PrivilegeLevel level = getNextLevel(current);
00519 
00520     if (hpstate.red || (tl == MaxTL - 1)) {
00521         getREDVector(5, PC, NPC);
00522         doREDFault(tc, tt);
00523         // This changes the hpstate and pstate, so we need to make sure we
00524         // save the old version on the trap stack in doREDFault.
00525         enterREDState(tc);
00526     } else if (tl == MaxTL) {
00527         panic("Should go to error state here.. crap\n");
00528         // Do error_state somehow?
00529         // Probably inject a WDR fault using the interrupt mechanism.
00530         // What should the PC and NPC be set to?
00531     } else if (tl > MaxPTL && level == Privileged) {
00532         // guest_watchdog fault
00533         doNormalFault(tc, trapType(), true);
00534         getHyperVector(tc, PC, NPC, 2);
00535     } else if (level == Hyperprivileged ||
