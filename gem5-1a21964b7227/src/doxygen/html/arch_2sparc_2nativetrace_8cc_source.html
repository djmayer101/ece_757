00051 void
00052 Trace::SparcNativeTrace::check(NativeTraceRecord *record)
00053 {
00054     ThreadContext *tc = record->getThread();
00055 
00056     uint64_t regVal, realRegVal;
00057 
00058     // Integer registers
00059 
00060     // I doubt a real SPARC will describe more integer registers than this.
00061     assert(SparcISA::NumIntArchRegs == 32);
00062     const char **regName = intRegNames;
00063     for (int i = 0; i < SparcISA::NumIntArchRegs; i++) {
00064         regVal = tc->readIntReg(i);
00065         read(&realRegVal, sizeof(realRegVal));
00066         realRegVal = SparcISA::gtoh(realRegVal);
00067         checkReg(*(regName++), regVal, realRegVal);
00068     }
00069 
00070     SparcISA::PCState pc = tc->pcState();
00071     // PC
00072     read(&realRegVal, sizeof(realRegVal));
00073     realRegVal = SparcISA::gtoh(realRegVal);
00074     regVal = pc.npc();
00075     checkReg("pc", regVal, realRegVal);
00076 
00077     // NPC
00078     read(&realRegVal, sizeof(realRegVal));
00079     realRegVal = SparcISA::gtoh(realRegVal);
00080     pc.nnpc();
00081     checkReg("npc", regVal, realRegVal);
00082 
00083     // CCR
00084     read(&realRegVal, sizeof(realRegVal));
00085     realRegVal = SparcISA::gtoh(realRegVal);
00086     regVal = tc->readIntReg(SparcISA::NumIntArchRegs + 2);
00087     checkReg("ccr", regVal, realRegVal);
00088 }
