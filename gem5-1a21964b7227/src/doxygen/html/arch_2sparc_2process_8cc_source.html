00070 void
00071 SparcLiveProcess::handleTrap(int trapNum, ThreadContext *tc)
00072 {
00073     PCState pc = tc->pcState();
00074     switch (trapNum) {
00075       case 0x01: // Software breakpoint
00076         warn("Software breakpoint encountered at pc %#x.\n", pc.pc());
00077         break;
00078       case 0x02: // Division by zero
00079         warn("Software signaled a division by zero at pc %#x.\n", pc.pc());
00080         break;
00081       case 0x03: // Flush window trap
00082         flushWindows(tc);
00083         break;
00084       case 0x04: // Clean windows
00085         warn("Ignoring process request for clean register "
00086                 "windows at pc %#x.\n", pc.pc());
00087         break;
00088       case 0x05: // Range check
00089         warn("Software signaled a range check at pc %#x.\n", pc.pc());
00090         break;
00091       case 0x06: // Fix alignment
00092         warn("Ignoring process request for os assisted unaligned accesses "
00093                 "at pc %#x.\n", pc.pc());
00094         break;
00095       case 0x07: // Integer overflow
00096         warn("Software signaled an integer overflow at pc %#x.\n", pc.pc());
00097         break;
00098       case 0x32: // Get integer condition codes
00099         warn("Ignoring process request to get the integer condition codes "
00100                 "at pc %#x.\n", pc.pc());
00101         break;
00102       case 0x33: // Set integer condition codes
00103         warn("Ignoring process request to set the integer condition codes "
00104                 "at pc %#x.\n", pc.pc());
00105         break;
00106       default:
00107         panic("Unimplemented trap to operating system: trap number %#x.\n", trapNum);
00108     }
00109 }

00438 {
00439     IntReg Cansave = tc->readIntReg(NumIntArchRegs + 3);
00440     IntReg Canrestore = tc->readIntReg(NumIntArchRegs + 4);
00441     IntReg Otherwin = tc->readIntReg(NumIntArchRegs + 6);
00442     MiscReg CWP = tc->readMiscReg(MISCREG_CWP);
00443     MiscReg origCWP = CWP;
00444     CWP = (CWP + Cansave + 2) % NWindows;
00445     while (NWindows - 2 - Cansave != 0) {
00446         if (Otherwin) {
00447             panic("Otherwin non-zero.\n");
00448         } else {
00449             tc->setMiscReg(MISCREG_CWP, CWP);
00450             // Do the stores
00451             IntReg sp = tc->readIntReg(StackPointerReg);
00452             for (int index = 16; index < 32; index++) {
00453                 uint32_t regVal = tc->readIntReg(index);
00454                 regVal = htog(regVal);
00455                 if (!tc->getMemProxy().tryWriteBlob(
00456                         sp + (index - 16) * 4, (uint8_t *)&regVal, 4)) {
00457                     warn("Failed to save register to the stack when "
00458                             "flushing windows.\n");
00459                 }
00460             }
00461             Canrestore--;
00462             Cansave++;
00463             CWP = (CWP + 1) % NWindows;
00464         }
00465     }
00466     tc->setIntReg(NumIntArchRegs + 3, Cansave);
00467     tc->setIntReg(NumIntArchRegs + 4, Canrestore);

00534 void
00535 SparcLiveProcess::setSyscallReturn(ThreadContext *tc,
00536         SyscallReturn return_value)
00537 {
00538     // check for error condition.  SPARC syscall convention is to
00539     // indicate success/failure in reg the carry bit of the ccr
00540     // and put the return value itself in the standard return value reg ().
00541     PSTATE pstate = tc->readMiscRegNoEffect(MISCREG_PSTATE);
00542     if (return_value.successful()) {
00543         // no error, clear XCC.C
00544         tc->setIntReg(NumIntArchRegs + 2,
00545                 tc->readIntReg(NumIntArchRegs + 2) & 0xEE);
00546         IntReg val = return_value.value();
00547         if (pstate.am)
00548             val = bits(val, 31, 0);
00549         tc->setIntReg(ReturnValueReg, val);
