00150 // RemoteGDB::acc
00151 //
00152 //      Determine if the mapping at va..(va+len) is valid.
00153 //
00154 bool
00155 RemoteGDB::acc(Addr va, size_t len)
00156 {
00157     //@Todo In NetBSD, this function checks if all addresses
00158     // from va to va + len have valid page map entries. Not
00159     // sure how this will work for other OSes or in general.
00160     if (FullSystem) {
00161         if (va)
00162             return true;
00163         return false;
00164     } else {
00165         TlbEntry entry;
00166         // Check to make sure the first byte is mapped into the processes
00167         // address space.
00168         if (context->getProcessPtr()->pTable->lookup(va, entry))
00169             return true;
00170         return false;
00171     }
00172 }

00175 // RemoteGDB::getregs
00176 //
00177 //      Translate the kernel debugger register format into
00178 //      the GDB register format.
00179 void
00180 RemoteGDB::getregs()
00181 {
00182     memset(gdbregs.regs, 0, gdbregs.size);
00183 
00184     PCState pc = context->pcState();
00185     PSTATE pstate = context->readMiscReg(MISCREG_PSTATE);
00186 
00187     if (pstate.am) {
00188         uint32_t *regs;
00189         regs = (uint32_t*)gdbregs.regs;
00190         regs[Reg32Pc] = htobe((uint32_t)pc.pc());
00191         regs[Reg32Npc] = htobe((uint32_t)pc.npc());
00192         for (int x = RegG0; x <= RegI0 + 7; x++)
00193             regs[x] = htobe((uint32_t)context->readIntReg(x - RegG0));
00194 
00195         regs[Reg32Y] = htobe((uint32_t)context->readIntReg(NumIntArchRegs + 1));
00196         regs[Reg32Psr] = htobe((uint32_t)pstate);
00197         regs[Reg32Fsr] = htobe((uint32_t)context->readMiscReg(MISCREG_FSR));
00198         regs[Reg32Csr] = htobe((uint32_t)context->readIntReg(NumIntArchRegs + 2));
00199     } else {
00200         gdbregs.regs[RegPc] = htobe(pc.pc());
00201         gdbregs.regs[RegNpc] = htobe(pc.npc());
00202         for (int x = RegG0; x <= RegI0 + 7; x++)
00203             gdbregs.regs[x] = htobe(context->readIntReg(x - RegG0));
00204 
00205         gdbregs.regs[RegFsr] = htobe(context->readMiscReg(MISCREG_FSR));
00206         gdbregs.regs[RegFprs] = htobe(context->readMiscReg(MISCREG_FPRS));
00207         gdbregs.regs[RegY] = htobe(context->readIntReg(NumIntArchRegs + 1));
00208         gdbregs.regs[RegState] = htobe(
00209             context->readMiscReg(MISCREG_CWP) |
00210             pstate << 8 |
00211             context->readMiscReg(MISCREG_ASI) << 24 |
00212             context->readIntReg(NumIntArchRegs + 2) << 32);
00213     }
00214 
00215     DPRINTF(GDBRead, "PC=%#x\n", gdbregs.regs[RegPc]);
00216 
00217     // Floating point registers are left at 0 in netbsd
00218     // All registers other than the pc, npc and int regs
00219     // are ignored as well.
00220 }

00223 // RemoteGDB::setregs
00224 //
00225 //      Translate the GDB register format into the kernel
00226 //      debugger register format.
00227 //
00228 void
00229 RemoteGDB::setregs()
00230 {
00231     PCState pc;
00232     pc.pc(gdbregs.regs[RegPc]);
00233     pc.npc(gdbregs.regs[RegNpc]);
00234     pc.nnpc(pc.npc() + sizeof(MachInst));
00235     pc.upc(0);
00236     pc.nupc(1);
00237     context->pcState(pc);
00238     for (int x = RegG0; x <= RegI0 + 7; x++)
00239         context->setIntReg(x - RegG0, gdbregs.regs[x]);
00240     // Only the integer registers, pc and npc are set in netbsd
00241 }
