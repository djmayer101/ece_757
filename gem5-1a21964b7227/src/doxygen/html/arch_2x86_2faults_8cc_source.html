00053     void X86FaultBase::invoke(ThreadContext * tc, StaticInstPtr inst)
00054     {
00055         if (!FullSystem) {
00056             FaultBase::invoke(tc, inst);
00057             return;
00058         }
00059 
00060         PCState pcState = tc->pcState();
00061         Addr pc = pcState.pc();
00062         DPRINTF(Faults, "RIP %#x: vector %d: %s\n",
00063                 pc, vector, describe());
00064         using namespace X86ISAInst::RomLabels;
00065         HandyM5Reg m5reg = tc->readMiscRegNoEffect(MISCREG_M5_REG);
00066         MicroPC entry;
00067         if (m5reg.mode == LongMode) {
00068             if (isSoft()) {
00069                 entry = extern_label_longModeSoftInterrupt;
00070             } else {
00071                 entry = extern_label_longModeInterrupt;
00072             }
00073         } else {
00074             entry = extern_label_legacyModeInterrupt;
00075         }
00076         tc->setIntReg(INTREG_MICRO(1), vector);
00077         tc->setIntReg(INTREG_MICRO(7), pc);
00078         if (errorCode != (uint64_t)(-1)) {
00079             if (m5reg.mode == LongMode) {
00080                 entry = extern_label_longModeInterruptWithError;
00081             } else {
00082                 panic("Legacy mode interrupts with error codes "
00083                         "aren't implementde.\n");
00084             }
00085             // Software interrupts shouldn't have error codes. If one
00086             // does, there would need to be microcode to set it up.
00087             assert(!isSoft());
00088             tc->setIntReg(INTREG_MICRO(15), errorCode);
00089         }
00090         pcState.upc(romMicroPC(entry));
00091         pcState.nupc(romMicroPC(entry) + 1);
00092         tc->pcState(pcState);
00093     }

00135     void PageFault::invoke(ThreadContext * tc, StaticInstPtr inst)
00136     {
00137         if (FullSystem) {
00138             HandyM5Reg m5reg = tc->readMiscRegNoEffect(MISCREG_M5_REG);
00139             X86FaultBase::invoke(tc);
00140             /*
00141              * If something bad happens while trying to enter the page fault
00142              * handler, I'm pretty sure that's a double fault and then all
00143              * bets are off. That means it should be safe to update this
00144              * state now.
00145              */
00146             if (m5reg.mode == LongMode) {
00147                 tc->setMiscReg(MISCREG_CR2, addr);
00148             } else {
00149                 tc->setMiscReg(MISCREG_CR2, (uint32_t)addr);
00150             }
00151         } else {
00152             PageFaultErrorCode code = errorCode;
00153             const char *modeStr = "";
00154             if (code.fetch)
00155                 modeStr = "execute";
00156             else if (code.write)
00157                 modeStr = "write";
00158             else
00159                 modeStr = "read";
00160             panic("Tried to %s unmapped address %#x.\n", modeStr, addr);
00161         }
00162     }
