00080 void
00081 TLB::evictLRU()
00082 {
00083     // Find the entry with the lowest (and hence least recently updated)
00084     // sequence number.
00085 
00086     unsigned lru = 0;
00087     for (unsigned i = 1; i < size; i++) {
00088         if (tlb[i].lruSeq < tlb[lru].lruSeq)
00089             lru = i;
00090     }
00091 
00092     assert(tlb[lru].trieHandle);
00093     trie.remove(tlb[lru].trieHandle);
00094     tlb[lru].trieHandle = NULL;
00095     freeList.push_back(&tlb[lru]);
00096 }

00098 TlbEntry *
00099 TLB::insert(Addr vpn, TlbEntry &entry)
00100 {
00101     // If somebody beat us to it, just use that existing entry.
00102     TlbEntry *newEntry = trie.lookup(vpn);
00103     if (newEntry) {
00104         assert(newEntry->vaddr == vpn);
00105         return newEntry;
00106     }
00107 
00108     if (freeList.empty())
00109         evictLRU();
00110 
00111     newEntry = freeList.front();
00112     freeList.pop_front();
00113 
00114     *newEntry = entry;
00115     newEntry->lruSeq = nextSeq();
00116     newEntry->vaddr = vpn;
00117     newEntry->trieHandle =
00118     trie.insert(vpn, TlbEntryTrie::MaxBits - entry.logBytes, newEntry);
00119     return newEntry;

00174 Fault
00175 TLB::translateInt(RequestPtr req, ThreadContext *tc)
00176 {
00177     DPRINTF(TLB, "Addresses references internal memory.\n");
00178     Addr vaddr = req->getVaddr();
00179     Addr prefix = (vaddr >> 3) & IntAddrPrefixMask;
00180     if (prefix == IntAddrPrefixCPUID) {
00181         panic("CPUID memory space not yet implemented!\n");
00182     } else if (prefix == IntAddrPrefixMSR) {
00183         vaddr = (vaddr >> 3) & ~IntAddrPrefixMask;
00184         req->setFlags(Request::MMAPPED_IPR);
00185 
00186         MiscRegIndex regNum;
00187         if (!msrAddrToIndex(regNum, vaddr))
00188             return new GeneralProtection(0);
00189 
00190         //The index is multiplied by the size of a MiscReg so that
00191         //any memory dependence calculations will not see these as
00192         //overlapping.
00193         req->setPaddr((Addr)regNum * sizeof(MiscReg));
00194         return NoFault;
00195     } else if (prefix == IntAddrPrefixIO) {
00196         // TODO If CPL > IOPL or in virtual mode, check the I/O permission
00197         // bitmap in the TSS.
00198 
00199         Addr IOPort = vaddr & ~IntAddrPrefixMask;
00200         // Make sure the address fits in the expected 16 bit IO address
00201         // space.
00202         assert(!(IOPort & ~0xFFFF));
00203         if (IOPort == 0xCF8 && req->getSize() == 4) {
00204             req->setFlags(Request::MMAPPED_IPR);
00205             req->setPaddr(MISCREG_PCI_CONFIG_ADDRESS * sizeof(MiscReg));
00206         } else if ((IOPort & ~mask(2)) == 0xCFC) {
00207             req->setFlags(Request::UNCACHEABLE);
00208             Addr configAddress =
00209                 tc->readMiscRegNoEffect(MISCREG_PCI_CONFIG_ADDRESS);
00210             if (bits(configAddress, 31, 31)) {
00211                 req->setPaddr(PhysAddrPrefixPciConfig |
00212                         mbits(configAddress, 30, 2) |
00213                         (IOPort & mask(2)));
00214             } else {
00215                 req->setPaddr(PhysAddrPrefixIO | IOPort);
00216             }
00217         } else {
00218             req->setFlags(Request::UNCACHEABLE);
00219             req->setPaddr(PhysAddrPrefixIO | IOPort);
00220         }
00221         return NoFault;
00222     } else {
00223         panic("Access to unrecognized internal address space %#x.\n",
00224                 prefix);
00225     }
00226 }

00229 TLB::translate(RequestPtr req, ThreadContext *tc, Translation *translation,
00230         Mode mode, bool &delayedResponse, bool timing)
00231 {
00232     uint32_t flags = req->getFlags();
00233     int seg = flags & SegmentFlagMask;
00234     bool storeCheck = flags & (StoreCheck << FlagShift);
00235 
00236     delayedResponse = false;
00237 
00238     // If this is true, we're dealing with a request to a non-memory address
00239     // space.
00240     if (seg == SEGMENT_REG_MS) {
00241         return translateInt(req, tc);
00242     }
00243 
00244     Addr vaddr = req->getVaddr();
00245     DPRINTF(TLB, "Translating vaddr %#x.\n", vaddr);
00246 
00247     HandyM5Reg m5Reg = tc->readMiscRegNoEffect(MISCREG_M5_REG);
00248 
00249     // If protected mode has been enabled...
00250     if (m5Reg.prot) {
00251         DPRINTF(TLB, "In protected mode.\n");
00252         // If we're not in 64-bit mode, do protection/limit checks
00253         if (m5Reg.mode != LongMode) {
00254             DPRINTF(TLB, "Not in long mode. Checking segment protection.\n");
00255             // Check for a NULL segment selector.
00256             if (!(seg == SEGMENT_REG_TSG || seg == SYS_SEGMENT_REG_IDTR ||
00257                         seg == SEGMENT_REG_HS || seg == SEGMENT_REG_LS)
00258                     && !tc->readMiscRegNoEffect(MISCREG_SEG_SEL(seg)))
00259                 return new GeneralProtection(0);
00260             bool expandDown = false;
00261             SegAttr attr = tc->readMiscRegNoEffect(MISCREG_SEG_ATTR(seg));
00262             if (seg >= SEGMENT_REG_ES && seg <= SEGMENT_REG_HS) {
00263                 if (!attr.writable && (mode == Write || storeCheck))
00264                     return new GeneralProtection(0);
00265                 if (!attr.readable && mode == Read)
00266                     return new GeneralProtection(0);
00267                 expandDown = attr.expandDown;
00268 
00269             }
00270             Addr base = tc->readMiscRegNoEffect(MISCREG_SEG_BASE(seg));
00271             Addr limit = tc->readMiscRegNoEffect(MISCREG_SEG_LIMIT(seg));
00272             bool sizeOverride = (flags & (AddrSizeFlagBit << FlagShift));
00273             unsigned logSize = sizeOverride ? (unsigned)m5Reg.altAddr
00274                                             : (unsigned)m5Reg.defAddr;
00275             int size = (1 << logSize) * 8;
00276             Addr offset = bits(vaddr - base, size - 1, 0);
00277             Addr endOffset = offset + req->getSize() - 1;
00278             if (expandDown) {
00279                 DPRINTF(TLB, "Checking an expand down segment.\n");
00280                 warn_once("Expand down segments are untested.\n");
00281                 if (offset <= limit || endOffset <= limit)
00282                     return new GeneralProtection(0);
00283             } else {
00284                 if (offset > limit || endOffset > limit)
00285                     return new GeneralProtection(0);
00286             }
00287         }
00288         if (m5Reg.submode != SixtyFourBitMode ||
00289                 (flags & (AddrSizeFlagBit << FlagShift)))
00290             vaddr &= mask(32);
00291         // If paging is enabled, do the translation.
00292         if (m5Reg.paging) {
00293             DPRINTF(TLB, "Paging enabled.\n");
00294             // The vaddr already has the segment base applied.
00295             TlbEntry *entry = lookup(vaddr);
00296             if (!entry) {
00297                 if (FullSystem) {
00298                     Fault fault = walker->start(tc, translation, req, mode);
00299                     if (timing || fault != NoFault) {
00300                         // This gets ignored in atomic mode.
00301                         delayedResponse = true;
00302                         return fault;
00303                     }
00304                     entry = lookup(vaddr);
00305                     assert(entry);
00306                 } else {
00307                     DPRINTF(TLB, "Handling a TLB miss for "
00308                             "address %#x at pc %#x.\n",
00309                             vaddr, tc->instAddr());
00310 
00311                     Process *p = tc->getProcessPtr();
00312                     TlbEntry newEntry;
00313                     bool success = p->pTable->lookup(vaddr, newEntry);
00314                     if (!success && mode != Execute) {
00315                         // Check if we just need to grow the stack.
00316                         if (p->fixupStackFault(vaddr)) {
00317                             // If we did, lookup the entry for the new page.
00318                             success = p->pTable->lookup(vaddr, newEntry);
00319                         }
00320                     }
00321                     if (!success) {
00322                         return new PageFault(vaddr, true, mode, true, false);
00323                     } else {
00324                         Addr alignedVaddr = p->pTable->pageAlign(vaddr);
00325                         DPRINTF(TLB, "Mapping %#x to %#x\n", alignedVaddr,
00326                                 newEntry.pageStart());
00327                         entry = insert(alignedVaddr, newEntry);
00328                     }
00329                     DPRINTF(TLB, "Miss was serviced.\n");
00330                 }
00331             }
00332 
00333             DPRINTF(TLB, "Entry found with paddr %#x, "
00334                     "doing protection checks.\n", entry->paddr);
00335             // Do paging protection checks.
00336             bool inUser = (m5Reg.cpl == 3 &&
00337                     !(flags & (CPL0FlagBit << FlagShift)));
00338             CR0 cr0 = tc->readMiscRegNoEffect(MISCREG_CR0);
00339             bool badWrite = (!entry->writable && (inUser || cr0.wp));
00340             if ((inUser && !entry->user) || (mode == Write && badWrite)) {
00341                 // The page must have been present to get into the TLB in
00342                 // the first place. We'll assume the reserved bits are
00343                 // fine even though we're not checking them.
00344                 return new PageFault(vaddr, true, mode, inUser, false);
00345             }
00346             if (storeCheck && badWrite) {
00347                 // This would fault if this were a write, so return a page
00348                 // fault that reflects that happening.
00349                 return new PageFault(vaddr, true, Write, inUser, false);
00350             }
00351 
00352             Addr paddr = entry->paddr | (vaddr & mask(entry->logBytes));
00353             DPRINTF(TLB, "Translated %#x -> %#x.\n", vaddr, paddr);
00354             req->setPaddr(paddr);
00355             if (entry->uncacheable)
00356                 req->setFlags(Request::UNCACHEABLE);
00357         } else {
00358             //Use the address which already has segmentation applied.
00359             DPRINTF(TLB, "Paging disabled.\n");
00360             DPRINTF(TLB, "Translated %#x -> %#x.\n", vaddr, vaddr);
00361             req->setPaddr(vaddr);
00362         }
00363     } else {
00364         // Real mode
00365         DPRINTF(TLB, "In real mode.\n");
00366         DPRINTF(TLB, "Translated %#x -> %#x.\n", vaddr, vaddr);
00367         req->setPaddr(vaddr);
00368     }
00369     // Check for an access to the local APIC
00370     if (FullSystem) {
00371         LocalApicBase localApicBase =
00372             tc->readMiscRegNoEffect(MISCREG_APIC_BASE);
00373         Addr baseAddr = localApicBase.base * PageBytes;
00374         Addr paddr = req->getPaddr();
00375         if (baseAddr <= paddr && baseAddr + PageBytes > paddr) {
00376             // The Intel developer's manuals say the below restrictions apply,
00377             // but the linux kernel, because of a compiler optimization, breaks
00378             // them.
00379             /*
00380             // Check alignment
00381             if (paddr & ((32/8) - 1))
00382                 return new GeneralProtection(0);
00383             // Check access size
00384             if (req->getSize() != (32/8))
00385                 return new GeneralProtection(0);
00386             */
00387             // Force the access to be uncacheable.
00388             req->setFlags(Request::UNCACHEABLE);
00389             req->setPaddr(x86LocalAPICAddress(tc->contextId(),
00390                         paddr - baseAddr));
00391         }
00392     }
00393     return NoFault;
00394 }
