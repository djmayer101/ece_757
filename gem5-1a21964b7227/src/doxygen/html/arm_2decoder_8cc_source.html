00042 void
00043 Decoder::process()
00044 {
00045     // emi is typically ready, with some caveats below...
00046     instDone = true;
00047 
00048     if (!emi.thumb) {
00049         emi.instBits = data;
00050         emi.sevenAndFour = bits(data, 7) && bits(data, 4);
00051         emi.isMisc = (bits(data, 24, 23) == 0x2 &&
00052                       bits(data, 20) == 0);
00053         consumeBytes(4);
00054         DPRINTF(Decoder, "Arm inst: %#x.\n", (uint64_t)emi);
00055     } else {
00056         uint16_t word = (data >> (offset * 8));
00057         if (bigThumb) {
00058             // A 32 bit thumb inst is half collected.
00059             emi.instBits = emi.instBits | word;
00060             bigThumb = false;
00061             consumeBytes(2);
00062             DPRINTF(Decoder, "Second half of 32 bit Thumb: %#x.\n",
00063                     emi.instBits);
00064         } else {
00065             uint16_t highBits = word & 0xF800;
00066             if (highBits == 0xE800 || highBits == 0xF000 ||
00067                     highBits == 0xF800) {
00068                 // The start of a 32 bit thumb inst.
00069                 emi.bigThumb = 1;
00070                 if (offset == 0) {
00071                     // We've got the whole thing.
00072                     emi.instBits = (data >> 16) | (data << 16);
00073                     DPRINTF(Decoder, "All of 32 bit Thumb: %#x.\n",
00074                             emi.instBits);
00075                     consumeBytes(4);
00076                 } else {
00077                     // We only have the first half word.
00078                     DPRINTF(Decoder,
00079                             "First half of 32 bit Thumb.\n");
00080                     emi.instBits = (uint32_t)word << 16;
00081                     bigThumb = true;
00082                     consumeBytes(2);
00083                     // emi not ready yet.
00084                     instDone = false;
00085                 }
00086             } else {
00087                 // A 16 bit thumb inst.
00088                 consumeBytes(2);
00089                 emi.instBits = word;
00090                 // Set the condition code field artificially.
00091                 emi.condCode = COND_UC;
00092                 DPRINTF(Decoder, "16 bit Thumb: %#x.\n",
00093                         emi.instBits);
00094                 if (bits(word, 15, 8) == 0xbf &&
00095                         bits(word, 3, 0) != 0x0) {
00096                     foundIt = true;
00097                     itBits = bits(word, 7, 0);
00098                     DPRINTF(Decoder,
00099                             "IT detected, cond = %#x, mask = %#x\n",
00100                             itBits.cond, itBits.mask);
00101                 }
00102             }
00103         }
00104     }
00105 }
