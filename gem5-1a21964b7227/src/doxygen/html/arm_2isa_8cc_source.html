00197 MiscReg
00198 ISA::readMiscReg(int misc_reg, ThreadContext *tc)
00199 {
00200     ArmSystem *arm_sys;
00201 
00202     if (misc_reg == MISCREG_CPSR) {
00203         CPSR cpsr = miscRegs[misc_reg];
00204         PCState pc = tc->pcState();
00205         cpsr.j = pc.jazelle() ? 1 : 0;
00206         cpsr.t = pc.thumb() ? 1 : 0;
00207         return cpsr;
00208     }
00209     if (misc_reg >= MISCREG_CP15_UNIMP_START)
00210         panic("Unimplemented CP15 register %s read.\n",
00211               miscRegName[misc_reg]);
00212 
00213     switch (misc_reg) {
00214       case MISCREG_MPIDR:
00215         arm_sys = dynamic_cast<ArmSystem*>(tc->getSystemPtr());
00216         assert(arm_sys);
00217 
00218         if (arm_sys->multiProc) {
00219             return 0x80000000 | // multiprocessor extensions available
00220                    tc->cpuId();
00221         } else {
00222             return 0x80000000 |  // multiprocessor extensions available
00223                    0x40000000 |  // in up system
00224                    tc->cpuId();
00225         }
00226         break;
00227       case MISCREG_CLIDR:
00228         warn_once("The clidr register always reports 0 caches.\n");
00229         warn_once("clidr LoUIS field of 0b001 to match current "
00230                   "ARM implementations.\n");
00231         return 0x00200000;
00232       case MISCREG_CCSIDR:
00233         warn_once("The ccsidr register isn't implemented and "
00234                 "always reads as 0.\n");
00235         break;
00236       case MISCREG_CTR:
00237         {
00238             //all caches have the same line size in gem5
00239             //4 byte words in ARM
00240             unsigned lineSizeWords =
00241                 tc->getCpuPtr()->getInstPort().peerBlockSize() / 4;
00242             unsigned log2LineSizeWords = 0;
00243 
00244             while (lineSizeWords >>= 1) {
00245                 ++log2LineSizeWords;
00246             }
00247 
00248             CTR ctr = 0;
00249             //log2 of minimun i-cache line size (words)
00250             ctr.iCacheLineSize = log2LineSizeWords;
00251             //b11 - gem5 uses pipt
00252             ctr.l1IndexPolicy = 0x3;
00253             //log2 of minimum d-cache line size (words)
00254             ctr.dCacheLineSize = log2LineSizeWords;
00255             //log2 of max reservation size (words)
00256             ctr.erg = log2LineSizeWords;
00257             //log2 of max writeback size (words)
00258             ctr.cwg = log2LineSizeWords;
00259             //b100 - gem5 format is ARMv7
00260             ctr.format = 0x4;
00261 
00262             return ctr;
00263         }
00264       case MISCREG_ACTLR:
00265         warn("Not doing anything for miscreg ACTLR\n");
00266         break;
00267       case MISCREG_PMCR:
00268       case MISCREG_PMCCNTR:
00269       case MISCREG_PMSELR:
00270         warn("Not doing anything for read to miscreg %s\n",
00271                 miscRegName[misc_reg]);
00272         break;
00273       case MISCREG_CPSR_Q:
00274         panic("shouldn't be reading this register seperately\n");
00275       case MISCREG_FPSCR_QC:
00276         return readMiscRegNoEffect(MISCREG_FPSCR) & ~FpscrQcMask;
00277       case MISCREG_FPSCR_EXC:
00278         return readMiscRegNoEffect(MISCREG_FPSCR) & ~FpscrExcMask;
00279       case MISCREG_L2CTLR:
00280         {
00281             // mostly unimplemented, just set NumCPUs field from sim and return
00282             L2CTLR l2ctlr = 0;
00283             // b00:1CPU to b11:4CPUs
00284             l2ctlr.numCPUs = tc->getSystemPtr()->numContexts() - 1;
00285             return l2ctlr;
00286         }
00287       case MISCREG_DBGDIDR:
00288         /* For now just implement the version number.
00289          * Return 0 as we don't support debug architecture yet.
00290          */
00291         return 0;
00292       case MISCREG_DBGDSCR_INT:
00293         return 0;
00294     }
00295     return readMiscRegNoEffect(misc_reg);
