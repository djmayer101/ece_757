00065 uint64_t
00066 getArgument(ThreadContext *tc, int &number, uint16_t size, bool fp)
00067 {
00068     if (!FullSystem) {
00069         panic("getArgument() only implemented for full system mode.\n");
00070         M5_DUMMY_RETURN
00071     }
00072 
00073     if (size == (uint16_t)(-1))
00074         size = ArmISA::MachineBytes;
00075     if (fp)
00076         panic("getArgument(): Floating point arguments not implemented\n");
00077 
00078     if (number < NumArgumentRegs) {
00079         // If the argument is 64 bits, it must be in an even regiser
00080         // number. Increment the number here if it isn't even.
00081         if (size == sizeof(uint64_t)) {
00082             if ((number % 2) != 0)
00083                 number++;
00084             // Read the two halves of the data. Number is inc here to
00085             // get the second half of the 64 bit reg.
00086             uint64_t tmp;
00087             tmp = tc->readIntReg(number++);
00088             tmp |= tc->readIntReg(number) << 32;
00089             return tmp;
00090         } else {
00091            return tc->readIntReg(number);
00092         }
00093     } else {
00094         Addr sp = tc->readIntReg(StackPointerReg);
00095         FSTranslatingPortProxy &vp = tc->getVirtProxy();
00096         uint64_t arg;
00097         if (size == sizeof(uint64_t)) {
00098             // If the argument is even it must be aligned
00099             if ((number % 2) != 0)
00100                 number++;
00101             arg = vp.read<uint64_t>(sp +
00102                     (number-NumArgumentRegs) * sizeof(uint32_t));
00103             // since two 32 bit args == 1 64 bit arg, increment number
00104             number++;
00105         } else {
00106             arg = vp.read<uint32_t>(sp +
00107                            (number-NumArgumentRegs) * sizeof(uint32_t));
00108         }
00109         return arg;
00110     }
00111 }

00127 void
00128 copyRegs(ThreadContext *src, ThreadContext *dest)
00129 {
00130     for (int i = 0; i < TheISA::NumIntRegs; i++)
00131         dest->setIntRegFlat(i, src->readIntRegFlat(i));
00132 
00133     for (int i = 0; i < TheISA::NumFloatRegs; i++)
00134         dest->setFloatRegFlat(i, src->readFloatRegFlat(i));
00135 
00136     for (int i = 0; i < TheISA::NumMiscRegs; i++)
00137         dest->setMiscRegNoEffect(i, src->readMiscRegNoEffect(i));
00138 
00139     // setMiscReg "with effect" will set the misc register mapping correctly.
00140     // e.g. updateRegMap(val)
00141     dest->setMiscReg(MISCREG_CPSR, src->readMiscRegNoEffect(MISCREG_CPSR));
00142 
00143     // Copy over the PC State
00144     dest->pcState(src->pcState());
00145 
00146     // Invalidate the tlb misc register cache
00147     dest->getITBPtr()->invalidateMiscReg();
00148     dest->getDTBPtr()->invalidateMiscReg();
