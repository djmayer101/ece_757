00067 //
00068 // This implements a cprintf based panic() function.  panic() should
00069 // be called when something happens that should never ever happen
00070 // regardless of what the user does (i.e., an acutal m5 bug).  panic()
00071 // calls abort which can dump core or enter the debugger.
00072 //
00073 //
00074 #define panic(...) exit_message("panic", -1, __VA_ARGS__)
00075 
00076 //
00077 // This implements a cprintf based fatal() function.  fatal() should
00078 // be called when the simulation cannot continue due to some condition
00079 // that is the user's fault (bad configuration, invalid arguments,
00080 // etc.) and not a simulator bug.  fatal() calls exit(1), i.e., a
00081 // "normal" exit with an error code, as opposed to abort() like
00082 // panic() does.
00083 //
00084 #define fatal(...) exit_message("fatal", 1, __VA_ARGS__)

00104 // Only print the message the first time this expression is
00105 // encountered.  i.e.  This doesn't check the string itself and
00106 // prevent duplicate strings, this prevents the statement from
00107 // happening more than once. So, even if the arguments change and that
00108 // would have resulted in a different message thoes messages would be
00109 // supressed.
00110 #define base_message_once(...) do {                     \
00111         static bool once = false;                       \
00112         if (!once) {                                    \
00113             base_message(__VA_ARGS__);                  \
00114             once = true;                                \
00115         }                                               \
00116     } while (0)
