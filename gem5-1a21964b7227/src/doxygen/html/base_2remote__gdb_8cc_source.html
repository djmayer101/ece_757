00362 // Send a packet to gdb
00363 void
00364 BaseRemoteGDB::send(const char *bp)
00365 {
00366     const char *p;
00367     uint8_t csum, c;
00368 
00369     DPRINTF(GDBSend, "send:  %s\n", bp);
00370 
00371     do {
00372         p = bp;
00373         //Start sending a packet
00374         putbyte(GDBStart);
00375         //Send the contents, and also keep a check sum.
00376         for (csum = 0; (c = *p); p++) {
00377             putbyte(c);
00378             csum += c;
00379         }
00380         //Send the ending character.
00381         putbyte(GDBEnd);
00382         //Sent the checksum.
00383         putbyte(i2digit(csum >> 4));
00384         putbyte(i2digit(csum));
00385         //Try transmitting over and over again until the other end doesn't send an
00386         //error back.
00387     } while ((c = getbyte() & 0x7f) == GDBBadP);
00388 }

00390 // Receive a packet from gdb
00391 int
00392 BaseRemoteGDB::recv(char *bp, int maxlen)
00393 {
00394     char *p;
00395     int c, csum;
00396     int len;
00397 
00398     do {
00399         p = bp;
00400         csum = len = 0;
00401         //Find the beginning of a packet
00402         while ((c = getbyte()) != GDBStart)
00403             ;
00404 
00405         //Read until you find the end of the data in the packet, and keep
00406         //track of the check sum.
00407         while ((c = getbyte()) != GDBEnd && len < maxlen) {
00408             c &= 0x7f;
00409             csum += c;
00410             *p++ = c;
00411             len++;
00412         }
00413 
00414         //Mask the check sum, and terminate the command string.
00415         csum &= 0xff;
00416         *p = '\0';
00417 
00418         //If the command was too long, report an error.
00419         if (len >= maxlen) {
00420             putbyte(GDBBadP);
00421             continue;
00422         }
00423 
00424         //Bring in the checksum. If the check sum matches, csum will be 0.
00425         csum -= digit2i(getbyte()) * 16;
00426         csum -= digit2i(getbyte());
00427 
00428         //If the check sum was correct
00429         if (csum == 0) {
00430             //Report that the packet was received correctly
00431             putbyte(GDBGoodP);
00432             // Sequence present?
00433             if (bp[2] == ':') {
00434                 putbyte(bp[0]);
00435                 putbyte(bp[1]);
00436                 len -= 3;
00437                 memcpy(bp, bp+3, len);
00438             }
00439             break;
00440         }
00441         //Otherwise, report that there was a mistake.
00442         putbyte(GDBBadP);
00443     } while (1);
00444 
00445     DPRINTF(GDBRecv, "recv:  %s: %s\n", gdb_command(*bp), bp);
00446 
00447     return (len);
00448 }

00623 // This function does all command processing for interfacing to a
00624 // remote gdb.  Note that the error codes are ignored by gdb at
00625 // present, but might eventually become meaningful. (XXX) It might
00626 // makes sense to use POSIX errno values, because that is what the
00627 // gdb/remote.c functions want to return.
00628 bool
00629 BaseRemoteGDB::trap(int type)
00630 {
00631     uint64_t val;
00632     size_t datalen, len;
00633     char data[GDBPacketBufLen + 1];
00634     char *buffer;
00635     size_t bufferSize;
00636     const char *p;
00637     char command, subcmd;
00638     string var;
00639     bool ret;
00640 
00641     if (!attached)
00642         return false;
00643 
00644     bufferSize = gdbregs.bytes() * 2 + 256;
00645     buffer = (char*)malloc(bufferSize);
00646 
00647     DPRINTF(GDBMisc, "trap: PC=%s\n", context->pcState());
00648 
00649     clearSingleStep();
00650 
00651     /*
00652      * The first entry to this function is normally through
00653      * a breakpoint trap in kgdb_connect(), in which case we
00654      * must advance past the breakpoint because gdb will not.
00655      *
00656      * On the first entry here, we expect that gdb is not yet
00657      * listening to us, so just enter the interaction loop.
00658      * After the debugger is "active" (connected) it will be
00659      * waiting for a "signaled" message from us.
00660      */
00661     if (!active)
00662         active = true;
00663     else
00664         // Tell remote host that an exception has occurred.
00665         snprintf((char *)buffer, bufferSize, "S%02x", type);
00666         send(buffer);
00667 
00668     // Stick frame regs into our reg cache.
00669     getregs();
00670
