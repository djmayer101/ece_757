00869 Fault
00870 BaseDynInst<Impl>::readMem(Addr addr, uint8_t *data,
00871                            unsigned size, unsigned flags)
00872 {
00873     instFlags[ReqMade] = true;
00874     Request *req = NULL;
00875     Request *sreqLow = NULL;
00876     Request *sreqHigh = NULL;
00877 
00878     if (instFlags[ReqMade] && translationStarted()) {
00879         req = savedReq;
00880         sreqLow = savedSreqLow;
00881         sreqHigh = savedSreqHigh;
00882     } else {
00883         req = new Request(asid, addr, size, flags, masterId(), this->pc.instAddr(),
00884                           thread->contextId(), threadNumber);
00885 
00886         // Only split the request if the ISA supports unaligned accesses.
00887         if (TheISA::HasUnalignedMemAcc) {
00888             splitRequest(req, sreqLow, sreqHigh);
00889         }
00890         initiateTranslation(req, sreqLow, sreqHigh, NULL, BaseTLB::Read);
00891     }
00892 
00893     if (translationCompleted()) {
00894         if (fault == NoFault) {

00975 BaseDynInst<Impl>::splitRequest(RequestPtr req, RequestPtr &sreqLow,
00976                                 RequestPtr &sreqHigh)
00977 {
00978     // Check to see if the request crosses the next level block boundary.
00979     unsigned block_size = cpu->getDataPort().peerBlockSize();
00980     Addr addr = req->getVaddr();
00981     Addr split_addr = roundDown(addr + req->getSize() - 1, block_size);
00982     assert(split_addr <= addr || split_addr - addr < block_size);
00983 
00984     // Spans two blocks.
00985     if (split_addr > addr) {
