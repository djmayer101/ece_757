00062 void
00063 Bitmap::write(std::ostream *bmp) const
00064 {
00065     assert(data);
00066 
00067     // header is always the same for a bitmap object; compute the info once per
00068     //   bitmap object
00069     if (!headerBuffer) {
00070         // For further information see:
00071         //   http://en.wikipedia.org/wiki/BMP_file_format
00072         Magic magic = {{'B','M'}};
00073         Header header = {
00074             static_cast<uint32_t>(sizeof(VideoConvert::Rgb8888)) *
00075             width * height, 0, 0, 54};
00076         Info info = {static_cast<uint32_t>(sizeof(Info)), width, height, 1,
00077                      static_cast<uint32_t>(sizeof(VideoConvert::Rgb8888)) * 8,
00078                      0, static_cast<uint32_t>(sizeof(VideoConvert::Rgb8888)) *
00079                      width * height, 1, 1, 0, 0};
00080 
00081         char *p = headerBuffer = new char[sizeofHeaderBuffer];
00082         memcpy(p, &magic, sizeof(Magic));
00083         p += sizeof(Magic);
00084         memcpy(p, &header, sizeof(Header));
00085         p += sizeof(Header);
00086         memcpy(p, &info,   sizeof(Info));
00087     }
00088 
00089     // 1.  write the header
00090     bmp->write(headerBuffer, sizeofHeaderBuffer);
00091 
00092     // 2.  write the bitmap data
00093     uint8_t *tmp = vc.convert(data);
00094     uint32_t *tmp32 = (uint32_t*)tmp;
00095 
00096     // BMP start store data left to right starting with the bottom row
00097     // so we need to do some creative flipping
00098     for (int i = height - 1; i >= 0; i--)
00099         for (int j = 0; j < width; j++)
00100             bmp->write((char*)&tmp32[i * width + j], sizeof(uint32_t));
00101 
00102     bmp->flush();
00103 
00104     delete [] tmp;
00105 }
