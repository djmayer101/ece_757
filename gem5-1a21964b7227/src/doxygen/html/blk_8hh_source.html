00273     void clearLoadLocks(Request *req = NULL)
00274     {
00275         if (!req) {
00276             // No request, invaldate all locks to this line
00277             lockList.clear();
00278         } else {
00279             // Only invalidate locks that overlap with this request
00280             std::list<Lock>::iterator lock_itr = lockList.begin();
00281             while (lock_itr != lockList.end()) {
00282                 if (lock_itr->overlapping(req)) {
00283                     lock_itr = lockList.erase(lock_itr);
00284                 } else {
00285                     ++lock_itr;
00286                 }
00287             }

00296     bool checkWrite(PacketPtr pkt)
00297     {
00298         Request *req = pkt->req;
00299         if (pkt->isLLSC()) {
00300             // it's a store conditional... have to check for matching
00301             // load locked.
00302             bool success = false;
00303 
00304             for (std::list<Lock>::iterator i = lockList.begin();
00305                  i != lockList.end(); ++i)
00306             {
00307                 if (i->matchesContext(req)) {
00308                     // it's a store conditional, and as far as the memory
00309                     // system can tell, the requesting context's lock is
00310                     // still valid.
00311                     success = true;
00312                     break;
00313                 }
00314             }
00315 
00316             req->setExtraData(success ? 1 : 0);
00317             clearLoadLocks(req);
00318             return success;
00319         } else {
00320             // for *all* stores (conditional or otherwise) we have to
00321             // clear the list of load-locks as they're all invalid now.
00322             clearLoadLocks(req);
00323             return true;
