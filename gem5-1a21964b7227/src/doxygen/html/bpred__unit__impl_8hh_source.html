00131 BPredUnit::predict(StaticInstPtr &inst, const InstSeqNum &seqNum,
00132                    TheISA::PCState &pc, ThreadID tid)
00133 {
00134     // See if branch predictor predicts taken.
00135     // If so, get its target addr either from the BTB or the RAS.
00136     // Save off record of branch stuff so the RAS can be fixed
00137     // up once it's done.
00138 
00139     bool pred_taken = false;
00140     TheISA::PCState target = pc;
00141 
00142     ++lookups;
00143 
00144     void *bp_history = NULL;
00145 
00146     if (inst->isUncondCtrl()) {
00147         DPRINTF(Branch, "[tid:%i]: Unconditional control.\n", tid);
00148         pred_taken = true;
00149         // Tell the BP there was an unconditional branch.
00150         uncondBranch(bp_history);
00151     } else {
00152         ++condPredicted;
00153         pred_taken = lookup(pc.instAddr(), bp_history);
00154 
00155         DPRINTF(Branch, "[tid:%i]: [sn:%i] Branch predictor"
00156                 " predicted %i for PC %s\n", tid, seqNum,  pred_taken, pc);
00157     }
00158 
00159     DPRINTF(Branch, "[tid:%i]: [sn:%i] Creating prediction history "
00160             "for PC %s\n", tid, seqNum, pc);
00161 
00162     PredictorHistory predict_record(seqNum, pc.instAddr(),
00163                                     pred_taken, bp_history, tid);
00164 
00165     // Now lookup in the BTB or RAS.
00166     if (pred_taken) {
00167         if (inst->isReturn()) {
00168             ++usedRAS;
00169             predict_record.wasReturn = true;
00170             // If it's a function return call, then look up the address
00171             // in the RAS.
00172             TheISA::PCState rasTop = RAS[tid].top();
00173             target = TheISA::buildRetPC(pc, rasTop);
00174 
00175             // Record the top entry of the RAS, and its index.
00176             predict_record.usedRAS = true;
00177             predict_record.RASIndex = RAS[tid].topIdx();
00178             predict_record.RASTarget = rasTop;
00179 
00180             RAS[tid].pop();
00181 
00182             DPRINTF(Branch, "[tid:%i]: Instruction %s is a return, "
00183                     "RAS predicted target: %s, RAS index: %i.\n",
00184                     tid, pc, target, predict_record.RASIndex);
00185         } else {
00186             ++BTBLookups;
00187 
00188             if (inst->isCall()) {
00189                 RAS[tid].push(pc);
00190                 predict_record.pushedRAS = true;
00191 
00192                 // Record that it was a call so that the top RAS entry can
00193                 // be popped off if the speculation is incorrect.
00194                 predict_record.wasCall = true;
00195 
00196                 DPRINTF(Branch, "[tid:%i]: Instruction %s was a "
00197                         "call, adding %s to the RAS index: %i.\n",
00198                         tid, pc, pc, RAS[tid].topIdx());
00199             }
00200 
00201             if (BTB.valid(pc.instAddr(), tid)) {
00202                 ++BTBHits;
00203 
00204                 // If it's not a return, use the BTB to get the target addr.
00205                 target = BTB.lookup(pc.instAddr(), tid);
00206 
00207                 DPRINTF(Branch, "[tid:%i]: Instruction %s predicted"
00208                         " target is %s.\n", tid, pc, target);
00209 
00210             } else {
00211                 DPRINTF(Branch, "[tid:%i]: BTB doesn't have a "
00212                         "valid entry.\n",tid);
00213                 pred_taken = false;
00214                 // The Direction of the branch predictor is altered because the
00215                 // BTB did not have an entry
00216                 // The predictor needs to be updated accordingly
00217                 if (!inst->isCall() && !inst->isReturn()) {
00218                       btbUpdate(pc.instAddr(), bp_history);
00219                       DPRINTF(Branch, "[tid:%i]:[sn:%i] btbUpdate"
00220                               " called for %s\n", tid, seqNum, pc);
00221                 } else if (inst->isCall() && !inst->isUncondCtrl()) {
00222                       RAS[tid].pop();
00223                       predict_record.pushedRAS = false;
00224                 }
00225                 TheISA::advancePC(target, inst);
00226             }
00227         }
00228     } else {
00229         if (inst->isReturn()) {
00230            predict_record.wasReturn = true;
00231         }
00232         TheISA::advancePC(target, inst);
00233     }
00234 
00235     pc = target;
00236 
00237     predHist[tid].push_front(predict_record);
00238 
00239     DPRINTF(Branch, "[tid:%i]: [sn:%i]: History entry added."
00240             "predHist.size(): %i\n", tid, seqNum, predHist[tid].size());
00241 
00242     return pred_taken;
00243 }

00399                      pred_hist.front().seqNum, pred_hist.front().pc);
00400              RAS[tid].pop();
00401         }
00402 
00403         // This call should delete the bpHistory.
00404         squash(pred_hist.front().bpHistory);
00405 
00406         DPRINTF(Branch, "[tid:%i]: Removing history for [sn:%i] "
00407                 "PC %s.\n", tid, pred_hist.front().seqNum,
00408                 pred_hist.front().pc);
00409 
00410         pred_hist.pop_front();
00411 
00412         DPRINTF(Branch, "[tid:%i]: predHist.size(): %i\n",
00413                 tid, predHist[tid].size());

00423     // all the branches that have been seen up until this branch and
00424     // fix up everything.
00425     // NOTE: This should be call conceivably in 2 scenarios:
00426     // (1) After an branch is executed, it updates its status in the ROB
00427     //     The commit stage then checks the ROB update and sends a signal to
00428     //     the fetch stage to squash history after the mispredict
00429     // (2) In the decode stage, you can find out early if a unconditional
00430     //     PC-relative, branch was predicted incorrectly. If so, a signal
00431     //     to the fetch stage is sent to squash history after the mispredict
00432 
00433     History &pred_hist = predHist[tid];
00434 
00435     ++condIncorrect;
00436

00438             "setting target to %s.\n", tid, squashed_sn, corrTarget);

00440     // Squash All Branches AFTER this mispredicted branch
00441     squash(squashed_sn, tid);
00442 
00443     // If there's a squash due to a syscall, there may not be an entry
00444     // corresponding to the squash.  In that case, don't bother trying to
00445     // fix up the entry.
00446     if (!pred_hist.empty()) {

00480         } else {
