00066 void
00067 BranchPredictor::execute(int slot_num)
00068 {
00069     ResourceRequest* bpred_req = reqs[slot_num];
00070     DynInstPtr inst = bpred_req->inst;
00071     if (inst->fault != NoFault) {
00072         DPRINTF(InOrderBPred,
00073                 "[tid:%i]: [sn:%i]: Detected %s fault @ %x. Forwarding to "
00074                 "next stage.\n", inst->readTid(), inst->seqNum, inst->fault->name(),
00075                 inst->pcState());
00076         bpred_req->done();
00077         return;
00078     }
00079 
00080     if (!inst->isControl()) {
00081         DPRINTF(Resource, "Ignoring %s, not a control inst.\n",
00082                 inst->instName());
00083         bpred_req->done();
00084         return;
00085     }
00086 
00087     ThreadID tid = inst->readTid();
00088     InstSeqNum seq_num = inst->seqNum;
00089     switch (bpred_req->cmd)
00090     {
00091       case PredictBranch:
00092         {
00093             if (inst->seqNum > cpu->squashSeqNum[tid] &&
00094                 curTick() == cpu->lastSquashCycle[tid]) {
00095                 DPRINTF(InOrderStage, "[tid:%u]: [sn:%i]: squashed, "
00096                         "skipping prediction \n", tid, inst->seqNum);
00097             } else {
00098                 TheISA::PCState instPC = inst->pcState();
00099                 TheISA::PCState pred_PC = inst->pcState();
00100                 TheISA::advancePC(pred_PC, inst->staticInst);
00101 
00102                 if (inst->isControl()) {
00103                     // If not, the pred_PC be updated to pc+8
00104                     // If predicted, the pred_PC will be updated to new target
00105                     // value
00106                     bool predict_taken = branchPred->predictInOrder(
00107                                             inst->staticInst, inst->seqNum,
00108                                             inst->asid, instPC, pred_PC, tid);
00109 
00110                     if (predict_taken) {
00111                         DPRINTF(InOrderBPred, "[tid:%i]: [sn:%i]: Branch "
00112                                 "predicted true.\n", tid, seq_num);
00113                         predictedTaken++;
00114                     } else {
00115                         DPRINTF(InOrderBPred, "[tid:%i]: [sn:%i]: Branch "
00116                                 "predicted false.\n", tid, seq_num);
00117                         predictedNotTaken++;
00118                     }
00119 
00120                     inst->setBranchPred(predict_taken);
00121                 }
00122 
00123                 //@todo: Check to see how hw_rei is handled here...how does
00124                 //PC,NPC get updated to compare mispredict against???
00125                 inst->setPredTarg(pred_PC);
00126                 DPRINTF(InOrderBPred, "[tid:%i]: [sn:%i]: %s Predicted PC is "
00127                         "%s.\n", tid, seq_num, inst->instName(), pred_PC);
00128             }
00129 
00130             bpred_req->done();
00131         }
00132         break;
00133 
00134       case UpdatePredictor:
00135         {
00136             if (inst->seqNum > cpu->squashSeqNum[tid] &&
00137                 curTick() == cpu->lastSquashCycle[tid]) {
00138                 DPRINTF(InOrderStage, "[tid:%u]: [sn:%i]: squashed, "
00139                         "skipping branch predictor update \n",
00140                         tid, inst->seqNum);
00141             } else {
00142                 DPRINTF(InOrderBPred, "[tid:%i]: [sn:%i]: Updating "
00143                         "Branch Predictor.\n",
00144                         tid, seq_num);
00145 
00146 
00147                 branchPred->update(seq_num, tid);
00148             }
00149 
00150             bpred_req->done();
00151         }
00152         break;
00153 
00154       default:
00155         fatal("Unrecognized command to %s", resName);
00156     }
00157 }

00160 BranchPredictor::squash(DynInstPtr inst, int squash_stage,
00161                         InstSeqNum squash_seq_num, ThreadID tid)
00162 {
00163     InstSeqNum bpred_squash_num = inst->seqNum;
00164     DPRINTF(InOrderBPred, "[tid:%i][sn:%i] Squashing...\n", tid,
00165             bpred_squash_num);
00166 
00167     // update due to branch resolution
00168     if (squash_stage >= ThePipeline::BackEndStartStage) {
00169         branchPred->squash(bpred_squash_num, inst->pcState(),
00170                            inst->pcState().branching(), tid);
00171     } else {
00172     // update due to predicted taken branch
00173         branchPred->squash(bpred_squash_num, tid);
