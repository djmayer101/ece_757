00152 bool
00153 Bridge::BridgeSlavePort::recvTimingReq(PacketPtr pkt)
00154 {
00155     DPRINTF(Bridge, "recvTimingReq: %s addr 0x%x\n",
00156             pkt->cmdString(), pkt->getAddr());
00157 
00158     // ensure we do not have something waiting to retry
00159     if(retryReq)
00160         return false;
00161 
00162     DPRINTF(Bridge, "Response queue size: %d outresp: %d\n",
00163             transmitList.size(), outstandingResponses);
00164 
00165     if (masterPort.reqQueueFull()) {
00166         DPRINTF(Bridge, "Request queue full\n");
00167         retryReq = true;
00168     } else if (pkt->needsResponse()) {
00169         if (respQueueFull()) {
00170             DPRINTF(Bridge, "Response queue full\n");
00171             retryReq = true;
00172         } else {
00173             DPRINTF(Bridge, "Reserving space for response\n");
00174             assert(outstandingResponses != respQueueLimit);
00175             ++outstandingResponses;
00176             retryReq = false;
00177 
00178             // @todo: We need to pay for this and not just zero it out
00179             pkt->busFirstWordDelay = pkt->busLastWordDelay = 0;
00180 
00181             masterPort.schedTimingReq(pkt, bridge.clockEdge(delay));
00182         }
00183     }
00184 
00185     // remember that we are now stalling a packet and that we have to
00186     // tell the sending master to retry once space becomes available,
00187     // we make no distinction whether the stalling is due to the
00188     // request queue or response queue being full
00189     return !retryReq;
00190 }

00204 {
00205     // If we expect to see a response, we need to restore the source
00206     // and destination field that is potentially changed by a second
00207     // bus
00208     if (!pkt->memInhibitAsserted() && pkt->needsResponse()) {
00209         // Update the sender state so we can deal with the response
00210         // appropriately
00211         pkt->pushSenderState(new RequestState(pkt->getSrc()));

00213 
00214     // If we're about to put this packet at the head of the queue, we
00215     // need to schedule an event to do the transmit.  Otherwise there
00216     // should already be an event scheduled for sending the head
00217     // packet.

00219         bridge.schedule(sendEvent, when);
00220     }
