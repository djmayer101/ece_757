00036 DefaultBTB::DefaultBTB(unsigned _numEntries,
00037                        unsigned _tagBits,
00038                        unsigned _instShiftAmt)
00039     : numEntries(_numEntries),
00040       tagBits(_tagBits),
00041       instShiftAmt(_instShiftAmt)
00042 {
00043     DPRINTF(Fetch, "BTB: Creating BTB object.\n");
00044 
00045     if (!isPowerOf2(numEntries)) {
00046         fatal("BTB entries is not a power of 2!");
00047     }
00048 
00049     btb.resize(numEntries);
00050 
00051     for (unsigned i = 0; i < numEntries; ++i) {
00052         btb[i].valid = false;
00053     }
00054 
00055     idxMask = numEntries - 1;
00056 
00057     tagMask = (1 << tagBits) - 1;
00058 
00059     tagShiftAmt = instShiftAmt + floorLog2(numEntries);
00060 }

00085 bool
00086 DefaultBTB::valid(Addr instPC, ThreadID tid)
00087 {
00088     unsigned btb_idx = getIndex(instPC);
00089 
00090     Addr inst_tag = getTag(instPC);
00091 
00092     assert(btb_idx < numEntries);
00093 
00094     if (btb[btb_idx].valid
00095         && inst_tag == btb[btb_idx].tag
00096         && btb[btb_idx].tid == tid) {
00097         return true;
00098     } else {
00099         return false;
00100     }
00101 }
