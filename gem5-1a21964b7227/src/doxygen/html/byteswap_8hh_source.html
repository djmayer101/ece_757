00063 //These functions actually perform the swapping for parameters
00064 //of various bit lengths
00065 inline uint64_t
00066 swap_byte64(uint64_t x)
00067 {
00068 #if defined(__linux__)
00069     return bswap_64(x);
00070 #elif defined(__APPLE__)
00071     return OSSwapInt64(x);
00072 #else
00073     return  (uint64_t)((((uint64_t)(x) & 0xff) << 56) |
00074             ((uint64_t)(x) & 0xff00ULL) << 40 |
00075             ((uint64_t)(x) & 0xff0000ULL) << 24 |
00076             ((uint64_t)(x) & 0xff000000ULL) << 8 |
00077             ((uint64_t)(x) & 0xff00000000ULL) >> 8 |
00078             ((uint64_t)(x) & 0xff0000000000ULL) >> 24 |
00079             ((uint64_t)(x) & 0xff000000000000ULL) >> 40 |
00080             ((uint64_t)(x) & 0xff00000000000000ULL) >> 56) ;
00081 #endif
00082 }

00111 // This function lets the compiler figure out how to call the
00112 // swap_byte functions above for different data types.  Since the
00113 // sizeof() values are known at compile time, it should inline to a
00114 // direct call to the right swap_byteNN() function.
00115 template <typename T>
00116 inline T swap_byte(T x) {
00117     if (sizeof(T) == 8)
00118         return swap_byte64((uint64_t)x);
00119     else if (sizeof(T) == 4)
00120         return swap_byte32((uint32_t)x);
00121     else if (sizeof(T) == 2)
00122         return swap_byte16((uint16_t)x);
00123     else if (sizeof(T) == 1)
00124         return x;
00125     else
00126         panic("Can't byte-swap values larger than 64 bits");
00127 }
