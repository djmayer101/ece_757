00653         // upper cache will be responding
00654         if (pkt->isInvalidate()) {
00655             BlkType *blk = tags->findBlock(pkt->getAddr());
00656             if (blk && blk->isValid()) {
00657                 tags->invalidate(blk);
00658                 blk->invalidate();
00659                 DPRINTF(Cache, "rcvd mem-inhibited %s on 0x%x: invalidating\n",
00660                         pkt->cmdString(), pkt->getAddr());
00661             }
00662             if (!last_level_cache) {
00663                 DPRINTF(Cache, "forwarding mem-inhibited %s on 0x%x\n",
00664                         pkt->cmdString(), pkt->getAddr());
00665                 lat += ticksToCycles(memSidePort->sendAtomic(pkt));
00666             }
00667         } else {
00668             DPRINTF(Cache, "rcvd mem-inhibited %s on 0x%x: not responding\n",
00669                     pkt->cmdString(), pkt->getAddr());
00670         }
00671 
00672         return lat;
00673     }
00674 
00675     // should assert here that there are no outstanding MSHRs or
00676     // writebacks... that would mean that someone used an atomic
00677     // access in timing mode
00678 
00679     BlkType *blk = NULL;
00680     PacketList writebacks;
00681 
00682     if (!access(pkt, blk, lat, writebacks)) {
00683         // MISS
00684         PacketPtr bus_pkt = getBusPacket(pkt, blk, pkt->needsExclusive());
00685 
00686         bool is_forward = (bus_pkt == NULL);
00687 
00688         if (is_forward) {
00689             // just forwarding the same request to the next level
00690             // no local cache operation involved
00691             bus_pkt = pkt;
00692         }
00693 
00694         DPRINTF(Cache, "Sending an atomic %s for %x\n",
00695                 bus_pkt->cmdString(), bus_pkt->getAddr());
00696 
00697 #if TRACING_ON
00698         CacheBlk::State old_state = blk ? blk->status : 0;
00699 #endif
00700 
00701         lat += ticksToCycles(memSidePort->sendAtomic(bus_pkt));
00702 
00703         DPRINTF(Cache, "Receive response: %s for addr %x in state %i\n",
00704                 bus_pkt->cmdString(), bus_pkt->getAddr(), old_state);
00705 
00706         // If packet was a forward, the response (if any) is already
00707         // in place in the bus_pkt == pkt structure, so we don't need
00708         // to do anything.  Otherwise, use the separate bus_pkt to
00709         // generate response to pkt and then delete it.
00710         if (!is_forward) {
00711             if (pkt->needsResponse()) {
00712                 assert(bus_pkt->isResponse());
00713                 if (bus_pkt->isError()) {
00714                     pkt->makeAtomicResponse();
00715                     pkt->copyError(bus_pkt);
00716                 } else if (bus_pkt->isRead() ||
00717                            bus_pkt->cmd == MemCmd::UpgradeResp) {
00718                     // we're updating cache state to allow us to
00719                     // satisfy the upstream request from the cache
00720                     blk = handleFill(bus_pkt, blk, writebacks);
00721                     satisfyCpuSideRequest(pkt, blk);
00722                 } else {
00723                     // we're satisfying the upstream request without
00724                     // modifying cache state, e.g., a write-through
00725                     pkt->makeAtomicResponse();
00726                 }
00727             }
00728             delete bus_pkt;
00729         }
00730     }
00731 
00732     // Note that we don't invoke the prefetcher at all in atomic mode.
00733     // It's not clear how to do it properly, particularly for
00734     // prefetchers that aggressively generate prefetch candidates and
00735     // rely on bandwidth contention to throttle them; these will tend
00736     // to pollute the cache in atomic mode since there is no bandwidth
00737     // contention.  If we ever do want to enable prefetching in atomic
00738     // mode, though, this is the place to do it... see timingAccess()
00739     // for an example (though we'd want to issue the prefetch(es)
00740     // immediately rather than calling requestMemSideBus() as we do
00741     // there).
00742 
00743     // Handle writebacks if needed
00744     while (!writebacks.empty()){
00745         PacketPtr wbPkt = writebacks.front();
00746         memSidePort->sendAtomic(wbPkt);
00747         writebacks.pop_front();
00748         delete wbPkt;
00749     }
00750 
00751     // We now have the block one way or another (hit or completed miss)
00752 
00753     if (pkt->needsResponse()) {
00754         pkt->makeAtomicResponse();
00755     }
00756 
00757     return lat;
00758 }

00763 Cache<TagStore>::functionalAccess(PacketPtr pkt, bool fromCpuSide)

00765     if (system->bypassCaches()) {

00767         // shouldn't happen in bypass mode.
00768         assert(fromCpuSide);

00771         // so we don't need to check if we need to update anything.
00772         memSidePort->sendFunctional(pkt);

00774     }
00775

00786     // needs to be found.  As a result we always update the request if
00787     // we have it, but only declare it satisfied if we are the owner.

00790     bool have_data = blk && blk->isValid()

00811     pkt->popLabel();
00812

00814         pkt->makeResponse();

00816         // if it came as a request from the CPU side then make sure it
00817         // continues towards the memory side
00818         if (fromCpuSide) {
00819             memSidePort->sendFunctional(pkt);
00820         } else if (forwardSnoops && cpuSidePort->isSnooping()) {

00822             // and we should only forward it if we are forwarding snoops

00840     assert(pkt->isResponse());

00844     bool is_error = pkt->isError();

00846     assert(mshr);
00847

00850                 "cmd: %s\n", pkt->getAddr(), pkt->cmdString());
00851     }
00852 
00853     DPRINTF(Cache, "Handling response to %x\n", pkt->getAddr());
00854 
00855     MSHRQueue *mq = mshr->queue;
00856     bool wasFull = mq->isFull();
00857 
00858     if (mshr == noTargetMSHR) {
00859         // we always clear at least one target
00860         clearBlocked(Blocked_NoTargets);
00861         noTargetMSHR = NULL;
00862     }
00863 
00864     // Initial target is used just for stats
00865     MSHR::Target *initial_tgt = mshr->getTarget();
00866     BlkType *blk = tags->findBlock(pkt->getAddr());
00867     int stats_cmd_idx = initial_tgt->pkt->cmdToIndex();
00868     Tick miss_latency = curTick() - initial_tgt->recvTime;
00869     PacketList writebacks;
00870 
00871     if (pkt->req->isUncacheable()) {
00872         assert(pkt->req->masterId() < system->maxMasters());
00873         mshr_uncacheable_lat[stats_cmd_idx][pkt->req->masterId()] +=
00874             miss_latency;
00875     } else {
00876         assert(pkt->req->masterId() < system->maxMasters());
00877         mshr_miss_latency[stats_cmd_idx][pkt->req->masterId()] +=
00878             miss_latency;
00879     }
00880 
00881     bool is_fill = !mshr->isForward &&
00882         (pkt->isRead() || pkt->cmd == MemCmd::UpgradeResp);
00883 
00884     if (is_fill && !is_error) {
00885         DPRINTF(Cache, "Block for addr %x being updated in Cache\n",
00886                 pkt->getAddr());
00887 
00888         // give mshr a chance to do some dirty work
00889         mshr->handleFill(pkt, blk);
00890 
00891         blk = handleFill(pkt, blk, writebacks);
00892         assert(blk != NULL);
00893     }
00894 
00895     // First offset for critical word first calculations
00896     int initial_offset = 0;
00897 
00898     if (mshr->hasTargets()) {
00899         initial_offset = mshr->getTarget()->pkt->getOffset(blkSize);
00900     }
00901 
00902     while (mshr->hasTargets()) {
00903         MSHR::Target *target = mshr->getTarget();
00904 
00905         switch (target->source) {
00906           case MSHR::Target::FromCPU:
00907             Tick completion_time;
00908             if (is_fill) {
00909                 satisfyCpuSideRequest(target->pkt, blk,
00910                                       true, mshr->hasPostDowngrade());
00911                 // How many bytes past the first request is this one
00912                 int transfer_offset =
00913                     target->pkt->getOffset(blkSize) - initial_offset;
00914                 if (transfer_offset < 0) {
00915                     transfer_offset += blkSize;
00916                 }
00917 
00918                 // If critical word (no offset) return first word time.
00919                 // responseLatency is the latency of the return path
00920                 // from lower level caches/memory to an upper level cache or
00921                 // the core.
00922                 completion_time = curTick() + responseLatency * clockPeriod() +
00923                     (transfer_offset ? pkt->busLastWordDelay :
00924                      pkt->busFirstWordDelay);
00925 
00926                 assert(!target->pkt->req->isUncacheable());
00927 
00928                 assert(target->pkt->req->masterId() < system->maxMasters());
00929                 missLatency[target->pkt->cmdToIndex()][target->pkt->req->masterId()] +=
00930                     completion_time - target->recvTime;
00931             } else if (pkt->cmd == MemCmd::UpgradeFailResp) {
00932                 // failed StoreCond upgrade
00933                 assert(target->pkt->cmd == MemCmd::StoreCondReq ||
00934                        target->pkt->cmd == MemCmd::StoreCondFailReq ||
00935                        target->pkt->cmd == MemCmd::SCUpgradeFailReq);
00936                 // responseLatency is the latency of the return path
00937                 // from lower level caches/memory to an upper level cache or
00938                 // the core.
00939                 completion_time = curTick() + responseLatency * clockPeriod() +
00940                     pkt->busLastWordDelay;
00941                 target->pkt->req->setExtraData(0);
00942             } else {
00943                 // not a cache fill, just forwarding response
00944                 // responseLatency is the latency of the return path
00945                 // from lower level cahces/memory to the core.
00946                 completion_time = curTick() + responseLatency * clockPeriod() +
00947                     pkt->busLastWordDelay;
00948                 if (pkt->isRead() && !is_error) {
00949                     target->pkt->setData(pkt->getPtr<uint8_t>());
00950                 }
00951             }
00952             target->pkt->makeTimingResponse();
00953             // if this packet is an error copy that to the new packet
00954             if (is_error)
00955                 target->pkt->copyError(pkt);
00956             if (target->pkt->cmd == MemCmd::ReadResp &&
00957                 (pkt->isInvalidate() || mshr->hasPostInvalidate())) {
00958                 // If intermediate cache got ReadRespWithInvalidate,
00959                 // propagate that.  Response should not have
00960                 // isInvalidate() set otherwise.
00961                 target->pkt->cmd = MemCmd::ReadRespWithInvalidate;
00962             }
00963             // reset the bus additional time as it is now accounted for
00964             target->pkt->busFirstWordDelay = target->pkt->busLastWordDelay = 0;
00965             cpuSidePort->schedTimingResp(target->pkt, completion_time);
00966             break;
00967 
00968           case MSHR::Target::FromPrefetcher:
00969             assert(target->pkt->cmd == MemCmd::HardPFReq);
00970             if (blk)
00971                 blk->status |= BlkHWPrefetched;
00972             delete target->pkt->req;
00973             delete target->pkt;
00974             break;
00975 
00976           case MSHR::Target::FromSnoop:
00977             // I don't believe that a snoop can be in an error state
00978             assert(!is_error);
00979             // response to snoop request
00980             DPRINTF(Cache, "processing deferred snoop...\n");
00981             assert(!(pkt->isInvalidate() && !mshr->hasPostInvalidate()));
00982             handleSnoop(target->pkt, blk, true, true,
00983                         mshr->hasPostInvalidate());
00984             break;
00985 
00986           default:
00987             panic("Illegal target->source enum %d\n", target->source);
00988         }
00989 
00990         mshr->popTarget();
00991     }
00992 
00993     if (blk && blk->isValid()) {
00994         if (pkt->isInvalidate() || mshr->hasPostInvalidate()) {
00995             assert(blk != tempBlock);
00996             tags->invalidate(blk);
00997             blk->invalidate();
00998         } else if (mshr->hasPostDowngrade()) {
00999             blk->status &= ~BlkWritable;
01000         }
01001     }
01002 
01003     if (mshr->promoteDeferredTargets()) {
01004         // avoid later read getting stale data while write miss is
01005         // outstanding.. see comment in timingAccess()
01006         if (blk) {
01007             blk->status &= ~BlkReadable;
01008         }
01009         mq = mshr->queue;
01010         mq->markPending(mshr);
01011         requestMemSideBus((RequestCause)mq->index, curTick() +
01012                           pkt->busLastWordDelay);
01013     } else {
01014         mq->deallocate(mshr);
01015         if (wasFull && !mq->isFull()) {
01016             clearBlocked((BlockedCause)mq->index);
01017         }
01018     }
01019 
01020     // copy writebacks to write buffer
01021     while (!writebacks.empty()) {
01022         PacketPtr wbPkt = writebacks.front();
01023         allocateWriteBuffer(wbPkt, time, true);
01024         writebacks.pop_front();
01025     }
01026     // if we used temp block, clear it out
01027     if (blk == tempBlock) {
01028         if (blk->isDirty()) {
01029             allocateWriteBuffer(writebackBlk(blk), time, true);
01030         }
01031         blk->invalidate();
01032     }
01033 
01034     delete pkt;
01035 }

01178 // Note that the reason we return a list of writebacks rather than
01179 // inserting them directly in the write buffer is that this function
01180 // is called by both atomic and timing-mode accesses, and in atomic
01181 // mode we don't mess with the write buffer (we just perform the
01182 // writebacks atomically once the original request is complete).
01183 template<class TagStore>
01184 typename Cache<TagStore>::BlkType*
01185 Cache<TagStore>::handleFill(PacketPtr pkt, BlkType *blk,
01186                             PacketList &writebacks)
01187 {
01188     Addr addr = pkt->getAddr();
01189 #if TRACING_ON
01190     CacheBlk::State old_state = blk ? blk->status : 0;
01191 #endif
01192 
01193     if (blk == NULL) {
01194         // better have read new data...
01195         assert(pkt->hasData());
01196         // need to do a replacement
01197         blk = allocateBlock(addr, writebacks);
01198         if (blk == NULL) {
01199             // No replaceable block... just use temporary storage to
01200             // complete the current request and then get rid of it
01201             assert(!tempBlock->isValid());
01202             blk = tempBlock;
01203             tempBlock->set = tags->extractSet(addr);
01204             tempBlock->tag = tags->extractTag(addr);
01205             DPRINTF(Cache, "using temp block for %x\n", addr);
01206         } else {
01207             int id = pkt->req->masterId();
01208             tags->insertBlock(pkt->getAddr(), blk, id);
01209         }
01210 
01211         // we should never be overwriting a valid block
01212         assert(!blk->isValid());
01213     } else {
01214         // existing block... probably an upgrade
01215         assert(blk->tag == tags->extractTag(addr));
01216         // either we're getting new data or the block should already be valid
01217         assert(pkt->hasData() || blk->isValid());
01218         // don't clear block status... if block is already dirty we
01219         // don't want to lose that
01220     }
01221 
01222     blk->status |= BlkValid | BlkReadable;
01223 
01224     if (!pkt->sharedAsserted()) {
01225         blk->status |= BlkWritable;
01226         // If we got this via cache-to-cache transfer (i.e., from a
01227         // cache that was an owner) and took away that owner's copy,
01228         // then we need to write it back.  Normally this happens
01229         // anyway as a side effect of getting a copy to write it, but
01230         // there are cases (such as failed store conditionals or
01231         // compare-and-swaps) where we'll demand an exclusive copy but
01232         // end up not writing it.
01233         if (pkt->memInhibitAsserted())
01234             blk->status |= BlkDirty;
01235     }
01236 
01237     DPRINTF(Cache, "Block addr %x moving from state %i to %i\n",
01238             addr, old_state, blk->status);
01239 
01240     // if we got new data, copy it in
01241     if (pkt->isRead()) {
01242         std::memcpy(blk->data, pkt->getPtr<uint8_t>(), blkSize);
01243     }
01244 
01245     blk->whenReady = curTick() + pkt->busLastWordDelay;
01246 
01247     return blk;
01248 }

01257 template<class TagStore>

01264     // already made a copy...

01273     }
01274     if (pkt->cmd == MemCmd::ReadResp && pending_inval) {

01278         // the latter case MemInhibit will keep the invalidation from

01289 Cache<TagStore>::handleSnoop(PacketPtr pkt, BlkType *blk,
01290                              bool is_timing, bool is_deferred,
01291                              bool pending_inval)
01292 {
01293     // deferred snoops can only happen in timing mode
01294     assert(!(is_deferred && !is_timing));
01295     // pending_inval only makes sense on deferred snoops
01296     assert(!(pending_inval && !is_deferred));
01297     assert(pkt->isRequest());
01298 
01299     // the packet may get modified if we or a forwarded snooper
01300     // responds in atomic mode, so remember a few things about the
01301     // original packet up front
01302     bool invalidate = pkt->isInvalidate();
01303     bool M5_VAR_USED needs_exclusive = pkt->needsExclusive();
01304 
01305     if (forwardSnoops) {
01306         // first propagate snoop upward to see if anyone above us wants to
01307         // handle it.  save & restore packet src since it will get
01308         // rewritten to be relative to cpu-side bus (if any)
01309         bool alreadyResponded = pkt->memInhibitAsserted();
01310         if (is_timing) {
01311             Packet snoopPkt(pkt, true);  // clear flags
01312             snoopPkt.setExpressSnoop();
01313             snoopPkt.pushSenderState(new ForwardResponseRecord(pkt->getSrc()));
01314             // the snoop packet does not need to wait any additional
01315             // time
01316             snoopPkt.busFirstWordDelay = snoopPkt.busLastWordDelay = 0;
01317             cpuSidePort->sendTimingSnoopReq(&snoopPkt);
01318             if (snoopPkt.memInhibitAsserted()) {
01319                 // cache-to-cache response from some upper cache
01320                 assert(!alreadyResponded);
01321                 pkt->assertMemInhibit();
01322             } else {
01323                 delete snoopPkt.popSenderState();
01324             }
01325             if (snoopPkt.sharedAsserted()) {
01326                 pkt->assertShared();
01327             }
01328         } else {
01329             cpuSidePort->sendAtomicSnoop(pkt);
01330             if (!alreadyResponded && pkt->memInhibitAsserted()) {
01331                 // cache-to-cache response from some upper cache:
01332                 // forward response to original requester
01333                 assert(pkt->isResponse());
01334             }
01335         }
01336     }
01337 
01338     if (!blk || !blk->isValid()) {
01339         return;
01340     }
01341 
01342     // we may end up modifying both the block state and the packet (if
01343     // we respond in atomic mode), so just figure out what to do now
01344     // and then do it later
01345     bool respond = blk->isDirty() && pkt->needsResponse();
01346     bool have_exclusive = blk->isWritable();
01347 
01348     if (pkt->isRead() && !invalidate) {
01349         assert(!needs_exclusive);
01350         pkt->assertShared();
01351         int bits_to_clear = BlkWritable;
01352         const bool haveOwnershipState = true; // for now
01353         if (!haveOwnershipState) {
01354             // if we don't support pure ownership (dirty && !writable),
01355             // have to clear dirty bit here, assume memory snarfs data
01356             // on cache-to-cache xfer
01357             bits_to_clear |= BlkDirty;
01358         }
01359         blk->status &= ~bits_to_clear;
01360     }
01361 
01362     DPRINTF(Cache, "snooped a %s request for addr %x, %snew state is %i\n",
01363             pkt->cmdString(), blockAlign(pkt->getAddr()),
01364             respond ? "responding, " : "", invalidate ? 0 : blk->status);
01365 
01366     if (respond) {
01367         assert(!pkt->memInhibitAsserted());
01368         pkt->assertMemInhibit();
01369         if (have_exclusive) {
01370             pkt->setSupplyExclusive();
01371         }
01372         if (is_timing) {
01373             doTimingSupplyResponse(pkt, blk->data, is_deferred, pending_inval);
01374         } else {
01375             pkt->makeAtomicResponse();
01376             pkt->setDataFromBlock(blk->data, blkSize);
01377         }
01378     } else if (is_timing && is_deferred) {
01379         // if it's a deferred timing snoop then we've made a copy of
01380         // the packet, and so if we're not using that copy to respond
01381         // then we need to delete it here.
01382         delete pkt;
01383     }
01384 
01385     // Do this last in case it deallocates block data or something
01386     // like that
01387     if (invalidate) {
01388         assert(blk != tempBlock);
01389         tags->invalidate(blk);
01390         blk->invalidate();
01391     }
01392 }

01398 {
01399     // Snoops shouldn't happen when bypassing caches
01400     assert(!system->bypassCaches());

01402     // Note that some deferred snoops don't have requests, since the
01403     // original access may have already completed
01404     if ((pkt->req && pkt->req->isUncacheable()) ||
01405         pkt->cmd == MemCmd::Writeback) {
01406         //Can't get a hit on an uncacheable address
01407         //Revisit this for multi level coherence
01408         return;
01409     }
01410 
01411     BlkType *blk = tags->findBlock(pkt->getAddr());
01412 
01413     Addr blk_addr = blockAlign(pkt->getAddr());
01414     MSHR *mshr = mshrQueue.findMatch(blk_addr);
01415 
01416     // Let the MSHR itself track the snoop and decide whether we want
01417     // to go ahead and do the regular cache snoop
01418     if (mshr && mshr->handleSnoop(pkt, order++)) {
01419         DPRINTF(Cache, "Deferring snoop on in-service MSHR to blk %x\n",
01420                 blk_addr);
01421         if (mshr->getNumTargets() > numTarget)
01422             warn("allocating bonus target for snoop"); //handle later
01423         return;
01424     }
01425 
01426     //We also need to check the writeback buffers and handle those
01427     std::vector<MSHR *> writebacks;
01428     if (writeBuffer.findMatches(blk_addr, writebacks)) {
01429         DPRINTF(Cache, "Snoop hit in writeback to addr: %x\n",
01430                 pkt->getAddr());
01431 
01432         //Look through writebacks for any non-uncachable writes, use that
01433         if (writebacks.size()) {
01434             // We should only ever find a single match
01435             assert(writebacks.size() == 1);
01436             mshr = writebacks[0];
01437             assert(!mshr->isUncacheable());
01438             assert(mshr->getNumTargets() == 1);
01439             PacketPtr wb_pkt = mshr->getTarget()->pkt;
01440             assert(wb_pkt->cmd == MemCmd::Writeback);
01441 
01442             assert(!pkt->memInhibitAsserted());
01443             pkt->assertMemInhibit();
01444             if (!pkt->needsExclusive()) {
01445                 pkt->assertShared();
01446                 // the writeback is no longer the exclusive copy in the system
01447                 wb_pkt->clearSupplyExclusive();
01448             } else {
01449                 // if we're not asserting the shared line, we need to
01450                 // invalidate our copy.  we'll do that below as long as
01451                 // the packet's invalidate flag is set...
01452                 assert(pkt->isInvalidate());
01453             }
01454             doTimingSupplyResponse(pkt, wb_pkt->getPtr<uint8_t>(),
01455                                    false, false);
01456 
01457             if (pkt->isInvalidate()) {
01458                 // Invalidation trumps our writeback... discard here
01459                 markInService(mshr);
01460                 delete wb_pkt;
01461             }
01462         } // writebacks.size()
01463     }
01464 
01465     // If this was a shared writeback, there may still be
01466     // other shared copies above that require invalidation.
01467     // We could be more selective and return here if the
01468     // request is non-exclusive or if the writeback is
01469     // exclusive.
01470     handleSnoop(pkt, blk, true, false, false);

01502 MSHR *
01503 Cache<TagStore>::getNextMSHR()
01504 {
01505     // Check both MSHR queue and write buffer for potential requests
01506     MSHR *miss_mshr  = mshrQueue.getNextMSHR();
01507     MSHR *write_mshr = writeBuffer.getNextMSHR();
01508 
01509     // Now figure out which one to send... some cases are easy
01510     if (miss_mshr && !write_mshr) {
01511         return miss_mshr;
01512     }
01513     if (write_mshr && !miss_mshr) {
01514         return write_mshr;
01515     }
01516 
01517     if (miss_mshr && write_mshr) {
01518         // We have one of each... normally we favor the miss request
01519         // unless the write buffer is full
01520         if (writeBuffer.isFull() && writeBuffer.inServiceEntries == 0) {
01521             // Write buffer is full, so we'd like to issue a write;
01522             // need to search MSHR queue for conflicting earlier miss.
01523             MSHR *conflict_mshr =
01524                 mshrQueue.findPending(write_mshr->addr, write_mshr->size);
01525 
01526             if (conflict_mshr && conflict_mshr->order < write_mshr->order) {
01527                 // Service misses in order until conflict is cleared.
01528                 return conflict_mshr;
01529             }
01530 
01531             // No conflicts; issue write
01532             return write_mshr;
01533         }
01534 
01535         // Write buffer isn't full, but need to check it for
01536         // conflicting earlier writeback
01537         MSHR *conflict_mshr =
01538             writeBuffer.findPending(miss_mshr->addr, miss_mshr->size);
01539         if (conflict_mshr) {
01540             // not sure why we don't check order here... it was in the
01541             // original code but commented out.
01542 
01543             // The only way this happens is if we are
01544             // doing a write and we didn't have permissions
01545             // then subsequently saw a writeback (owned got evicted)
01546             // We need to make sure to perform the writeback first
01547             // To preserve the dirty data, then we can issue the write
01548 
01549             // should we return write_mshr here instead?  I.e. do we
01550             // have to flush writes in order?  I don't think so... not
01551             // for Alpha anyway.  Maybe for x86?
01552             return conflict_mshr;
01553         }
01554 
01555         // No conflicts; issue read
01556         return miss_mshr;
01557     }
01558 
01559     // fall through... no pending requests.  Try a prefetch.
01560     assert(!miss_mshr && !write_mshr);
01561     if (prefetcher && !mshrQueue.isFull()) {
01562         // If we have a miss queue slot, we can try a prefetch
01563         PacketPtr pkt = prefetcher->getPacket();
01564         if (pkt) {
01565             Addr pf_addr = blockAlign(pkt->getAddr());
01566             if (!tags->findBlock(pf_addr) && !mshrQueue.findMatch(pf_addr) &&
01567                                              !writeBuffer.findMatch(pf_addr)) {
01568                 // Update statistic on number of prefetches issued
01569                 // (hwpf_mshr_misses)
01570                 assert(pkt->req->masterId() < system->maxMasters());
01571                 mshr_misses[pkt->cmdToIndex()][pkt->req->masterId()]++;
01572                 // Don't request bus, since we already have it
01573                 return allocateMissBuffer(pkt, curTick(), false);
01574             } else {
01575                 // free the request and packet
01576                 delete pkt->req;
01577                 delete pkt;
01578             }
01579         }
01580     }
01581 
01582     return NULL;

01587 PacketPtr
01588 Cache<TagStore>::getTimingPacket()
01589 {
01590     MSHR *mshr = getNextMSHR();
01591 
01592     if (mshr == NULL) {
01593         return NULL;
01594     }
01595 
01596     // use request from 1st target
01597     PacketPtr tgt_pkt = mshr->getTarget()->pkt;
01598     PacketPtr pkt = NULL;
01599 
01600     if (tgt_pkt->cmd == MemCmd::SCUpgradeFailReq ||
01601         tgt_pkt->cmd == MemCmd::StoreCondFailReq) {
01602         // SCUpgradeReq or StoreCondReq saw invalidation while queued
01603         // in MSHR, so now that we are getting around to processing
01604         // it, just treat it as if we got a failure response
01605         pkt = new Packet(tgt_pkt);
01606         pkt->cmd = MemCmd::UpgradeFailResp;
01607         pkt->senderState = mshr;
01608         pkt->busFirstWordDelay = pkt->busLastWordDelay = 0;
01609         recvTimingResp(pkt);
01610         return NULL;
01611     } else if (mshr->isForwardNoResponse()) {
01612         // no response expected, just forward packet as it is
01613         assert(tags->findBlock(mshr->addr) == NULL);
01614         pkt = tgt_pkt;
01615     } else {
01616         BlkType *blk = tags->findBlock(mshr->addr);
01617 
01618         if (tgt_pkt->cmd == MemCmd::HardPFReq) {
01619             // It might be possible for a writeback to arrive between
01620             // the time the prefetch is placed in the MSHRs and when
01621             // it's selected to send... if so, this assert will catch
01622             // that, and then we'll have to figure out what to do.
01623             assert(blk == NULL);
01624 
01625             // We need to check the caches above us to verify that they don't have
01626             // a copy of this block in the dirty state at the moment. Without this
01627             // check we could get a stale copy from memory  that might get used
01628             // in place of the dirty one.
01629             PacketPtr snoop_pkt = new Packet(tgt_pkt, true);
01630             snoop_pkt->setExpressSnoop();
01631             snoop_pkt->senderState = mshr;
01632             cpuSidePort->sendTimingSnoopReq(snoop_pkt);
01633 
01634             if (snoop_pkt->memInhibitAsserted()) {
01635                 markInService(mshr, snoop_pkt);
01636                 DPRINTF(Cache, "Upward snoop of prefetch for addr %#x hit\n",
01637                         tgt_pkt->getAddr());
01638                 delete snoop_pkt;
01639                 return NULL;
01640             }
01641             delete snoop_pkt;
01642         }
01643 
01644         pkt = getBusPacket(tgt_pkt, blk, mshr->needsExclusive());
01645 
01646         mshr->isForward = (pkt == NULL);
01647 
01648         if (mshr->isForward) {
01649             // not a cache block request, but a response is expected
01650             // make copy of current packet to forward, keep current
01651             // copy for response handling
01652             pkt = new Packet(tgt_pkt);
01653             pkt->allocate();
01654             if (pkt->isWrite()) {
01655                 pkt->setData(tgt_pkt->getPtr<uint8_t>());
01656             }
01657         }
01658     }
01659 
01660     assert(pkt != NULL);
01661     pkt->senderState = mshr;
01662     return pkt;
01663 }

01815     // if we have a response packet waiting we have to start with that
01816     if (deferredPacketReady()) {

01819     } else {

01829

01831

01834                 if (!mshr->isForwardNoResponse()) {
01835                     // we are awaiting a retry, but we
01836                     // delete the packet and will be creating a new packet
01837                     // when we get the opportunity
01838                     delete pkt;
01839                 }
01840                 // note that we have now masked any requestBus and
01841                 // schedSendEvent (we will wait for a retry before
01842                 // doing anything), and this is so even if we do not
01843                 // care about this packet and might override it before
01844                 // it gets retried
01845             } else {
