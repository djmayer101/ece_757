00294     assert(cache_req);

00299         cache_req->instIdx == inst->curSkedEntry->idx) {

00308                inst->curSkedEntry->idx != CacheUnit::InitSecondSplitWrite) {
00309         // If same command, just check to see if memory access was completed

00357     ThreadID tid = inst->readTid();

00361     //@todo: HACK: the DTB expects the correct PC in the ThreadContext
00362     //       but how if the memory accesses are speculative? Shouldn't

00364     ThreadContext *tc = cpu->thread[tid]->getTC();
00365     PCState old_pc = tc->pcState();
00366     tc->pcState() = inst->pcState();
00367 
00368     inst->fault =
00369         _tlb->translateAtomic(cache_req->memReq, tc, tlb_mode);
00370     tc->pcState() = old_pc;
00371 
00372     if (inst->fault != NoFault) {
00373         DPRINTF(InOrderTLB, "[tid:%i]: %s encountered while translating "
00374                 "addr:%08p for [sn:%i].\n", tid, inst->fault->name(),
00375                 cache_req->memReq->getVaddr(), inst->seqNum);
00376 
00377         tlbBlocked[tid] = true;
00378         tlbBlockSeqNum[tid] = inst->seqNum;
00379 
00380         // Make sure nothing gets executed until after this faulting
00381         // instruction gets handled.
00382         inst->setSerializeAfter();
00383 
00384         // Mark it as complete so it can pass through next stage.
00385         // Fault Handling will happen at commit/graduation
00386         cache_req->setCompleted();
00387     } else {
00388         DPRINTF(InOrderTLB, "[tid:%i]: [sn:%i] virt. addr %08p translated "
00389                 "to phys. addr:%08p.\n", tid, inst->seqNum,
00390                 cache_req->memReq->getVaddr(),
00391                 cache_req->memReq->getPaddr());

00402 CacheUnit::read(DynInstPtr inst, Addr addr,
00403                 uint8_t *data, unsigned size, unsigned flags)
00404 {
00405     CacheReqPtr cache_req = dynamic_cast<CacheReqPtr>(findRequest(inst));
00406     assert(cache_req && "Can't Find Instruction for Read!");
00407 
00408     // The block size of our peer
00409     unsigned blockSize = cacheBlkSize;
00410 
00411     //The size of the data we're trying to read.
00412     int fullSize = size;
00413     inst->totalSize = size;
00414 
00415     if (inst->traceData) {
00416         inst->traceData->setAddr(addr);
00417     }
00418 
00419     if (inst->split2ndAccess) {     
00420         size = inst->split2ndSize;
00421         cache_req->splitAccess = true;        
00422         cache_req->split2ndAccess = true;
00423         
00424         DPRINTF(InOrderCachePort, "[sn:%i] Split Read Access (2 of 2) for "
00425                 "(%#x, %#x).\n", inst->seqNum, inst->getMemAddr(),
00426                 inst->split2ndAddr);
00427     }  
00428     
00429 
00430     //The address of the second part of this access if it needs to be split
00431     //across a cache line boundary.
00432     Addr secondAddr = roundDown(addr + size - 1, blockSize);
00433 
00434     
00435     if (secondAddr > addr && !inst->split2ndAccess) {
00436 
00437         if (!inst->splitInst) {
00438             DPRINTF(InOrderCachePort, "%i: sn[%i] Split Read Access (1 of 2) for "
00439                     "(%#x, %#x).\n", curTick(), inst->seqNum, addr, secondAddr);
00440 
00441             unsigned stage_num = cache_req->getStageNum();
00442             unsigned cmd = inst->curSkedEntry->cmd;
00443 
00444             // 1. Make A New Inst. Schedule w/Split Read/Complete Entered on
00445             // the schedule
00446             // ==============================
00447             // 2. Reassign curSkedPtr to current command (InitiateRead) on new
00448             // schedule
00449             // ==============================
00450             inst->splitInst = true;
00451             inst->setBackSked(cpu->createBackEndSked(inst));
00452             inst->curSkedEntry = inst->backSked->find(stage_num, cmd);
00453         } else {
00454             DPRINTF(InOrderCachePort, "[tid:%i] [sn:%i] Retrying Split Read "
00455                     "Access (1 of 2) for (%#x, %#x).\n", inst->readTid(),
00456                     inst->seqNum, addr, secondAddr);
00457         }
00458 
00459         // Save All "Total" Split Information
00460         // ==============================
00461         inst->splitMemData = new uint8_t[size];
00462 
00463         // Split Information for First Access
00464         // ==============================
00465         size = secondAddr - addr;
00466         cache_req->splitAccess = true;
00467 
00468         // Split Information for Second Access
00469         // ==============================
00470         inst->split2ndSize = addr + fullSize - secondAddr;
00471         inst->split2ndAddr = secondAddr;            
00472         inst->split2ndDataPtr = inst->splitMemData + size;
00473         inst->split2ndFlags = flags;        
00474     }
00475     
00476     doTLBAccess(inst, cache_req, size, flags, TheISA::TLB::Read);
00477 
00478     if (inst->fault == NoFault) {
00479         if (!cache_req->splitAccess) {            
00480             cache_req->reqData = new uint8_t[size];
00481             doCacheAccess(inst, NULL);
00482         } else {
00483             if (!inst->split2ndAccess) {                
00484                 cache_req->reqData = inst->splitMemData;
00485             } else {
00486                 cache_req->reqData = inst->split2ndDataPtr;                
00487             }
00488             
00489             doCacheAccess(inst, NULL, cache_req);            
00490         }        
00491     }
00492 
00493     return inst->fault;
00494 }

00643     ThreadID tid = inst->readTid();
00644     std::string acc_type = "write";
00645 #endif
00646 
00647     switch (cache_req->cmd)
00648     {
00649 
00650       case InitiateReadData:
00651 #if TRACING_ON
00652         acc_type = "read";
00653 #endif        
00654       case InitiateWriteData:
00655         if (cachePortBlocked) {
00656             DPRINTF(InOrderCachePort, "Cache Port Blocked. Cannot Access\n");
00657             cache_req->done(false);
00658             return;
00659         }
00660 
00661         DPRINTF(InOrderCachePort,
00662                 "[tid:%u]: [sn:%i] Initiating data %s access to %s for "
00663                 "addr. %08p\n", tid, inst->seqNum, acc_type, name(),
00664                 cache_req->inst->getMemAddr());
00665 
00666         inst->setCurResSlot(slot_num);
00667 
00668         if (inst->isDataPrefetch() || inst->isInstPrefetch()) {
00669             inst->execute();
00670         } else {
00671             inst->initiateAcc();
00672         }
00673         
00674         break;
00675 
00676       case InitSecondSplitRead:
00677         DPRINTF(InOrderCachePort,
00678                 "[tid:%u]: [sn:%i] Initiating split data read access to %s "
00679                 "for addr. %08p\n", tid, inst->seqNum, name(),
00680                 cache_req->inst->split2ndAddr);
00681         inst->split2ndAccess = true;
00682         assert(inst->split2ndAddr != 0);
00683         read(inst, inst->split2ndAddr, &inst->split2ndData,
00684              inst->totalSize, inst->split2ndFlags);
00685         break;
00686 
00687       case InitSecondSplitWrite:
00688         DPRINTF(InOrderCachePort,
00689                 "[tid:%u]: [sn:%i] Initiating split data write access to %s "
00690                 "for addr. %08p\n", tid, inst->seqNum, name(),
00691                 cache_req->inst->getMemAddr());
00692 
00693         inst->split2ndAccess = true;
00694         assert(inst->split2ndAddr != 0);
00695         write(inst, &inst->split2ndData, inst->totalSize,
00696               inst->split2ndAddr, inst->split2ndFlags, NULL);
00697         break;
00698 
00699       case CompleteReadData:
00700         DPRINTF(InOrderCachePort,
00701                 "[tid:%i]: [sn:%i]: Trying to Complete Data Read Access\n",
00702                 tid, inst->seqNum);
00703 
00704 
00705         //@todo: timing translations need to check here...
00706         assert(!inst->isInstPrefetch() && "Can't Handle Inst. Prefecthes");
00707         if (cache_req->isMemAccComplete() || inst->isDataPrefetch()) {
00708             finishCacheUnitReq(inst, cache_req);
00709         } else {
00710             DPRINTF(InOrderStall, "STALL: [tid:%i]: Data miss from %08p\n",
00711                     tid, cache_req->inst->getMemAddr());
00712             cache_req->setCompleted(false);
00713             cache_req->setMemStall(true);            
00714         }
00715         break;
00716 
00717       case CompleteWriteData:
00718         {
00719             DPRINTF(InOrderCachePort,
00720                     "[tid:%i]: [sn:%i]: Trying to Complete Data Write Access\n",
00721                     tid, inst->seqNum);
00722 
00723 
00724             //@todo: check that timing translation is finished here
00725             RequestPtr mem_req = cache_req->memReq;
00726             if (mem_req->isCondSwap() || mem_req->isLLSC() || mem_req->isSwap()) {
00727                 DPRINTF(InOrderCachePort, "Detected Conditional Store Inst.\n");
00728 
00729                 if (!cache_req->isMemAccComplete()) {
00730                     DPRINTF(InOrderStall, "STALL: [tid:%i]: Data miss from %08p\n",
00731                             tid, cache_req->inst->getMemAddr());
00732                     cache_req->setCompleted(false);
00733                     cache_req->setMemStall(true);
00734                     return;
00735                 } else {
00736                     DPRINTF(InOrderStall, "Mem Acc Completed\n");
00737                 }
00738             }
00739 
00740             if (cache_req->isMemAccPending()) {
00741                 DPRINTF(InOrderCachePort, "Store Instruction Pending Completion.\n");
00742                 cache_req->dataPkt->reqData = cache_req->reqData;
00743                 cache_req->dataPkt->memReq = cache_req->memReq;
00744             } else
00745                 DPRINTF(InOrderCachePort, "Store Instruction Finished Completion.\n");
00746 
00747             //@todo: if split inst save data
00748             finishCacheUnitReq(inst, cache_req);
00749         }
00750         break;
00751 
00752       case CompleteSecondSplitRead:
00753         DPRINTF(InOrderCachePort,
00754                 "[tid:%i]: [sn:%i]: Trying to Complete Split Data Read "
00755                 "Access\n", tid, inst->seqNum);
00756 
00757         //@todo: check that timing translation is finished here
00758         assert(!inst->isInstPrefetch() && "Can't Handle Inst. Prefecthes");
00759         if (cache_req->isMemAccComplete() || inst->isDataPrefetch()) {
00760             finishCacheUnitReq(inst, cache_req);
00761         } else {
00762             DPRINTF(InOrderStall, "STALL: [tid:%i]: Data miss from %08p\n",
00763                     tid, cache_req->inst->split2ndAddr);
00764             cache_req->setCompleted(false);
00765             cache_req->setMemStall(true);            
00766         }
00767         break;
00768 
00769       case CompleteSecondSplitWrite:
00770         DPRINTF(InOrderCachePort,
00771                 "[tid:%i]: [sn:%i]: Trying to Complete Split Data Write "
00772                 "Access\n", tid, inst->seqNum);
00773         //@todo: illegal to have a unaligned cond.swap or llsc?
00774         assert(!cache_req->memReq->isSwap() && !cache_req->memReq->isCondSwap()
00775                && !cache_req->memReq->isLLSC());
00776 
00777         if (cache_req->isMemAccPending()) {
00778             cache_req->dataPkt->reqData = cache_req->reqData;
00779             cache_req->dataPkt->memReq = cache_req->memReq;
00780         }
00781 
00782         //@todo: check that timing translation is finished here
00783         finishCacheUnitReq(inst, cache_req);
00784         break;
00785         
00786       default:
00787         fatal("Unrecognized command to %s", resName);
00788     }
00789 }

00826 void
00827 CacheUnit::doCacheAccess(DynInstPtr inst, uint64_t *write_res,
00828                          CacheReqPtr split_req)
00829 {
00830     Fault fault = NoFault;
00831 #if TRACING_ON
00832     ThreadID tid = inst->readTid();
00833 #endif
00834     bool do_access = true;  // flag to suppress cache access
00835 
00836     // Special Handling if this is a split request
00837     CacheReqPtr cache_req;
00838     if (split_req == NULL)
00839         cache_req = dynamic_cast<CacheReqPtr>(reqs[inst->getCurResSlot()]);
00840     else {
00841         cache_req = split_req;
00842         assert(0);
00843     }
00844 
00845     // Make a new packet inside the CacheRequest object
00846     assert(cache_req);
00847     buildDataPacket(cache_req);
00848 
00849     // Special Handling for LL/SC or Compare/Swap
00850      bool is_write = cache_req->dataPkt->isWrite();
00851      RequestPtr mem_req = cache_req->dataPkt->req;
00852      if (is_write) {
00853          DPRINTF(InOrderCachePort,
00854                  "[tid:%u]: [sn:%i]: Storing data: %s\n",
00855                  tid, inst->seqNum,
00856                  printMemData(cache_req->dataPkt->getPtr<uint8_t>(),
00857                               cache_req->dataPkt->getSize()));
00858 
00859         if (mem_req->isCondSwap()) {
00860              assert(write_res);
00861              cache_req->memReq->setExtraData(*write_res);
00862          }
00863         if (mem_req->isLLSC()) {
00864             assert(cache_req->inst->isStoreConditional());
00865             DPRINTF(InOrderCachePort, "Evaluating Store Conditional access\n");
00866             do_access = TheISA::handleLockedWrite(inst.get(), mem_req);
00867         }
00868      }
00869 
00870     // Finally, go ahead and make the access if we can...
00871     DPRINTF(InOrderCachePort,
00872             "[tid:%i] [sn:%i] attempting to access cache for addr %08p\n",
00873             tid, inst->seqNum, cache_req->dataPkt->getAddr());
00874 
00875     if (do_access) {
00876         if (!cachePort->sendTimingReq(cache_req->dataPkt)) {
00877             DPRINTF(InOrderCachePort,
00878                     "[tid:%i] [sn:%i] cannot access cache, because port "
00879                     "is blocked. now waiting to retry request\n", tid, 
00880                     inst->seqNum);
00881             delete cache_req->dataPkt;
00882             cache_req->dataPkt = NULL;
00883 
00884             delete cache_req->memReq;
00885             cache_req->memReq = NULL;
00886 
00887             cache_req->done(false);
00888             cachePortBlocked = true;
00889         } else {
00890             DPRINTF(InOrderCachePort,
00891                     "[tid:%i] [sn:%i] is now waiting for cache response\n",
00892                     tid, inst->seqNum);
00893             cache_req->setCompleted();
00894             cache_req->setMemAccPending();
00895             cachePortBlocked = false;
00896         }
00897     } else if (mem_req->isLLSC()){
00898         // Store-Conditional instructions complete even if they "failed"
00899         assert(cache_req->inst->isStoreConditional());
00900         cache_req->setCompleted(true);
00901 
00902         DPRINTF(LLSC,
00903                 "[tid:%i]: T%i Ignoring Failed Store Conditional Access\n",
00904                 tid, tid);
00905 
00906         processCacheCompletion(cache_req->dataPkt);
00907     } else {
00908         delete cache_req->dataPkt;
00909         cache_req->dataPkt = NULL;
00910 
00911         delete cache_req->memReq;
00912         cache_req->memReq = NULL;
00913 
00914         // Make cache request again since access due to
00915         // inability to access
00916         DPRINTF(InOrderStall, "STALL: \n");
00917         cache_req->done(false);
00918     }
00919 
00920 }

00922 bool
00923 CacheUnit::processSquash(CacheReqPacket *cache_pkt)
00924 {
00925     // The resource may no longer be actively servicing this
00926     // packet. Scenarios like a store that has been sent to the
00927     // memory system or access that's been squashed. If that's
00928     // the case, we can't access the request slot because it
00929     // will be either invalid or servicing another request.
00930     if (!cache_pkt->hasSlot) {
00931         DPRINTF(InOrderCachePort,
00932                 "%x does not have a slot in unit, ignoring.\n",
00933                 cache_pkt->getAddr());
00934 
00935         if (cache_pkt->reqData) {
00936             delete [] cache_pkt->reqData;
00937             cache_pkt->reqData = NULL;
00938         }
00939 
00940         if (cache_pkt->memReq) {
00941             delete cache_pkt->memReq;
00942             cache_pkt->memReq = NULL;
00943         }
00944 
00945         delete cache_pkt;
00946         cache_pkt = NULL;
00947         cpu->wakeCPU();
00948         return true;
00949     } else {
00950         DPRINTF(InOrderCachePort, "%x has slot %i\n",
00951                 cache_pkt->getAddr(), cache_pkt->cacheReq->getSlot());
00952     }
00953 
00954 
00955     // It's possible that the request is squashed but the
00956     // packet is still acknowledged by the resource. Squashes
00957     // should happen at the end of the cycles and trigger the
00958     // code above, but if not, this would handle any timing
00959     // variations due to diff. user parameters.
00960     if (cache_pkt->cacheReq->isSquashed()) {
00961         DPRINTF(InOrderCachePort,
00962                 "Ignoring completion of squashed access, [tid:%i] [sn:%i]\n",
00963                 cache_pkt->cacheReq->getInst()->readTid(),
00964                 cache_pkt->cacheReq->getInst()->seqNum);
00965 
00966         cache_pkt->cacheReq->setMemAccPending(false);
00967         cache_pkt->cacheReq->freeSlot();
00968         delete cache_pkt;
00969         cache_pkt = NULL;
00970         cpu->wakeCPU();
00971         return true;
00972     }
00973 
00974 
00975     return false;
00976 }

00978 void
00979 CacheUnit::processCacheCompletion(PacketPtr pkt)
00980 {
00981     //@todo: use packet sender state instead of deriving from packet class to
00982     //  get special state
00983     CacheReqPacket* cache_pkt = dynamic_cast<CacheReqPacket*>(pkt);
00984     assert(cache_pkt);
00985 
00986     DPRINTF(InOrderCachePort, "Finished request for %x\n", pkt->getAddr());
00987 
00988     if (processSquash(cache_pkt))
00989         return;
00990 
00991     CacheRequest *cache_req = dynamic_cast<CacheReqPtr>(
00992         findRequest(cache_pkt->cacheReq->getInst(), cache_pkt->instIdx));
00993 
00994     if (!cache_req) {
00995         panic("[tid:%u]: [sn:%i]: Can't find slot for cache access to "
00996               "addr. %08p\n", cache_pkt->cacheReq->getInst()->readTid(),
00997               cache_pkt->cacheReq->getInst()->seqNum,
00998               cache_pkt->cacheReq->getInst()->getMemAddr());
00999     }
01000     
01001     assert(cache_req);
01002     assert(cache_req == cache_pkt->cacheReq);
01003 
01004     DPRINTF(InOrderCachePort,
01005             "[tid:%u]: [sn:%i]: [slot:%i] Waking from cache access (vaddr.%08p, paddr:%08p)\n",
01006             cache_pkt->cacheReq->getInst()->readTid(),
01007             cache_pkt->cacheReq->getInst()->seqNum,
01008             cache_req->getSlot(),
01009             cache_pkt->req->getVaddr(),
01010             cache_pkt->req->getPaddr());
01011 
01012     // Get resource request info
01013     unsigned stage_num = cache_req->getStageNum();
01014     DynInstPtr inst = cache_req->inst;
01015     ThreadID tid = cache_req->inst->readTid();
01016 
01017     assert(!cache_req->isSquashed());
01018     assert(inst->staticInst && inst->isMemRef());
01019 
01020 
01021     DPRINTF(InOrderCachePort,
01022             "[tid:%u]: [sn:%i]: Processing cache access\n",
01023             tid, inst->seqNum);
01024 
01025     PacketPtr split_pkt = NULL;
01026     if (inst->splitInst) {
01027         inst->splitFinishCnt++;
01028 
01029         if (inst->splitFinishCnt == 2) {
01030             cache_req->memReq->setVirt(0/*inst->tid*/,
01031                                        inst->getMemAddr(),
01032                                        inst->totalSize,
01033                                        0,
01034                                        cpu->dataMasterId(),
01035                                        0);
01036 
01037             split_pkt = new Packet(cache_req->memReq, cache_req->pktCmd);
01038             split_pkt->dataStatic(inst->splitMemData);
01039 
01040             DPRINTF(InOrderCachePort, "Completing Split Access.\n");
01041             inst->completeAcc(split_pkt);
01042         }
01043     } else {
01044         inst->completeAcc(cache_pkt);
01045     }
01046 
01047     inst->setExecuted();
01048 
01049     if (inst->isLoad()) {
01050         assert(cache_pkt->isRead());
01051 
01052         if (cache_pkt->req->isLLSC()) {
01053             DPRINTF(InOrderCachePort,
01054                     "[tid:%u]: Handling Load-Linked for [sn:%u]\n",
01055                     tid, inst->seqNum);
01056             TheISA::handleLockedRead(inst.get(), cache_pkt->req);
01057         }
01058 
01059         DPRINTF(InOrderCachePort,
01060                 "[tid:%u]: [sn:%i]: Bytes loaded were: %s\n",
01061                 tid, inst->seqNum,
01062                 (split_pkt) ? printMemData(split_pkt->getPtr<uint8_t>(),
01063                                            split_pkt->getSize()) :
01064                               printMemData(cache_pkt->getPtr<uint8_t>(),
01065                                            cache_pkt->getSize()));
01066     } else if(inst->isStore()) {
01067         assert(cache_pkt->isWrite());
01068 
01069         DPRINTF(InOrderCachePort,
01070                 "[tid:%u]: [sn:%i]: Bytes stored were: %s\n",
01071                 tid, inst->seqNum,
01072                 (split_pkt) ? printMemData(split_pkt->getPtr<uint8_t>(),
01073                                            split_pkt->getSize()) :
01074                               printMemData(cache_pkt->getPtr<uint8_t>(),
01075                                            cache_pkt->getSize()));
01076     }
01077 
01078 
01079     if (split_pkt) {
01080         delete split_pkt;
01081         split_pkt = NULL;
01082     }
01083 
01084     cache_req->setMemAccPending(false);
01085     cache_req->setMemAccCompleted();
01086 
01087     if (cache_req->isMemStall() &&
01088         cpu->threadModel == InOrderCPU::SwitchOnCacheMiss) {
01089         DPRINTF(InOrderCachePort, "[tid:%u] Waking up from Cache Miss.\n",
01090                 tid);
01091             
01092         cpu->activateContext(tid);
01093             
01094         DPRINTF(ThreadModel, "Activating [tid:%i] after return from cache"
01095                 "miss.\n", tid);
01096     }
01097         
01098     // Wake up the CPU (if it went to sleep and was waiting on this
01099     // completion event).
01100     cpu->wakeCPU();
01101 
01102     DPRINTF(Activity, "[tid:%u] Activating %s due to cache completion\n",
01103             tid, cpu->pipelineStage[stage_num]->name());
01104 
01105     cpu->switchToActive(stage_num);
01106 }

01217             req_ptr->getInst()->seqNum > squash_seq_num) {
01218

01224                 DPRINTF(AddrDep, "Request for [tid:%i] [sn:%i] already "

01227                         req_ptr->getInst()->seqNum);
