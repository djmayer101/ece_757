00272

00274         if (fault != NoFault || secondAddr <= addr)
00275         {
00276             if (fault != NoFault && was_prefetch) {
00277               fault = NoFault;
00278             }
00279             break;
00280         }
00281 
00282         //Update size and access address
00283         size = addr + fullSize - secondAddr;
00284         //And access the right address.

00287
