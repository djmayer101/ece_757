00070         curMacroStaticInst = StaticInst::nullStaticInstPtr;

00482 void
00483 Checker<Impl>::validateExecution(DynInstPtr &inst)
00484 {
00485     uint64_t checker_val;
00486     uint64_t inst_val;
00487     int idx = -1;
00488     bool result_mismatch = false;
00489 
00490     if (inst->isUnverifiable()) {
00491         // Unverifiable instructions assume they were executed
00492         // properly by the CPU. Grab the result from the
00493         // instruction and write it to the register.
00494         copyResult(inst, 0, idx);
00495     } else if (inst->numDestRegs() > 0 && !result.empty()) {
00496         DPRINTF(Checker, "Dest regs %d, number of checker dest regs %d\n",
00497                          inst->numDestRegs(), result.size());
00498         for (int i = 0; i < inst->numDestRegs() && !result.empty(); i++) {
00499             result.front().get(checker_val);
00500             result.pop();
00501             inst_val = 0;
00502             inst->template popResult<uint64_t>(inst_val);
00503             if (checker_val != inst_val) {
00504                 result_mismatch = true;
00505                 idx = i;
00506                 break;
00507             }
00508         }
00509     } // Checker CPU checks all the saved results in the dyninst passed by
00510       // the cpu model being checked against the saved results present in
00511       // the static inst executed in the Checker.  Sometimes the number
00512       // of saved results differs between the dyninst and static inst, but
00513       // this is ok and not a bug.  May be worthwhile to try and correct this.
00514 
00515     if (result_mismatch) {
00516         warn("%lli: Instruction results do not match! (Values may not "
00517              "actually be integers) Inst: %#x, checker: %#x",
00518              curTick(), inst_val, checker_val);
00519 
00520         // It's useful to verify load values from memory, but in MP
00521         // systems the value obtained at execute may be different than
00522         // the value obtained at completion.  Similarly DMA can
00523         // present the same problem on even UP systems.  Thus there is
00524         // the option to only warn on loads having a result error.
00525         // The load/store queue in Detailed CPU can also cause problems
00526         // if load/store forwarding is allowed.
00527         if (inst->isLoad() && warnOnlyOnLoadError) {
00528             copyResult(inst, inst_val, idx);
00529         } else {
00530             handleError(inst);
00531         }
00532     }
00533 
00534     if (inst->nextInstAddr() != thread->nextInstAddr()) {
00535         warn("%lli: Instruction next PCs do not match! Inst: %#x, "
00536              "checker: %#x",
00537              curTick(), inst->nextInstAddr(), thread->nextInstAddr());
00538         handleError(inst);
00539     }
00540 
00541     // Checking side effect registers can be difficult if they are not
00542     // checked simultaneously with the execution of the instruction.
00543     // This is because other valid instructions may have modified
00544     // these registers in the meantime, and their values are not
00545     // stored within the DynInst.
00546     while (!miscRegIdxs.empty()) {
00547         int misc_reg_idx = miscRegIdxs.front();
00548         miscRegIdxs.pop();

00550         if (inst->tcBase()->readMiscRegNoEffect(misc_reg_idx) !=

00552             warn("%lli: Misc reg idx %i (side effect) does not match! "
00553                  "Inst: %#x, checker: %#x",
00554                  curTick(), misc_reg_idx,
00555                  inst->tcBase()->readMiscRegNoEffect(misc_reg_idx),
00556                  thread->readMiscRegNoEffect(misc_reg_idx));
00557             handleError(inst);
00558         }
00559     }
