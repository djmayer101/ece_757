00081     void update() const
00082     {
00083         // both tick and cycle are up-to-date and we are done, note
00084         // that the >= is important as it captures cases where tick
00085         // has already passed curTick()
00086         if (tick >= curTick())
00087             return;
00088 
00089         // optimise for the common case and see if the tick should be
00090         // advanced by a single clock period
00091         tick += clock;
00092         ++cycle;
00093 
00094         // see if we are done at this point
00095         if (tick >= curTick())
00096             return;
00097 
00098         // if not, we have to recalculate the cycle and tick, we
00099         // perform the calculations in terms of relative cycles to
00100         // allow changes to the clock period in the future
00101         Cycles elapsedCycles(divCeil(curTick() - tick, clock));
00102         cycle += elapsedCycles;
00103         tick += elapsedCycles * clock;
00104     }
