00083 uint8_t
00084 X86ISA::Cmos::readRegister(uint8_t reg)
00085 {
00086     assert(reg < numRegs);
00087     uint8_t val;
00088     if (reg <= 0xD) {
00089         val = rtc.readData(reg);
00090         DPRINTF(CMOS,
00091             "Reading CMOS RTC reg %x as %x.\n", reg, val);
00092     } else {
00093         val = regs[reg];
00094         DPRINTF(CMOS,
00095             "Reading non-volitile CMOS address %x as %x.\n", reg, val);
00096     }
00097     return val;
00098 }

00102 {
00103     assert(reg < numRegs);
00104     if (reg <= 0xD) {
00105         DPRINTF(CMOS, "Writing CMOS RTC reg %x with %x.\n",

00109         DPRINTF(CMOS, "Writing non-volitile CMOS address %x with %x.\n",
