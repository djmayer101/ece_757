00115 CoherentBus::recvTimingReq(PacketPtr pkt, PortID slave_port_id)
00116 {
00117     // determine the source port based on the id
00118     SlavePort *src_port = slavePorts[slave_port_id];
00119 
00120     // remember if the packet is an express snoop
00121     bool is_express_snoop = pkt->isExpressSnoop();
00122 
00123     // test if the bus should be considered occupied for the current
00124     // port, and exclude express snoops from the check
00125     if (!is_express_snoop && !reqLayer.tryTiming(src_port)) {
00126         DPRINTF(CoherentBus, "recvTimingReq: src %s %s 0x%x BUSY\n",
00127                 src_port->name(), pkt->cmdString(), pkt->getAddr());
00128         return false;
00129     }
00130 
00131     DPRINTF(CoherentBus, "recvTimingReq: src %s %s expr %d 0x%x\n",
00132             src_port->name(), pkt->cmdString(), is_express_snoop,
00133             pkt->getAddr());
00134 
00135     // set the source port for routing of the response
00136     pkt->setSrc(slave_port_id);
00137 
00138     calcPacketTiming(pkt);
00139     Tick packetFinishTime = pkt->busLastWordDelay + curTick();
00140 
00141     // uncacheable requests need never be snooped
00142     if (!pkt->req->isUncacheable() && !system->bypassCaches()) {
00143         // the packet is a memory-mapped request and should be
00144         // broadcasted to our snoopers but the source
00145         forwardTiming(pkt, slave_port_id);
00146     }
00147 
00148     // remember if we add an outstanding req so we can undo it if
00149     // necessary, if the packet needs a response, we should add it
00150     // as outstanding and express snoops never fail so there is
00151     // not need to worry about them
00152     bool add_outstanding = !is_express_snoop && pkt->needsResponse();
00153 
00154     // keep track that we have an outstanding request packet
00155     // matching this request, this is used by the coherency
00156     // mechanism in determining what to do with snoop responses
00157     // (in recvTimingSnoop)
00158     if (add_outstanding) {
00159         // we should never have an exsiting request outstanding
00160         assert(outstandingReq.find(pkt->req) == outstandingReq.end());
00161         outstandingReq.insert(pkt->req);
00162     }
00163 
00164     // since it is a normal request, determine the destination
00165     // based on the address and attempt to send the packet
00166     bool success = masterPorts[findPort(pkt->getAddr())]->sendTimingReq(pkt);
00167 
00168     // if this is an express snoop, we are done at this point
00169     if (is_express_snoop) {
00170         assert(success);
00171     } else {
00172         // for normal requests, check if successful
00173         if (!success)  {
00174             // inhibited packets should never be forced to retry
00175             assert(!pkt->memInhibitAsserted());
00176 
00177             // if it was added as outstanding and the send failed, then
00178             // erase it again
00179             if (add_outstanding)
00180                 outstandingReq.erase(pkt->req);
00181 
00182             // undo the calculation so we can check for 0 again
00183             pkt->busFirstWordDelay = pkt->busLastWordDelay = 0;
00184 
00185             DPRINTF(CoherentBus, "recvTimingReq: src %s %s 0x%x RETRY\n",
00186                     src_port->name(), pkt->cmdString(), pkt->getAddr());
00187 
00188             // update the bus state and schedule an idle event
00189             reqLayer.failedTiming(src_port, clockEdge(Cycles(headerCycles)));
00190         } else {
00191             // update the bus state and schedule an idle event
00192             reqLayer.succeededTiming(packetFinishTime);
00193         }
00194     }
00195 
00196     return success;
00197 }

00279 
00280     // get the destination from the packet
00281     PortID dest = pkt->getDest();
00282 
00283     // responses are never express snoops
00284     assert(!pkt->isExpressSnoop());
00285 
00286     calcPacketTiming(pkt);
00287     Tick packetFinishTime = pkt->busLastWordDelay + curTick();
00288 
00289     // determine if the response is from a snoop request we
00290     // created as the result of a normal request (in which case it
00291     // should be in the outstandingReq), or if we merely forwarded
00292     // someone else's snoop request
00293     if (outstandingReq.find(pkt->req) == outstandingReq.end()) {
00294         // this is a snoop response to a snoop request we
00295         // forwarded, e.g. coming from the L1 and going to the L2
00296         // this should be forwarded as a snoop response
00297         bool success M5_VAR_USED = masterPorts[dest]->sendTimingSnoopResp(pkt);
00298         assert(success);
00299     } else {
00300         // we got a snoop response on one of our slave ports,
00301         // i.e. from a coherent master connected to the bus, and
00302         // since we created the snoop request as part of
00303         // recvTiming, this should now be a normal response again
00304         outstandingReq.erase(pkt->req);
00305 
00306         // this is a snoop response from a coherent master, with a
00307         // destination field set on its way through the bus as
00308         // request, hence it should never go back to where the
00309         // snoop response came from, but instead to where the
00310         // original request came from
00311         assert(slave_port_id != dest);
00312 
00313         // as a normal response, it should go back to a master
00314         // through one of our slave ports
00315         bool success M5_VAR_USED = slavePorts[dest]->sendTimingResp(pkt);
00316 
00317         // currently it is illegal to block responses... can lead
00318         // to deadlock
00319         assert(success);
00320     }
00321 
00322     snoopRespLayer.succeededTiming(packetFinishTime);
00323 
00324     return true;

00364     MemCmd snoop_response_cmd = MemCmd::InvalidCmd;
00365     Tick snoop_response_latency = 0;

00384     if (snoop_response_cmd != MemCmd::InvalidCmd) {
00385         // no one else should have responded

00417 std::pair<MemCmd, Tick>
00418 CoherentBus::forwardAtomic(PacketPtr pkt, PortID exclude_slave_port_id)
00419 {
00420     // the packet may be changed on snoops, record the original
00421     // command to enable us to restore it between snoops so that
00422     // additional snoops can take place properly
00423     MemCmd orig_cmd = pkt->cmd;
00424     MemCmd snoop_response_cmd = MemCmd::InvalidCmd;
00425     Tick snoop_response_latency = 0;
00426 
00427     // snoops should only happen if the system isn't bypassing caches
00428     assert(!system->bypassCaches());
00429 
00430     for (SlavePortIter s = snoopPorts.begin(); s != snoopPorts.end(); ++s) {
00431         SlavePort *p = *s;
00432         // we could have gotten this request from a snooping master
00433         // (corresponding to our own slave port that is also in
00434         // snoopPorts) and should not send it back to where it came
00435         // from
00436         if (exclude_slave_port_id == InvalidPortID ||
00437             p->getId() != exclude_slave_port_id) {
00438             Tick latency = p->sendAtomicSnoop(pkt);
00439             // in contrast to a functional access, we have to keep on
00440             // going as all snoopers must be updated even if we get a
00441             // response
00442             if (pkt->isResponse()) {
00443                 // response from snoop agent
00444                 assert(pkt->cmd != orig_cmd);
00445                 assert(pkt->memInhibitAsserted());
00446                 // should only happen once
00447                 assert(snoop_response_cmd == MemCmd::InvalidCmd);
00448                 // save response state
00449                 snoop_response_cmd = pkt->cmd;
00450                 snoop_response_latency = latency;
00451                 // restore original packet state for remaining snoopers
00452                 pkt->cmd = orig_cmd;
00453             }
00454         }
00455     }
00456 
00457     // the packet is restored as part of the loop and any potential
00458     // snoop response is part of the returned pair
00459     return std::make_pair(snoop_response_cmd, snoop_response_latency);
