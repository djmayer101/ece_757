00303             panic("Monitor got a response without monitor sender state\n");
00304 
00305         // Restore the sate
00306         pkt->senderState = commReceivedState->predecessor;
00307     }
00308 
00309     // Attempt to send the packet
00310     bool successful = slavePort.sendTimingResp(pkt);
00311 
00312     if (!stats.disableLatencyHists) {
00313         // If packet successfully send, sample value of latency,
00314         // afterwards delete sender state, otherwise restore state
00315         if (successful) {
00316             latency = curTick() - commReceivedState->transmitTime;
00317             DPRINTF(CommMonitor, "Latency: %d\n", latency);
00318             delete commReceivedState;
00319         } else {
00320             // Don't delete anything and let the packet look like we
00321             // did not touch it
00322             pkt->senderState = commReceivedState;
00323         }
00324     }
00325 
00326     if (successful && isRead) {
00327         // Decrement number of outstanding read requests
00328         DPRINTF(CommMonitor, "Received read response\n");
00329         if (!stats.disableOutstandingHists) {
00330             assert(stats.outstandingReadReqs != 0);
00331             --stats.outstandingReadReqs;
00332         }
00333 
00334         if (!stats.disableLatencyHists) {
00335             stats.readLatencyHist.sample(latency);
00336         }
00337 
00338         // Update the bandwidth stats based on responses for reads
00339         if (!stats.disableBandwidthHists) {
00340             stats.readBytes += size;
00341             stats.totalReadBytes += size;
00342         }
00343 
00344     } else if (successful && isWrite) {
00345         // Decrement number of outstanding write requests
00346         DPRINTF(CommMonitor, "Received write response\n");
00347         if (!stats.disableOutstandingHists) {
00348             assert(stats.outstandingWriteReqs != 0);
00349             --stats.outstandingWriteReqs;
00350         }
00351 
00352         if (!stats.disableLatencyHists) {
00353             stats.writeLatencyHist.sample(latency);
00354         }
00355     } else if (successful) {
00356         DPRINTF(CommMonitor, "Received non read/write response\n");

00543 void
00544 CommMonitor::samplePeriodic()
00545 {
00546     // the periodic stats update runs on the granularity of sample
00547     // periods, but in combination with this there may also be a
00548     // external resets and dumps of the stats (through schedStatEvent)
00549     // causing the stats themselves to capture less than a sample
00550     // period
00551 
00552     // only capture if we have not reset the stats during the last
00553     // sample period
00554     if (simTicks.value() >= samplePeriodTicks) {
00555         if (!stats.disableTransactionHists) {
00556             stats.readTransHist.sample(stats.readTrans);
00557             stats.writeTransHist.sample(stats.writeTrans);
00558         }
00559 
00560         if (!stats.disableBandwidthHists) {
00561             stats.readBandwidthHist.sample(stats.readBytes / samplePeriod);
00562             stats.writeBandwidthHist.sample(stats.writtenBytes / samplePeriod);
00563         }
00564 
00565         if (!stats.disableOutstandingHists) {
00566             stats.outstandingReadsHist.sample(stats.outstandingReadReqs);
00567             stats.outstandingWritesHist.sample(stats.outstandingWriteReqs);
00568         }
00569     }
00570 
00571     // reset the sampled values
00572     stats.readTrans = 0;
00573     stats.writeTrans = 0;
00574 
00575     stats.readBytes = 0;
00576     stats.writtenBytes = 0;
00577 
00578     schedule(samplePeriodicEvent, curTick() + samplePeriodTicks);
00579 }
