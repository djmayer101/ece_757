00396 template <class Impl>
00397 bool
00398 DefaultCommit<Impl>::isDrained() const
00399 {
00400     /* Make sure no one is executing microcode. There are two reasons
00401      * for this:
00402      * - Hardware virtualized CPUs can't switch into the middle of a
00403      *   microcode sequence.
00404      * - The current fetch implementation will most likely get very
00405      *   confused if it tries to start fetching an instruction that
00406      *   is executing in the middle of a ucode sequence that changes
00407      *   address mappings. This can happen on for example x86.
00408      */
00409     for (ThreadID tid = 0; tid < numThreads; tid++) {
00410         if (pc[tid].microPC() != 0)
00411             return false;
00412     }
00413 
00414     /* Make sure that all instructions have finished committing before
00415      * declaring the system as drained. We want the pipeline to be
00416      * completely empty when we declare the CPU to be drained. This
00417      * makes debugging easier since CPU handover and restoring from a
00418      * checkpoint with a different CPU should have the same timing.
00419      */
00420     return rob->isEmpty() &&
00421         interrupt == NoFault;
00422 }

00546 template <class Impl>
00547 void
00548 DefaultCommit<Impl>::squashAll(ThreadID tid)
00549 {
00550     // If we want to include the squashing instruction in the squash,
00551     // then use one older sequence number.
00552     // Hopefully this doesn't mess things up.  Basically I want to squash
00553     // all instructions of this thread.
00554     InstSeqNum squashed_inst = rob->isEmpty() ?
00555         lastCommitedSeqNum[tid] : rob->readHeadInst(tid)->seqNum - 1;
00556 
00557     // All younger instructions will be squashed. Set the sequence
00558     // number as the youngest instruction in the ROB (0 in this case.
00559     // Hopefully nothing breaks.)
00560     youngestSeqNum[tid] = lastCommitedSeqNum[tid];
00561 
00562     rob->squash(squashed_inst, tid);
00563     changedROBNumEntries[tid] = true;
00564 
00565     // Send back the sequence number of the squashed instruction.
00566     toIEW->commitInfo[tid].doneSeqNum = squashed_inst;
00567 
00568     // Send back the squash signal to tell stages that they should
00569     // squash.
00570     toIEW->commitInfo[tid].squash = true;
00571 
00572     // Send back the rob squashing signal so other stages know that
00573     // the ROB is in the process of squashing.
00574     toIEW->commitInfo[tid].robSquashing = true;
00575 
00576     toIEW->commitInfo[tid].mispredictInst = NULL;
00577     toIEW->commitInfo[tid].squashInst = NULL;
00578 
00579     toIEW->commitInfo[tid].pc = pc[tid];

00650 {

00652     _nextStatus = Inactive;
00653 
00654     if (activeThreads->empty())
00655         return;
00656 
00657     list<ThreadID>::iterator threads = activeThreads->begin();
00658     list<ThreadID>::iterator end = activeThreads->end();
00659 
00660     // Check if any of the threads are done squashing.  Change the
00661     // status if they are done.
00662     while (threads != end) {
00663         ThreadID tid = *threads++;
00664 
00665         // Clear the bit saying if the thread has committed stores
00666         // this cycle.
00667         committedStores[tid] = false;
00668 
00669         if (commitStatus[tid] == ROBSquashing) {
00670

00672                 commitStatus[tid] = Running;

00675                         " insts this cycle.\n", tid);
00676                 rob->doSquash(tid);
00677                 toIEW->commitInfo[tid].robSquashing = true;
00678                 wroteToTimeBuffer = true;
00679             }
00680         }
00681     }
00682 
00683     commit();
00684 
00685     markCompletedInsts();
00686 
00687     threads = activeThreads->begin();
00688 
00689     while (threads != end) {
00690         ThreadID tid = *threads++;
00691 
00692         if (!rob->isEmpty(tid) && rob->readHeadInst(tid)->readyToCommit()) {
00693             // The ROB has more instructions it can commit. Its next status
00694             // will be active.
00695             _nextStatus = Active;
00696 
00697             DynInstPtr inst = rob->readHeadInst(tid);
00698 
00699             DPRINTF(Commit,"[tid:%i]: Instruction [sn:%lli] PC %s is head of"
00700                     " ROB and ready to commit\n",
00701                     tid, inst->seqNum, inst->pcState());
00702 
00703         } else if (!rob->isEmpty(tid)) {
00704             DynInstPtr inst = rob->readHeadInst(tid);
00705 
00706             DPRINTF(Commit,"[tid:%i]: Can't commit, Instruction [sn:%lli] PC "
00707                     "%s is head of ROB and not ready\n",
00708                     tid, inst->seqNum, inst->pcState());
00709         }
00710 
00711         DPRINTF(Commit, "[tid:%i]: ROB has %d insts & %d free entries.\n",
00712                 tid, rob->countInsts(tid), rob->numFreeEntries(tid));
00713     }
00714 
00715 
00716     if (wroteToTimeBuffer) {
00717         DPRINTF(Activity, "Activity This Cycle.\n");
00718         cpu->activityThisCycle();
00719     }
00720 
00721     updateStatus();
00722 }

00724 template <class Impl>
00725 void
00726 DefaultCommit<Impl>::handleInterrupt()
00727 {
00728     // Verify that we still have an interrupt to handle
00729     if (!cpu->checkInterrupts(cpu->tcBase(0))) {
00730         DPRINTF(Commit, "Pending interrupt is cleared by master before "
00731                 "it got handled. Restart fetching from the orig path.\n");
00732         toIEW->commitInfo[0].clearInterrupt = true;
00733         interrupt = NoFault;
00734         avoidQuiesceLiveLock = true;
00735         return;
00736     }
00737 
00738     // Wait until all in flight instructions are finished before enterring
00739     // the interrupt.
00740     if (canHandleInterrupts && cpu->instList.empty()) {
00741         // Squash or record that I need to squash this cycle if
00742         // an interrupt needed to be handled.
00743         DPRINTF(Commit, "Interrupt detected.\n");
00744 
00745         // Clear the interrupt now that it's going to be handled
00746         toIEW->commitInfo[0].clearInterrupt = true;
00747 
00748         assert(!thread[0]->noSquashFromTC);
00749         thread[0]->noSquashFromTC = true;
00750 
00751         if (cpu->checker) {
00752             cpu->checker->handlePendingInt();
00753         }
00754 
00755         // CPU will handle interrupt.
00756         cpu->processInterrupts(interrupt);
00757 
00758         thread[0]->noSquashFromTC = false;
00759 
00760         commitStatus[0] = TrapPending;
00761 
00762         // Generate trap squash event.
00763         generateTrapEvent(0);
00764 
00765         interrupt = NoFault;
00766         avoidQuiesceLiveLock = false;
00767     } else {
00768         DPRINTF(Commit, "Interrupt pending: instruction is %sin "
00769                 "flight, ROB is %sempty\n",
00770                 canHandleInterrupts ? "not " : "",
00771                 cpu->instList.empty() ? "" : "not " );
00772     }
00773 }

00775 template <class Impl>
00776 void
00777 DefaultCommit<Impl>::propagateInterrupt()
00778 {
00779     if (commitStatus[0] == TrapPending || interrupt || trapSquash[0] ||
00780             tcSquash[0])
00781         return;
00782 
00783     // Process interrupts if interrupts are enabled, not in PAL
00784     // mode, and no other traps or external squashes are currently
00785     // pending.
00786     // @todo: Allow other threads to handle interrupts.
00787 
00788     // Get any interrupt that happened
00789     interrupt = cpu->getInterrupts();
00790 
00791     // Tell fetch that there is an interrupt pending.  This
00792     // will make fetch wait until it sees a non PAL-mode PC,
00793     // at which point it stops fetching instructions.
00794     if (interrupt != NoFault)
00795         toIEW->commitInfo[0].interruptPending = true;
00796 }

00798 template <class Impl>
00799 void
00800 DefaultCommit<Impl>::commit()
00801 {
00802     if (FullSystem) {
00803         // Check if we have a interrupt and get read to handle it
00804         if (cpu->checkInterrupts(cpu->tcBase(0)))
00805             propagateInterrupt();
00806     }
00807 
00809     // Check for any possible squashes, handle them first
00811     list<ThreadID>::iterator threads = activeThreads->begin();
00812     list<ThreadID>::iterator end = activeThreads->end();
00813 
00814     while (threads != end) {
00815         ThreadID tid = *threads++;
00816 
00817         // Not sure which one takes priority.  I think if we have
00818         // both, that's a bad sign.
00819         if (trapSquash[tid] == true) {
00820             assert(!tcSquash[tid]);
00821             squashFromTrap(tid);
00822         } else if (tcSquash[tid] == true) {
00823             assert(commitStatus[tid] != TrapPending);
00824             squashFromTC(tid);
00825         } else if (commitStatus[tid] == SquashAfterPending) {
00826             // A squash from the previous cycle of the commit stage (i.e.,
00827             // commitInsts() called squashAfter) is pending. Squash the
00828             // thread now.
00829             squashFromSquashAfter(tid);
00830         }
00831 
00832         // Squashed sequence number must be older than youngest valid
00833         // instruction in the ROB. This prevents squashes from younger
00834         // instructions overriding squashes from older instructions.
00835         if (fromIEW->squash[tid] &&
00836             commitStatus[tid] != TrapPending &&
00837             fromIEW->squashedSeqNum[tid] <= youngestSeqNum[tid]) {
00838 
00839             if (fromIEW->mispredictInst[tid]) {
00840                 DPRINTF(Commit,
00841                     "[tid:%i]: Squashing due to branch mispred PC:%#x [sn:%i]\n",
00842                     tid,
00843                     fromIEW->mispredictInst[tid]->instAddr(),
00844                     fromIEW->squashedSeqNum[tid]);
00845             } else {
00846                 DPRINTF(Commit,
00847                     "[tid:%i]: Squashing due to order violation [sn:%i]\n",
00848                     tid, fromIEW->squashedSeqNum[tid]);
00849             }
00850 
00851             DPRINTF(Commit, "[tid:%i]: Redirecting to PC %#x\n",
00852                     tid,
00853                     fromIEW->pc[tid].nextInstAddr());
00854 
00855             commitStatus[tid] = ROBSquashing;
00856 
00857             // If we want to include the squashing instruction in the squash,
00858             // then use one older sequence number.
00859             InstSeqNum squashed_inst = fromIEW->squashedSeqNum[tid];
00860 
00861             if (fromIEW->includeSquashInst[tid] == true) {
00862                 squashed_inst--;
00863             }
00864 
00865             // All younger instructions will be squashed. Set the sequence
00866             // number as the youngest instruction in the ROB.
00867             youngestSeqNum[tid] = squashed_inst;
00868 
00869             rob->squash(squashed_inst, tid);
00870             changedROBNumEntries[tid] = true;
00871 
00872             toIEW->commitInfo[tid].doneSeqNum = squashed_inst;
00873 
00874             toIEW->commitInfo[tid].squash = true;
00875 
00876             // Send back the rob squashing signal so other stages know that
00877             // the ROB is in the process of squashing.
00878             toIEW->commitInfo[tid].robSquashing = true;
00879 
00880             toIEW->commitInfo[tid].mispredictInst =
00881                 fromIEW->mispredictInst[tid];
00882             toIEW->commitInfo[tid].branchTaken =
00883                 fromIEW->branchTaken[tid];
00884             toIEW->commitInfo[tid].squashInst =
00885                                     rob->findInst(tid, squashed_inst);
00886             if (toIEW->commitInfo[tid].mispredictInst) {
00887                 if (toIEW->commitInfo[tid].mispredictInst->isUncondCtrl()) {
00888                      toIEW->commitInfo[tid].branchTaken = true;
00889                 }
00890             }
00891 
00892             toIEW->commitInfo[tid].pc = fromIEW->pc[tid];
00893 
00894             if (toIEW->commitInfo[tid].mispredictInst) {
00895                 ++branchMispredicts;
00896             }
00897         }
00898 
00899     }
00900 
00901     setNextStatus();
00902 
00903     if (squashCounter != numThreads) {
00904         // If we're not currently squashing, then get instructions.
00905         getInsts();
00906 
00907         // Try to commit any instructions.
00908         commitInsts();
00909     }
00910 
00911     //Check for any activity
00912     threads = activeThreads->begin();
00913 
00914     while (threads != end) {
00915         ThreadID tid = *threads++;
00916 
00917         if (changedROBNumEntries[tid]) {
00918             toIEW->commitInfo[tid].usedROB = true;
00919             toIEW->commitInfo[tid].freeROBEntries = rob->numFreeEntries(tid);
00920 
00921             wroteToTimeBuffer = true;
00922             changedROBNumEntries[tid] = false;
00923             if (rob->isEmpty(tid))
00924                 checkEmptyROB[tid] = true;
00925         }
00926 
00927         // ROB is only considered "empty" for previous stages if: a)
00928         // ROB is empty, b) there are no outstanding stores, c) IEW
00929         // stage has received any information regarding stores that
00930         // committed.
00931         // c) is checked by making sure to not consider the ROB empty
00932         // on the same cycle as when stores have been committed.
00933         // @todo: Make this handle multi-cycle communication between
00934         // commit and IEW.
00935         if (checkEmptyROB[tid] && rob->isEmpty(tid) &&
00936             !iewStage->hasStoresToWB(tid) && !committedStores[tid]) {
00937             checkEmptyROB[tid] = false;
00938             toIEW->commitInfo[tid].usedROB = true;
00939             toIEW->commitInfo[tid].emptyROB = true;
00940             toIEW->commitInfo[tid].freeROBEntries = rob->numFreeEntries(tid);
00941             wroteToTimeBuffer = true;
00942         }
00943 
00944     }
00945 }

00947 template <class Impl>
00948 void
00949 DefaultCommit<Impl>::commitInsts()
00950 {
00952     // Handle commit
00953     // Note that commit will be handled prior to putting new
00954     // instructions in the ROB so that the ROB only tries to commit
00955     // instructions it has in this current cycle, and not instructions
00956     // it is writing in during this cycle.  Can't commit and squash
00957     // things at the same time...
00959 
00960     DPRINTF(Commit, "Trying to commit instructions in the ROB.\n");
00961 
00962     unsigned num_committed = 0;
00963 
00964     DynInstPtr head_inst;
00965 
00966     // Commit as many instructions as possible until the commit bandwidth
00967     // limit is reached, or it becomes impossible to commit any more.
00968     while (num_committed < commitWidth) {
00969         // Check for any interrupt that we've already squashed for
00970         // and start processing it.
00971         if (interrupt != NoFault)
00972             handleInterrupt();
00973 
00974         int commit_thread = getCommittingThread();
00975 
00976         if (commit_thread == -1 || !rob->isHeadReady(commit_thread))
00977             break;
00978 
00979         head_inst = rob->readHeadInst(commit_thread);
00980 
00981         ThreadID tid = head_inst->threadNumber;
00982 
00983         assert(tid == commit_thread);
00984 
00985         DPRINTF(Commit, "Trying to commit head instruction, [sn:%i] [tid:%i]\n",
00986                 head_inst->seqNum, tid);
00987 
00988         // If the head instruction is squashed, it is ready to retire
00989         // (be removed from the ROB) at any time.
00990         if (head_inst->isSquashed()) {
00991 
00992             DPRINTF(Commit, "Retiring squashed instruction from "
00993                     "ROB.\n");
00994 
00995             rob->retireHead(commit_thread);
00996 
00997             ++commitSquashedInsts;
00998 
00999             // Record that the number of ROB entries has changed.
01000             changedROBNumEntries[tid] = true;
01001         } else {
01002             pc[tid] = head_inst->pcState();
01003 
01004             // Increment the total number of non-speculative instructions
01005             // executed.
01006             // Hack for now: it really shouldn't happen until after the
01007             // commit is deemed to be successful, but this count is needed
01008             // for syscalls.
01009             thread[tid]->funcExeInst++;
01010 
01011             // Try to commit the head instruction.
01012             bool commit_success = commitHead(head_inst, num_committed);
01013 
01014             if (commit_success) {
01015                 ++num_committed;
01016 
01017                 changedROBNumEntries[tid] = true;
01018 
01019                 // Set the doneSeqNum to the youngest committed instruction.
01020                 toIEW->commitInfo[tid].doneSeqNum = head_inst->seqNum;
01021 
01022                 if (tid == 0) {
01023                     canHandleInterrupts =  (!head_inst->isDelayedCommit()) &&
01024                                            ((THE_ISA != ALPHA_ISA) ||
01025                                              (!(pc[0].instAddr() & 0x3)));
01026                 }
01027 
01028                 // Updates misc. registers.
01029                 head_inst->updateMiscRegs();
01030 
01031                 cpu->traceFunctions(pc[tid].instAddr());
01032 
01033                 TheISA::advancePC(pc[tid], head_inst->staticInst);
01034 
01035                 // Keep track of the last sequence number commited
01036                 lastCommitedSeqNum[tid] = head_inst->seqNum;
01037 
01038                 // If this is an instruction that doesn't play nicely with
01039                 // others squash everything and restart fetch
01040                 if (head_inst->isSquashAfter())
01041                     squashAfter(tid, head_inst);
01042 
01043                 if (drainPending) {
01044                     DPRINTF(Drain, "Draining: %i:%s\n", tid, pc[tid]);
01045                     if (pc[tid].microPC() == 0 && interrupt == NoFault) {
01046                         squashAfter(tid, head_inst);
01047                         cpu->commitDrained(tid);
01048                     }
01049                 }
01050 
01051                 int count = 0;
01052                 Addr oldpc;
01053                 // Debug statement.  Checks to make sure we're not
01054                 // currently updating state while handling PC events.
01055                 assert(!thread[tid]->noSquashFromTC && !thread[tid]->trapPending);
01056                 do {
01057                     oldpc = pc[tid].instAddr();
01058                     cpu->system->pcEventQueue.service(thread[tid]->getTC());
01059                     count++;
01060                 } while (oldpc != pc[tid].instAddr());
01061                 if (count > 1) {
01062                     DPRINTF(Commit,
01063                             "PC skip function event, stopping commit\n");
01064                     break;
01065                 }
01066 
01067                 // Check if an instruction just enabled interrupts and we've
01068                 // previously had an interrupt pending that was not handled
01069                 // because interrupts were subsequently disabled before the
01070                 // pipeline reached a place to handle the interrupt. In that
01071                 // case squash now to make sure the interrupt is handled.
01072                 //
01073                 // If we don't do this, we might end up in a live lock situation
01074                 if (!interrupt  && avoidQuiesceLiveLock &&
01075                    (!head_inst->isMicroop() || head_inst->isLastMicroop()) &&
01076                    cpu->checkInterrupts(cpu->tcBase(0)))
01077                     squashAfter(tid, head_inst);
01078             } else {
01079                 DPRINTF(Commit, "Unable to commit head instruction PC:%s "
01080                         "[tid:%i] [sn:%i].\n",
01081                         head_inst->pcState(), tid ,head_inst->seqNum);
01082                 break;
01083             }
01084         }
01085     }
01086 
01087     DPRINTF(CommitRate, "%i\n", num_committed);
01088     numCommittedDist.sample(num_committed);
01089 
01090     if (num_committed == commitWidth) {
01091         commitEligibleSamples++;
01092     }
01093 }

01095 template <class Impl>
01096 bool
01097 DefaultCommit<Impl>::commitHead(DynInstPtr &head_inst, unsigned inst_num)
01098 {
01099     assert(head_inst);
01100 
01101     ThreadID tid = head_inst->threadNumber;
01102 
01103     // If the instruction is not executed yet, then it will need extra
01104     // handling.  Signal backwards that it should be executed.
01105     if (!head_inst->isExecuted()) {
01106         // Keep this number correct.  We have not yet actually executed
01107         // and committed this instruction.
01108         thread[tid]->funcExeInst--;
01109 
01110         if (head_inst->isNonSpeculative() ||
01111             head_inst->isStoreConditional() ||
01112             head_inst->isMemBarrier() ||
01113             head_inst->isWriteBarrier()) {
01114 
01115             DPRINTF(Commit, "Encountered a barrier or non-speculative "
01116                     "instruction [sn:%lli] at the head of the ROB, PC %s.\n",
01117                     head_inst->seqNum, head_inst->pcState());
01118 
01119             if (inst_num > 0 || iewStage->hasStoresToWB(tid)) {
01120                 DPRINTF(Commit, "Waiting for all stores to writeback.\n");
01121                 return false;
01122             }
01123 
01124             toIEW->commitInfo[tid].nonSpecSeqNum = head_inst->seqNum;
01125 
01126             // Change the instruction so it won't try to commit again until
01127             // it is executed.
01128             head_inst->clearCanCommit();
01129 
01130             ++commitNonSpecStalls;
01131 
01132             return false;
01133         } else if (head_inst->isLoad()) {
01134             if (inst_num > 0 || iewStage->hasStoresToWB(tid)) {
01135                 DPRINTF(Commit, "Waiting for all stores to writeback.\n");
01136                 return false;
01137             }
01138 
01139             assert(head_inst->uncacheable());
01140             DPRINTF(Commit, "[sn:%lli]: Uncached load, PC %s.\n",
01141                     head_inst->seqNum, head_inst->pcState());
01142 
01143             // Send back the non-speculative instruction's sequence
01144             // number.  Tell the lsq to re-execute the load.
01145             toIEW->commitInfo[tid].nonSpecSeqNum = head_inst->seqNum;
01146             toIEW->commitInfo[tid].uncached = true;
01147             toIEW->commitInfo[tid].uncachedLoad = head_inst;
01148 
01149             head_inst->clearCanCommit();
01150 
01151             return false;
01152         } else {
01153             panic("Trying to commit un-executed instruction "
01154                   "of unknown type!\n");
01155         }
01156     }
01157 
01158     if (head_inst->isThreadSync()) {
01159         // Not handled for now.
01160         panic("Thread sync instructions are not handled yet.\n");
01161     }
01162 
01163     // Check if the instruction caused a fault.  If so, trap.
01164     Fault inst_fault = head_inst->getFault();
01165 
01166     // Stores mark themselves as completed.
01167     if (!head_inst->isStore() && inst_fault == NoFault) {
01168         head_inst->setCompleted();
01169     }
01170 
01171     // Use checker prior to updating anything due to traps or PC
01172     // based events.
01173     if (cpu->checker) {
01174         cpu->checker->verify(head_inst);
01175     }
01176 
01177     if (inst_fault != NoFault) {
01178         DPRINTF(Commit, "Inst [sn:%lli] PC %s has a fault\n",
01179                 head_inst->seqNum, head_inst->pcState());
01180 
01181         if (iewStage->hasStoresToWB(tid) || inst_num > 0) {
01182             DPRINTF(Commit, "Stores outstanding, fault must wait.\n");
01183             return false;
01184         }
01185 
01186         head_inst->setCompleted();
01187 
01188         if (cpu->checker) {
01189             // Need to check the instruction before its fault is processed
01190             cpu->checker->verify(head_inst);
01191         }
01192 
01193         assert(!thread[tid]->noSquashFromTC);
01194 
01195         // Mark that we're in state update mode so that the trap's
01196         // execution doesn't generate extra squashes.
01197         thread[tid]->noSquashFromTC = true;
01198 
01199         // Execute the trap.  Although it's slightly unrealistic in
01200         // terms of timing (as it doesn't wait for the full timing of
01201         // the trap event to complete before updating state), it's
01202         // needed to update the state as soon as possible.  This
01203         // prevents external agents from changing any specific state
01204         // that the trap need.
01205         cpu->trap(inst_fault, tid, head_inst->staticInst);
01206 
01207         // Exit state update mode to avoid accidental updating.
01208         thread[tid]->noSquashFromTC = false;
01209 
01210         commitStatus[tid] = TrapPending;
01211 
01212         DPRINTF(Commit, "Committing instruction with fault [sn:%lli]\n",
01213             head_inst->seqNum);
01214         if (head_inst->traceData) {
01215             if (DTRACE(ExecFaulting)) {
01216                 head_inst->traceData->setFetchSeq(head_inst->seqNum);
01217                 head_inst->traceData->setCPSeq(thread[tid]->numOp);
01218                 head_inst->traceData->dump();
01219             }
01220             delete head_inst->traceData;
01221             head_inst->traceData = NULL;
01222         }
01223 
01224         // Generate trap squash event.
01225         generateTrapEvent(tid);
01226         return false;
01227     }
01228 
01229     updateComInstStats(head_inst);
01230 
01231     if (FullSystem) {
01232         if (thread[tid]->profile) {
01233             thread[tid]->profilePC = head_inst->instAddr();
01234             ProfileNode *node = thread[tid]->profile->consume(
01235                     thread[tid]->getTC(), head_inst->staticInst);
01236 
01237             if (node)
01238                 thread[tid]->profileNode = node;
01239         }
01240         if (CPA::available()) {
01241             if (head_inst->isControl()) {
01242                 ThreadContext *tc = thread[tid]->getTC();
01243                 CPA::cpa()->swAutoBegin(tc, head_inst->nextInstAddr());
01244             }
01245         }
01246     }
01247     DPRINTF(Commit, "Committing instruction with [sn:%lli] PC %s\n",
01248             head_inst->seqNum, head_inst->pcState());
01249     if (head_inst->traceData) {
01250         head_inst->traceData->setFetchSeq(head_inst->seqNum);
01251         head_inst->traceData->setCPSeq(thread[tid]->numOp);
01252         head_inst->traceData->dump();
01253         delete head_inst->traceData;
01254         head_inst->traceData = NULL;
01255     }
01256     if (head_inst->isReturn()) {
01257         DPRINTF(Commit,"Return Instruction Committed [sn:%lli] PC %s \n",
01258                         head_inst->seqNum, head_inst->pcState());
01259     }
01260 
01261     // Update the commit rename map
01262     for (int i = 0; i < head_inst->numDestRegs(); i++) {
01263         renameMap[tid]->setEntry(head_inst->flattenedDestRegIdx(i),
01264                                  head_inst->renamedDestRegIdx(i));
01265     }
01266 
01267     // Finally clear the head ROB entry.
01268     rob->retireHead(tid);
01269 
01270 #if TRACING_ON
01271     if (DTRACE(O3PipeView)) {
01272         head_inst->commitTick = curTick() - head_inst->fetchTick;
01273     }
01274 #endif
01275 
01276     // If this was a store, record it for this cycle.
01277     if (head_inst->isStore())
01278         committedStores[tid] = true;
01279 
01280     // Return true to indicate that we have committed an instruction.
01281     return true;
01282 }

01284 template <class Impl>
01285 void
01286 DefaultCommit<Impl>::getInsts()
01287 {
01288     DPRINTF(Commit, "Getting instructions from Rename stage.\n");
01289 
01290     // Read any renamed instructions and place them into the ROB.
01291     int insts_to_process = std::min((int)renameWidth, fromRename->size);
01292 
01293     for (int inst_num = 0; inst_num < insts_to_process; ++inst_num) {
01294         DynInstPtr inst;
01295

01297         ThreadID tid = inst->threadNumber;
01298 
01299         if (!inst->isSquashed() &&
01300             commitStatus[tid] != ROBSquashing &&
01301             commitStatus[tid] != TrapPending) {
01302             changedROBNumEntries[tid] = true;
01303

01305                     inst->pcState(), inst->seqNum, tid);
01306 
01307             rob->insertInst(inst);
01308 
01309             assert(rob->getThreadEntries(tid) <= rob->getMaxEntries(tid));
01310 
01311             youngestSeqNum[tid] = inst->seqNum;
01312         } else {
01313             DPRINTF(Commit, "Instruction PC %s [sn:%i] [tid:%i] was "
01314                     "squashed, skipping.\n",
01315                     inst->pcState(), inst->seqNum, tid);
01316         }
01317     }

01343 template <class Impl>
01344 void
01345 DefaultCommit<Impl>::markCompletedInsts()
