00138     if (cr.command.start_dma()) {
00139         assert(!busy);
00140         cr.status.dma_transfer_status(0);
00141         nextState = DescriptorFetch;

00153         if (busy)

00364 void
00365 CopyEngine::CopyEngineChannel::channelWrite(Packet *pkt, Addr daddr, int size)
00366 {
00367     switch (daddr) {
00368       case CHAN_CONTROL:
00369         assert(size == sizeof(uint16_t));
00370         int old_int_disable;
00371         old_int_disable = cr.ctrl.interrupt_disable();
00372         cr.ctrl(pkt->get<uint16_t>());
00373         if (cr.ctrl.interrupt_disable())
00374             cr.ctrl.interrupt_disable(0);
00375         else
00376             cr.ctrl.interrupt_disable(old_int_disable);
00377         break;
00378       case CHAN_STATUS:
00379         assert(size == sizeof(uint64_t));
00380         DPRINTF(DMACopyEngine, "Warning, ignorning write to register %x\n",
00381                     daddr);
00382         break;
00383       case CHAN_CHAINADDR:
00384         assert(size == sizeof(uint64_t) || size == sizeof(uint32_t));
00385         if (size == sizeof(uint64_t))
00386             cr.descChainAddr = pkt->get<uint64_t>();
00387         else
00388             cr.descChainAddr =  (uint64_t)pkt->get<uint32_t>() |
00389                 (cr.descChainAddr & ~mask(32));
00390         DPRINTF(DMACopyEngine, "Chain Address %x\n", cr.descChainAddr);
00391         break;
00392       case CHAN_CHAINADDR_HIGH:
00393         assert(size == sizeof(uint32_t));
00394         cr.descChainAddr =  ((uint64_t)pkt->get<uint32_t>() <<32) |
00395             (cr.descChainAddr & mask(32));
00396         DPRINTF(DMACopyEngine, "Chain Address %x\n", cr.descChainAddr);
00397         break;
00398       case CHAN_COMMAND:
00399         assert(size == sizeof(uint8_t));
00400         cr.command(pkt->get<uint8_t>());
00401         recvCommand();
00402         break;
00403       case CHAN_CMPLNADDR:
00404         assert(size == sizeof(uint64_t) || size == sizeof(uint32_t));
00405         if (size == sizeof(uint64_t))
00406             cr.completionAddr = pkt->get<uint64_t>();
00407         else
00408             cr.completionAddr =  pkt->get<uint32_t>() |
00409                 (cr.completionAddr & ~mask(32));
00410         break;
00411       case CHAN_CMPLNADDR_HIGH:
00412         assert(size == sizeof(uint32_t));
00413         cr.completionAddr =  ((uint64_t)pkt->get<uint32_t>() <<32) |
00414             (cr.completionAddr & mask(32));
00415         break;
00416       case CHAN_ERROR:
00417         assert(size == sizeof(uint32_t));
00418         cr.error(~pkt->get<uint32_t>() & cr.error());
00419         break;
00420       default:
00421         panic("Read request to unknown channel register number: (%d)%#x\n",
00422                 channelId, daddr);
00423     }
