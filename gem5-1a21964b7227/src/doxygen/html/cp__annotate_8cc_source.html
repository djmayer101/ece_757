00348 void
00349 CPA::swBegin(System *sys, int cpuid, std::string st, uint64_t frame, bool expl,
00350         int flags)
00351 {
00352     int x = 0;
00353     int len;
00354     while (ignoreSymbols[x].len)
00355     {
00356         len = ignoreSymbols[x].len;
00357         if (!st.compare(0,len, ignoreSymbols[x].symbol, len))
00358             return;
00359         x++;
00360     }
00361 
00362     int sysi = getSys(sys);
00363     StackId sid = StackId(sysi, frame);
00364     // if expl is true suspend symbol table based states
00365     if (!smStack[sid].size())
00366         return;
00367     if (!expl && swExpl[sid])
00368         return;
00369     if (expl)
00370         swExpl[sid] = true;
00371     DPRINTFS(AnnotateVerbose, sys, "SwBegin: %s sysi: %d\n", st, sysi);
00372     int smi = smStack[sid].back();
00373     int sti = getSt(smMap[smi-1].second.first, st);
00374     if (lastState[smi] != sti) {
00375         lastState[smi] = sti;
00376         add(OP_BEGIN, flags, cpuid, smi, sti);
00377     }

00614     DPRINTF(AnnotateQ, "swSq: %s [%#x] cur size: %d bytes: %d, new size: %d\n",
00615              q, id, qSize[qi-1], qBytes[qi-1], size);
00616 
00617     if (FL_RESET & flags) {
00618         DPRINTF(AnnotateQ, "Resetting Queue %s\n", q);
00619         add(OP_SIZE_QUEUE, FL_NONE, tc->contextId(), smi, qi, 0);
00620         qData[qi-1].clear();
00621         qSize[qi-1] = 0;
00622         qBytes[qi-1] = 0;
00623     }
00624 
00625     if (qBytes[qi-1] < size)
00626         doQ(sys, FL_NONE, tc->contextId(), smi, q, qi, size - qBytes[qi-1]);
00627     else if (qBytes[qi-1] > size) {
00628         DPRINTF(AnnotateQ, "removing for resize of queue %s\n", q);
00629         add(OP_SIZE_QUEUE, FL_NONE, tc->contextId(), smi, qi, size);
00630         if (size <= 0) {
00631             qData[qi-1].clear();
00632             qSize[qi-1] = 0;
00633             qBytes[qi-1] = 0;
00634             return;
00635         }
00636         int need = qBytes[qi-1] - size;
00637         qBytes[qi-1] = size;
00638         while (need > 0) {
00639             int32_t tail_bytes = qData[qi-1].back()->data;
00640             if (qSize[qi-1] <= 0 || qBytes[qi-1] < 0) {
00641                 dump(true);
00642                 dumpKey();
00643                 fatal("Queue %s had inconsistancy when doing size queue!\n", q);
00644             }
00645             if (tail_bytes > need) {
00646                 qData[qi-1].back()->data -= need;
00647                 need = 0;
00648             } else if (tail_bytes == need) {
00649                 qData[qi-1].pop_back();
00650                 qSize[qi-1]--;
00651                 need = 0;
00652             } else {
00653                 qData[qi-1].pop_back();
00654                 qSize[qi-1]--;
00655                 need -= tail_bytes;
00656             }
00657         }
00658     }
00659 }

00989 void
00990 CPA::doDq(System *sys, int flags, int cpuid, int sm,
00991                string q, int qi, int count)
00992 {
00993 
00994     StringWrap name(sys->name());
00995     if (count == -1) {
00996         add(OP_DEQUEUE, flags, cpuid, sm, qi, count);
00997         qData[qi-1].clear();
00998         qSize[qi-1] = 0;
00999         qBytes[qi-1] = 0;
01000         DPRINTF(AnnotateQ, "Dequeing all data in queue %s size now %d/%d\n",
01001                 q, qSize[qi-1], qBytes[qi-1]);
01002         return;
01003     }
01004 
01005     assert(count > 0);
01006     if (qSize[qi-1] <= 0 || qBytes[qi-1] <= 0 || !qData[qi-1].size()) {
01007         dump(true);
01008         dumpKey();
01009         fatal("Queue %s dequing with no data available in queue!\n",
01010                 q);
01011     }
01012     assert(qSize[qi-1] >= 0);
01013     assert(qBytes[qi-1] >= 0);
01014     assert(qData[qi-1].size());
01015 
01016     int32_t need = count;
01017     qBytes[qi-1] -= count;
01018     if (qBytes[qi-1] < 0) {
01019         dump(true);
01020         dumpKey();
01021         fatal("Queue %s dequing with no bytes available in queue!\n",
01022                 q);
01023     }
01024 
01025     while (need > 0) {
01026         int32_t head_bytes = qData[qi-1].front()->data;
01027         if (qSize[qi-1] <= 0 || qBytes[qi-1] < 0) {
01028             dump(true);
01029             dumpKey();
01030             fatal("Queue %s dequing with nothing in queue!\n",
01031                 q);
01032         }
01033 
01034         if (head_bytes > need) {
01035             qData[qi-1].front()->data -= need;
01036             need = 0;
01037         } else if (head_bytes == need) {
01038             qData[qi-1].pop_front();
01039             qSize[qi-1]--;
01040             need = 0;
01041         } else {
01042             qData[qi-1].pop_front();
01043             qSize[qi-1]--;
01044             need -= head_bytes;
01045         }
01046     }
01047 
01048     add(OP_DEQUEUE, flags, cpuid, sm, qi, count);
01049     DPRINTF(AnnotateQ, "Dequeing in queue %s size now %d/%d\n",
01050             q, qSize[qi-1], qBytes[qi-1]);
01051 }
