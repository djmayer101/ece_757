00080     CachePage *
00081     getPage(Addr addr)
00082     {
00083         Addr page_addr = addr & ~(TheISA::PageBytes - 1);
00084 
00085         // Check against recent lookups.
00086         if (recent[0] != pageMap.end()) {
00087             if (recent[0]->first == page_addr)
00088                 return recent[0]->second;
00089             if (recent[1] != pageMap.end() &&
00090                     recent[1]->first == page_addr) {
00091                 update(recent[1]);
00092                 // recent[1] has just become recent[0].
00093                 return recent[0]->second;
00094             }
00095         }
00096 
00097         // Actually look in the has_map.
00098         PageIt it = pageMap.find(page_addr);
00099         if (it != pageMap.end()) {
00100             update(it);
00101             return it->second;
00102         }
00103 
00104         // Didn't find an existing page, so add a new one.
00105         CachePage *newPage = new CachePage;
00106         page_addr = page_addr & ~(TheISA::PageBytes - 1);
00107         typename PageMap::value_type to_insert(page_addr, newPage);
00108         update(pageMap.insert(to_insert).first);
00109         return newPage;
00110     }
