00070     uint64_t
00071     stringToRegister(const char *str)
00072     {
00073         uint64_t reg = 0;
00074         for (int pos = 3; pos >=0; pos--) {
00075             reg <<= 8;
00076             reg |= str[pos];
00077         }
00078         return reg;

00082     doCpuid(ThreadContext * tc, uint32_t function,
00083             uint32_t index, CpuidResult &result)
00084     {
00085         uint16_t family = bits(function, 31, 16);
00086         uint16_t funcNum = bits(function, 15, 0);
00087         if (family == 0x8000) {
00088             // The extended functions
00089             switch (funcNum) {
00090               case VendorAndLargestExtFunc:
00091                 assert(vendorStringSize >= 12);
00092                 result = CpuidResult(
00093                         0x80000000 + NumExtendedCpuidFuncs - 1,
00094                         stringToRegister(vendorString),
00095                         stringToRegister(vendorString + 4),
00096                         stringToRegister(vendorString + 8));
00097                 break;
00098               case FamilyModelSteppingBrandFeatures:
00099                 result = CpuidResult(0x00020f51, 0x00000405,
00100                                      0xe3d3fbff, 0x00000001);
00101                 break;
00102               case NameString1:
00103               case NameString2:
00104               case NameString3:
00105                 {
00106                     // Zero fill anything beyond the end of the string. This
00107                     // should go away once the string is a vetted parameter.
00108                     char cleanName[nameStringSize];
00109                     memset(cleanName, '\0', nameStringSize);
00110                     strncpy(cleanName, nameString, nameStringSize);
00111 
00112                     int offset = (funcNum - NameString1) * 16;
00113                     assert(nameStringSize >= offset + 16);
00114                     result = CpuidResult(
00115                             stringToRegister(cleanName + offset + 0),
00116                             stringToRegister(cleanName + offset + 4),
00117                             stringToRegister(cleanName + offset + 12),
00118                             stringToRegister(cleanName + offset + 8));
00119                 }
00120                 break;
00121               case L1CacheAndTLB:
00122                 result = CpuidResult(0xff08ff08, 0xff20ff20,
00123                                      0x40020140, 0x40020140);
00124                 break;
00125               case L2L3CacheAndL2TLB:
00126                 result = CpuidResult(0x00000000, 0x42004200,
00127                                      0x00000000, 0x04008140);
00128                 break;
00129               case APMInfo:
00130                 result = CpuidResult(0x80000018, 0x68747541,
00131                                      0x69746e65, 0x444d4163);
00132                 break;
00133 /*            case LongModeAddressSize:
00134               case SVMInfo:
00135               case TLB1GBPageInfo:
00136               case PerformanceInfo:*/
00137               default:
00138                 warn("x86 cpuid: unimplemented function %u", funcNum);
00139                 return false;
00140             }
00141         } else if(family == 0x0000) {
00142             // The standard functions
00143             switch (funcNum) {
00144               case VendorAndLargestStdFunc:
00145                 assert(vendorStringSize >= 12);
00146                 result = CpuidResult(
00147                         NumStandardCpuidFuncs - 1,
00148                         stringToRegister(vendorString),
00149                         stringToRegister(vendorString + 4),
00150                         stringToRegister(vendorString + 8));
00151                 break;
00152               case FamilyModelStepping:
00153                 result = CpuidResult(0x00020f51, 0x00000805,
00154                                      0xe7dbfbff, 0x00000001);
00155                 break;
00156               default:
00157                 warn("x86 cpuid: unimplemented function %u", funcNum);
00158                 return false;
00159             }
00160         } else {
00161             warn("x86 cpuid: unknown family %#x", family);
00162             return false;
