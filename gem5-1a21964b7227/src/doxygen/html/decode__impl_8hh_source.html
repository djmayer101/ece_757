00352     decodeStatus[tid] = Squashing;

00498 template <class Impl>
00499 bool
00500 DefaultDecode<Impl>::checkSignalsAndUpdate(ThreadID tid)
00501 {
00502     // Check if there's a squash signal, squash if there is.
00503     // Check stall signals, block if necessary.
00504     // If status was blocked
00505     //     Check if stall conditions have passed
00506     //         if so then go to unblocking
00507     // If status was Squashing
00508     //     check if squashing is not high.  Switch to running this cycle.
00509 
00510     // Update the per thread stall statuses.
00511     readStallSignals(tid);
00512 
00513     // Check squash signals from commit.
00514     if (fromCommit->commitInfo[tid].squash) {
00515 
00516         DPRINTF(Decode, "[tid:%u]: Squashing instructions due to squash "
00517                 "from commit.\n", tid);
00518 
00519         squash(tid);
00520 
00521         return true;
00522     }
00523 
00524     // Check ROB squash signals from commit.
00525     if (fromCommit->commitInfo[tid].robSquashing) {
00526         DPRINTF(Decode, "[tid:%u]: ROB is still squashing.\n", tid);
00527 
00528         // Continue to squash.
00529         decodeStatus[tid] = Squashing;
00530 
00531         return true;
00532     }
00533 
00534     if (checkStall(tid)) {
00535         return block(tid);
00536     }
00537 
00538     if (decodeStatus[tid] == Blocked) {
00539         DPRINTF(Decode, "[tid:%u]: Done blocking, switching to unblocking.\n",
00540                 tid);
00541 
00542         decodeStatus[tid] = Unblocking;
00543 
00544         unblock(tid);
00545 
00546         return true;
00547     }
00548 
00549     if (decodeStatus[tid] == Squashing) {
00550         // Switch status to running if decode isn't being told to block or
00551         // squash this cycle.
00552         DPRINTF(Decode, "[tid:%u]: Done squashing, switching to running.\n",
00553                 tid);
00554 
00555         decodeStatus[tid] = Running;
00556 
00557         return false;
00558     }
00559 
00560     // If we've reached this point, we have not gotten any signals that
00561     // cause decode to change its status.  Decode remains the same as before.
00562     return false;
00563 }

00601 template<class Impl>
00602 void
00603 DefaultDecode<Impl>::decode(bool &status_change, ThreadID tid)
00604 {
00605     // If status is Running or idle,
00606     //     call decodeInsts()
00607     // If status is Unblocking,
00608     //     buffer any instructions coming from fetch
00609     //     continue trying to empty skid buffer
00610     //     check if stall conditions have passed
00611 
00612     if (decodeStatus[tid] == Blocked) {
00613         ++decodeBlockedCycles;
00614     } else if (decodeStatus[tid] == Squashing) {
00615         ++decodeSquashCycles;
00616     }
00617 
00618     // Decode should try to decode as many instructions as its bandwidth
00619     // will allow, as long as it is not currently blocked.
00620     if (decodeStatus[tid] == Running ||
00621         decodeStatus[tid] == Idle) {
00622         DPRINTF(Decode, "[tid:%u]: Not blocked, so attempting to run "
00623                 "stage.\n",tid);
00624 
00625         decodeInsts(tid);
00626     } else if (decodeStatus[tid] == Unblocking) {
00627         // Make sure that the skid buffer has something in it if the
00628         // status is unblocking.
00629         assert(!skidsEmpty());
00630 
00631         // If the status was unblocking, then instructions from the skid
00632         // buffer were used.  Remove those instructions and handle
00633         // the rest of unblocking.
00634         decodeInsts(tid);
00635 
00636         if (fetchInstsValid()) {
00637             // Add the current inputs to the skid buffer so they can be
00638             // reprocessed when this stage unblocks.
00639             skidInsert(tid);
00640         }
00641 
00642         status_change = unblock(tid) || status_change;
00643     }
00644 }

00697         }

00699         // Also check if instructions have no source registers.  Mark
00700         // them as ready to issue at any time.  Not sure if this check
00701         // should exist here or at a later stage; however it doesn't matter
00702         // too much for function correctness.
00703         if (inst->numSrcRegs() == 0) {
00704             inst->setCanIssue();
00705         }
00706 
00707         // This current instruction is valid, so add it into the decode
00708         // queue.  The next instruction may not be valid, so check to
00709         // see if branches were predicted correctly.
00710         toRename->insts[toRenameIndex] = inst;
00711 
00712         ++(toRename->size);
00713         ++toRenameIndex;
00714         ++decodeDecodedInsts;
00715         --insts_available;
00716 
00717 #if TRACING_ON
00718         if (DTRACE(O3PipeView)) {
00719             inst->decodeTick = curTick() - inst->fetchTick;
00720         }
00721 #endif
00722 
00723         // Ensure that if it was predicted as a branch, it really is a
00724         // branch.
00725         if (inst->readPredTaken() && !inst->isControl()) {
00726             panic("Instruction predicted as a branch!");
00727 
00728             ++decodeControlMispred;
00729 
00730             // Might want to set some sort of boolean and just do
00731             // a check at the end
00732             squash(inst, inst->threadNumber);
00733 
00734             break;
00735         }
00736 
00737         // Go ahead and compute any PC-relative branches.
00738         if (inst->isDirectCtrl() && inst->isUncondCtrl()) {
00739             ++decodeBranchResolved;
00740 
00741             if (!(inst->branchTarget() == inst->readPredTarg())) {
00742                 ++decodeBranchMispred;
00743 
00744                 // Might want to set some sort of boolean and just do
00745                 // a check at the end
00746                 squash(inst, inst->threadNumber);
00747                 TheISA::PCState target = inst->branchTarget();
00748 
00749                 DPRINTF(Decode, "[sn:%i]: Updating predictions: PredPC: %s\n",
00750                         inst->seqNum, target);
00751                 //The micro pc after an instruction level branch should be 0
00752                 inst->setPredTarg(target);
00753                 break;
00754             }
00755         }
00756     }
00757 
00758     // If we didn't process all instructions, then we will need to block
00759     // and put all those instructions into the skid buffer.
00760     if (!insts_to_decode.empty()) {
00761         block(tid);
00762     }
00763 
00764     // Record that decode has written to the time buffer for activity
00765     // tracking.
00766     if (toRenameIndex) {
00767         wroteToTimeBuffer = true;
00768     }
