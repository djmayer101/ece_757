00052 void
00053 DecodeUnit::execute(int slot_num)
00054 {
00055     ResourceRequest* decode_req = reqs[slot_num];
00056     DynInstPtr inst = reqs[slot_num]->inst;
00057 
00058     switch (decode_req->cmd)
00059     {
00060       case DecodeInst:
00061         {
00062 
00063             if (inst->fault != NoFault) {
00064                 inst->setBackSked(cpu->faultSked);
00065                 DPRINTF(InOrderDecode,"[tid:%i]: Fault found for instruction [sn:%i]\n",
00066                         inst->readTid(), inst->seqNum);
00067             } else {
00068                 assert(!inst->staticInst->isMacroop());
00069                 inst->setBackSked(cpu->createBackEndSked(inst));
00070                 DPRINTF(InOrderDecode,"Decoded instruction [sn:%i]: %s : 0x%x\n",
00071                         inst->seqNum, inst->instName(),
00072                         inst->staticInst->machInst);
00073             }
00074 
00075             if (inst->backSked != NULL) {
00076                 DPRINTF(InOrderDecode,
00077                     "[tid:%i]: Back End Schedule created for %s  [sn:%i].\n",
00078                         inst->readTid(), inst->instName(), inst->seqNum);
00079                 decode_req->done();
00080             } else {
00081                 DPRINTF(Resource,
00082                     "[tid:%i] Static Inst not available to decode.\n",
00083                         inst->readTid());
00084                 DPRINTF(Resource,
00085                     "Unable to create schedule for instruction [sn:%i] \n",
00086                     inst->seqNum);
00087                 DPRINTF(InOrderStall, "STALL: \n");
00088                 decode_req->done(false);
00089             }
00090         }
00091         break;
00092 
00093       default:
00094         fatal("Unrecognized command to %s", resName);
00095     }
00096 }
