00058 void
00059 DmaPort::handleResp(PacketPtr pkt, Tick delay)
00060 {
00061     // should always see a response with a sender state
00062     assert(pkt->isResponse());
00063 
00064     // get the DMA sender state
00065     DmaReqState *state = dynamic_cast<DmaReqState*>(pkt->senderState);
00066     assert(state);
00067 
00068     DPRINTF(DMA, "Received response %s for addr: %#x size: %d nb: %d,"  \
00069             " tot: %d sched %d\n",
00070             pkt->cmdString(), pkt->getAddr(), pkt->req->getSize(),
00071             state->numBytes, state->totBytes,
00072             state->completionEvent ?
00073             state->completionEvent->scheduled() : 0);
00074 
00075     assert(pendingCount != 0);
00076     pendingCount--;
00077 
00078     // update the number of bytes received based on the request rather
00079     // than the packet as the latter could be rounded up to line sizes
00080     state->numBytes += pkt->req->getSize();
00081     assert(state->totBytes >= state->numBytes);
00082 
00083     // if we have reached the total number of bytes for this DMA
00084     // request, then signal the completion and delete the sate
00085     if (state->totBytes == state->numBytes) {
00086         if (state->completionEvent) {
00087             delay += state->delay;
00088             device->schedule(state->completionEvent, curTick() + delay);
00089         }
00090         delete state;
00091     }
00092 
00093     // delete the request that we created and also the packet
00094     delete pkt->req;
00095     delete pkt;
00096 
00097     // we might be drained at this point, if so signal the drain event
00098     if (pendingCount == 0 && drainManager) {
00099         drainManager->signalDrainDone();
00100         drainManager = NULL;
00101     }
00102 }

00155 void
00156 DmaPort::dmaAction(Packet::Command cmd, Addr addr, int size, Event *event,
00157                    uint8_t *data, Tick delay, Request::Flags flag)
00158 {
00159     // one DMA request sender state for every action, that is then
00160     // split into many requests and packets based on the block size,
00161     // i.e. cache line size
00162     DmaReqState *reqState = new DmaReqState(event, size, delay);
00163 
00164     DPRINTF(DMA, "Starting DMA for addr: %#x size: %d sched: %d\n", addr, size,
00165             event ? event->scheduled() : -1);
00166     for (ChunkGenerator gen(addr, size, peerBlockSize());
00167          !gen.done(); gen.next()) {
00168         Request *req = new Request(gen.addr(), gen.size(), flag, masterId);
00169         PacketPtr pkt = new Packet(req, cmd);
00170 
00171         // Increment the data pointer on a write
00172         if (data)
00173             pkt->dataStatic(data + gen.complete());
00174 
00175         pkt->senderState = reqState;
00176 
00177         DPRINTF(DMA, "--Queuing DMA for addr: %#x size: %d\n", gen.addr(),
00178                 gen.size());
00179         queueDma(pkt);
00180     }
00181 
00182     // in zero time also initiate the sending of the packets we have
00183     // just created, for atomic this involves actually completing all
00184     // the requests
00185     sendDma();
00186 }

00198 void
00199 DmaPort::trySendTimingReq()
00200 {
00201     // send the first packet on the transmit list and schedule the
00202     // following send if it is successful
00203     PacketPtr pkt = transmitList.front();
00204 
00205     DPRINTF(DMA, "Trying to send %s addr %#x\n", pkt->cmdString(),
00206             pkt->getAddr());
00207 
00208     inRetry = !sendTimingReq(pkt);
00209     if (!inRetry) {
00210         transmitList.pop_front();
00211         DPRINTF(DMA, "-- Done\n");
00212         // if there is more to do, then do so
00213         if (!transmitList.empty())
00214             // this should ultimately wait for as many cycles as the
00215             // device needs to send the packet, but currently the port
00216             // does not have any known width so simply wait a single
00217             // cycle
00218             device->schedule(sendEvent, device->clockEdge(Cycles(1)));
00219     } else {
00220         DPRINTF(DMA, "-- Failed, waiting for retry\n");
00221     }
00222 
00223     DPRINTF(DMA, "TransmitList: %d, inRetry: %d\n",
00224             transmitList.size(), inRetry);
00225 }

00227 void
00228 DmaPort::sendDma()
00229 {
00230     // some kind of selcetion between access methods
00231     // more work is going to have to be done to make
00232     // switching actually work
00233     assert(transmitList.size());
00234 
00235     if (sys->isTimingMode()) {
00236         // if we are either waiting for a retry or are still waiting
00237         // after sending the last packet, then do not proceed
00238         if (inRetry || sendEvent.scheduled()) {
00239             DPRINTF(DMA, "Can't send immediately, waiting to send\n");
00240             return;
00241         }
00242 
00243         trySendTimingReq();
00244     } else if (sys->isAtomicMode()) {
00245         // send everything there is to send in zero time
00246         while (!transmitList.empty()) {
00247             PacketPtr pkt = transmitList.front();
00248             transmitList.pop_front();
00249 
00250             DPRINTF(DMA, "Sending  DMA for addr: %#x size: %d\n",
00251                     pkt->req->getPaddr(), pkt->req->getSize());
00252             Tick lat = sendAtomic(pkt);
00253 
00254             handleResp(pkt, lat);
00255         }
00256     } else
00257         panic("Unknown memory mode.");
00258 }
