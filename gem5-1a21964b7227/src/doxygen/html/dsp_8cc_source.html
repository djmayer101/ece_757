00113 MipsISA::signExtend(uint64_t value, int32_t fmt)
00114 {
00115     int32_t signpos = SIMD_NBITS[fmt];
00116     uint64_t sign = uint64_t(1) << (signpos - 1);
00117     uint64_t ones = ~(0ULL);
00118 
00119     if (value & sign)
00120         value |= (ones << signpos); // extend with ones
00121     else
00122         value &= (ones >> (64 - signpos)); // extend with zeros
00123 
00124     return value;
00125 }
