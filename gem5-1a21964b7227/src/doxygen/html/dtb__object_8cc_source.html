00091 bool
00092 DtbObject::addBootCmdLine(const char* _args, size_t len)
00093 {
00094     const char* root_path = "/";
00095     const char* node_name = "chosen";
00096     const char* full_path_node_name = "/chosen";
00097     const char* property_name = "bootargs";
00098 
00099     // Make a new buffer that has extra space to add nodes/properties
00100     int newLen = 2*this->len;
00101     uint8_t* fdt_buf_w_space = new uint8_t[newLen];
00102     // Copy and unpack flattened device tree into new buffer
00103     int ret = fdt_open_into((void*)fileData, (void*)fdt_buf_w_space, (newLen));
00104     if (ret < 0) {
00105         warn("Error resizing buffer of flattened device tree, "
00106              "errno: %d\n", ret);
00107         delete [] fdt_buf_w_space;
00108         return false;
00109     }
00110 
00111     // First try finding the /chosen node in the dtb
00112     int offset = fdt_path_offset((void*)fdt_buf_w_space, full_path_node_name);
00113     if (offset < 0) {
00114         // try adding the node by walking dtb tree to proper insertion point
00115         offset = fdt_path_offset((void*)fdt_buf_w_space, root_path);
00116         offset = fdt_add_subnode((void*)fdt_buf_w_space, offset, node_name);
00117         offset = fdt_path_offset((void*)fdt_buf_w_space, full_path_node_name);
00118         if (offset < 0) {
00119             warn("Error finding or adding \"chosen\" subnode to flattened "
00120                  "device tree, errno: %d\n", offset);
00121             delete [] fdt_buf_w_space;
00122             return false;
00123         }
00124     }
00125 
00126     // Set the bootargs property in the /chosen node
00127     ret = fdt_setprop((void*)fdt_buf_w_space, offset, property_name,
00128                       (const void*)_args, len+1);
00129     if (ret < 0) {
00130         warn("Error setting \"bootargs\" property to flattened device tree, "
00131              "errno: %d\n", ret);
00132         delete [] fdt_buf_w_space;
00133         return false;
00134     }
00135 
00136     // Repack the dtb for kernel use
00137     ret = fdt_pack((void*)fdt_buf_w_space);
00138     if (ret < 0) {
00139         warn("Error re-packing flattened device tree structure, "
00140              "errno: %d\n", ret);
00141         delete [] fdt_buf_w_space;
00142         return false;
00143     }
00144 
00145     text.size = newLen;
00146     text.fileImage = fdt_buf_w_space;
00147 
00148     // clean up old buffer and set to new fdt blob
00149     munmap(fileData, this->len);
00150     fileData = fdt_buf_w_space;
00151     fileDataMmapped = false;
00152     this->len = newLen;
00153 
00154     return true;
00155 }
