00047 void EmulEnv::doModRM(const ExtMachInst & machInst)
00048 {
00049     assert(machInst.modRM.mod != 3);
00050     //Use the SIB byte for addressing if the modrm byte calls for it.
00051     if (machInst.modRM.rm == 4 && machInst.addrSize != 2) {
00052         scale = 1 << machInst.sib.scale;
00053         index = machInst.sib.index | (machInst.rex.x << 3);
00054         base = machInst.sib.base | (machInst.rex.b << 3);
00055         //In this special case, we don't use a base. The displacement also
00056         //changes, but that's managed by the decoder.
00057         if (machInst.sib.base == INTREG_RBP && machInst.modRM.mod == 0)
00058             base = NUM_INTREGS;
00059         //In -this- special case, we don't use an index.
00060         if (index == INTREG_RSP)
00061             index = NUM_INTREGS;
00062     } else {
00063         if (machInst.addrSize == 2) {
00064             unsigned rm = machInst.modRM.rm;
00065             if (rm <= 3) {
00066                 scale = 1;
00067                 if (rm < 2) {
00068                     base = INTREG_RBX;
00069                 } else {
00070                     base = INTREG_RBP;
00071                 }
00072                 index = (rm % 2) ? INTREG_RDI : INTREG_RSI;
00073             } else {
00074                 scale = 0;
00075                 switch (rm) {
00076                   case 4:
00077                     base = INTREG_RSI;
00078                     break;
00079                   case 5:
00080                     base = INTREG_RDI;
00081                     break;
00082                   case 6:
00083                     base = INTREG_RBP;
00084                     break;
00085                   case 7:
00086                     base = INTREG_RBX;
00087                     break;
00088                 }
00089             }
00090         } else {
00091             scale = 0;
00092             base = machInst.modRM.rm | (machInst.rex.b << 3);
00093             if (machInst.modRM.mod == 0 && machInst.modRM.rm == 5) {
00094                 //Since we need to use a different encoding of this
00095                 //instruction anyway, just ignore the base in those cases
00096                 base = NUM_INTREGS;
00097             }
00098         }
00099     }
00100     //Figure out what segment to use. This won't be entirely accurate since
00101     //the presence of a displacement is supposed to make the instruction
00102     //default to the data segment.
00103     if ((base != INTREG_RBP && base != INTREG_RSP) || machInst.dispSize) {
00104         seg = SEGMENT_REG_DS;
00105         //Handle any segment override that might have been in the instruction
00106         int segFromInst = machInst.legacy.seg;
00107         if (segFromInst)
00108             seg = (SegmentRegIndex)(segFromInst - 1);
00109     } else {
00110         seg = SEGMENT_REG_SS;
00111     }
00112 }
