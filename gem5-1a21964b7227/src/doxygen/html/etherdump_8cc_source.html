00075 void
00076 EtherDump::init()
00077 {
00078     struct pcap_file_header hdr;
00079     hdr.magic = TCPDUMP_MAGIC;
00080     hdr.version_major = PCAP_VERSION_MAJOR;
00081     hdr.version_minor = PCAP_VERSION_MINOR;
00082 
00083     hdr.thiszone = 0;

00085     hdr.sigfigs = 0;
00086     hdr.linktype = DLT_EN10MB;

00088     stream->write(reinterpret_cast<char *>(&hdr), sizeof(hdr));

00093 void
00094 EtherDump::dumpPacket(EthPacketPtr &packet)
00095 {
00096     pcap_pkthdr pkthdr;
00097     pkthdr.seconds = curTick() / SimClock::Int::s;
00098     pkthdr.microseconds = (curTick() / SimClock::Int::us) % ULL(1000000);
00099     pkthdr.caplen = std::min(packet->length, maxlen);
00100     pkthdr.len = packet->length;
00101     stream->write(reinterpret_cast<char *>(&pkthdr), sizeof(pkthdr));
00102     stream->write(reinterpret_cast<char *>(packet->data), pkthdr.caplen);
00103     stream->flush();
00104 }
