00170 bool
00171 EtherLink::Link::transmit(EthPacketPtr pkt)
00172 {
00173     if (busy()) {
00174         DPRINTF(Ethernet, "packet not sent, link busy\n");
00175         return false;
00176     }
00177 
00178     DPRINTF(Ethernet, "packet sent: len=%d\n", pkt->length);
00179     DDUMP(EthernetData, pkt->data, pkt->length);
00180 
00181     packet = pkt;
00182     Tick delay = (Tick)ceil(((double)pkt->length * ticksPerByte) + 1.0);
00183     if (delayVar != 0)
00184         delay += random_mt.random<Tick>(0, delayVar);
00185 
00186     DPRINTF(Ethernet, "scheduling packet: delay=%d, (rate=%f)\n",
00187             delay, ticksPerByte);
00188     parent->schedule(doneEvent, curTick() + delay);
00189 
00190     return true;
00191 }
