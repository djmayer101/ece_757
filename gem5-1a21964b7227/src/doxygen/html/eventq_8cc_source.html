00078 Event *
00079 Event::insertBefore(Event *event, Event *curr)
00080 {
00081     // Either way, event will be the top element in the 'in bin' list
00082     // which is the pointer we need in order to look into the list, so
00083     // we need to insert that into the bin list.
00084     if (!curr || *event < *curr) {
00085         // Insert the event before the current list since it is in the future.
00086         event->nextBin = curr;
00087         event->nextInBin = NULL;
00088     } else {
00089         // Since we're on the correct list, we need to point to the next list
00090         event->nextBin = curr->nextBin;  // curr->nextBin can now become stale
00091 
00092         // Insert event at the top of the stack
00093         event->nextInBin = curr;
00094     }
00095 
00096     return event;
00097 }

00100 EventQueue::insert(Event *event)
00101 {
00102     // Deal with the head case
00103     if (!head || *event <= *head) {
00104         head = Event::insertBefore(event, head);
00105         return;
00106     }
00107 
00108     // Figure out either which 'in bin' list we are on, or where a new list
00109     // needs to be inserted
00110     Event *prev = head;
00111     Event *curr = head->nextBin;
00112     while (curr && *curr < *event) {
00113         prev = curr;
00114         curr = curr->nextBin;
00115     }
00116 
00117     // Note: this operation may render all nextBin pointers on the
00118     // prev 'in bin' list stale (except for the top one)
00119     prev->nextBin = Event::insertBefore(event, curr);
00120 }

00122 Event *
00123 Event::removeItem(Event *event, Event *top)
00124 {
00125     Event *curr = top;
00126     Event *next = top->nextInBin;
00127 
00128     // if we removed the top item, we need to handle things specially
00129     // and just remove the top item, fixing up the next bin pointer of
00130     // the new top item
00131     if (event == top) {
00132         if (!next)
00133             return top->nextBin;
00134         next->nextBin = top->nextBin;
00135         return next;
00136     }
00137 
00138     // Since we already checked the current element, we're going to
00139     // keep checking event against the next element.
00140     while (event != next) {
00141         if (!next)
00142             panic("event not found!");
00143 
00144         curr = next;
00145         next = next->nextInBin;
00146     }
00147 
00148     // remove next from the 'in bin' list since it's what we're looking for
00149     curr->nextInBin = next->nextInBin;
00150     return top;
00151 }

00153 void
00154 EventQueue::remove(Event *event)
00155 {
00156     if (head == NULL)
00157         panic("event not found!");
00158 
00159     // deal with an event on the head's 'in bin' list (event has the same
00160     // time as the head)
00161     if (*head == *event) {
00162         head = Event::removeItem(event, head);
00163         return;
00164     }
00165 
00166     // Find the 'in bin' list that this event belongs on
00167     Event *prev = head;
00168     Event *curr = head->nextBin;
00169     while (curr && *curr < *event) {
00170         prev = curr;
00171         curr = curr->nextBin;
00172     }
00173 
00174     if (!curr || *curr != *event)
00175         panic("event not found!");
00176 
00177     // curr points to the top item of the the correct 'in bin' list, when
00178     // we remove an item, it returns the new top item (which may be
00179     // unchanged)
00180     prev->nextBin = Event::removeItem(event, curr);

00183 Event *
00184 EventQueue::serviceOne()
00185 {
00186     Event *event = head;
00187     Event *next = head->nextInBin;
00188     event->flags.clear(Event::Scheduled);
00189 
00190     if (next) {
00191         // update the next bin pointer since it could be stale
00192         next->nextBin = head->nextBin;
00193 
00194         // pop the stack
00195         head = next;
00196     } else {
00197         // this was the only element on the 'in bin' list, so get rid of
00198         // the 'in bin' list and point to the next bin list
00199         head = head->nextBin;
00200     }
00201 
00202     // handle action
00203     if (!event->squashed()) {
00204         // forward current cycle to the time when this event occurs.
00205         setCurTick(event->when());
00206 
00207         event->process();
00208         if (event->isExitEvent()) {
00209             assert(!event->flags.isSet(Event::AutoDelete) ||
00210                    !event->flags.isSet(Event::IsMainQueue)); // would be silly
00211             return event;
00212         }
00213     } else {
00214         event->flags.clear(Event::Squashed);
00215     }
00216 
00217     if (event->flags.isSet(Event::AutoDelete) && !event->scheduled())
00218         delete event;
00219 
00220     return NULL;
00221 }

00232 void
00233 Event::unserialize(Checkpoint *cp, const string &section)
00234 {
00235     if (scheduled())
00236         mainEventQueue.deschedule(this);
00237 
00238     UNSERIALIZE_SCALAR(_when);
00239     UNSERIALIZE_SCALAR(_priority);
00240 
00241     short _flags;
00242     UNSERIALIZE_SCALAR(_flags);
00243 
00244     // Old checkpoints had no concept of the Initialized flag
00245     // so restoring from old checkpoints always fail.
00246     // Events are initialized on construction but original code 
00247     // "flags = _flags" would just overwrite the initialization. 
00248     // So, read in the checkpoint flags, but then set the Initialized 
00249     // flag on top of it in order to avoid failures.
00250     assert(initialized());
00251     flags = _flags;
00252     flags.set(Initialized);
00253 
00254     // need to see if original event was in a scheduled, unsquashed
00255     // state, but don't want to restore those flags in the current
00256     // object itself (since they aren't immediately true)
00257     bool wasScheduled = flags.isSet(Scheduled) && !flags.isSet(Squashed);
00258     flags.clear(Squashed | Scheduled);
00259 
00260     if (wasScheduled) {
00261         DPRINTF(Config, "rescheduling at %d\n", _when);
00262         mainEventQueue.schedule(this, _when);
00263     }
00264 }
