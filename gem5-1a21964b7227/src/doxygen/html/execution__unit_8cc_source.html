00087 void
00088 ExecutionUnit::execute(int slot_num)
00089 {
00090     ResourceRequest* exec_req = reqs[slot_num];
00091     DynInstPtr inst = reqs[slot_num]->inst;
00092     if (inst->fault != NoFault) {
00093         DPRINTF(InOrderExecute,
00094                 "[tid:%i]: [sn:%i]: Detected %s fault @ %x. Forwarding to "
00095                 "next stage.\n", inst->readTid(), inst->seqNum, inst->fault->name(),
00096                 inst->pcState());
00097         exec_req->done();
00098         return;
00099     }
00100 
00101     Fault fault = NoFault;
00102     Tick cur_tick = curTick();
00103     unsigned stage_num = exec_req->getStageNum();
00104     ThreadID tid = inst->readTid();
00105 #if TRACING_ON
00106     InstSeqNum seq_num = inst->seqNum;
00107 #endif
00108 
00109     switch (exec_req->cmd)
00110     {
00111       case ExecuteInst:
00112         {
00113             if (inst->isNop()) {
00114                 DPRINTF(InOrderExecute, "[tid:%i] [sn:%i] [PC:%s] Ignoring execution"
00115                         "of %s.\n", inst->readTid(), seq_num, inst->pcState(),
00116                         inst->instName());
00117                 inst->setExecuted();
00118                 exec_req->done();
00119                 return;
00120             } else {
00121                 DPRINTF(InOrderExecute, "[tid:%i] Executing [sn:%i] [PC:%s] %s.\n",
00122                         inst->readTid(), seq_num, inst->pcState(), inst->instName());
00123             }
00124 
00125             if (cur_tick != lastExecuteTick) {
00126                 lastExecuteTick = cur_tick;
00127             }
00128 
00129             //@todo: handle address generation here
00130             assert(!inst->isMemRef());
00131 
00132             if (inst->isControl()) {
00133                 if (lastControlTick == cur_tick) {
00134                     DPRINTF(InOrderExecute, "Can not Execute More than One Control "
00135                             "Inst Per Cycle. Blocking Request.\n");
00136                     exec_req->done(false);
00137                     return;
00138                 }
00139                 lastControlTick = curTick();
00140 
00141                 // Evaluate Branch
00142                 fault = inst->execute();
00143 
00144                 // Should unconditional control , pc relative count as an
00145                 // execution??? Probably not.
00146                 executions++;
00147 
00148                 if (fault == NoFault) {
00149                     inst->setExecuted();
00150 
00151                     if (inst->mispredicted()) {
00152                         assert(inst->isControl());
00153 
00154                         // Set up Squash Generated By this Misprediction
00155                         TheISA::PCState pc = inst->pcState();
00156                         TheISA::advancePC(pc, inst->staticInst);
00157                         inst->setPredTarg(pc);
00158                         inst->setSquashInfo(stage_num);
00159                         setupSquash(inst, stage_num, tid);
00160 
00161                         DPRINTF(InOrderExecute, "[tid:%i]: [sn:%i] Squashing from "
00162                                 "stage %i. Redirecting  fetch to %s.\n", tid,
00163                                 inst->seqNum, stage_num, pc);
00164                         DPRINTF(InOrderStall, "STALL: [tid:%i]: Branch"
00165                                 " misprediction at %s\n", tid, inst->pcState());
00166 
00167                         if (inst->predTaken()) {
00168                             predictedTakenIncorrect++;
00169                             DPRINTF(InOrderExecute, "[tid:%i] [sn:%i] %s ..."
00170                                     "PC %s ... Mispredicts! "
00171                                     "(Prediction: Taken)\n",
00172                                     tid, inst->seqNum,
00173                                     inst->staticInst->disassemble(
00174                                         inst->instAddr()),
00175                                     inst->pcState());
00176                         } else {
00177                             predictedNotTakenIncorrect++;
00178                             DPRINTF(InOrderExecute, "[tid:%i] [sn:%i] %s ..."
00179                                     "PC %s ... Mispredicts! "
00180                                     "(Prediction: Not Taken)\n",
00181                                     tid, inst->seqNum,
00182                                     inst->staticInst->disassemble(
00183                                         inst->instAddr()),
00184                                     inst->pcState());
00185                         }
00186                         predictedIncorrect++;
00187                     } else {
00188                         DPRINTF(InOrderExecute, "[tid:%i]: [sn:%i]: Prediction"
00189                                 "Correct.\n", inst->readTid(), seq_num);
00190                         predictedCorrect++;
00191                     }
00192 
00193                     exec_req->done();
00194                 } else {
00195                     DPRINTF(Fault, "[tid:%i]:[sn:%i]: Fault %s found\n",
00196                             inst->readTid(), inst->seqNum, fault->name());
00197                     inst->fault = fault;
00198                     exec_req->done();
00199                 }
00200             } else {
00201                 // Regular ALU instruction
00202                 fault = inst->execute();
00203                 executions++;
00204 
00205                 if (fault == NoFault) {
00206                     inst->setExecuted();
00207 
00208 #if TRACING_ON
00209                     for (int didx = 0; didx < inst->numDestRegs(); didx++)
00210                         if (inst->resultType(didx) == InOrderDynInst::Float ||
00211                             inst->resultType(didx) == InOrderDynInst::FloatBits ||
00212                             inst->resultType(didx) == InOrderDynInst::Double)
00213                             DPRINTF(InOrderExecute, "[tid:%i]: [sn:%i]: Dest result %i "
00214                                     "of FP execution is %08f (%x).\n", inst->readTid(),
00215                                     seq_num, didx, inst->readFloatResult(didx),
00216                                     inst->readFloatBitsResult(didx));
00217                         else
00218                             DPRINTF(InOrderExecute, "[tid:%i]: [sn:%i]: Dest result %i "
00219                                     "of Int execution is 0x%x.\n", inst->readTid(),
00220                                     seq_num, didx, inst->readIntResult(didx));
00221 #endif
00222 
00223                     if (!FullSystem) {
00224                         // The Syscall might change the PC, so conservatively
00225                         // squash everything behing it
00226                         if (inst->isSyscall()) {
00227                             inst->setSquashInfo(stage_num);
00228                             setupSquash(inst, stage_num, tid);
00229                         }
00230                     }
00231                 } else {
00232                     DPRINTF(InOrderExecute, "[tid:%i]: [sn:%i]: had a %s "
00233                             "fault.\n", inst->readTid(), seq_num, fault->name());
00234                     DPRINTF(Fault, "[tid:%i]:[sn:%i]: Fault %s found\n",
00235                             inst->readTid(), inst->seqNum, fault->name());
00236                     inst->fault = fault;
00237                 }
00238 
00239                 exec_req->done();
00240             }
00241         }
00242         break;
00243 
00244       default:
00245         fatal("Unrecognized command to %s", resName);
00246     }
00247 }
