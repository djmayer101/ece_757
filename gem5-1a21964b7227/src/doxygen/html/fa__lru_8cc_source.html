00046     : blkSize(_blkSize), size(_size), hitLatency(hit_latency)
00047 {

00051     if (!(hitLatency > 0))
00052         fatal("Access latency in cycles must be at least one cycle");
00053     if (!isPowerOf2(size))
00054         fatal("Cache Size must be power of 2 for now");
00055 
00056     // Track all cache sizes from 128K up by powers of 2
00057     numCaches = floorLog2(size) - 17;
00058     if (numCaches >0){
00059         cacheBoundaries = new FALRUBlk *[numCaches];
00060         cacheMask = (1 << numCaches) - 1;
00061     } else {
00062         cacheMask = 0;
00063     }
00064 
00065     warmedUp = false;
00066     warmupBound = size/blkSize;
00067     numBlocks = size/blkSize;
00068 
00069     blks = new FALRUBlk[numBlocks];

00071     tail = &(blks[numBlocks-1]);
00072 
00073     head->prev = NULL;
00074     head->next = &(blks[1]);
00075     head->inCache = cacheMask;
00076 
00077     tail->prev = &(blks[numBlocks-2]);
00078     tail->next = NULL;
00079     tail->inCache = 0;
00080 
00081     unsigned index = (1 << 17) / blkSize;
00082     unsigned j = 0;

00084     for (unsigned i = 1; i < numBlocks - 1; i++) {
00085         blks[i].inCache = flags;

00091         }
00092         blks[i].prev = &(blks[i-1]);
00093         blks[i].next = &(blks[i+1]);

00095     }

00098     //assert(check());
