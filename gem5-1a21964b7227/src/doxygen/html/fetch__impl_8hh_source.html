00370 void
00371 DefaultFetch<Impl>::processCacheCompletion(PacketPtr pkt)
00372 {
00373     ThreadID tid = pkt->req->threadId();
00374 
00375     DPRINTF(Fetch, "[tid:%u] Waking up from cache miss.\n", tid);
00376     assert(!cpu->switchedOut());
00377 
00378     // Only change the status if it's still waiting on the icache access
00379     // to return.
00380     if (fetchStatus[tid] != IcacheWaitResponse ||
00381         pkt->req != memReq[tid]) {
00382         ++fetchIcacheSquashes;
00383         delete pkt->req;
00384         delete pkt;
00385         return;
00386     }
00387 
00388     memcpy(cacheData[tid], pkt->getPtr<uint8_t>(), cacheBlkSize);
00389     cacheDataValid[tid] = true;
00390 
00391     // Wake up the CPU (if it went to sleep and was waiting on
00392     // this completion event).
00393     cpu->wakeCPU();
00394 
00395     DPRINTF(Activity, "[tid:%u] Activating fetch due to cache completion\n",
00396             tid);
00397 
00398     switchToActive();
00399 
00400     // Only switch to IcacheAccessComplete if we're not stalled as well.
00401     if (checkStall(tid)) {
00402         fetchStatus[tid] = Blocked;
00403     } else {
00404         fetchStatus[tid] = IcacheAccessComplete;
00405     }
00406 
00407     // Reset the mem req to NULL.
00408     delete pkt->req;
00409     delete pkt;
00410     memReq[tid] = NULL;

00443 template <class Impl>
00444 bool
00445 DefaultFetch<Impl>::isDrained() const
00446 {
00447     /* Make sure that threads are either idle of that the commit stage
00448      * has signaled that draining has completed by setting the drain
00449      * stall flag. This effectively forces the pipeline to be disabled
00450      * until the whole system is drained (simulation may continue to
00451      * drain other components).
00452      */
00453     for (ThreadID i = 0; i < numThreads; ++i) {
00454         if (!(fetchStatus[i] == Idle ||
00455               (fetchStatus[i] == Blocked && stalls[i].drain)))
00456             return false;
00457     }
00458 
00459     /* The pipeline might start up again in the middle of the drain
00460      * cycle if the finish translation event is scheduled, so make
00461      * sure that's not the case.
00462      */
00463     return !finishTranslationEvent.scheduled();
00464 }

00525 {
00526     // Do branch prediction check here.
00527     // A bit of a misnomer...next_PC is actually the current PC until
00528     // this function updates it.
00529     bool predict_taken;
00530 
00531     if (!inst->isControl()) {
00532         TheISA::advancePC(nextPC, inst->staticInst);

00544                 tid, inst->seqNum, nextPC);

00546         DPRINTF(Fetch, "[tid:%i]: [sn:%i]:Branch predicted to be not taken.\n",

00551             tid, inst->seqNum, nextPC);

00554 
00555     ++fetchedBranches;
00556 
00557     if (predict_taken) {
00558         ++predictedBranches;
00559     }
00560

00612 void
00613 DefaultFetch<Impl>::finishTranslation(Fault fault, RequestPtr mem_req)
00614 {
00615     ThreadID tid = mem_req->threadId();
00616     Addr block_PC = mem_req->getVaddr();
00617 
00618     assert(!cpu->switchedOut());
00619 
00620     // Wake up CPU if it was idle
00621     cpu->wakeCPU();
00622 
00623     if (fetchStatus[tid] != ItlbWait || mem_req != memReq[tid] ||
00624         mem_req->getVaddr() != memReq[tid]->getVaddr()) {
00625         DPRINTF(Fetch, "[tid:%i] Ignoring itlb completed after squash\n",
00626                 tid);
00627         ++fetchTlbSquashes;
00628         delete mem_req;
00629         return;
00630     }
00631 
00632 
00633     // If translation was successful, attempt to read the icache block.
00634     if (fault == NoFault) {
00635         // Check that we're not going off into random memory
00636         // If we have, just wait around for commit to squash something and put
00637         // us on the right track
00638         if (!cpu->system->isMemAddr(mem_req->getPaddr())) {
00639             warn("Address %#x is outside of physical memory, stopping fetch\n",
00640                     mem_req->getPaddr());
00641             fetchStatus[tid] = NoGoodAddr;
00642             delete mem_req;
00643             memReq[tid] = NULL;
00644             return;
00645         }
00646 
00647         // Build packet here.
00648         PacketPtr data_pkt = new Packet(mem_req, MemCmd::ReadReq);
00649         data_pkt->dataDynamicArray(new uint8_t[cacheBlkSize]);
00650 
00651         cacheDataPC[tid] = block_PC;
00652         cacheDataValid[tid] = false;
00653         DPRINTF(Fetch, "Fetch: Doing instruction read.\n");
00654 
00655         fetchedCacheLines++;
00656 
00657         // Access the cache.
00658         if (!cpu->getInstPort().sendTimingReq(data_pkt)) {
00659             assert(retryPkt == NULL);
00660             assert(retryTid == InvalidThreadID);
00661             DPRINTF(Fetch, "[tid:%i] Out of MSHRs!\n", tid);
00662 
00663             fetchStatus[tid] = IcacheWaitRetry;
00664             retryPkt = data_pkt;
00665             retryTid = tid;
00666             cacheBlocked = true;
00667         } else {
00668             DPRINTF(Fetch, "[tid:%i]: Doing Icache access.\n", tid);
00669             DPRINTF(Activity, "[tid:%i]: Activity: Waiting on I-cache "
00670                     "response.\n", tid);
00671 
00672             lastIcacheStall[tid] = curTick();
00673             fetchStatus[tid] = IcacheWaitResponse;
00674         }
00675     } else {
00676         if (!(numInst < fetchWidth)) {
00677             assert(!finishTranslationEvent.scheduled());
00678             finishTranslationEvent.setFault(fault);
00679             finishTranslationEvent.setReq(mem_req);
00680             cpu->schedule(finishTranslationEvent,
00681                           cpu->clockEdge(Cycles(1)));
00682             return;
00683         }
00684         DPRINTF(Fetch, "[tid:%i] Got back req with addr %#x but expected %#x\n",
00685                 tid, mem_req->getVaddr(), memReq[tid]->getVaddr());
00686         // Translation faulted, icache request won't be sent.
00687         delete mem_req;
00688         memReq[tid] = NULL;
00689 
00690         // Send the fault to commit.  This thread will not do anything
00691         // until commit handles the fault.  The only other way it can
00692         // wake up is if a squash comes along and changes the PC.
00693         TheISA::PCState fetchPC = pc[tid];
00694 
00695         DPRINTF(Fetch, "[tid:%i]: Translation faulted, building noop.\n", tid);
00696         // We will use a nop in ordier to carry the fault.
00697         DynInstPtr instruction = buildInst(tid,
00698                 decoder[tid]->decode(TheISA::NoopMachInst, fetchPC.instAddr()),
00699                 NULL, fetchPC, fetchPC, false);
00700 
00701         instruction->setPredTarg(fetchPC);
00702         instruction->fault = fault;
00703         wroteToTimeBuffer = true;
00704 
00705         DPRINTF(Activity, "Activity this cycle.\n");
00706         cpu->activityThisCycle();
00707 
00708         fetchStatus[tid] = TrapPending;
00709 
00710         DPRINTF(Fetch, "[tid:%i]: Blocked, need to handle the trap.\n", tid);
00711         DPRINTF(Fetch, "[tid:%i]: fault (%s) detected @ PC %s.\n",
00712                 tid, fault->name(), pc[tid]);
00713     }
00714     _status = updateFetchStatus();
00715 }

00717 template <class Impl>
00718 inline void
00719 DefaultFetch<Impl>::doSquash(const TheISA::PCState &newPC,
00720                              const DynInstPtr squashInst, ThreadID tid)
00721 {
00722     DPRINTF(Fetch, "[tid:%i]: Squashing, setting PC to: %s.\n",
00723             tid, newPC);
00724 
00725     pc[tid] = newPC;
00726     fetchOffset[tid] = 0;
00727     if (squashInst && squashInst->pcState().instAddr() == newPC.instAddr())
00728         macroop[tid] = squashInst->macroop;
00729     else
00730         macroop[tid] = NULL;
00731     decoder[tid]->reset();
00732 
00733     // Clear the icache miss if it's outstanding.
00734     if (fetchStatus[tid] == IcacheWaitResponse) {
00735         DPRINTF(Fetch, "[tid:%i]: Squashing outstanding Icache miss.\n",
00736                 tid);
00737         memReq[tid] = NULL;
00738     } else if (fetchStatus[tid] == ItlbWait) {
00739         DPRINTF(Fetch, "[tid:%i]: Squashing outstanding ITLB miss.\n",
00740                 tid);
00741         memReq[tid] = NULL;
00742     }
00743 
00744     // Get rid of the retrying packet if it was from this thread.
00745     if (retryTid == tid) {
00746         assert(cacheBlocked);
00747         if (retryPkt) {
00748             delete retryPkt->req;
00749             delete retryPkt;
00750         }
00751         retryPkt = NULL;
00752         retryTid = InvalidThreadID;
00753     }
00754 
00755     fetchStatus[tid] = Squashing;
00756 
00757     // microops are being squashed, it is not known wheather the
00758     // youngest non-squashed microop was  marked delayed commit
00759     // or not. Setting the flag to true ensures that the
00760     // interrupts are not handled when they cannot be, though
00761     // some opportunities to handle interrupts may be missed.
00762     delayedCommit[tid] = true;
00763 
00764     ++fetchSquashCycles;
00765 }

00866 template <class Impl>
00867 void
00868 DefaultFetch<Impl>::tick()
00869 {
00870     list<ThreadID>::iterator threads = activeThreads->begin();
00871     list<ThreadID>::iterator end = activeThreads->end();
00872     bool status_change = false;
00873 
00874     wroteToTimeBuffer = false;
00875 
00876     for (ThreadID i = 0; i < Impl::MaxThreads; ++i) {
00877         issuePipelinedIfetch[i] = false;
00878     }
00879 
00880     while (threads != end) {
00881         ThreadID tid = *threads++;
00882 
00883         // Check the signals for each thread to determine the proper status
00884         // for each thread.
00885         bool updated_status = checkSignalsAndUpdate(tid);
00886         status_change =  status_change || updated_status;
00887     }
00888 
00889     DPRINTF(Fetch, "Running stage.\n");
00890 
00891     if (FullSystem) {
00892         if (fromCommit->commitInfo[0].interruptPending) {
00893             interruptPending = true;
00894         }
00895 
00896         if (fromCommit->commitInfo[0].clearInterrupt) {
00897             interruptPending = false;
00898         }
00899     }
00900 
00901     for (threadFetched = 0; threadFetched < numFetchingThreads;
00902          threadFetched++) {
00903         // Fetch each of the actively fetching threads.
00904         fetch(status_change);
00905     }
00906 
00907     // Record number of instructions fetched this cycle for distribution.
00908     fetchNisnDist.sample(numInst);
00909 
00910     if (status_change) {
00911         // Change the fetch stage status if there was a status change.
00912         _status = updateFetchStatus();
00913     }
00914 
00915     // If there was activity this cycle, inform the CPU of it.
00916     if (wroteToTimeBuffer || cpu->contextSwitch) {
00917         DPRINTF(Activity, "Activity this cycle.\n");
00918 
00919         cpu->activityThisCycle();
00920     }
00921 
00922     // Issue the next I-cache request if possible.
00923     for (ThreadID i = 0; i < Impl::MaxThreads; ++i) {
00924         if (issuePipelinedIfetch[i]) {
00925             pipelineIcacheAccesses(i);
00926         }
00927     }
00928 
00929     // Reset the number of the instruction we've fetched.
00930     numInst = 0;
00931 }

00933 template <class Impl>
00934 bool
00935 DefaultFetch<Impl>::checkSignalsAndUpdate(ThreadID tid)
00936 {
00937     // Update the per thread stall statuses.
00938     if (fromDecode->decodeBlock[tid]) {
00939         stalls[tid].decode = true;
00940     }
00941 
00942     if (fromDecode->decodeUnblock[tid]) {
00943         assert(stalls[tid].decode);
00944         assert(!fromDecode->decodeBlock[tid]);
00945         stalls[tid].decode = false;
00946     }
00947 
00948     if (fromRename->renameBlock[tid]) {
00949         stalls[tid].rename = true;
00950     }
00951 
00952     if (fromRename->renameUnblock[tid]) {
00953         assert(stalls[tid].rename);
00954         assert(!fromRename->renameBlock[tid]);
00955         stalls[tid].rename = false;
00956     }
00957 
00958     if (fromIEW->iewBlock[tid]) {
00959         stalls[tid].iew = true;
00960     }
00961 
00962     if (fromIEW->iewUnblock[tid]) {
00963         assert(stalls[tid].iew);
00964         assert(!fromIEW->iewBlock[tid]);
00965         stalls[tid].iew = false;
00966     }
00967 
00968     if (fromCommit->commitBlock[tid]) {
00969         stalls[tid].commit = true;
00970     }
00971 
00972     if (fromCommit->commitUnblock[tid]) {
00973         assert(stalls[tid].commit);
00974         assert(!fromCommit->commitBlock[tid]);
00975         stalls[tid].commit = false;
00976     }
00977 
00978     // Check squash signals from commit.
00979     if (fromCommit->commitInfo[tid].squash) {
00980 
00981         DPRINTF(Fetch, "[tid:%u]: Squashing instructions due to squash "
00982                 "from commit.\n",tid);
00983         // In any case, squash.
00984         squash(fromCommit->commitInfo[tid].pc,
00985                fromCommit->commitInfo[tid].doneSeqNum,
00986                fromCommit->commitInfo[tid].squashInst, tid);
00987 
00988         // If it was a branch mispredict on a control instruction, update the
00989         // branch predictor with that instruction, otherwise just kill the
00990         // invalid state we generated in after sequence number
00991         if (fromCommit->commitInfo[tid].mispredictInst &&
00992             fromCommit->commitInfo[tid].mispredictInst->isControl()) {
00993             branchPred->squash(fromCommit->commitInfo[tid].doneSeqNum,
00994                               fromCommit->commitInfo[tid].pc,
00995                               fromCommit->commitInfo[tid].branchTaken,
00996                               tid);
00997         } else {
00998             branchPred->squash(fromCommit->commitInfo[tid].doneSeqNum,
00999                               tid);
01000         }
01001 
01002         return true;
01003     } else if (fromCommit->commitInfo[tid].doneSeqNum) {
01004         // Update the branch predictor if it wasn't a squashed instruction
01005         // that was broadcasted.
01006         branchPred->update(fromCommit->commitInfo[tid].doneSeqNum, tid);
01007     }
01008 
01009     // Check ROB squash signals from commit.
01010     if (fromCommit->commitInfo[tid].robSquashing) {
01011         DPRINTF(Fetch, "[tid:%u]: ROB is still squashing.\n", tid);
01012 
01013         // Continue to squash.
01014         fetchStatus[tid] = Squashing;
01015 
01016         return true;
01017     }
01018 
01019     // Check squash signals from decode.
01020     if (fromDecode->decodeInfo[tid].squash) {
01021         DPRINTF(Fetch, "[tid:%u]: Squashing instructions due to squash "
01022                 "from decode.\n",tid);
01023 
01024         // Update the branch predictor.
01025         if (fromDecode->decodeInfo[tid].branchMispredict) {
01026             branchPred->squash(fromDecode->decodeInfo[tid].doneSeqNum,
01027                               fromDecode->decodeInfo[tid].nextPC,
01028                               fromDecode->decodeInfo[tid].branchTaken,
01029                               tid);
01030         } else {
01031             branchPred->squash(fromDecode->decodeInfo[tid].doneSeqNum,
01032                               tid);
01033         }
01034 
01035         if (fetchStatus[tid] != Squashing) {
01036 
01037             DPRINTF(Fetch, "Squashing from decode with PC = %s\n",
01038                 fromDecode->decodeInfo[tid].nextPC);
01039             // Squash unless we're already squashing
01040             squashFromDecode(fromDecode->decodeInfo[tid].nextPC,
01041                              fromDecode->decodeInfo[tid].squashInst,
01042                              fromDecode->decodeInfo[tid].doneSeqNum,
01043                              tid);
01044 
01045             return true;
01046         }
01047     }
01048 
01049     if (checkStall(tid) &&
01050         fetchStatus[tid] != IcacheWaitResponse &&
01051         fetchStatus[tid] != IcacheWaitRetry &&
01052         fetchStatus[tid] != ItlbWait) {
01053         DPRINTF(Fetch, "[tid:%i]: Setting to blocked\n",tid);
01054 
01055         fetchStatus[tid] = Blocked;
01056 
01057         return true;
01058     }
01059 
01060     if (fetchStatus[tid] == Blocked ||
01061         fetchStatus[tid] == Squashing) {
01062         // Switch status to running if fetch isn't being told to block or
01063         // squash this cycle.
01064         DPRINTF(Fetch, "[tid:%i]: Done squashing, switching to running.\n",
01065                 tid);
01066 
01067         fetchStatus[tid] = Running;
01068 
01069         return true;
01070     }
01071 
01072     // If we've reached this point, we have not gotten any signals that
01073     // cause fetch to change its status.  Fetch remains the same as before.
01074     return false;
01075 }

01127 template<class Impl>
01128 void
01129 DefaultFetch<Impl>::fetch(bool &status_change)
01130 {
01132     // Start actual fetch
01134     ThreadID tid = getFetchingThread(fetchPolicy);
01135 
01136     assert(!cpu->switchedOut());
01137 
01138     if (tid == InvalidThreadID) {
01139         // Breaks looping condition in tick()
01140         threadFetched = numFetchingThreads;
01141 
01142         if (numThreads == 1) {  // @todo Per-thread stats
01143             profileStall(0);
01144         }
01145 
01146         return;
01147     }
01148 
01149     DPRINTF(Fetch, "Attempting to fetch from [tid:%i]\n", tid);
01150 
01151     // The current PC.
01152     TheISA::PCState thisPC = pc[tid];
01153 
01154     Addr pcOffset = fetchOffset[tid];
01155     Addr fetchAddr = (thisPC.instAddr() + pcOffset) & BaseCPU::PCMask;
01156 
01157     bool inRom = isRomMicroPC(thisPC.microPC());
01158 
01159     // If returning from the delay of a cache miss, then update the status
01160     // to running, otherwise do the cache access.  Possibly move this up
01161     // to tick() function.
01162     if (fetchStatus[tid] == IcacheAccessComplete) {
01163         DPRINTF(Fetch, "[tid:%i]: Icache miss is complete.\n", tid);
01164 
01165         fetchStatus[tid] = Running;
01166         status_change = true;
01167     } else if (fetchStatus[tid] == Running) {
01168         // Align the fetch PC so its at the start of a cache block.
01169         Addr block_PC = icacheBlockAlignPC(fetchAddr);
01170 
01171         // If buffer is no longer valid or fetchAddr has moved to point
01172         // to the next cache block, AND we have no remaining ucode
01173         // from a macro-op, then start fetch from icache.
01174         if (!(cacheDataValid[tid] && block_PC == cacheDataPC[tid])
01175             && !inRom && !macroop[tid]) {
01176             DPRINTF(Fetch, "[tid:%i]: Attempting to translate and read "
01177                     "instruction, starting at PC %s.\n", tid, thisPC);
01178 
01179             fetchCacheLine(fetchAddr, tid, thisPC.instAddr());
01180 
01181             if (fetchStatus[tid] == IcacheWaitResponse)
01182                 ++icacheStallCycles;
01183             else if (fetchStatus[tid] == ItlbWait)
01184                 ++fetchTlbCycles;
01185             else
01186                 ++fetchMiscStallCycles;
01187             return;
01188         } else if ((checkInterrupt(thisPC.instAddr()) && !delayedCommit[tid])) {
01189             // Stall CPU if an interrupt is posted and we're not issuing
01190             // an delayed commit micro-op currently (delayed commit instructions
01191             // are not interruptable by interrupts, only faults)
01192             ++fetchMiscStallCycles;
01193             DPRINTF(Fetch, "[tid:%i]: Fetch is stalled!\n", tid);
01194             return;
01195         }
01196     } else {
01197         if (fetchStatus[tid] == Idle) {
01198             ++fetchIdleCycles;
01199             DPRINTF(Fetch, "[tid:%i]: Fetch is idle!\n", tid);
01200         }
01201 
01202         // Status is Idle, so fetch should do nothing.
01203         return;
01204     }
01205 
01206     ++fetchCycles;
01207 
01208     TheISA::PCState nextPC = thisPC;
01209 
01210     StaticInstPtr staticInst = NULL;
01211     StaticInstPtr curMacroop = macroop[tid];
01212 
01213     // If the read of the first instruction was successful, then grab the
01214     // instructions from the rest of the cache line and put them into the
01215     // queue heading to decode.
01216 
01217     DPRINTF(Fetch, "[tid:%i]: Adding instructions to queue to "
01218             "decode.\n", tid);
01219 
01220     // Need to keep track of whether or not a predicted branch
01221     // ended this fetch block.
01222     bool predictedBranch = false;
01223 
01224     TheISA::MachInst *cacheInsts =
01225         reinterpret_cast<TheISA::MachInst *>(cacheData[tid]);
01226 
01227     const unsigned numInsts = cacheBlkSize / instSize;
01228     unsigned blkOffset = (fetchAddr - cacheDataPC[tid]) / instSize;
01229 
01230     // Loop through instruction memory from the cache.
01231     // Keep issuing while fetchWidth is available and branch is not
01232     // predicted taken
01233     while (numInst < fetchWidth && !predictedBranch) {
01234 
01235         // We need to process more memory if we aren't going to get a
01236         // StaticInst from the rom, the current macroop, or what's already
01237         // in the decoder.
01238         bool needMem = !inRom && !curMacroop &&
01239             !decoder[tid]->instReady();
01240         fetchAddr = (thisPC.instAddr() + pcOffset) & BaseCPU::PCMask;
01241         Addr block_PC = icacheBlockAlignPC(fetchAddr);
01242 
01243         if (needMem) {
01244             // If buffer is no longer valid or fetchAddr has moved to point
01245             // to the next cache block then start fetch from icache.
01246             if (!cacheDataValid[tid] || block_PC != cacheDataPC[tid])
01247                 break;
01248 
01249             if (blkOffset >= numInsts) {
01250                 // We need to process more memory, but we've run out of the
01251                 // current block.
01252                 break;
01253             }
01254 
01255             if (ISA_HAS_DELAY_SLOT && pcOffset == 0) {
01256                 // Walk past any annulled delay slot instructions.
01257                 Addr pcAddr = thisPC.instAddr() & BaseCPU::PCMask;
01258                 while (fetchAddr != pcAddr && blkOffset < numInsts) {
01259                     blkOffset++;
01260                     fetchAddr += instSize;
01261                 }
01262                 if (blkOffset >= numInsts)
01263                     break;
01264             }
01265 
01266             MachInst inst = TheISA::gtoh(cacheInsts[blkOffset]);
01267             decoder[tid]->moreBytes(thisPC, fetchAddr, inst);
01268 
01269             if (decoder[tid]->needMoreBytes()) {
01270                 blkOffset++;
01271                 fetchAddr += instSize;
01272                 pcOffset += instSize;
01273             }
01274         }
01275 
01276         // Extract as many instructions and/or microops as we can from
01277         // the memory we've processed so far.
01278         do {
01279             if (!(curMacroop || inRom)) {
01280                 if (decoder[tid]->instReady()) {
01281                     staticInst = decoder[tid]->decode(thisPC);
01282 
01283                     // Increment stat of fetched instructions.
01284                     ++fetchedInsts;
01285 
01286                     if (staticInst->isMacroop()) {
01287                         curMacroop = staticInst;
01288                     } else {
01289                         pcOffset = 0;
01290                     }
01291                 } else {
01292                     // We need more bytes for this instruction so blkOffset and
01293                     // pcOffset will be updated
01294                     break;
01295                 }
01296             }
01297             // Whether we're moving to a new macroop because we're at the
01298             // end of the current one, or the branch predictor incorrectly
01299             // thinks we are...
01300             bool newMacro = false;
01301             if (curMacroop || inRom) {
01302                 if (inRom) {
01303                     staticInst = cpu->microcodeRom.fetchMicroop(
01304                             thisPC.microPC(), curMacroop);
01305                 } else {
01306                     staticInst = curMacroop->fetchMicroop(thisPC.microPC());
01307                 }
01308                 newMacro |= staticInst->isLastMicroop();
01309             }
01310 
01311             DynInstPtr instruction =
01312                 buildInst(tid, staticInst, curMacroop,
01313                           thisPC, nextPC, true);
01314 
01315             numInst++;
01316 
01317 #if TRACING_ON
01318             if (DTRACE(O3PipeView)) {

01362         DPRINTF(Fetch, "[tid:%i]: Done fetching, reached fetch bandwidth "
01363                 "for this cycle.\n", tid);
01364     } else if (blkOffset >= cacheBlkSize) {
01365         DPRINTF(Fetch, "[tid:%i]: Done fetching, reached the end of cache "
01366                 "block.\n", tid);
01367     }
01368 
01369     macroop[tid] = curMacroop;
01370     fetchOffset[tid] = pcOffset;
01371

01374     }
01375

01379     // a state that would preclude fetching
