00076 void
00077 FetchSeqUnit::execute(int slot_num)
00078 {
00079     ResourceRequest* fs_req = reqs[slot_num];
00080     DynInstPtr inst = fs_req->inst;
00081     ThreadID tid = inst->readTid();
00082     int stage_num = fs_req->getStageNum();
00083 
00084     if (inst->fault != NoFault) {
00085         DPRINTF(InOrderFetchSeq,
00086                 "[tid:%i]: [sn:%i]: Detected %s fault @ %x. Forwarding to "
00087                 "next stage.\n", tid, inst->seqNum, inst->fault->name(),
00088                 inst->pcState());
00089         fs_req->done();
00090         return;
00091     }
00092 
00093     switch (fs_req->cmd)
00094     {
00095       case AssignNextPC:
00096         {
00097             DPRINTF(InOrderFetchSeq, "[tid:%i]: Current PC is %s\n", tid,
00098                     pc[tid]);
00099 
00100             if (pcValid[tid]) {
00101                 inst->pcState(pc[tid]);
00102                 inst->setMemAddr(pc[tid].instAddr());
00103 
00104                 // Advance to next PC (typically PC + 4)
00105                 pc[tid].advance();
00106 
00107                 inst->setSeqNum(cpu->getAndIncrementInstSeq(tid));
00108 
00109                 DPRINTF(InOrderFetchSeq, "[tid:%i]: Assigning [sn:%i] to "
00110                         "PC %s\n", tid, inst->seqNum, inst->pcState());
00111 
00112                 fs_req->done();
00113             } else {
00114                 DPRINTF(InOrderStall, "STALL: [tid:%i]: NPC not valid\n", tid);
00115                 fs_req->done(false);
00116             }
00117         }
00118         break;
00119 
00120       case UpdateTargetPC:
00121         {
00122             assert(!inst->isCondDelaySlot()  &&
00123                    "Not Handling Conditional Delay Slot");
00124 
00125             if (inst->isControl()) {
00126                 if (inst->isReturn() && !inst->predTaken()) {
00127                     // If it's a return, then we must wait for resolved address.
00128                     // The Predictor will mark a return a false as "not taken"
00129                     // if there is no RAS entry
00130                     DPRINTF(InOrderFetchSeq, "[tid:%d]: Setting block signal "
00131                             "for stage %i.\n",
00132                             tid, stage_num);
00133                     cpu->pipelineStage[stage_num]->
00134                         toPrevStages->stageBlock[stage_num][tid] = true;
00135                     pcValid[tid] = false;
00136                     pcBlockStage[tid] = stage_num;
00137                 } else if (inst->predTaken()) {
00138                     // Taken Control
00139                     inst->setSquashInfo(stage_num);
00140                     setupSquash(inst, stage_num, tid);
00141 
00142                     DPRINTF(InOrderFetchSeq, "[tid:%i] Setting up squash to "
00143                             "start from stage %i, after [sn:%i].\n",
00144                             tid, stage_num, inst->squashSeqNum);
00145                 }
00146             } else {
00147                 DPRINTF(InOrderFetchSeq, "[tid:%i]: [sn:%i]: Ignoring branch "
00148                         "target update since then is not a control "
00149                         "instruction.\n", tid, inst->seqNum);
00150             }
00151 
00152             fs_req->done();
00153         }
00154         break;
00155 
00156       default:
00157         fatal("Unrecognized command to %s", resName);
00158     }
00159 }

00162 FetchSeqUnit::squash(DynInstPtr inst, int squash_stage,
00163                      InstSeqNum squash_seq_num, ThreadID tid)
00164 {
00165     DPRINTF(InOrderFetchSeq, "[tid:%i]: Updating due to squash from %s (%s) "
00166             "stage %i.\n", tid, inst->instName(), inst->pcState(),
00167             squash_stage);
00168 
00169     if (lastSquashCycle[tid] == curTick() &&
00170         squashSeqNum[tid] <= squash_seq_num) {
00171         DPRINTF(InOrderFetchSeq, "[tid:%i]: Ignoring squash from stage %i, "
00172                 "since there is an outstanding squash that is older.\n",
00173                 tid, squash_stage);
00174     } else {
00175         squashSeqNum[tid] = squash_seq_num;
00176         lastSquashCycle[tid] = curTick();
00177 
00178         if (inst->staticInst) {
00179             if (inst->fault != NoFault) {
00180                 // A Trap Caused This Fault and will update the pc state
00181                 // when done trapping
00182                 DPRINTF(InOrderFetchSeq, "[tid:%i] Blocking due to fault @ "
00183                         "[sn:%i].%s %s \n", tid, inst->seqNum,
00184                         inst->instName(), inst->pcState());
00185                 pcValid[tid] = false;
00186             } else {
00187                 TheISA::PCState nextPC;
00188                 assert(inst->staticInst);
00189                 if (inst->isControl()) {
00190                     nextPC = inst->readPredTarg();
00191 
00192                     // If we are already fetching this PC then advance to next PC
00193                     // =======
00194                     // This should handle ISAs w/delay slots and annulled delay
00195                     // slots to figure out which is the next PC to fetch after
00196                     // a mispredict
00197                     DynInstPtr bdelay_inst = NULL;
00198                     ListIt bdelay_it;
00199                     if (inst->onInstList) {
00200                         bdelay_it = inst->getInstListIt();
00201                         bdelay_it++;
00202                     } else {
00203                         InstSeqNum branch_delay_num = inst->seqNum + 1;
00204                         bdelay_it = cpu->findInst(branch_delay_num, tid);
00205                     }
00206 
00207                     if (bdelay_it != cpu->instList[tid].end()) {
00208                         bdelay_inst = (*bdelay_it);
00209                     }
00210 
00211                     if (bdelay_inst) {
00212                         if (bdelay_inst->pc.instAddr() == nextPC.instAddr()) {
00213                             bdelay_inst->pc = nextPC;
00214                             advancePC(nextPC, inst->staticInst);
00215                             DPRINTF(InOrderFetchSeq, "Advanced PC to %s\n", nextPC);
00216                         }
00217                     }
00218                 } else {
00219                     nextPC = inst->pcState();
00220                     advancePC(nextPC, inst->staticInst);
00221                 }
00222 
00223 
00224                 DPRINTF(InOrderFetchSeq, "[tid:%i]: Setting PC to %s.\n",
00225                         tid, nextPC);
00226                 pc[tid] = nextPC;
00227 
00228                 // Unblock Any Stages Waiting for this information to be updated ...
00229                 if (!pcValid[tid]) {
00230                     DPRINTF(InOrderFetchSeq, "[tid:%d]: Setting unblock signal "
00231                             "for stage %i.\n",
00232                             tid, pcBlockStage[tid]);
00233 
00234                     // Need to use "fromNextStages" instead of "toPrevStages"
00235                     // because the timebuffer will have already have advanced
00236                     // in the tick function and this squash function will happen after
00237                     // the tick
00238                     cpu->pipelineStage[pcBlockStage[tid]]->
00239                         fromNextStages->stageUnblock[pcBlockStage[tid]][tid] = true;
00240                 }
00241 
00242                 pcValid[tid] = true;
00243             }
00244         }
00245     }
00246 
00247     Resource::squash(inst, squash_stage, squash_seq_num, tid);
00248 }
