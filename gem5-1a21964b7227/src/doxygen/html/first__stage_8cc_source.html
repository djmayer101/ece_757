00064 FirstStage::squash(InstSeqNum squash_seq_num, ThreadID tid)

00066     // Set status to squashing.
00067     //stageStatus[tid] = Squashing;
00068 
00069     // Clear the instruction list and skid buffer in case they have any
00070     // insts in them.
00071     DPRINTF(InOrderStage, "Removing instructions from stage instruction "
00072             "list.\n");
00073     while (!skidBuffer[tid].empty()) {
00074         if (skidBuffer[tid].front()->seqNum <= squash_seq_num) {
00075             DPRINTF(InOrderStage,"[tid:%i]: Cannot remove [sn:%i] because "
00076                     "it's <= squashing seqNum %i.\n",
00077                     tid,
00078                     skidBuffer[tid].front()->seqNum,
00079                     squash_seq_num);
00080 
00081             DPRINTF(InOrderStage, "[tid:%i]: Cannot remove incoming "
00082                     "instructions before delay slot [sn:%i]. %i insts"
00083                     "left.\n", tid, squash_seq_num,
00084                     skidBuffer[tid].size());
00085             break;
00086         }
00087         DPRINTF(InOrderStage, "[tid:%i]: Removing instruction, [sn:%i] "
00088                 "PC %s.\n", tid, skidBuffer[tid].front()->seqNum,
00089                 skidBuffer[tid].front()->pc);
00090         skidBuffer[tid].pop_front();
00091     }
00092 
00093     // Now that squash has propagated to the first stage,
00094     // Alert CPU to remove instructions from the CPU instruction list.
00095     // @todo: Move this to the CPU object.
00096     cpu->removeInstsUntil(squash_seq_num, tid);

00113 {
00114     list<ThreadID>::iterator threads = activeThreads->begin();
00115 
00116     //Check stall and squash signals.
00117     while (threads != activeThreads->end()) {
00118         ThreadID tid = *threads++;
00119         status_change =  checkSignalsAndUpdate(tid) || status_change;
00120     }
00121 
00122     while (instsProcessed < stageWidth)  {
00123         ThreadID tid = getFetchingThread(fetchPolicy);
00124 
00125         if (tid >= 0) {
00126             DPRINTF(InOrderStage, "Processing [tid:%i]\n",tid);
00127             processThread(status_change, tid);
00128             DPRINTF(InOrderStage, "Done Processing [tid:%i]\n",tid);
00129         } else {
00130             DPRINTF(InOrderStage, "No more threads to fetch from.\n");
00131             break;
00132         }
00133     }
00134 
00135     if (instsProcessed > 0) {
00136         ++runCycles;
00137         idle = false;        
00138     } else {
00139         ++idleCycles;
00140         idle = true;        
00141     }

00255     list<ThreadID>::iterator pri_iter = fetchPriorityList->begin();
00256     list<ThreadID>::iterator end      = fetchPriorityList->end();
00257 
00258     ThreadID high_pri;

00261         high_pri = *pri_iter;

00266             stageStatus[high_pri] == Idle ||
