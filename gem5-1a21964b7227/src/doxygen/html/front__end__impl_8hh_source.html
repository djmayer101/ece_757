00309 FrontEnd<Impl>::tick()
00310 {
00311     if (switchedOut)
00312         return;
00313 
00314     for (int insts_to_queue = numInstsReady[-latency];
00315          !instBuffer.empty() && insts_to_queue;
00316          --insts_to_queue)
00317     {
00318         DPRINTF(FE, "Transferring instruction [sn:%lli] to the feBuffer\n",
00319                 instBuffer.front()->seqNum);
00320         feBuffer.push_back(instBuffer.front());
00321         instBuffer.pop_front();
00322     }
00323 
00324     numInstsReady.advance();
00325 
00326     // @todo: Maybe I want to just have direct communication...
00327     if (fromCommit->doneSeqNum) {
00328         branchPred.update(fromCommit->doneSeqNum, 0);
00329     }
00330 
00331     IFQCount += instBufferSize;
00332     IFQFcount += instBufferSize == maxInstBufferSize;
00333 
00334     // Fetch cache line
00335     if (status == IcacheAccessComplete) {
00336         cacheBlkValid = true;
00337 
00338         status = Running;
00339 //        if (barrierInst)
00340 //            status = SerializeBlocked;
00341         if (freeRegs <= 0)
00342             status = RenameBlocked;
00343         checkBE();
00344     } else if (status == IcacheWaitResponse || status == IcacheWaitRetry) {
00345         DPRINTF(FE, "Still in Icache wait.\n");
00346         icacheStallCycles++;
00347         return;
00348     }
00349 
00350     if (status == RenameBlocked || status == SerializeBlocked ||
00351         status == TrapPending || status == BEBlocked) {
00352         // Will cause a one cycle bubble between changing state and
00353         // restarting.
00354         DPRINTF(FE, "In blocked status.\n");
00355 
00356         fetchBlockedCycles++;
00357 
00358         if (status == SerializeBlocked) {
00359             dispatchSerializeStallCycles++;
00360         }
00361         updateStatus();
00362         return;
00363     } else if (status == QuiescePending) {
00364         DPRINTF(FE, "Waiting for quiesce to execute or get squashed.\n");
00365         return;
00366     } else if (status != IcacheAccessComplete) {
00367         if (fetchCacheLineNextCycle) {
00368             Fault fault = fetchCacheLine();
00369             if (fault != NoFault) {
00370                 handleFault(fault);
00371                 fetchFault = fault;
00372                 return;
00373             }
00374             fetchCacheLineNextCycle = false;
00375         }
00376         // If miss, stall until it returns.
00377         if (status == IcacheWaitResponse || status == IcacheWaitRetry) {
00378             // Tell CPU to not tick me for now.
00379             return;
00380         }
00381     }
00382 
00383     fetchCycles++;
00384 
00385     int num_inst = 0;
00386 
00387     // Otherwise loop and process instructions.
00388     // One way to hack infinite width is to set width and maxInstBufferSize
00389     // both really high.  Inelegant, but probably will work.
00390     while (num_inst < width &&
00391            instBufferSize < maxInstBufferSize) {
00392         // Get instruction from cache line.
00393         DynInstPtr inst = getInstFromCacheline();
00394 
00395         if (!inst) {
00396             // PC is no longer in the cache line, end fetch.
00397             // Might want to check this at the end of the cycle so that
00398             // there's no cycle lost to checking for a new cache line.
00399             DPRINTF(FE, "Need to get new cache line\n");
00400             fetchCacheLineNextCycle = true;
00401             break;
00402         }
00403 
00404         processInst(inst);
00405 
00406         if (status == SerializeBlocked) {
00407             break;
00408         }
00409 
00410         // Possibly push into a time buffer that estimates the front end
00411         // latency
00412         instBuffer.push_back(inst);
00413         ++instBufferSize;
00414         numInstsReady[0]++;
00415         ++num_inst;
00416 
00417         if (inst->isQuiesce()) {
00418             status = QuiescePending;
00419             break;
00420         }
00421 
00422         if (inst->predTaken()) {
00423             // Start over with tick?
00424             break;
00425         } else if (freeRegs <= 0) {
00426             DPRINTF(FE, "Ran out of free registers to rename to!\n");
00427             status = RenameBlocked;
00428             break;
00429         } else if (serializeNext) {
00430             break;
00431         }
00432     }
00433 
00434     fetchNisnDist.sample(num_inst);
00435     checkBE();
00436 
00437     DPRINTF(FE, "Num insts processed: %i, Inst Buffer size: %i, Free "
00438             "Regs %i\n", num_inst, instBufferSize, freeRegs);
00439 }
