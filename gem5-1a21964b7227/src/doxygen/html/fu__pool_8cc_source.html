00176 int
00177 FUPool::getUnit(OpClass capability)
00178 {
00179     //  If this pool doesn't have the specified capability,
00180     //  return this information to the caller
00181     if (!capabilityList[capability])
00182         return -2;
00183 
00184     int fu_idx = fuPerCapList[capability].getFU();
00185     int start_idx = fu_idx;
00186 
00187     // Iterate through the circular queue if needed, stopping if we've reached
00188     // the first element again.
00189     while (unitBusy[fu_idx]) {
00190         fu_idx = fuPerCapList[capability].getFU();
00191         if (fu_idx == start_idx) {
00192             // No FU available
00193             return -1;
00194         }
00195     }
00196 
00197     assert(fu_idx < numFU);
00198 
00199     unitBusy[fu_idx] = true;
00200 
00201     return fu_idx;
00202 }
