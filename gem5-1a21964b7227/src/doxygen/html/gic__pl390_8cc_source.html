00298                 // this is a software interrupt from another CPU
00299                 if (!cpuSgiPending[active_int])
00300                     panic("Interrupt %d active but no CPU generated it?\n",
00301                             active_int);

00303                     // See which CPU generated the interrupt
00304                     uint8_t cpugen =
00305                         bits(cpuSgiPending[active_int], 7 + 8 * x, 8 * x);

00573 void
00574 Pl390::updateIntState(int hint)
00575 {
00576     for (int cpu = 0; cpu < CPU_MAX; cpu++) {
00577         if (!cpuEnabled[cpu])
00578             continue;
00579         if (cpu >= sys->numContexts())
00580             break;
00581 
00582         /*@todo use hint to do less work. */
00583         int highest_int = SPURIOUS_INT;
00584         // Priorities below that set in ICCPMR can be ignored
00585         uint8_t highest_pri = cpuPriority[cpu];
00586 
00587         // Check SGIs
00588         for (int swi = 0; swi < SGI_MAX; swi++) {
00589             if (!cpuSgiPending[swi])
00590                 continue;
00591             if (cpuSgiPending[swi] & genSwiMask(cpu))
00592                 if (highest_pri > bankedIntPriority[cpu][swi]) {
00593                     highest_pri = bankedIntPriority[cpu][swi];
00594                     highest_int = swi;
00595                 }
00596         }
00597 
00598         // Check PPIs
00599         if (cpuPpiPending[cpu]) {
00600         for (int ppi = 0; ppi < PPI_MAX; ppi++) {
00601             if (cpuPpiPending[cpu] & (1 << ppi))
00602                 if (highest_pri > bankedIntPriority[cpu][SGI_MAX + ppi]) {
00603                     highest_pri = bankedIntPriority[cpu][SGI_MAX + ppi];
00604                     highest_int = SGI_MAX + ppi;
00605                 }
00606             }
00607         }
00608 
00609         bool mp_sys = sys->numRunningContexts() > 1;
00610         // Check other ints
00611         for (int x = 0; x < (itLines/INT_BITS_MAX); x++) {
00612             if (intEnabled[x] & pendingInt[x]) {
00613                 for (int y = 0; y < INT_BITS_MAX; y++) {
00614                    uint32_t int_nm = x * INT_BITS_MAX + y;
00615                    DPRINTF(GIC, "Checking for interrupt# %d \n",int_nm);
00616                     /* Set current pending int as highest int for current cpu
00617                        if the interrupt's priority higher than current prioirty
00618                        and if currrent cpu is the target (for mp configs only)
00619                      */
00620                     if ((bits(intEnabled[x], y) & bits(pendingInt[x], y)) &&
00621                         (intPriority[int_nm] < highest_pri))
00622                         if ( (!mp_sys) || (cpuTarget[int_nm] & (1 << cpu))) {
00623                             highest_pri = intPriority[int_nm];
00624                             highest_int = int_nm;
00625                         }
00626                 }
00627             }
00628         }
00629 
00630         cpuHighestInt[cpu] = highest_int;
00631 
00632         if (highest_int == SPURIOUS_INT)
00633             continue;
00634 
00635         /* @todo make this work for more than one cpu, need to handle 1:N, N:N
00636          * models */
00637         if (enabled && cpuEnabled[cpu] && (highest_pri < cpuPriority[cpu]) &&
00638             !(activeInt[intNumToWord(highest_int)]
00639             & (1 << intNumToBit(highest_int)))) {
00640 
00641             DPRINTF(Interrupt, "Posting interrupt %d to cpu%d\n", highest_int,
00642                     cpu);
00643             postInt(cpu, curTick() + intLatency);
00644         }
00645     }
00646 }

00648 void
00649 Pl390::updateRunPri()
00650 {
00651     for (int cpu = 0; cpu < CPU_MAX; cpu++) {
00652         if (!cpuEnabled[cpu])
00653             continue;
00654         uint8_t maxPriority = 0xff;
00655         for (int i = 0; i < itLines; i++){
00656             if (i < SGI_MAX) {
00657                 if ((cpuSgiActive[i] & genSwiMask(cpu)) &&
00658                         (bankedIntPriority[cpu][i] < maxPriority))
00659                     maxPriority = bankedIntPriority[cpu][i];
00660             } else if (i < (SGI_MAX + PPI_MAX)) {
00661                 if ((cpuPpiActive[cpu] & ( 1 << (i - SGI_MAX))) &&
00662                         (bankedIntPriority[cpu][i] < maxPriority))
00663                     maxPriority = bankedIntPriority[cpu][i];
00664 
00665             } else {
00666                 if (activeInt[intNumToWord(i)] & (1 << intNumToBit(i)))
00667                     if (intPriority[i] < maxPriority)
00668                         maxPriority = intPriority[i];
00669             }
00670         }
00671         iccrpr[cpu] = maxPriority;
00672     }
