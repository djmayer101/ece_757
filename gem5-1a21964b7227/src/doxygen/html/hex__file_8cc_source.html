00042 /*
00043  * Load a Hex File into memory. Currently only used with MIPS
00044  * BARE_IRON mode. A hex file consists of [Address Data] tuples that
00045  * get directly loaded into physical memory. The address specified is
00046  * a word address (i.e., to get the byte address, shift left by 2) The
00047  * data is a full 32-bit hex value.
00048 */
00049 HexFile::HexFile(const string _filename)
00050     : filename(_filename)
00051 {
00052     fp = fopen(filename.c_str(), "r");
00053     if (fp == NULL)
00054         panic("Unable to open %s\n", filename.c_str());
00055 }

00063 bool
00064 HexFile::loadSections(PortProxy& memProxy)
00065 {
00066     char Line[64];
00067     Addr MemAddr;
00068     uint32_t Data;
00069     while (!feof(fp)) {
00070         char *ret = fgets(Line, sizeof(Line), fp);
00071         if (!ret)
00072             panic("malformed file");
00073         parseLine(Line, &MemAddr, &Data);
00074         if (MemAddr != 0) {
00075             // Now, write to memory
00076             memProxy.writeBlob(MemAddr << 2, (uint8_t *)&Data, sizeof(Data));
00077         }
00078     }
00079     return true;
00080 }

00082 void
00083 HexFile::parseLine(char *Str, Addr *A, uint32_t *D)
00084 {
00085     int i = 0;
00086     bool Flag = false;
00087     *A = 0;
00088     *D = 0;
00089     int Digit = 0;
00090     unsigned Number = 0;
00091 
00092     /* Skip white spaces */
00093     while (Str[i] != '\0' && Str[i]==' ')
00094         i++;
00095 
00096     /* Ok, we're at some character...process things */
00097     while (Str[i] != '\0') {
00098         if (Str[i] >= '0' && Str[i] <= '9') {
00099             Digit = Str[i] - '0';
00100         } else if (Str[i] >= 'a' && Str[i] <= 'f') {
00101             Digit = Str[i] - 'a' + 10;
00102         } else if (Str[i] >= 'A' && Str[i] <= 'F') {
00103           Digit=Str[i]-'A'+10;
00104         } else if (Str[i] == ' ' || Str[i] == '\n') {
00105             if (Number == 0)
00106                 return;
00107             if (Flag == false) {
00108                 *A = Number;
00109                 Number = 0;
00110                 Flag = true;
00111             } else {
00112                 *D = Number;
00113                 return;
00114             }
00115         } else {
00116             // Ok, we've encountered a non-hex character, cannot be a
00117             // valid line, skip and return 0's
00118             *A = 0;
00119             *D = 0;
00120             return;
00121         }
00122 
00123         Number <<= 4;
00124         Number += Digit;
00125         i++;
00126     }
00127 
00128     if (Flag != true) {
00129         *A = 0;
00130         *D = 0;
00131     } else {
00132         *D = Number;
00133     }
00134 }
