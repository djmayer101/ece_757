00111 void
00112 X86ISA::I82094AA::writeReg(uint8_t offset, uint32_t value)
00113 {
00114     if (offset == 0x0) {
00115         id = bits(value, 31, 24);
00116     } else if (offset == 0x1) {
00117         // The IOAPICVER register is read only.
00118     } else if (offset == 0x2) {
00119         arbId = bits(value, 31, 24);
00120     } else if (offset >= 0x10 && offset <= (0x10 + TableSize * 2)) {
00121         int index = (offset - 0x10) / 2;
00122         if (offset % 2) {
00123             redirTable[index].topDW = value;
00124             redirTable[index].topReserved = 0;
00125         } else {
00126             redirTable[index].bottomDW = value;
00127             redirTable[index].bottomReserved = 0;
00128         }
00129     } else {
00130         warn("Access to undefined I/O APIC register %#x.\n", offset);
00131     }
00132     DPRINTF(I82094AA,
00133             "Wrote %#x to I/O APIC register %#x .\n", value, offset);
00134 }

00136 uint32_t
00137 X86ISA::I82094AA::readReg(uint8_t offset)
00138 {
00139     uint32_t result = 0;
00140     if (offset == 0x0) {
00141         result = id << 24;
00142     } else if (offset == 0x1) {
00143         result = ((TableSize - 1) << 16) | APICVersion;
00144     } else if (offset == 0x2) {
00145         result = arbId << 24;
00146     } else if (offset >= 0x10 && offset <= (0x10 + TableSize * 2)) {
00147         int index = (offset - 0x10) / 2;
00148         if (offset % 2) {
00149             result = redirTable[index].topDW;
00150         } else {
00151             result = redirTable[index].bottomDW;
00152         }
00153     } else {
00154         warn("Access to undefined I/O APIC register %#x.\n", offset);
00155     }
00156     DPRINTF(I82094AA,
00157             "Read %#x from I/O APIC register %#x.\n", result, offset);
00158     return result;
00159 }

00161 void
00162 X86ISA::I82094AA::signalInterrupt(int line)
00163 {
00164     DPRINTF(I82094AA, "Received interrupt %d.\n", line);
00165     assert(line < TableSize);
00166     RedirTableEntry entry = redirTable[line];
00167     if (entry.mask) {
00168         DPRINTF(I82094AA, "Entry was masked.\n");
00169         return;
00170     } else {
00171         TriggerIntMessage message = 0;
00172         message.destination = entry.dest;
00173         if (entry.deliveryMode == DeliveryMode::ExtInt) {
00174             assert(extIntPic);
00175             message.vector = extIntPic->getVector();
00176         } else {
00177             message.vector = entry.vector;
00178         }
00179         message.deliveryMode = entry.deliveryMode;
00180         message.destMode = entry.destMode;
00181         message.level = entry.polarity;
00182         message.trigger = entry.trigger;
00183         ApicList apics;
00184         int numContexts = sys->numContexts();
00185         if (message.destMode == 0) {
00186             if (message.deliveryMode == DeliveryMode::LowestPriority) {
00187                 panic("Lowest priority delivery mode from the "
00188                         "IO APIC aren't supported in physical "
00189                         "destination mode.\n");
00190             }
00191             if (message.destination == 0xFF) {
00192                 for (int i = 0; i < numContexts; i++) {
00193                     apics.push_back(i);
00194                 }
00195             } else {
00196                 apics.push_back(message.destination);
00197             }
00198         } else {
00199             for (int i = 0; i < numContexts; i++) {
00200                 Interrupts *localApic = sys->getThreadContext(i)->
00201                     getCpuPtr()->getInterruptController();
00202                 if ((localApic->readReg(APIC_LOGICAL_DESTINATION) >> 24) &
00203                         message.destination) {
00204                     apics.push_back(localApic->getInitialApicId());
00205                 }
00206             }
00207             if (message.deliveryMode == DeliveryMode::LowestPriority &&
00208                     apics.size()) {
00209                 // The manual seems to suggest that the chipset just does
00210                 // something reasonable for these instead of actually using
00211                 // state from the local APIC. We'll just rotate an offset
00212                 // through the set of APICs selected above.
00213                 uint64_t modOffset = lowestPriorityOffset % apics.size();
00214                 lowestPriorityOffset++;
00215                 ApicList::iterator apicIt = apics.begin();
00216                 while (modOffset--) {
00217                     apicIt++;
00218                     assert(apicIt != apics.end());
00219                 }
00220                 int selected = *apicIt;
00221                 apics.clear();
00222                 apics.push_back(selected);
00223             }
00224         }
00225         intMasterPort.sendMessage(apics, message, sys->isTimingMode());
00226     }
00227 }
