00850 void
00851 IGbE::DescCache<T>::writeback(Addr aMask)
00852 {
00853     int curHead = descHead();
00854     int max_to_wb = usedCache.size();
00855 
00856     // Check if this writeback is less restrictive that the previous
00857     // and if so setup another one immediately following it
00858     if (wbOut) {
00859         if (aMask < wbAlignment) {
00860             moreToWb = true;
00861             wbAlignment = aMask;
00862         }
00863         DPRINTF(EthernetDesc,
00864                 "Writing back already in process, returning\n");
00865         return;
00866     }
00867 
00868     moreToWb = false;
00869     wbAlignment = aMask;
00870    
00871 
00872     DPRINTF(EthernetDesc, "Writing back descriptors head: %d tail: "
00873             "%d len: %d cachePnt: %d max_to_wb: %d descleft: %d\n",
00874             curHead, descTail(), descLen(), cachePnt, max_to_wb,
00875             descLeft());
00876 
00877     if (max_to_wb + curHead >= descLen()) {
00878         max_to_wb = descLen() - curHead;
00879         moreToWb = true;
00880         // this is by definition aligned correctly
00881     } else if (wbAlignment != 0) {
00882         // align the wb point to the mask
00883         max_to_wb = max_to_wb & ~wbAlignment;
00884     }
00885 
00886     DPRINTF(EthernetDesc, "Writing back %d descriptors\n", max_to_wb);
00887 
00888     if (max_to_wb <= 0) {
00889         if (usedCache.size())
00890             igbe->anBegin(annSmWb, "Wait Alignment", CPA::FL_WAIT);
00891         else
00892             igbe->anWe(annSmWb, annUsedCacheQ);
00893         return;
00894     }
00895 
00896     wbOut = max_to_wb;
00897 
00898     assert(!wbDelayEvent.scheduled()); 
00899     igbe->schedule(wbDelayEvent, curTick() + igbe->wbDelay);
00900     igbe->anBegin(annSmWb, "Prepare Writeback Desc");
00901 }

00935 {

00938     if (curFetching) {
00939         DPRINTF(EthernetDesc,

00959             igbe->anWf(annSmFetch, annDescQ);

00962     }

00970             max_to_fetch, descLeft());

00974         return;

01006 template<class T>
01007 void
01008 IGbE::DescCache<T>::fetchComplete()
01009 {
01010     T *newDesc;
01011     igbe->anBegin(annSmFetch, "Fetch Complete");
01012     for (int x = 0; x < curFetching; x++) {
01013         newDesc = new T;
01014         memcpy(newDesc, &fetchBuf[x], sizeof(T));
01015         unusedCache.push_back(newDesc);
01016         igbe->anDq(annSmFetch, annUnusedDescQ);
01017         igbe->anQ(annSmFetch, annUnusedCacheQ);
01018         igbe->anQ(annSmFetch, annDescQ);
01019     }
01020 
01021 
01022 #ifndef NDEBUG
01023     int oldCp = cachePnt;
01024 #endif
01025 
01026     cachePnt += curFetching;

01029         cachePnt = 0;
01030 
01031     curFetching = 0;
01032 
01033     DPRINTF(EthernetDesc, "Fetching complete cachePnt %d -> %d\n",
01034             oldCp, cachePnt);
01035 
01036     if ((descTail() >= cachePnt ? (descTail() - cachePnt) : (descLen() -
01037                                                              cachePnt)) == 0)
01038     {
01039         igbe->anWe(annSmFetch, annUnusedDescQ);
01040     } else if (!(size - usedCache.size() - unusedCache.size())) {
01041         igbe->anWf(annSmFetch, annDescQ);
01042     } else {
01043         igbe->anBegin(annSmFetch, "Wait", CPA::FL_WAIT);
01044     }
01045

01048 }

01053 {

01055     igbe->anBegin(annSmWb, "Finish Writeback");
01056 
01057     long  curHead = descHead();

01060 #endif
01061             
01062     for (int x = 0; x < wbOut; x++) {
01063         assert(usedCache.size());
01064         delete usedCache[0];

01252                 packet->length, igbe->regs.srrctl(), buf_len);

01255                        packet->length, &pktEvent, packet->data,
01256                        igbe->rxWriteDelay);
01257         desc->adv_wb.header_len = htole(0);
01258         desc->adv_wb.sph = htole(0);
01259         desc->adv_wb.pkt_len = htole((uint16_t)(pktPtr->length));
01260         break;
01261       case RXDT_ADV_SPLIT_A:
01262         int split_point;
01263         
01264         buf_len = igbe->regs.rctl.lpe() ? igbe->regs.srrctl.bufLen() :
01265             igbe->regs.rctl.descSize();

01271                 igbe->regs.srrctl(), desc->adv_read.hdr, hdr_len,
01272                 desc->adv_read.pkt, buf_len);
01273 
01274         split_point = hsplit(pktPtr);
01275 
01276         if (packet->length <= hdr_len) {
01277             bytesCopied = packet->length;
01278             assert(pkt_offset == 0);
01279             DPRINTF(EthernetDesc, "Hdr split: Entire packet in header\n");
01280             igbe->dmaWrite(pciToDma(desc->adv_read.hdr),
01281                            packet->length, &pktEvent, packet->data,
01282                            igbe->rxWriteDelay);
01283             desc->adv_wb.header_len = htole((uint16_t)packet->length);
01284             desc->adv_wb.sph = htole(0);
01285             desc->adv_wb.pkt_len = htole(0);
01286         } else if (split_point) {
01287             if (pkt_offset) {
01288                 // we are only copying some data, header/data has already been
01289                 // copied
01290                 int max_to_copy =
01291                     std::min(packet->length - pkt_offset, buf_len);
01292                 bytesCopied += max_to_copy;
01293                 DPRINTF(EthernetDesc,
01294                         "Hdr split: Continuing data buffer copy\n");
01295                 igbe->dmaWrite(pciToDma(desc->adv_read.pkt),
01296                                max_to_copy, &pktEvent,
01297                                packet->data + pkt_offset, igbe->rxWriteDelay);
01298                 desc->adv_wb.header_len = htole(0);
01299                 desc->adv_wb.pkt_len = htole((uint16_t)max_to_copy);
01300                 desc->adv_wb.sph = htole(0);
01301             } else {
01302                 int max_to_copy =
01303                     std::min(packet->length - split_point, buf_len);
01304                 bytesCopied += max_to_copy + split_point;
01305                 
01306                 DPRINTF(EthernetDesc, "Hdr split: splitting at %d\n",
01307                         split_point);
01308                 igbe->dmaWrite(pciToDma(desc->adv_read.hdr),
01309                                split_point, &pktHdrEvent,
01310                                packet->data, igbe->rxWriteDelay);
01311                 igbe->dmaWrite(pciToDma(desc->adv_read.pkt),
01312                                max_to_copy, &pktDataEvent,
01313                                packet->data + split_point, igbe->rxWriteDelay);
01314                 desc->adv_wb.header_len = htole(split_point);
01315                 desc->adv_wb.sph = 1;
01316                 desc->adv_wb.pkt_len = htole((uint16_t)(max_to_copy));
01317             }
01318         } else {
01319             panic("Header split not fitting within header buffer or "
01320                   "undecodable packet not fitting in header unsupported\n");
01321         }
01322         break;
01323       default:
01324         panic("Unimplemnted RX receive buffer type: %d\n",
01325               igbe->regs.srrctl.desctype());
01326     }

01333 {
01334     assert(unusedCache.size());

01336     desc = unusedCache.front();
01337 
01338     igbe->anBegin("RXS", "Update Desc");
01339 
01340     uint16_t crcfixup = igbe->regs.rctl.secrc() ? 0 : 4 ;

01345             (uint16_t)(pktPtr->length + crcfixup));

01353     uint8_t err = 0;
01354     uint16_t ext_err = 0;
01355     uint16_t csum = 0;
01356     uint16_t ptype = 0;
01357     uint16_t ip_id = 0;
01358 
01359     assert(bytesCopied <= pktPtr->length);
01360     if (bytesCopied == pktPtr->length)
01361         status |= RXDS_EOP;
01362 
01363     IpPtr ip(pktPtr);
01364 
01365     if (ip) {
01366         DPRINTF(EthernetDesc, "Proccesing Ip packet with Id=%d\n", ip->id());
01367         ptype |= RXDP_IPV4;
01368         ip_id = ip->id();
01369 
01370         if (igbe->regs.rxcsum.ipofld()) {
01371             DPRINTF(EthernetDesc, "Checking IP checksum\n");
01372             status |= RXDS_IPCS;
01373             csum = htole(cksum(ip));
01374             igbe->rxIpChecksums++;

01376                 err |= RXDE_IPE;
01377                 ext_err |= RXDEE_IPE;
01378                 DPRINTF(EthernetDesc, "Checksum is bad!!\n");
01379             }
01380         }
01381         TcpPtr tcp(ip);
01382         if (tcp && igbe->regs.rxcsum.tuofld()) {
01383             DPRINTF(EthernetDesc, "Checking TCP checksum\n");
01384             status |= RXDS_TCPCS;
01385             ptype |= RXDP_TCP;
01386             csum = htole(cksum(tcp));
01387             igbe->rxTcpChecksums++;
01388             if (cksum(tcp) != 0) {
01389                 DPRINTF(EthernetDesc, "Checksum is bad!!\n");
01390                 err |= RXDE_TCPE;
01391                 ext_err |= RXDEE_TCPE;
01392             }
01393         }
01394 
01395         UdpPtr udp(ip);
01396         if (udp && igbe->regs.rxcsum.tuofld()) {
01397             DPRINTF(EthernetDesc, "Checking UDP checksum\n");
01398             status |= RXDS_UDPCS;
01399             ptype |= RXDP_UDP;
01400             csum = htole(cksum(udp));
01401             igbe->rxUdpChecksums++;
01402             if (cksum(udp) != 0) {
01403                 DPRINTF(EthernetDesc, "Checksum is bad!!\n");
01404                 ext_err |= RXDEE_TCPE;
01405                 err |= RXDE_TCPE;
01406             }
01407         }
01408     } else { // if ip
01409         DPRINTF(EthernetSM, "Proccesing Non-Ip packet\n");
01410     }
01411 
01412     switch (igbe->regs.srrctl.desctype()) {
01413       case RXDT_LEGACY:
01414         desc->legacy.len = htole((uint16_t)(pktPtr->length + crcfixup));
01415         desc->legacy.status = htole(status);
01416         desc->legacy.errors = htole(err);
01417         // No vlan support at this point... just set it to 0
01418         desc->legacy.vlan = 0;
01419         break;
01420       case RXDT_ADV_SPLIT_A:
01421       case RXDT_ADV_ONEBUF:
01422         desc->adv_wb.rss_type = htole(0);
01423         desc->adv_wb.pkt_type = htole(ptype);
01424         if (igbe->regs.rxcsum.pcsd()) {
01425             // no rss support right now
01426             desc->adv_wb.rss_hash = htole(0);
01427         } else {
01428             desc->adv_wb.id = htole(ip_id);
01429             desc->adv_wb.csum = htole(csum);
01430         }
01431         desc->adv_wb.status = htole(status);
01432         desc->adv_wb.errors = htole(ext_err);
01433         // no vlan support
01434         desc->adv_wb.vlan_tag = htole(0); 
01435         break;
01436       default:
01437         panic("Unimplemnted RX receive buffer type %d\n",
01438               igbe->regs.srrctl.desctype());
01439     }
01440 
01441     DPRINTF(EthernetDesc, "Descriptor complete w0: %#x w1: %#x\n",
01442             desc->adv_read.pkt, desc->adv_read.hdr);
01443 
01444     if (bytesCopied == pktPtr->length) {
01445         DPRINTF(EthernetDesc,
01446                 "Packet completely written to descriptor buffers\n");
01447         // Deal with the rx timer interrupts
01448         if (igbe->regs.rdtr.delay()) {
01449             Tick delay = igbe->regs.rdtr.delay() * igbe->intClock();
01450             DPRINTF(EthernetSM, "RXS: Scheduling DTR for %d\n", delay);
01451             igbe->reschedule(igbe->rdtrEvent, curTick() + delay);
01452         }
01453 
01454         if (igbe->regs.radv.idv()) {
01455             Tick delay = igbe->regs.radv.idv() * igbe->intClock();
01456             DPRINTF(EthernetSM, "RXS: Scheduling ADV for %d\n", delay);
01457             if (!igbe->radvEvent.scheduled()) {
01458                 igbe->schedule(igbe->radvEvent, curTick() + delay);
01459             }
01460         }
01461 
01462         // if neither radv or rdtr, maybe itr is set...
01463         if (!igbe->regs.rdtr.delay() && !igbe->regs.radv.idv()) {
01464             DPRINTF(EthernetSM,
01465                     "RXS: Receive interrupt delay disabled, posting IT_RXT\n");
01466             igbe->postInterrupt(IT_RXT);
01467         }
01468 
01469         // If the packet is small enough, interrupt appropriately
01470         // I wonder if this is delayed or not?!
01471         if (pktPtr->length <= igbe->regs.rsrpd.idv()) {
01472             DPRINTF(EthernetSM,
01473                     "RXS: Posting IT_SRPD beacuse small packet received\n");
01474             igbe->postInterrupt(IT_SRPD);
01475         }
01476         bytesCopied = 0;
01477     }
01478 
01479     pktPtr = NULL;
01480     igbe->checkDrain();
01481     enableSm();
01482     pktDone = true;
01483 
01484     igbe->anBegin("RXS", "Done Updating Desc");
01485     DPRINTF(EthernetDesc, "Processing of this descriptor complete\n");
01486     igbe->anDq("RXS", annUnusedCacheQ);
01487     unusedCache.pop_front();
01488     igbe->anQ("RXS", annUsedCacheQ);
01489     usedCache.push_back(desc);
01490 }

01807     // no support for vlans
01808     assert(!TxdOp::vle(desc));
01809 
01810     // we only support single packet descriptors at this point
01811     if (!useTso)
01812         assert(TxdOp::eop(desc));
01813 
01814     // set that this packet is done
01815     if (TxdOp::rs(desc))
01816         TxdOp::setDd(desc);
01817 
01818     DPRINTF(EthernetDesc, "TxDescriptor data d1: %#llx d2: %#llx\n",
01819             desc->d1, desc->d2);
01820 
01821     if (useTso) {
01822         IpPtr ip(pktPtr);
01823         if (ip) {
01824             DPRINTF(EthernetDesc, "TSO: Modifying IP header. Id + %d\n",
01825                     tsoPkts);
01826             ip->id(ip->id() + tsoPkts++);
01827             ip->len(pktPtr->length - EthPtr(pktPtr)->size()); 
01828         
01829             TcpPtr tcp(ip);
01830             if (tcp) {
01831                 DPRINTF(EthernetDesc,
01832                         "TSO: Modifying TCP header. old seq %d + %d\n",
01833                         tcp->seq(), tsoPrevSeq);
01834                 tcp->seq(tcp->seq() + tsoPrevSeq);
01835                 if (tsoUsedLen != tsoTotalLen)
01836                     tcp->flags(tcp->flags() & ~9); // clear fin & psh
01837             }
01838             UdpPtr udp(ip);
01839             if (udp) {
01840                 DPRINTF(EthernetDesc, "TSO: Modifying UDP header.\n");
01841                 udp->len(pktPtr->length - EthPtr(pktPtr)->size());
01842             }
01843         }
01844         tsoPrevSeq = tsoUsedLen;
01845     }
01846 
01847     if (DTRACE(EthernetDesc)) {
01848         IpPtr ip(pktPtr);
01849         if (ip)
01850             DPRINTF(EthernetDesc, "Proccesing Ip packet with Id=%d\n",
01851                     ip->id());
01852         else
01853             DPRINTF(EthernetSM, "Proccesing Non-Ip packet\n");
01854     }
01855 
01856     // Checksums are only ofloaded for new descriptor types
01857     if (TxdOp::isData(desc) && ( TxdOp::ixsm(desc) || TxdOp::txsm(desc)) ) {
01858         DPRINTF(EthernetDesc, "Calculating checksums for packet\n");
01859         IpPtr ip(pktPtr);
01860         assert(ip);
01861         if (TxdOp::ixsm(desc)) {
01862             ip->sum(0);
01863             ip->sum(cksum(ip));
01864             igbe->txIpChecksums++;
01865             DPRINTF(EthernetDesc, "Calculated IP checksum\n");
01866         }
01867         if (TxdOp::txsm(desc)) {
01868             TcpPtr tcp(ip);
01869             UdpPtr udp(ip);
01870             if (tcp) {
01871                 tcp->sum(0);
01872                 tcp->sum(cksum(tcp));
01873                 igbe->txTcpChecksums++;
01874                 DPRINTF(EthernetDesc, "Calculated TCP checksum\n");
01875             } else if (udp) {
01876                 assert(udp);
01877                 udp->sum(0);
01878                 udp->sum(cksum(udp));
01879                 igbe->txUdpChecksums++;
01880                 DPRINTF(EthernetDesc, "Calculated UDP checksum\n");
01881             } else {
01882                 panic("Told to checksum, but don't know how\n");
01883             }
01884         }
01885     }
01886 
01887     if (TxdOp::ide(desc)) {
01888         // Deal with the rx timer interrupts
01889         DPRINTF(EthernetDesc, "Descriptor had IDE set\n");
01890         if (igbe->regs.tidv.idv()) {
01891             Tick delay = igbe->regs.tidv.idv() * igbe->intClock();
01892             DPRINTF(EthernetDesc, "setting tidv\n");
01893             igbe->reschedule(igbe->tidvEvent, curTick() + delay, true);
01894         }
01895 
01896         if (igbe->regs.tadv.idv() && igbe->regs.tidv.idv()) {
01897             Tick delay = igbe->regs.tadv.idv() * igbe->intClock();
01898             DPRINTF(EthernetDesc, "setting tadv\n");
01899             if (!igbe->tadvEvent.scheduled()) {
01900                 igbe->schedule(igbe->tadvEvent, curTick() + delay);
01901             }
01902         }
01903     }
01904 
01905 
01906     if (!useTso ||  TxdOp::getLen(desc) == tsoDescBytesUsed) {
01907         DPRINTF(EthernetDesc, "Descriptor Done\n");
01908         igbe->anDq("TXS", annUnusedCacheQ);
01909         unusedCache.pop_front();
01910         igbe->anQ("TXS", annUsedCacheQ);
01911         usedCache.push_back(desc);
01912         tsoDescBytesUsed = 0;
01913     }
01914 
01915     if (useTso && tsoUsedLen == tsoTotalLen)
01916         useTso = false;
01917 
01918 
01919     DPRINTF(EthernetDesc,
01920             "------Packet of %d bytes ready for transmission-------\n",
01921             pktPtr->length);
01922     pktDone = true;
01923     pktWaiting = false;
01924     pktPtr = NULL;
01925     tsoPktHasHeader = false;
01926 
01927     if (igbe->regs.txdctl.wthresh() == 0) {
01928         igbe->anBegin("TXS", "Desc Writeback");
01929         DPRINTF(EthernetDesc, "WTHRESH == 0, writing back descriptor\n");
01930         writeback(0);
01931     } else if (!igbe->regs.txdctl.gran() && igbe->regs.txdctl.wthresh() <=
01932                descInBlock(usedCache.size())) {
01933         DPRINTF(EthernetDesc, "used > WTHRESH, writing back descriptor\n");
01934         igbe->anBegin("TXS", "Desc Writeback");
01935         writeback((igbe->cacheBlockSize()-1)>>4);
01936     } else if (igbe->regs.txdctl.wthresh() <= usedCache.size()) {
01937         DPRINTF(EthernetDesc, "used > WTHRESH, writing back descriptor\n");
01938         igbe->anBegin("TXS", "Desc Writeback");
01939         writeback((igbe->cacheBlockSize()-1)>>4);
01940     }
01941 
01942     enableSm();
01943     igbe->checkDrain();
01944 }

02112 void
02113 IGbE::txStateMachine()
02114 {
02115     if (!regs.tctl.en()) {
02116         txTick = false;
02117         DPRINTF(EthernetSM, "TXS: TX disabled, stopping ticking\n");
02118         return;
02119     }
02120 
02121     // If we have a packet available and it's length is not 0 (meaning it's not
02122     // a multidescriptor packet) put it in the fifo, otherwise an the next
02123     // iteration we'll get the rest of the data
02124     if (txPacket && txDescCache.packetAvailable()
02125         && !txDescCache.packetMultiDesc() && txPacket->length) {
02126         anQ("TXS", "TX FIFO Q");
02127         DPRINTF(EthernetSM, "TXS: packet placed in TX FIFO\n");
02128 #ifndef NDEBUG
02129         bool success =
02130 #endif
02131             txFifo.push(txPacket);
02132         txFifoTick = true && !drainManager;
02133         assert(success);
02134         txPacket = NULL;
02135         anBegin("TXS", "Desc Writeback");
02136         txDescCache.writeback((cacheBlockSize()-1)>>4);
02137         return;
02138     }
02139 
02140     // Only support descriptor granularity
02141     if (regs.txdctl.lwthresh() &&
02142         txDescCache.descLeft() < (regs.txdctl.lwthresh() * 8)) {
02143         DPRINTF(EthernetSM, "TXS: LWTHRESH caused posting of TXDLOW\n");
02144         postInterrupt(IT_TXDLOW);
02145     }
02146 
02147     if (!txPacket) {
02148         txPacket = new EthPacketData(16384);
02149     }
02150 
02151     if (!txDescCache.packetWaiting()) {
02152         if (txDescCache.descLeft() == 0) {
02153             postInterrupt(IT_TXQE);
02154             anBegin("TXS", "Desc Writeback");
02155             txDescCache.writeback(0);
02156             anBegin("TXS", "Desc Fetch");
02157             anWe("TXS", txDescCache.annUnusedCacheQ);
02158             txDescCache.fetchDescriptors();
02159             DPRINTF(EthernetSM, "TXS: No descriptors left in ring, forcing "
02160                     "writeback stopping ticking and posting TXQE\n");
02161             txTick = false;
02162             return;
02163         }
02164 
02165 
02166         if (!(txDescCache.descUnused())) {
02167             anBegin("TXS", "Desc Fetch");
02168             txDescCache.fetchDescriptors();
02169             anWe("TXS", txDescCache.annUnusedCacheQ);
02170             DPRINTF(EthernetSM, "TXS: No descriptors available in cache, "
02171                     "fetching and stopping ticking\n");
02172             txTick = false;
02173             return;
02174         }
02175         anPq("TXS", txDescCache.annUnusedCacheQ);
02176 
02177 
02178         txDescCache.processContextDesc();
02179         if (txDescCache.packetWaiting()) {
02180             DPRINTF(EthernetSM,
02181                     "TXS: Fetching TSO header, stopping ticking\n");
02182             txTick = false;
02183             return;
02184         }
02185 
02186         unsigned size = txDescCache.getPacketSize(txPacket);
02187         if (size > 0 && txFifo.avail() > size) {
02188             anRq("TXS", "TX FIFO Q");
02189             anBegin("TXS", "DMA Packet");
02190             DPRINTF(EthernetSM, "TXS: Reserving %d bytes in FIFO and "
02191                     "beginning DMA of next packet\n", size);
02192             txFifo.reserve(size);
02193             txDescCache.getPacketData(txPacket);
02194         } else if (size == 0) {
02195             DPRINTF(EthernetSM, "TXS: getPacketSize returned: %d\n", size);
02196             DPRINTF(EthernetSM,
02197                     "TXS: No packets to get, writing back used descriptors\n");
02198             anBegin("TXS", "Desc Writeback");
02199             txDescCache.writeback(0);
02200         } else {
02201             anWf("TXS", "TX FIFO Q");
02202             DPRINTF(EthernetSM, "TXS: FIFO full, stopping ticking until space "
02203                     "available in FIFO\n");
02204             txTick = false;
02205         }
02206 
02207 
02208         return;
02209     }
02210     DPRINTF(EthernetSM, "TXS: Nothing to do, stopping ticking\n");
02211     txTick = false;
02212 }

02262 void
02263 IGbE::rxStateMachine()
02264 {
02265     if (!regs.rctl.en()) {
02266         rxTick = false;
02267         DPRINTF(EthernetSM, "RXS: RX disabled, stopping ticking\n");
02268         return;
02269     }
02270 
02271     // If the packet is done check for interrupts/descriptors/etc
02272     if (rxDescCache.packetDone()) {
02273         rxDmaPacket = false;
02274         DPRINTF(EthernetSM, "RXS: Packet completed DMA to memory\n");
02275         int descLeft = rxDescCache.descLeft();
02276         DPRINTF(EthernetSM, "RXS: descLeft: %d rdmts: %d rdlen: %d\n",
02277                 descLeft, regs.rctl.rdmts(), regs.rdlen());
02278         switch (regs.rctl.rdmts()) {
02279           case 2: if (descLeft > .125 * regs.rdlen()) break;
02280           case 1: if (descLeft > .250 * regs.rdlen()) break;
02281           case 0: if (descLeft > .500 * regs.rdlen())  break;
02282             DPRINTF(Ethernet, "RXS: Interrupting (RXDMT) "
02283                     "because of descriptors left\n");
02284             postInterrupt(IT_RXDMT);
02285             break;
02286         }
02287 
02288         if (rxFifo.empty())
02289             rxDescCache.writeback(0);
02290 
02291         if (descLeft == 0) {
02292             anBegin("RXS", "Writeback Descriptors");
02293             rxDescCache.writeback(0);
02294             DPRINTF(EthernetSM, "RXS: No descriptors left in ring, forcing"
02295                     " writeback and stopping ticking\n");
02296             rxTick = false;
02297         }
02298 
02299         // only support descriptor granulaties
02300         assert(regs.rxdctl.gran());
02301 
02302         if (regs.rxdctl.wthresh() >= rxDescCache.descUsed()) {
02303             DPRINTF(EthernetSM,
02304                     "RXS: Writing back because WTHRESH >= descUsed\n");
02305             anBegin("RXS", "Writeback Descriptors");
02306             if (regs.rxdctl.wthresh() < (cacheBlockSize()>>4))
02307                 rxDescCache.writeback(regs.rxdctl.wthresh()-1);
02308             else
02309                 rxDescCache.writeback((cacheBlockSize()-1)>>4);
02310         }
02311 
02312         if ((rxDescCache.descUnused() < regs.rxdctl.pthresh()) &&
02313             ((rxDescCache.descLeft() - rxDescCache.descUnused()) >
02314              regs.rxdctl.hthresh())) {
02315             DPRINTF(EthernetSM, "RXS: Fetching descriptors because "
02316                     "descUnused < PTHRESH\n");
02317             anBegin("RXS", "Fetch Descriptors");
02318             rxDescCache.fetchDescriptors();
02319         }
02320 
02321         if (rxDescCache.descUnused() == 0) {
02322             anBegin("RXS", "Fetch Descriptors");
02323             rxDescCache.fetchDescriptors();
02324             anWe("RXS", rxDescCache.annUnusedCacheQ);
02325             DPRINTF(EthernetSM, "RXS: No descriptors available in cache, "
02326                     "fetching descriptors and stopping ticking\n");
02327             rxTick = false;
02328         }
02329         return;
02330     }
02331 
02332     if (rxDmaPacket) {
02333         DPRINTF(EthernetSM,
02334                 "RXS: stopping ticking until packet DMA completes\n");
02335         rxTick = false;
02336         return;
02337     }
02338 
02339     if (!rxDescCache.descUnused()) {
02340         anBegin("RXS", "Fetch Descriptors");
02341         rxDescCache.fetchDescriptors();
02342         anWe("RXS", rxDescCache.annUnusedCacheQ);
02343         DPRINTF(EthernetSM, "RXS: No descriptors available in cache, "
02344                 "stopping ticking\n");
02345         rxTick = false;
02346         DPRINTF(EthernetSM, "RXS: No descriptors available, fetching\n");
02347         return;
02348     }
02349     anPq("RXS", rxDescCache.annUnusedCacheQ);
02350 
02351     if (rxFifo.empty()) {
02352         anWe("RXS", "RX FIFO Q");
02353         DPRINTF(EthernetSM, "RXS: RxFIFO empty, stopping ticking\n");
02354         rxTick = false;
02355         return;
02356     }
02357     anPq("RXS", "RX FIFO Q");
02358     anBegin("RXS", "Get Desc");
02359 
02360     EthPacketPtr pkt;
02361     pkt = rxFifo.front();
02362 
02363 
02364     pktOffset = rxDescCache.writePacket(pkt, pktOffset);
02365     DPRINTF(EthernetSM, "RXS: Writing packet into memory\n");
02366     if (pktOffset == pkt->length) {
02367         anBegin( "RXS", "FIFO Dequeue");
02368         DPRINTF(EthernetSM, "RXS: Removing packet from FIFO\n");
02369         pktOffset = 0;
02370         anDq("RXS", "RX FIFO Q");
02371         rxFifo.pop();
02372     }
02373 
02374     DPRINTF(EthernetSM, "RXS: stopping ticking until packet DMA completes\n");
02375     rxTick = false;
02376     rxDmaPacket = true;
02377     anBegin("RXS", "DMA Packet");
02378 }
