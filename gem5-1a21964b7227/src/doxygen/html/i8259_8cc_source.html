00049 Tick
00050 X86ISA::I8259::read(PacketPtr pkt)
00051 {
00052     assert(pkt->getSize() == 1);
00053     switch(pkt->getAddr() - pioAddr)
00054     {
00055       case 0x0:
00056         if (readIRR) {
00057             DPRINTF(I8259, "Reading IRR as %#x.\n", IRR);
00058             pkt->set(IRR);
00059         } else {
00060             DPRINTF(I8259, "Reading ISR as %#x.\n", ISR);
00061             pkt->set(ISR);
00062         }
00063         break;
00064       case 0x1:
00065         DPRINTF(I8259, "Reading IMR as %#x.\n", IMR);
00066         pkt->set(IMR);
00067         break;
00068     }
00069     pkt->makeAtomicResponse();
00070     return latency;
00071 }

00073 Tick
00074 X86ISA::I8259::write(PacketPtr pkt)
00075 {
00076     assert(pkt->getSize() == 1);
00077     uint8_t val = pkt->get<uint8_t>();
00078     switch (pkt->getAddr() - pioAddr) {
00079       case 0x0:
00080         if (bits(val, 4)) {
00081             DPRINTF(I8259, "Received initialization command word 1.\n");
00082             IMR = 0;
00083             edgeTriggered = bits(val, 3);
00084             DPRINTF(I8259, "%s triggered mode.\n",
00085                     edgeTriggered ? "Edge" : "Level");
00086             cascadeMode = !bits(val, 1);
00087             DPRINTF(I8259, "%s mode.\n",
00088                     cascadeMode ? "Cascade" : "Single");
00089             expectICW4 = bits(val, 0);
00090             if (!expectICW4) {
00091                 autoEOI = false;
00092             }
00093             initControlWord = 1;
00094             DPRINTF(I8259, "Expecting %d more bytes.\n", expectICW4 ? 3 : 2);
00095         } else if (bits(val, 4, 3) == 0) {
00096             DPRINTF(I8259, "Received operation command word 2.\n");
00097             switch (bits(val, 7, 5)) {
00098               case 0x0:
00099                 DPRINTF(I8259,
00100                         "Subcommand: Rotate in auto-EOI mode (clear).\n");
00101                 break;
00102               case 0x1:
00103                 {
00104                     int line = findMsbSet(ISR);
00105                     DPRINTF(I8259, "Subcommand: Nonspecific EOI on line %d.\n",
00106                             line);
00107                     handleEOI(line);
00108                 }
00109                 break;
00110               case 0x2:
00111                 DPRINTF(I8259, "Subcommand: No operation.\n");
00112                 break;
00113               case 0x3:
00114                 {
00115                     int line = bits(val, 2, 0);
00116                     DPRINTF(I8259, "Subcommand: Specific EIO on line %d.\n",
00117                             line);
00118                     handleEOI(line);
00119                 }
00120                 break;
00121               case 0x4:
00122                 DPRINTF(I8259, "Subcommand: Rotate in auto-EOI mode (set).\n");
00123                 break;
00124               case 0x5:
00125                 DPRINTF(I8259, "Subcommand: Rotate on nonspecific EOI.\n");
00126                 break;
00127               case 0x6:
00128                 DPRINTF(I8259, "Subcommand: Set priority command.\n");
00129                 DPRINTF(I8259, "Lowest: IRQ%d   Highest IRQ%d.\n",
00130                         bits(val, 2, 0), (bits(val, 2, 0) + 1) % 8);
00131                 break;
00132               case 0x7:
00133                 DPRINTF(I8259, "Subcommand: Rotate on specific EOI.\n");
00134                 DPRINTF(I8259, "Lowest: IRQ%d   Highest IRQ%d.\n",
00135                         bits(val, 2, 0), (bits(val, 2, 0) + 1) % 8);
00136                 break;
00137             }
00138         } else if (bits(val, 4, 3) == 1) {
00139             DPRINTF(I8259, "Received operation command word 3.\n");
00140             if (bits(val, 7)) {
00141                 DPRINTF(I8259, "%s special mask mode.\n",
00142                         bits(val, 6) ? "Set" : "Clear");
00143             }
00144             if (bits(val, 1)) {
00145                 readIRR = bits(val, 0);
00146                 DPRINTF(I8259, "Read %s.\n", readIRR ? "IRR" : "ISR");
00147             }
00148         }
00149         break;
00150       case 0x1:
00151         switch (initControlWord) {
00152           case 0x0:
00153             DPRINTF(I8259, "Received operation command word 1.\n");
00154             DPRINTF(I8259, "Wrote IMR value %#x.\n", val);
00155             IMR = val;
00156             break;
00157           case 0x1:
00158             DPRINTF(I8259, "Received initialization command word 2.\n");
00159             vectorOffset = val & ~mask(3);
00160             DPRINTF(I8259, "Responsible for vectors %#x-%#x.\n",
00161                     vectorOffset, vectorOffset | mask(3));
00162             if (cascadeMode) {
00163                 initControlWord++;
00164             } else {
00165                 cascadeBits = 0;
00166                 initControlWord = 0;
00167             }
00168             break;
00169           case 0x2:
00170             DPRINTF(I8259, "Received initialization command word 3.\n");
00171             if (mode == Enums::I8259Master) {
00172                 DPRINTF(I8259, "Slaves attached to IRQs:%s%s%s%s%s%s%s%s\n",
00173                         bits(val, 0) ? " 0" : "",
00174                         bits(val, 1) ? " 1" : "",
00175                         bits(val, 2) ? " 2" : "",
00176                         bits(val, 3) ? " 3" : "",
00177                         bits(val, 4) ? " 4" : "",
00178                         bits(val, 5) ? " 5" : "",
00179                         bits(val, 6) ? " 6" : "",
00180                         bits(val, 7) ? " 7" : "");
00181                 cascadeBits = val;
00182             } else {
00183                 DPRINTF(I8259, "Slave ID is %d.\n", val & mask(3));
00184                 cascadeBits = val & mask(3);
00185             }
00186             if (expectICW4)
00187                 initControlWord++;
00188             else
00189                 initControlWord = 0;
00190             break;
00191           case 0x3:
00192             DPRINTF(I8259, "Received initialization command word 4.\n");
00193             if (bits(val, 4)) {
00194                 DPRINTF(I8259, "Special fully nested mode.\n");
00195             } else {
00196                 DPRINTF(I8259, "Not special fully nested mode.\n");
00197             }
00198             if (bits(val, 3) == 0) {
00199                 DPRINTF(I8259, "Nonbuffered.\n");
00200             } else if (bits(val, 2) == 0) {
00201                 DPRINTF(I8259, "Buffered.\n");
00202             } else {
00203                 DPRINTF(I8259, "Unrecognized buffer mode.\n");
00204             }
00205             autoEOI = bits(val, 1);
00206             DPRINTF(I8259, "%s End Of Interrupt.\n",
00207                     autoEOI ? "Automatic" : "Normal");
00208 
00209             DPRINTF(I8259, "%s mode.\n", bits(val, 0) ? "80x86" : "MCX-80/85");
00210             initControlWord = 0;
00211             break;
00212         }
00213         break;
00214     }
00215     pkt->makeAtomicResponse();
00216     return latency;
00217 }

00249     if (line >= NumLines)
00250         fatal("Line number %d doesn't exist. The max is %d.\n",

00282 int
00283 X86ISA::I8259::getVector()
00284 {
00285     /*
00286      * This code only handles one slave. Since that's how the PC platform
00287      * always uses the 8259 PIC, there shouldn't be any need for more. If
00288      * there -is- a need for more for some reason, "slave" can become a
00289      * vector of slaves.
00290      */
00291     int line = findMsbSet(IRR);
00292     IRR &= ~(1 << line);
00293     DPRINTF(I8259, "Interrupt %d was accepted.\n", line);
00294     if (autoEOI) {
00295         handleEOI(line);
00296     } else {
00297         ISR |= 1 << line;
00298     }
00299     if (slave && bits(cascadeBits, line)) {
00300         DPRINTF(I8259, "Interrupt was from slave who will "
00301                 "provide the vector.\n");
00302         return slave->getVector();
00303     }
00304     return line | vectorOffset;
00305 }
