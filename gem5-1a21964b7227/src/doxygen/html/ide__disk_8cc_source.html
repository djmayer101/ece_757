00730 {
00731     switch (devState) {
00732       case Device_Srst:
00733         if (action == ACT_SRST_SET) {
00734             // set the BSY bit
00735             status |= STATUS_BSY_BIT;
00736         } else if (action == ACT_SRST_CLEAR) {
00737             // clear the BSY bit
00738             status &= ~STATUS_BSY_BIT;
00739 
00740             // reset the device state
00741             reset(devID);
00742         }
00743         break;
00744 
00745       case Device_Idle_S:
00746         if (action == ACT_SELECT_WRITE && !isDEVSelect()) {
00747             devState = Device_Idle_NS;
00748         } else if (action == ACT_CMD_WRITE) {
00749             startCommand();
00750         }
00751 
00752         break;
00753 
00754       case Device_Idle_SI:
00755         if (action == ACT_SELECT_WRITE && !isDEVSelect()) {
00756             devState = Device_Idle_NS;
00757             intrClear();
00758         } else if (action == ACT_STAT_READ || isIENSet()) {
00759             devState = Device_Idle_S;
00760             intrClear();
00761         } else if (action == ACT_CMD_WRITE) {
00762             intrClear();
00763             startCommand();
00764         }
00765 
00766         break;
00767 
00768       case Device_Idle_NS:
00769         if (action == ACT_SELECT_WRITE && isDEVSelect()) {
00770             if (!isIENSet() && intrPending) {
00771                 devState = Device_Idle_SI;
00772                 intrPost();
00773             }
00774             if (isIENSet() || !intrPending) {
00775                 devState = Device_Idle_S;
00776             }
00777         }
00778         break;
00779 
00780       case Command_Execution:
00781         if (action == ACT_CMD_COMPLETE) {
00782             // clear the BSY bit
00783             setComplete();
00784 
00785             if (!isIENSet()) {
00786                 devState = Device_Idle_SI;
00787                 intrPost();
00788             } else {
00789                 devState = Device_Idle_S;
00790             }
00791         }
00792         break;
00793 
00794       case Prepare_Data_In:
00795         if (action == ACT_CMD_ERROR) {
00796             // clear the BSY bit
00797             setComplete();
00798 
00799             if (!isIENSet()) {
00800                 devState = Device_Idle_SI;
00801                 intrPost();
00802             } else {
00803                 devState = Device_Idle_S;
00804             }
00805         } else if (action == ACT_DATA_READY) {
00806             // clear the BSY bit
00807             status &= ~STATUS_BSY_BIT;
00808             // set the DRQ bit
00809             status |= STATUS_DRQ_BIT;
00810 
00811             // copy the data into the data buffer
00812             if (cmdReg.command == WDCC_IDENTIFY) {
00813                 // Reset the drqBytes for this block
00814                 drqBytesLeft = sizeof(struct ataparams);
00815 
00816                 memcpy((void *)dataBuffer, (void *)&driveID,
00817                        sizeof(struct ataparams));
00818             } else {
00819                 // Reset the drqBytes for this block
00820                 drqBytesLeft = SectorSize;
00821 
00822                 readDisk(curSector++, dataBuffer);
00823             }
00824 
00825             // put the first two bytes into the data register
00826             memcpy((void *)&cmdReg.data, (void *)dataBuffer,
00827                    sizeof(uint16_t));
00828 
00829             if (!isIENSet()) {
00830                 devState = Data_Ready_INTRQ_In;
00831                 intrPost();
00832             } else {
00833                 devState = Transfer_Data_In;
00834             }
00835         }
00836         break;
00837 
00838       case Data_Ready_INTRQ_In:
00839         if (action == ACT_STAT_READ) {
00840             devState = Transfer_Data_In;
00841             intrClear();
00842         }
00843         break;
00844 
00845       case Transfer_Data_In:
00846         if (action == ACT_DATA_READ_BYTE || action == ACT_DATA_READ_SHORT) {
00847             if (action == ACT_DATA_READ_BYTE) {
00848                 panic("DEBUG: READING DATA ONE BYTE AT A TIME!\n");
00849             } else {
00850                 drqBytesLeft -= 2;
00851                 cmdBytesLeft -= 2;
00852 
00853                 // copy next short into data registers
00854                 if (drqBytesLeft)
00855                     memcpy((void *)&cmdReg.data,
00856                            (void *)&dataBuffer[SectorSize - drqBytesLeft],
00857                            sizeof(uint16_t));
00858             }
00859 
00860             if (drqBytesLeft == 0) {
00861                 if (cmdBytesLeft == 0) {
00862                     // Clear the BSY bit
00863                     setComplete();
00864                     devState = Device_Idle_S;
00865                 } else {
00866                     devState = Prepare_Data_In;
00867                     // set the BSY_BIT
00868                     status |= STATUS_BSY_BIT;
00869                     // clear the DRQ_BIT
00870                     status &= ~STATUS_DRQ_BIT;
00871 
00874                     updateState(ACT_DATA_READY);
00875                 }
00876             }
00877         }
00878         break;
00879 
00880       case Prepare_Data_Out:
00881         if (action == ACT_CMD_ERROR || cmdBytesLeft == 0) {
00882             // clear the BSY bit
00883             setComplete();
00884 
00885             if (!isIENSet()) {
00886                 devState = Device_Idle_SI;
00887                 intrPost();
00888             } else {
00889                 devState = Device_Idle_S;
00890             }
00891         } else if (action == ACT_DATA_READY && cmdBytesLeft != 0) {
00892             // clear the BSY bit
00893             status &= ~STATUS_BSY_BIT;
00894             // set the DRQ bit
00895             status |= STATUS_DRQ_BIT;
00896 
00897             // clear the data buffer to get it ready for writes
00898             memset(dataBuffer, 0, MAX_DMA_SIZE);
00899 
00900             // reset the drqBytes for this block
00901             drqBytesLeft = SectorSize;
00902 
00903             if (cmdBytesLeft == cmdBytes || isIENSet()) {
00904                 devState = Transfer_Data_Out;
00905             } else {
00906                 devState = Data_Ready_INTRQ_Out;
00907                 intrPost();
00908             }
00909         }
00910         break;
00911 
00912       case Data_Ready_INTRQ_Out:
00913         if (action == ACT_STAT_READ) {
00914             devState = Transfer_Data_Out;
00915             intrClear();
00916         }
00917         break;
00918 
00919       case Transfer_Data_Out:
00920         if (action == ACT_DATA_WRITE_BYTE ||
00921             action == ACT_DATA_WRITE_SHORT) {
00922 
00923             if (action == ACT_DATA_READ_BYTE) {
00924                 panic("DEBUG: WRITING DATA ONE BYTE AT A TIME!\n");
00925             } else {
00926                 // copy the latest short into the data buffer
00927                 memcpy((void *)&dataBuffer[SectorSize - drqBytesLeft],
00928                        (void *)&cmdReg.data,
00929                        sizeof(uint16_t));
00930 
00931                 drqBytesLeft -= 2;
00932                 cmdBytesLeft -= 2;
00933             }
00934 
00935             if (drqBytesLeft == 0) {
00936                 // copy the block to the disk
00937                 writeDisk(curSector++, dataBuffer);
00938 
00939                 // set the BSY bit
00940                 status |= STATUS_BSY_BIT;
00941                 // set the seek bit
00942                 status |= STATUS_SEEK_BIT;
00943                 // clear the DRQ bit
00944                 status &= ~STATUS_DRQ_BIT;
00945 
00946                 devState = Prepare_Data_Out;
00947 
00950                 updateState(ACT_DATA_READY);
00951             }
00952         }
00953         break;
00954 
00955       case Prepare_Data_Dma:
00956         if (action == ACT_CMD_ERROR) {
00957             // clear the BSY bit
00958             setComplete();
00959 
00960             if (!isIENSet()) {
00961                 devState = Device_Idle_SI;
00962                 intrPost();
00963             } else {
00964                 devState = Device_Idle_S;
00965             }
00966         } else if (action == ACT_DMA_READY) {
00967             // clear the BSY bit
00968             status &= ~STATUS_BSY_BIT;
00969             // set the DRQ bit
00970             status |= STATUS_DRQ_BIT;
00971 
00972             devState = Transfer_Data_Dma;
00973 
00974             if (dmaState != Dma_Idle)
00975                 panic("Inconsistent DMA state, should be Dma_Idle\n");
00976 
00977             dmaState = Dma_Start;
00978             // wait for the write to the DMA start bit
00979         }
00980         break;
00981 
00982       case Transfer_Data_Dma:
00983         if (action == ACT_CMD_ERROR || action == ACT_DMA_DONE) {
00984             // clear the BSY bit
00985             setComplete();
00986             // set the seek bit
00987             status |= STATUS_SEEK_BIT;
00988             // clear the controller state for DMA transfer
00989             ctrl->setDmaComplete(this);
00990 
00991             if (!isIENSet()) {
00992                 devState = Device_Idle_SI;
00993                 intrPost();
00994             } else {
00995                 devState = Device_Idle_S;
00996             }
00997         }
00998         break;
00999 
01000       default:
01001         panic("Unknown IDE device state: %#x\n", devState);
01002     }
