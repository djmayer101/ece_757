00585 template<class Impl>
00586 void
00587 DefaultIEW<Impl>::instToCommit(DynInstPtr &inst)
00588 {
00589     // This function should not be called after writebackInsts in a
00590     // single cycle.  That will cause problems with an instruction
00591     // being added to the queue to commit without being processed by
00592     // writebackInsts prior to being sent to commit.
00593 
00594     // First check the time slot that this instruction will write
00595     // to.  If there are free write ports at the time, then go ahead
00596     // and write the instruction to that time.  If there are not,
00597     // keep looking back to see where's the first time there's a
00598     // free slot.
00599     while ((*iewQueue)[wbCycle].insts[wbNumInst]) {
00600         ++wbNumInst;
00601         if (wbNumInst == wbWidth) {
00602             ++wbCycle;
00603             wbNumInst = 0;
00604         }
00605 
00606         assert((wbCycle * wbWidth + wbNumInst) <= wbMax);
00607     }
00608 
00609     DPRINTF(IEW, "Current wb cycle: %i, width: %i, numInst: %i\nwbActual:%i\n",
00610             wbCycle, wbWidth, wbNumInst, wbCycle * wbWidth + wbNumInst);
00611     // Add finished instruction to queue to commit.
00612     (*iewQueue)[wbCycle].insts[wbNumInst] = inst;
00613     (*iewQueue)[wbCycle].size++;
00614 }

00679         ThreadID tid = *threads++;
00680

00689 void
00690 DefaultIEW<Impl>::updateStatus()
00691 {
00692     bool any_unblocking = false;
00693 
00694     list<ThreadID>::iterator threads = activeThreads->begin();
00695     list<ThreadID>::iterator end = activeThreads->end();
00696 
00697     while (threads != end) {
00698         ThreadID tid = *threads++;
00699

00701             any_unblocking = true;

00705

00707     // and there's no stores waiting to write back, and dispatch is not

00710     if (_status == Active && !instQueue.hasReadyInsts() &&
00711         !ldstQueue.willWB() && !any_unblocking) {

00713

00718                                        ldstQueue.willWB() ||
00719                                        any_unblocking)) {
00720         // Otherwise there is internal activity.  Set to active.
00721         DPRINTF(IEW, "IEW switching to active\n");
00722

00724 
00725         _status = Active;

00787 template <class Impl>
00788 void
00789 DefaultIEW<Impl>::checkSignalsAndUpdate(ThreadID tid)
00790 {
00791     // Check if there's a squash signal, squash if there is
00792     // Check stall signals, block if there is.
00793     // If status was Blocked
00794     //     if so then go to unblocking
00795     // If status was Squashing
00796     //     check if squashing is not high.  Switch to running this cycle.
00797 
00798     readStallSignals(tid);
00799 
00800     if (fromCommit->commitInfo[tid].squash) {
00801         squash(tid);
00802 
00803         if (dispatchStatus[tid] == Blocked ||
00804             dispatchStatus[tid] == Unblocking) {
00805             toRename->iewUnblock[tid] = true;
00806             wroteToTimeBuffer = true;
00807         }
00808 
00809         dispatchStatus[tid] = Squashing;
00810         fetchRedirect[tid] = false;
00811         return;
00812     }
00813 
00814     if (fromCommit->commitInfo[tid].robSquashing) {
00815         DPRINTF(IEW, "[tid:%i]: ROB is still squashing.\n", tid);
00816 
00817         dispatchStatus[tid] = Squashing;
00818         emptyRenameInsts(tid);
00819         wroteToTimeBuffer = true;
00820         return;
00821     }
00822 
00823     if (checkStall(tid)) {
00824         block(tid);
00825         dispatchStatus[tid] = Blocked;
00826         return;
00827     }
00828 
00829     if (dispatchStatus[tid] == Blocked) {
00830         // Status from previous cycle was blocked, but there are no more stall
00831         // conditions.  Switch over to unblocking.
00832         DPRINTF(IEW, "[tid:%i]: Done blocking, switching to unblocking.\n",
00833                 tid);
00834 
00835         dispatchStatus[tid] = Unblocking;
00836 
00837         unblock(tid);
00838 
00839         return;
00840     }
00841 
00842     if (dispatchStatus[tid] == Squashing) {
00843         // Switch status to running if rename isn't being told to block or
00844         // squash this cycle.
00845         DPRINTF(IEW, "[tid:%i]: Done squashing, switching to running.\n",
00846                 tid);
00847 
00848         dispatchStatus[tid] = Running;
00849 
00850         return;
00851     }
00852 }

00918 template<class Impl>
00919 void
00920 DefaultIEW<Impl>::dispatch(ThreadID tid)
00921 {
00922     // If status is Running or idle,
00923     //     call dispatchInsts()
00924     // If status is Unblocking,
00925     //     buffer any instructions coming from rename
00926     //     continue trying to empty skid buffer
00927     //     check if stall conditions have passed
00928 
00929     if (dispatchStatus[tid] == Blocked) {
00930         ++iewBlockCycles;
00931 
00932     } else if (dispatchStatus[tid] == Squashing) {
00933         ++iewSquashCycles;
00934     }
00935 
00936     // Dispatch should try to dispatch as many instructions as its bandwidth
00937     // will allow, as long as it is not currently blocked.
00938     if (dispatchStatus[tid] == Running ||
00939         dispatchStatus[tid] == Idle) {
00940         DPRINTF(IEW, "[tid:%i] Not blocked, so attempting to run "
00941                 "dispatch.\n", tid);
00942 
00943         dispatchInsts(tid);
00944     } else if (dispatchStatus[tid] == Unblocking) {
00945         // Make sure that the skid buffer has something in it if the
00946         // status is unblocking.
00947         assert(!skidsEmpty());
00948 
00949         // If the status was unblocking, then instructions from the skid
00950         // buffer were used.  Remove those instructions and handle
00951         // the rest of unblocking.
00952         dispatchInsts(tid);
00953 
00954         ++iewUnblockCycles;
00955 
00956         if (validInstsFromRename()) {
00957             // Add the current inputs to the skid buffer so they can be
00958             // reprocessed when this stage unblocks.
00959             skidInsert(tid);
00960         }
00961 
00962         unblock(tid);
00963     }
00964 }

00966 template <class Impl>
00967 void
00968 DefaultIEW<Impl>::dispatchInsts(ThreadID tid)
00969 {
00970     // Obtain instructions from skid buffer if unblocking, or queue from rename
00971     // otherwise.
00972     std::queue<DynInstPtr> &insts_to_dispatch =
00973         dispatchStatus[tid] == Unblocking ?
00974         skidBuffer[tid] : insts[tid];
00975 
00976     int insts_to_add = insts_to_dispatch.size();
00977 
00978     DynInstPtr inst;
00979     bool add_to_iq = false;
00980     int dis_num_inst = 0;
00981 
00982     // Loop through the instructions, putting them in the instruction
00983     // queue.
00984     for ( ; dis_num_inst < insts_to_add &&
00985               dis_num_inst < dispatchWidth;
00986           ++dis_num_inst)
00987     {
00988         inst = insts_to_dispatch.front();
00989 
00990         if (dispatchStatus[tid] == Unblocking) {
00991             DPRINTF(IEW, "[tid:%i]: Issue: Examining instruction from skid "
00992                     "buffer\n", tid);
00993         }
00994 
00995         // Make sure there's a valid instruction there.
00996         assert(inst);
00997 
00998         DPRINTF(IEW, "[tid:%i]: Issue: Adding PC %s [sn:%lli] [tid:%i] to "
00999                 "IQ.\n",
01000                 tid, inst->pcState(), inst->seqNum, inst->threadNumber);
01001 
01002         // Be sure to mark these instructions as ready so that the
01003         // commit stage can go ahead and execute them, and mark
01004         // them as issued so the IQ doesn't reprocess them.
01005 
01006         // Check for squashed instructions.
01007         if (inst->isSquashed()) {
01008             DPRINTF(IEW, "[tid:%i]: Issue: Squashed instruction encountered, "
01009                     "not adding to IQ.\n", tid);
01010 
01011             ++iewDispSquashedInsts;
01012 
01013             insts_to_dispatch.pop();
01014 
01015             //Tell Rename That An Instruction has been processed
01016             if (inst->isLoad() || inst->isStore()) {
01017                 toRename->iewInfo[tid].dispatchedToLSQ++;
01018             }
01019             toRename->iewInfo[tid].dispatched++;
01020 
01021             continue;
01022         }
01023 
01024         // Check for full conditions.
01025         if (instQueue.isFull(tid)) {
01026             DPRINTF(IEW, "[tid:%i]: Issue: IQ has become full.\n", tid);
01027 
01028             // Call function to start blocking.
01029             block(tid);
01030 
01031             // Set unblock to false. Special case where we are using
01032             // skidbuffer (unblocking) instructions but then we still
01033             // get full in the IQ.
01034             toRename->iewUnblock[tid] = false;
01035 
01036             ++iewIQFullEvents;
01037             break;
01038         } else if (ldstQueue.isFull(tid)) {
01039             DPRINTF(IEW, "[tid:%i]: Issue: LSQ has become full.\n",tid);
01040 
01041             // Call function to start blocking.
01042             block(tid);
01043 
01044             // Set unblock to false. Special case where we are using
01045             // skidbuffer (unblocking) instructions but then we still
01046             // get full in the IQ.
01047             toRename->iewUnblock[tid] = false;
01048 
01049             ++iewLSQFullEvents;
01050             break;
01051         }
01052 
01053         // Otherwise issue the instruction just fine.
01054         if (inst->isLoad()) {
01055             DPRINTF(IEW, "[tid:%i]: Issue: Memory instruction "
01056                     "encountered, adding to LSQ.\n", tid);
01057 
01058             // Reserve a spot in the load store queue for this
01059             // memory access.
01060             ldstQueue.insertLoad(inst);
01061 
01062             ++iewDispLoadInsts;
01063 
01064             add_to_iq = true;
01065 
01066             toRename->iewInfo[tid].dispatchedToLSQ++;
01067         } else if (inst->isStore()) {
01068             DPRINTF(IEW, "[tid:%i]: Issue: Memory instruction "
01069                     "encountered, adding to LSQ.\n", tid);
01070 
01071             ldstQueue.insertStore(inst);
01072 
01073             ++iewDispStoreInsts;
01074 
01075             if (inst->isStoreConditional()) {
01076                 // Store conditionals need to be set as "canCommit()"
01077                 // so that commit can process them when they reach the
01078                 // head of commit.
01079                 // @todo: This is somewhat specific to Alpha.
01080                 inst->setCanCommit();
01081                 instQueue.insertNonSpec(inst);
01082                 add_to_iq = false;
01083 
01084                 ++iewDispNonSpecInsts;
01085             } else {
01086                 add_to_iq = true;
01087             }
01088 
01089             toRename->iewInfo[tid].dispatchedToLSQ++;
01090         } else if (inst->isMemBarrier() || inst->isWriteBarrier()) {
01091             // Same as non-speculative stores.
01092             inst->setCanCommit();
01093             instQueue.insertBarrier(inst);
01094             add_to_iq = false;
01095         } else if (inst->isNop()) {
01096             DPRINTF(IEW, "[tid:%i]: Issue: Nop instruction encountered, "
01097                     "skipping.\n", tid);
01098 
01099             inst->setIssued();
01100             inst->setExecuted();
01101             inst->setCanCommit();
01102 
01103             instQueue.recordProducer(inst);
01104 
01105             iewExecutedNop[tid]++;
01106 
01107             add_to_iq = false;
01108         } else if (inst->isExecuted()) {
01109             assert(0 && "Instruction shouldn't be executed.\n");
01110             DPRINTF(IEW, "Issue: Executed branch encountered, "
01111                     "skipping.\n");
01112 
01113             inst->setIssued();
01114             inst->setCanCommit();
01115 
01116             instQueue.recordProducer(inst);
01117 
01118             add_to_iq = false;
01119         } else {
01120             add_to_iq = true;
01121         }
01122         if (inst->isNonSpeculative()) {
01123             DPRINTF(IEW, "[tid:%i]: Issue: Nonspeculative instruction "
01124                     "encountered, skipping.\n", tid);
01125 
01126             // Same as non-speculative stores.
01127             inst->setCanCommit();
01128 
01129             // Specifically insert it as nonspeculative.
01130             instQueue.insertNonSpec(inst);
01131 
01132             ++iewDispNonSpecInsts;
01133 
01134             add_to_iq = false;
01135         }
01136 
01137         // If the instruction queue is not full, then add the
01138         // instruction.
01139         if (add_to_iq) {
01140             instQueue.insert(inst);
01141         }
01142 
01143         insts_to_dispatch.pop();
01144 
01145         toRename->iewInfo[tid].dispatched++;
01146 
01147         ++iewDispatchedInsts;
01148 
01149 #if TRACING_ON
01150         inst->dispatchTick = curTick() - inst->fetchTick;
01151 #endif
01152     }
01153 
01154     if (!insts_to_dispatch.empty()) {
01155         DPRINTF(IEW,"[tid:%i]: Issue: Bandwidth Full. Blocking.\n", tid);
01156         block(tid);
01157         toRename->iewUnblock[tid] = false;
01158     }
01159 
01160     if (dispatchStatus[tid] == Idle && dis_num_inst) {
01161         dispatchStatus[tid] = Running;
01162 
01163         updatedQueues = true;
01164     }
01165 
01166     dis_num_inst = 0;
01167 }

01192 template <class Impl>
01193 void
01194 DefaultIEW<Impl>::executeInsts()
01195 {
01196     wbNumInst = 0;
01197     wbCycle = 0;
01198 
01199     list<ThreadID>::iterator threads = activeThreads->begin();
01200     list<ThreadID>::iterator end = activeThreads->end();
01201 
01202     while (threads != end) {
01203         ThreadID tid = *threads++;
01204         fetchRedirect[tid] = false;
01205     }
01206 
01207     // Uncomment this if you want to see all available instructions.
01208     // @todo This doesn't actually work anymore, we should fix it.
01209 //    printAvailableInsts();
01210 
01211     // Execute/writeback any instructions that are available.
01212     int insts_to_execute = fromIssue->size;
01213     int inst_num = 0;
01214     for (; inst_num < insts_to_execute;
01215           ++inst_num) {
01216 
01217         DPRINTF(IEW, "Execute: Executing instructions from IQ.\n");
01218 
01219         DynInstPtr inst = instQueue.getInstToExecute();
01220 
01221         DPRINTF(IEW, "Execute: Processing PC %s, [tid:%i] [sn:%i].\n",
01222                 inst->pcState(), inst->threadNumber,inst->seqNum);
01223 
01224         // Check if the instruction is squashed; if so then skip it
01225         if (inst->isSquashed()) {
01226             DPRINTF(IEW, "Execute: Instruction was squashed. PC: %s, [tid:%i]"
01227                          " [sn:%i]\n", inst->pcState(), inst->threadNumber,
01228                          inst->seqNum);
01229 
01230             // Consider this instruction executed so that commit can go
01231             // ahead and retire the instruction.
01232             inst->setExecuted();
01233 
01234             // Not sure if I should set this here or just let commit try to
01235             // commit any squashed instructions.  I like the latter a bit more.
01236             inst->setCanCommit();
01237 
01238             ++iewExecSquashedInsts;
01239 
01240             decrWb(inst->seqNum);
01241             continue;
01242         }
01243 
01244         Fault fault = NoFault;
01245 
01246         // Execute instruction.
01247         // Note that if the instruction faults, it will be handled
01248         // at the commit stage.
01249         if (inst->isMemRef()) {
01250             DPRINTF(IEW, "Execute: Calculating address for memory "
01251                     "reference.\n");
01252 
01253             // Tell the LDSTQ to execute this instruction (if it is a load).
01254             if (inst->isLoad()) {
01255                 // Loads will mark themselves as executed, and their writeback
01256                 // event adds the instruction to the queue to commit
01257                 fault = ldstQueue.executeLoad(inst);
01258 
01259                 if (inst->isTranslationDelayed() &&
01260                     fault == NoFault) {
01261                     // A hw page table walk is currently going on; the
01262                     // instruction must be deferred.
01263                     DPRINTF(IEW, "Execute: Delayed translation, deferring "
01264                             "load.\n");
01265                     instQueue.deferMemInst(inst);
01266                     continue;
01267                 }
01268 
01269                 if (inst->isDataPrefetch() || inst->isInstPrefetch()) {
01270                     inst->fault = NoFault;
01271                 }
01272             } else if (inst->isStore()) {
01273                 fault = ldstQueue.executeStore(inst);
01274 
01275                 if (inst->isTranslationDelayed() &&
01276                     fault == NoFault) {
01277                     // A hw page table walk is currently going on; the
01278                     // instruction must be deferred.
01279                     DPRINTF(IEW, "Execute: Delayed translation, deferring "
01280                             "store.\n");
01281                     instQueue.deferMemInst(inst);
01282                     continue;
01283                 }
01284 
01285                 // If the store had a fault then it may not have a mem req
01286                 if (fault != NoFault || inst->readPredicate() == false ||
01287                         !inst->isStoreConditional()) {
01288                     // If the instruction faulted, then we need to send it along
01289                     // to commit without the instruction completing.
01290                     // Send this instruction to commit, also make sure iew stage
01291                     // realizes there is activity.
01292                     inst->setExecuted();
01293                     instToCommit(inst);
01294                     activityThisCycle();
01295                 }
01296 
01297                 // Store conditionals will mark themselves as
01298                 // executed, and their writeback event will add the
01299                 // instruction to the queue to commit.
01300             } else {
01301                 panic("Unexpected memory type!\n");
01302             }
01303 
01304         } else {
01305             // If the instruction has already faulted, then skip executing it.
01306             // Such case can happen when it faulted during ITLB translation.
01307             // If we execute the instruction (even if it's a nop) the fault
01308             // will be replaced and we will lose it.
01309             if (inst->getFault() == NoFault) {
01310                 inst->execute();
01311                 if (inst->readPredicate() == false)
01312                     inst->forwardOldRegs();
01313             }
01314 
01315             inst->setExecuted();
01316 
01317             instToCommit(inst);
01318         }
01319 
01320         updateExeInstStats(inst);
01321 
01322         // Check if branch prediction was correct, if not then we need
01323         // to tell commit to squash in flight instructions.  Only
01324         // handle this if there hasn't already been something that
01325         // redirects fetch in this group of instructions.
01326 
01327         // This probably needs to prioritize the redirects if a different
01328         // scheduler is used.  Currently the scheduler schedules the oldest
01329         // instruction first, so the branch resolution order will be correct.
01330         ThreadID tid = inst->threadNumber;
01331 
01332         if (!fetchRedirect[tid] ||
01333             !toCommit->squash[tid] ||
01334             toCommit->squashedSeqNum[tid] > inst->seqNum) {
01335 
01336             // Prevent testing for misprediction on load instructions,
01337             // that have not been executed.
01338             bool loadNotExecuted = !inst->isExecuted() && inst->isLoad();
01339 
01340             if (inst->mispredicted() && !loadNotExecuted) {
01341                 fetchRedirect[tid] = true;
01342 
01343                 DPRINTF(IEW, "Execute: Branch mispredict detected.\n");
01344                 DPRINTF(IEW, "Predicted target was PC: %s.\n",
01345                         inst->readPredTarg());
01346                 DPRINTF(IEW, "Execute: Redirecting fetch to PC: %s.\n",
01347                         inst->pcState());
01348                 // If incorrect, then signal the ROB that it must be squashed.
01349                 squashDueToBranch(inst, tid);
01350 
01351                 if (inst->readPredTaken()) {
01352                     predictedTakenIncorrect++;
01353                 } else {
01354                     predictedNotTakenIncorrect++;
01355                 }
01356             } else if (ldstQueue.violation(tid)) {
01357                 assert(inst->isMemRef());
01358                 // If there was an ordering violation, then get the
01359                 // DynInst that caused the violation.  Note that this
01360                 // clears the violation signal.
01361                 DynInstPtr violator;
01362                 violator = ldstQueue.getMemDepViolator(tid);
01363 
01364                 DPRINTF(IEW, "LDSTQ detected a violation. Violator PC: %s "
01365                         "[sn:%lli], inst PC: %s [sn:%lli]. Addr is: %#x.\n",
01366                         violator->pcState(), violator->seqNum,
01367                         inst->pcState(), inst->seqNum, inst->physEffAddr);
01368 
01369                 fetchRedirect[tid] = true;
01370 
01371                 // Tell the instruction queue that a violation has occured.
01372                 instQueue.violation(inst, violator);
01373 
01374                 // Squash.
01375                 squashDueToMemOrder(violator, tid);
01376 
01377                 ++memOrderViolationEvents;
01378             } else if (ldstQueue.loadBlocked(tid) &&
01379                        !ldstQueue.isLoadBlockedHandled(tid)) {
01380                 fetchRedirect[tid] = true;
01381

01383                         "memory system is blocked.  PC: %s [sn:%lli]\n",
01384                         inst->pcState(), inst->seqNum);
01385

01387             }
01388         } else {
01389             // Reset any state associated with redirects that will not
01390             // be used.
01391             if (ldstQueue.violation(tid)) {
01392                 assert(inst->isMemRef());
01393 
01394                 DynInstPtr violator = ldstQueue.getMemDepViolator(tid);
01395 
01396                 DPRINTF(IEW, "LDSTQ detected a violation.  Violator PC: "
01397                         "%s, inst PC: %s.  Addr is: %#x.\n",
01398                         violator->pcState(), inst->pcState(),
01399                         inst->physEffAddr);
01400                 DPRINTF(IEW, "Violation will not be handled because "
01401                         "already squashing\n");
01402 
01403                 ++memOrderViolationEvents;
01404             }
01405             if (ldstQueue.loadBlocked(tid) &&
01406                 !ldstQueue.isLoadBlockedHandled(tid)) {
01407                 DPRINTF(IEW, "Load operation couldn't execute because the "
01408                         "memory system is blocked.  PC: %s [sn:%lli]\n",
01409                         inst->pcState(), inst->seqNum);
01410                 DPRINTF(IEW, "Blocked load will not be handled because "
01411                         "already squashing\n");

01416         }
01417     }
01418 
01419     // Update and record activity if we processed any instructions.
01420     if (inst_num) {
01421         if (exeStatus == Idle) {
01422             exeStatus = Running;
01423         }
01424 
01425         updatedQueues = true;
01426 
01427         cpu->activityThisCycle();
01428     }
01429 
01430     // Need to reset this in case a writeback event needs to write into the
01431     // iew queue.  That way the writeback event will write into the correct
01432     // spot in the queue.

01437 template <class Impl>
01438 void
01439 DefaultIEW<Impl>::writebackInsts()
01440 {
01441     // Loop through the head of the time buffer and wake any
01442     // dependents.  These instructions are about to write back.  Also
01443     // mark scoreboard that this instruction is finally complete.
01444     // Either have IEW have direct access to scoreboard, or have this
01445     // as part of backwards communication.
01446     for (int inst_num = 0; inst_num < wbWidth &&
01447              toCommit->insts[inst_num]; inst_num++) {
01448         DynInstPtr inst = toCommit->insts[inst_num];
01449         ThreadID tid = inst->threadNumber;
01450 
01451         DPRINTF(IEW, "Sending instructions to commit, [sn:%lli] PC %s.\n",
01452                 inst->seqNum, inst->pcState());
01453 
01454         iewInstsToCommit[tid]++;
01455 
01456         // Some instructions will be sent to commit without having
01457         // executed because they need commit to handle them.
01458         // E.g. Uncached loads have not actually executed when they
01459         // are first sent to commit.  Instead commit must tell the LSQ
01460         // when it's ready to execute the uncached load.
01461         if (!inst->isSquashed() && inst->isExecuted() && inst->getFault() == NoFault) {
01462             int dependents = instQueue.wakeDependents(inst);
01463 
01464             for (int i = 0; i < inst->numDestRegs(); i++) {
01465                 //mark as Ready
01466                 DPRINTF(IEW,"Setting Destination Register %i\n",
01467                         inst->renamedDestRegIdx(i));
01468                 scoreboard->setReg(inst->renamedDestRegIdx(i));
01469             }
01470 
01471             if (dependents) {
01472                 producerInst[tid]++;
01473                 consumerInst[tid]+= dependents;
01474             }
01475             writebackCount[tid]++;
01476         }
01477 
01478         decrWb(inst->seqNum);
01479     }
01480 }

01497     list<ThreadID>::iterator threads = activeThreads->begin();
01498     list<ThreadID>::iterator end = activeThreads->end();
01499

01501     while (threads != end) {
01502         ThreadID tid = *threads++;

01505 
01506         checkSignalsAndUpdate(tid);

01516         // (In actuality, this scheduling is for instructions that will
01517         // be executed next cycle.)
01518         instQueue.scheduleReadyInsts();

01521         // Not the best place for it, but this works (hopefully).
01522         issueToExecQueue.advance();

01526 
01527     if (updatedQueues || exeStatus == Running || updateLSQNextCycle) {
01528         exeStatus = Idle;

01532     }
01533 
01534     // Writeback any stores using any leftover bandwidth.
01535     ldstQueue.writebackStores();
01536 
01537     // Check the committed load/store signals to see if there's a load
01538     // or store to commit.  Also check if it's being told to execute a
01539     // nonspeculative instruction.
01540     // This is pretty inefficient...
01541 
01542     threads = activeThreads->begin();
01543     while (threads != end) {
01544         ThreadID tid = (*threads++);
01545 
01546         DPRINTF(IEW,"Processing [tid:%i]\n",tid);
01547 
01548         // Update structures based on instructions committed.
01549         if (fromCommit->commitInfo[tid].doneSeqNum != 0 &&
01550             !fromCommit->commitInfo[tid].squash &&
01551             !fromCommit->commitInfo[tid].robSquashing) {
01552 
01553             ldstQueue.commitStores(fromCommit->commitInfo[tid].doneSeqNum,tid);
01554 
01555             ldstQueue.commitLoads(fromCommit->commitInfo[tid].doneSeqNum,tid);
01556 
01557             updateLSQNextCycle = true;
01558             instQueue.commit(fromCommit->commitInfo[tid].doneSeqNum,tid);
01559         }
01560 
01561         if (fromCommit->commitInfo[tid].nonSpecSeqNum != 0) {
01562 
01563             //DPRINTF(IEW,"NonspecInst from thread %i",tid);

01565                 instQueue.replayMemInst(fromCommit->commitInfo[tid].uncachedLoad);

01567             } else {

01569                     fromCommit->commitInfo[tid].nonSpecSeqNum);
01570             }
01571         }
01572 
01573         if (broadcast_free_entries) {
01574             toFetch->iewInfo[tid].iqCount =
01575                 instQueue.getCount(tid);
01576             toFetch->iewInfo[tid].ldstqCount =
01577                 ldstQueue.getCount(tid);
01578 
01579             toRename->iewInfo[tid].usedIQ = true;
01580             toRename->iewInfo[tid].freeIQEntries =
01581                 instQueue.numFreeEntries();
01582             toRename->iewInfo[tid].usedLSQ = true;
01583             toRename->iewInfo[tid].freeLSQEntries =
01584                 ldstQueue.numFreeEntries(tid);
01585 
01586             wroteToTimeBuffer = true;
01587         }
01588 
01589         DPRINTF(IEW, "[tid:%i], Dispatch dispatched %i instructions.\n",
01590                 tid, toRename->iewInfo[tid].dispatched);
01591     }
01592 
01593     DPRINTF(IEW, "IQ has %i free entries (Can schedule: %i).  "
01594             "LSQ has %i free entries.\n",
01595             instQueue.numFreeEntries(), instQueue.hasReadyInsts(),
01596             ldstQueue.numFreeEntries());
01597 
01598     updateStatus();
01599 
01600     if (wroteToTimeBuffer) {
01601         DPRINTF(Activity, "Activity this cycle.\n");
01602         cpu->activityThisCycle();
01603     }
