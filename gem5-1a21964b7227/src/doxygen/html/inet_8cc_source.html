00079 void
00080 EthAddr::parse(const std::string &addr)
00081 {
00082     // the hack below is to make sure that ETH_ADDR_LEN is 6 otherwise
00083     // the sscanf function won't work.
00084     int bytes[ETH_ADDR_LEN == 6 ? ETH_ADDR_LEN : -1];
00085     if (sscanf(addr.c_str(), "%x:%x:%x:%x:%x:%x", &bytes[0], &bytes[1],
00086                &bytes[2], &bytes[3], &bytes[4], &bytes[5]) != ETH_ADDR_LEN) {
00087         memset(data, 0xff, ETH_ADDR_LEN);
00088         return;
00089     }
00090 
00091     for (int i = 0; i < ETH_ADDR_LEN; ++i) {
00092         if (bytes[i] & ~0xff) {
00093             memset(data, 0xff, ETH_ADDR_LEN);
00094             return;
00095         }
00096 
00097         data[i] = bytes[i];
00098     }
00099 }
