00182     Section *section = NULL;
00183 
00184     while (!f.eof()) {
00185         f >> ws; // Eat whitespace
00186         if (f.eof()) {
00187             break;
00188         }
00189 
00190         string line;
00191         getline(f, line);
00192         if (line.size() == 0)
00193             continue;
00194 
00195         eat_end_white(line);
00196         int last = line.size() - 1;
00197 
00198         if (line[0] == '[' && line[last] == ']') {
00199             string sectionName = line.substr(1, last - 1);
00200             eat_white(sectionName);
00201             section = addSection(sectionName);
00202             continue;
00203         }
00204 
00205         if (section == NULL)
00206             continue;
00207 
00208         if (!section->add(line))
00209             return false;
00210     }
00211 
00212     return true;
