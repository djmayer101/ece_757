00388     static const uint8_t INST_LOAD                          = 25;
00389     static const uint8_t INST_STORE                         = 24;
00390     static const uint8_t INST_CONTROL                       = 23;
00391     static const uint8_t INST_NONSPEC                       = 22;
00392     static const uint8_t INST_DEST_REGS                     = 18;
00393     static const uint8_t INST_SRC_REGS                      = 14;
00394     static const uint8_t INST_SPLIT_DATA                    = 13;

00396     inline SkedID genSkedID(DynInstPtr inst)
00397     {
00398         SkedID id = 0;
00399         id = (inst->opClass() << INST_OPCLASS) |
00400             (inst->isLoad() << INST_LOAD) |
00401             (inst->isStore() << INST_STORE) |
00402             (inst->isControl() << INST_CONTROL) |
00403             (inst->isNonSpeculative() << INST_NONSPEC) |
00404             (inst->numDestRegs() << INST_DEST_REGS) |
00405             (inst->numSrcRegs() << INST_SRC_REGS) |
00406             (inst->splitInst << INST_SPLIT_DATA);
00407         return id;
00408     }
