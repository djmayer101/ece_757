00140 {
00141     // Squash due to an external source
00142     // Not sure if this or an interrupt has higher priority
00143     if (squashPending) {
00144         squash(squashSeqNum, squashNextPC);
00145         return;
00146     }
00147 
00148     // if (interrupt) then set thread PC, stall front end, record that
00149     // I'm waiting for it to drain.  (for now just squash)
00150     if (FullSystem && (interruptBlocked || cpu->checkInterrupts(tc))) {
00151         if (!robEmpty()) {
00152             interruptBlocked = true;
00153         //AlphaDep
00154         } else if (robEmpty() && (PC & 0x3)) {
00155             // Will need to let the front end continue a bit until
00156             // we're out of pal mode.  Hopefully we never get into an
00157             // infinite loop...
00158             interruptBlocked = false;
00159         } else {
00160             interruptBlocked = false;
00161             checkInterrupts();
00162             return;
00163         }
00164     }
00165 
00166     if (status != DcacheMissLoadStall &&
00167         status != DcacheMissStoreStall) {
00168         for (int i = 0; i < width && (*instsAdded) < width; ++i) {
00169             DynInstPtr inst = frontEnd->getInst();
00170 
00171             if (!inst)
00172                 break;
00173 
00174             instList.push_back(inst);
00175 
00176             (*instsAdded)++;
00177         }
00178 
00179         if (faultFromFetch && robEmpty() && frontEnd->isEmpty()) {
00180             handleFault();
00181         } else {
00182             executeInsts();
00183         }
00184     }

00404         inst->clearDependents();

00408 
00409     frontEnd->addFreeRegs(freed_regs);
00410 
00411     for (int i = 0; i < latency+1; ++i) {
00412         numInstsToWB.advance();

00414 
00415     squashPending = false;
00416 
00417     // Probably want to make sure that this squash is the one that set the
00418     // thread into noSquashFromTC mode.
00419     thread->noSquashFromTC = false;

00421     // Tell front end to squash, reset PC to new one.
00422     frontEnd->squash(squash_num, next_PC);
