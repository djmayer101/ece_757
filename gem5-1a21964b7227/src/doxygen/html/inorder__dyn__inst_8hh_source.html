00477     bool finishSkedEntry()
00478     {
00479         curSkedEntry++;
00480 
00481         if (inFrontEnd && curSkedEntry == frontSked_end) {
00482             DPRINTF(InOrderDynInst, "[sn:%i] Switching to "
00483                   "back end schedule.\n", seqNum);
00484           assert(backSked != NULL);
00485           curSkedEntry.init(backSked);
00486           curSkedEntry = backSked->begin();
00487           inFrontEnd = false;
00488         } else if (!inFrontEnd && curSkedEntry == backSked_end) {
00489             return true;
00490         }
00491 
00492         DPRINTF(InOrderDynInst, "[sn:%i] Next Stage: %i "
00493                 "Next Resource: %i.\n", seqNum, curSkedEntry->stageNum,
00494                 curSkedEntry->resNum);
00495 
00496         return false;
00497     }
