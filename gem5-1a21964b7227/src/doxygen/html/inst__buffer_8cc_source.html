00072     switch (ib_req->cmd)
00073     {
00074       case ScheduleOrBypass:
00075         {
00076             int next_stage = stage_num + 1;
00077             int bypass_stage = stage_num + 2;

00081                 DPRINTF(InOrderInstBuffer, "[sn:%i] cannot bypass stage %i "

00083                         inst->seqNum, next_stage);

00086                 DPRINTF(InOrderInstBuffer, "[sn:%i] cannot bypass stage %i "
00087                         "because stage %i is blocking.\n",
00088                         inst->seqNum, next_stage);
00089                 do_bypass = false;

00091                       stageBufferAvail() <= 0) {
00092                 DPRINTF(InOrderInstBuffer, "[sn:%i] cannot bypass stage %i "
00093                         "because there is no room in stage %i incoming stage "
00094                         "buffer.\n", inst->seqNum, next_stage);
00095                 do_bypass = false;
00096             }
00097 
00098             if (!do_bypass) { // SCHEDULE USAGE OF BUFFER
00099                 DPRINTF(InOrderInstBuffer, "Scheduling [sn:%i] for buffer "
00100                         "insertion in stage %i\n",
00101                         inst->seqNum, next_stage);
00102 
00103                 // Add to schedule: Insert into buffer in next stage
00104                 int stage_pri = 20;
00105                 RSkedPtr insert_sked = (stage_num >= ThePipeline::BackEndStartStage) ?
00106                     inst->backSked : inst->frontSked;
00107 
00108                 insert_sked->push(new ScheduleEntry(next_stage,
00109                                                       stage_pri,
00110                                                       id,
00111                                                       InstBuffer::InsertInst));
00112 
00113                 // Add to schedule: Remove from buffer in next next (bypass)
00114                 // stage
00115                 stage_pri = 20;
00116                 RSkedPtr bypass_sked = (stage_num >= ThePipeline::BackEndStartStage) ?
00117                     inst->backSked : inst->frontSked;
00118 
00119                bypass_sked->push(new ScheduleEntry(bypass_stage,
00120                                                       stage_pri,
00121                                                       id,
00122                                                       InstBuffer::RemoveInst));
00123             } else {         // BYPASS BUFFER & NEXT STAGE
00124                 DPRINTF(InOrderInstBuffer, "Setting [sn:%i] to bypass stage "
00125                         "%i and enter stage %i.\n", inst->seqNum, next_stage,
00126                         bypass_stage);
00127                 inst->setNextStage(bypass_stage);
00128                 instsBypassed++;
00129             }
00130 
00131             ib_req->done();
00132         }
00133         break;
00134 
00135       case InsertInst:
00136         {
00137             bool inserted = false;
00138 
00139             if (instList.size() < width) {
00140                 DPRINTF(InOrderInstBuffer, "[tid:%i]: Inserting [sn:%i] into "
00141                         "buffer.\n", tid, inst->seqNum);
00142                 insert(inst);
00143                 inserted = true;
00144             } else {
00145                 DPRINTF(InOrderInstBuffer, "[tid:%i]: Denying [sn:%i] request "
00146                         "because buffer is full.\n", tid, inst->seqNum);
00147 
00148 
00149                 std::list<DynInstPtr>::iterator list_it = instList.begin();
00150                 std::list<DynInstPtr>::iterator list_end = instList.end();
00151 
00152                 while (list_it != list_end) {
00153                     DPRINTF(Resource,"Serving [tid:%i] [sn:%i].\n",
00154                             (*list_it)->readTid(), (*list_it)->seqNum);
00155                     list_it++;
00156                 }
00157             }
00158

00160         }
00161         break;
00162 
00163       case RemoveInst:
00164         {
00165             DPRINTF(InOrderInstBuffer, "[tid:%i]: Removing [sn:%i] from "
