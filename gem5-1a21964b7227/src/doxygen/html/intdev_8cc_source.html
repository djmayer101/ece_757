00045 void
00046 X86ISA::IntDev::IntMasterPort::sendMessage(ApicList apics,
00047                                            TriggerIntMessage message,
00048                                            bool timing)
00049 {
00050     ApicList::iterator apicIt;
00051     for (apicIt = apics.begin(); apicIt != apics.end(); apicIt++) {
00052         PacketPtr pkt = buildIntRequest(*apicIt, message);
00053         if (timing) {
00054             schedTimingReq(pkt, curTick() + latency);
00055             // The target handles cleaning up the packet in timing mode.
00056         } else {
00057             // ignore the latency involved in the atomic transaction
00058             sendAtomic(pkt);
00059             assert(pkt->isResponse());
00060             // also ignore the latency in handling the response
00061             recvResponse(pkt);
00062             delete pkt->req;
00063             delete pkt;
00064         }
00065     }
00066 }
