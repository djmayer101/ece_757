00065     }
00066

00068     if (_numSrcRegs > 0 && printSrcs) {
00069         if (_numDestRegs > 0 && printDest) {
00070             ss << ", ";

00072         printReg(ss, _srcRegIdx[0]);
00073         if (_numSrcRegs > 1 && printSecondSrc) {
