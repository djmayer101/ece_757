00160 void
00161 Intel8254Timer::Counter::write(const uint8_t data)
00162 {
00163     switch (write_byte) {
00164       case LSB:
00165         initial_count = (initial_count & 0xFF00) | data;
00166 
00167         if (event.scheduled())
00168             parent->deschedule(event);
00169         output_high = false;
00170         write_byte = MSB;
00171         break;
00172 
00173       case MSB:
00174         initial_count = (initial_count & 0x00FF) | (data << 8);
00175         // In the RateGen or SquareWave modes, the timer wraps around and
00176         // triggers on a value of 1, not 0.
00177         if (mode == RateGen || mode == SquareWave)
00178             period = initial_count - 1;
00179         else
00180             period = initial_count;
00181 
00182         if (period > 0)
00183             event.setTo(period);
00184 
00185         write_byte = LSB;
00186         break;
00187     }
00188 }
