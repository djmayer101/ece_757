00071 template<class T>
00072 uint8_t
00073 writeOutField(PortProxy& proxy, Addr addr, T val)
00074 {
00075     uint64_t guestVal = X86ISA::htog(val);
00076     proxy.writeBlob(addr, (uint8_t *)(&guestVal), sizeof(T));
00077 
00078     uint8_t checkSum = 0;
00079     while(guestVal) {
00080         checkSum += guestVal;
00081         guestVal >>= 8;
00082     }
00083     return checkSum;
00084 }

00086 uint8_t
00087 writeOutString(PortProxy& proxy, Addr addr, string str, int length)
00088 {
00089     char cleanedString[length + 1];
00090     cleanedString[length] = 0;
00091 
00092     if (str.length() > length) {
00093         memcpy(cleanedString, str.c_str(), length);
00094         warn("Intel MP configuration table string \"%s\" "
00095                 "will be truncated to \"%s\".\n", str, cleanedString);
00096     } else {
00097         memcpy(cleanedString, str.c_str(), str.length());
00098         memset(cleanedString + str.length(), 0, length - str.length());
00099     }
00100     proxy.writeBlob(addr, (uint8_t *)(&cleanedString), length);
00101 
00102     uint8_t checkSum = 0;
00103     for (int i = 0; i < length; i++)
00104         checkSum += cleanedString[i];
00105 
00106     return checkSum;
00107 }

00110 X86ISA::IntelMP::FloatingPointer::writeOut(PortProxy& proxy, Addr addr)
00111 {
00112     // Make sure that either a config table is present or a default
00113     // configuration was found but not both.
00114     if (!tableAddr && !defaultConfig)
00115         fatal("Either an MP configuration table or a default configuration "
00116                 "must be used.");
00117     if (tableAddr && defaultConfig)
00118         fatal("Both an MP configuration table and a default configuration "
00119                 "were set.");
00120 
00121     uint8_t checkSum = 0;
00122 
00123     proxy.writeBlob(addr, (uint8_t *)signature, 4);
00124     for (int i = 0; i < 4; i++)
00125         checkSum += signature[i];
00126 
00127     checkSum += writeOutField(proxy, addr + 4, tableAddr);
00128 
00129     // The length of the structure in paragraphs, aka 16 byte chunks.
00130     uint8_t length = 1;
00131     proxy.writeBlob(addr + 8, &length, 1);
00132     checkSum += length;
00133 
00134     proxy.writeBlob(addr + 9, &specRev, 1);
00135     checkSum += specRev;
00136 
00137     proxy.writeBlob(addr + 11, &defaultConfig, 1);
00138     checkSum += defaultConfig;
00139 
00140     uint32_t features2_5 = imcrPresent ? (1 << 7) : 0;
00141     checkSum += writeOutField(proxy, addr + 12, features2_5);
00142 
00143     checkSum = -checkSum;
00144     proxy.writeBlob(addr + 10, &checkSum, 1);
00145 
00146     return 16;

00191 Addr
00192 X86ISA::IntelMP::ConfigTable::writeOut(PortProxy& proxy, Addr addr)
00193 {
00194     uint8_t checkSum = 0;
00195 
00196     proxy.writeBlob(addr, (uint8_t *)signature, 4);
00197     for (int i = 0; i < 4; i++)
00198         checkSum += signature[i];
00199 
00200     // Base table length goes here but will be calculated later.
00201 
00202     proxy.writeBlob(addr + 6, (uint8_t *)(&specRev), 1);
00203     checkSum += specRev;
00204 
00205     // The checksum goes here but is still being calculated.
00206 
00207     checkSum += writeOutString(proxy, addr + 8, oemID, 8);
00208     checkSum += writeOutString(proxy, addr + 16, productID, 12);
00209 
00210     checkSum += writeOutField(proxy, addr + 28, oemTableAddr);
00211     checkSum += writeOutField(proxy, addr + 32, oemTableSize);
00212     checkSum += writeOutField(proxy, addr + 34, (uint16_t)baseEntries.size());
00213     checkSum += writeOutField(proxy, addr + 36, localApic);
00214 
00215     uint8_t reserved = 0;
00216     proxy.writeBlob(addr + 43, &reserved, 1);
00217     checkSum += reserved;
00218 
00219     vector<BaseConfigEntry *>::iterator baseEnt;
00220     uint16_t offset = 44;
00221     for (baseEnt = baseEntries.begin();
00222             baseEnt != baseEntries.end(); baseEnt++) {
00223         offset += (*baseEnt)->writeOut(proxy, addr + offset, checkSum);
00224     }
00225 
00226     // We've found the end of the base table this point.
00227     checkSum += writeOutField(proxy, addr + 4, offset);
00228 
00229     vector<ExtConfigEntry *>::iterator extEnt;
00230     uint16_t extOffset = 0;
00231     uint8_t extCheckSum = 0;
00232     for (extEnt = extEntries.begin();
00233             extEnt != extEntries.end(); extEnt++) {
00234         extOffset += (*extEnt)->writeOut(proxy,
00235                 addr + offset + extOffset, extCheckSum);
00236     }
00237 
00238     checkSum += writeOutField(proxy, addr + 40, extOffset);
00239     extCheckSum = -extCheckSum;
00240     checkSum += writeOutField(proxy, addr + 42, extCheckSum);
00241 
00242     // And now, we finally have the whole check sum completed.
00243     checkSum = -checkSum;
00244     writeOutField(proxy, addr + 7, checkSum);
00245 
00246     return offset + extOffset;
