00043 template <class T>
00044 inline bool
00045 isPrime(T n)
00046 {
00047     T i;
00048 
00049     if (n == 2 || n == 3)
00050         return true;
00051 
00052     // Don't try every odd number to prove if it is a prime.
00053     // Toggle between every 2nd and 4th number.
00054     // (This is because every 6th odd number is divisible by 3.)
00055     for (i = 5; i*i <= n; i += 6) {
00056         if (((n % i) == 0 ) || ((n % (i + 2)) == 0) ) {
00057             return false;
00058         }
00059     }
00060 
00061     return true;
00062 }

00080 {

00083 
00084     if (e == 0)
00085         return 1;
00086 
00087     uint64_t result = n;
00088     uint64_t old_result = 0;
00089     for (int x = 1; x < e; x++) {
00090         old_result = result;
00091         result *= n;
