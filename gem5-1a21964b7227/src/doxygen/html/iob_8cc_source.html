00089 void
00090 Iob::readIob(PacketPtr pkt)
00091 {
00092         Addr accessAddr = pkt->getAddr() - iobManAddr;
00093 
00094         if (accessAddr >= IntManAddr && accessAddr < IntManAddr + IntManSize) {
00095             int index = (accessAddr - IntManAddr) >> 3;
00096             uint64_t data = intMan[index].cpu << 8 | intMan[index].vector << 0;
00097             pkt->set(data);
00098             return;
00099         }
00100 
00101         if (accessAddr >= IntCtlAddr && accessAddr < IntCtlAddr + IntCtlSize) {
00102             int index = (accessAddr - IntCtlAddr) >> 3;
00103             uint64_t data = intCtl[index].mask  ? 1 << 2 : 0 |
00104                 intCtl[index].pend  ? 1 << 0 : 0;
00105             pkt->set(data);
00106             return;
00107         }
00108 
00109         if (accessAddr == JIntVecAddr) {
00110             pkt->set(jIntVec);
00111             return;
00112         }
00113 
00114         panic("Read to unknown IOB offset 0x%x\n", accessAddr);
00115 }

00182 void
00183 Iob::writeIob(PacketPtr pkt)
00184 {
00185         Addr accessAddr = pkt->getAddr() - iobManAddr;
00186         int index;
00187         uint64_t data;
00188 
00189         if (accessAddr >= IntManAddr && accessAddr < IntManAddr + IntManSize) {
00190             index = (accessAddr - IntManAddr) >> 3;
00191             data = pkt->get<uint64_t>();
00192             intMan[index].cpu = bits(data,12,8);
00193             intMan[index].vector = bits(data,5,0);
00194             DPRINTF(Iob, "Wrote IntMan %d cpu %d, vec %d\n", index,
00195                     intMan[index].cpu, intMan[index].vector);
00196             return;
00197         }
00198 
00199         if (accessAddr >= IntCtlAddr && accessAddr < IntCtlAddr + IntCtlSize) {
00200             index = (accessAddr - IntCtlAddr) >> 3;
00201             data = pkt->get<uint64_t>();
00202             intCtl[index].mask = bits(data,2,2);
00203             if (bits(data,1,1))
00204                 intCtl[index].pend = false;
00205             DPRINTF(Iob, "Wrote IntCtl %d pend %d cleared %d\n", index,
00206                     intCtl[index].pend, bits(data,2,2));
00207             return;
00208         }
00209 
00210         if (accessAddr == JIntVecAddr) {
00211             jIntVec = bits(pkt->get<uint64_t>(), 5,0);
00212             DPRINTF(Iob, "Wrote jIntVec %d\n", jIntVec);
00213             return;
00214         }
00215 
00216         if (accessAddr >= IntVecDisAddr && accessAddr < IntVecDisAddr + IntVecDisSize) {
00217             Type type;
00218             int cpu_id;
00219             int vector;
00220             index = (accessAddr - IntManAddr) >> 3;
00221             data = pkt->get<uint64_t>();
00222             type = (Type)bits(data,17,16);
00223             cpu_id = bits(data, 12,8);
00224             vector = bits(data,5,0);
00225             generateIpi(type,cpu_id, vector);
00226             return;
00227         }
00228 
00229         panic("Write to unknown IOB offset 0x%x\n", accessAddr);
00230 }

00273 void
00274 Iob::generateIpi(Type type, int cpu_id, int vector)
00275 {
00276     SparcISA::SparcFault<SparcISA::PowerOnReset> *por = new SparcISA::PowerOnReset();
00277     if (cpu_id >= sys->numContexts())
00278         return;
00279 
00280     switch (type) {
00281       case 0: // interrupt
00282         DPRINTF(Iob, "Generating interrupt because of I/O write to cpu: %d vec %d\n",
00283                 cpu_id, vector);
00284         ic->post(cpu_id, SparcISA::IT_INT_VEC, vector);
00285         break;
00286       case 1: // reset
00287         warn("Sending reset to CPU: %d\n", cpu_id);
00288         if (vector != por->trapType())
00289             panic("Don't know how to set non-POR reset to cpu\n");
00290         por->invoke(sys->threadContexts[cpu_id]);
00291         sys->threadContexts[cpu_id]->activate();
00292         break;
00293       case 2: // idle -- this means stop executing and don't wake on interrupts
00294         DPRINTF(Iob, "Idling CPU because of I/O write cpu: %d\n", cpu_id);
00295         sys->threadContexts[cpu_id]->halt();
00296         break;
00297       case 3: // resume
00298         DPRINTF(Iob, "Resuming CPU because of I/O write cpu: %d\n", cpu_id);
00299         sys->threadContexts[cpu_id]->activate();
00300         break;
00301       default:
00302         panic("Invalid type to generate ipi\n");
00303     }
00304 }

00307 Iob::receiveJBusInterrupt(int cpu_id, int source, uint64_t d0, uint64_t d1)
00308 {
00309     // If we are already dealing with an interrupt for that cpu we can't deal
00310     // with another one right now... come back later
00311     if (jIntBusy[cpu_id].busy)
00312         return false;
00313 
00314     DPRINTF(Iob, "Receiving jBus interrupt: %d for cpu %d vec %d\n",
00315             source, cpu_id, jIntVec);
00316 
00317     jIntBusy[cpu_id].busy = true;
00318     jIntBusy[cpu_id].source = source;
00319     jBusData0[cpu_id] = d0;
00320     jBusData1[cpu_id] = d1;
00321 
00322     ic->post(cpu_id, SparcISA::IT_INT_VEC, jIntVec);
00323     return true;
