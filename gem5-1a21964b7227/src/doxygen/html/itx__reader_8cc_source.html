00043 ITXReader::ITXReader(const string &name, const string &filename)
00044     : MemTraceReader(name)
00045 {
00046     if (strcmp((filename.c_str() + filename.length() -3), ".gz") == 0) {
00047         // Compressed file, need to use a pipe to gzip.
00048         stringstream buf;
00049         buf << "gzip -d -c " << filename << endl;
00050         trace = popen(buf.str().c_str(), "r");
00051     } else {
00052         trace = fopen(filename.c_str(), "rb");
00053     }
00054     if (!trace) {
00055         fatal("Can't open file %s", filename);
00056     }
00057     traceFormat = 0;
00058     int c;
00059     for (int i = 0; i < 4; ++i) {
00060         c = getc(trace);
00061         if (c == EOF) {
00062             fatal("Unexpected end of trace file.");
00063         }
00064         traceFormat |= (c & 0xff) << (8 * i);
00065     }
00066     if (traceFormat > 2)
00067         fatal("Invalid trace format.");
00068 }

00074     bool phys_val;
00075     do {
00076         int c = getc(trace);
00077         if (c != EOF) {
00078             // Decode first byte
00079             // phys_val<1> | type <2:0> | size <3:0>
00080             phys_val = c & 0x80;
00081             tmp_req->size = (c & 0x0f) + 1;
00082             int type = (c & 0x70) >> 4;
00083 
00084             // Could be a compressed instruction entry, expand if necessary
00085             if (type == ITXCodeComp) {
00086                 if (traceFormat != 2) {
00087                     fatal("Compressed code entry in non CompCode trace.");
00088                 }
00089                 if (!codeVirtValid) {
00090                     fatal("Corrupt CodeComp entry.");
00091                 }
00092 
00093                 tmp_req->vaddr = codeVirtAddr;
00094                 codeVirtAddr += tmp_req->size;
00095                 if (phys_val) {
00096                     if (!codePhysValid) {
00097                         fatal("Corrupt CodeComp entry.");
00098                     }
00099                     tmp_req->paddr = codePhysAddr;
00100                     if (((tmp_req->paddr & 0xfff) + tmp_req->size) & ~0xfff) {
00101                         // Crossed page boundary, next physical address is
00102                         // invalid
00103                         codePhysValid = false;
00104                     } else {
00105                         codePhysAddr += tmp_req->size;
00106                     }
00107                     assert(tmp_req->paddr >> 36 == 0);
00108                 } else {
00109                     codePhysValid = false;
00110                 }
00111                 type = ITXCode;
00112                 tmp_req->cmd = Read;
00113             } else {
00114                 // Normal entry
00115                 tmp_req->vaddr = 0;
00116                 for (int i = 0; i < 4; ++i) {
00117                     c = getc(trace);
00118                     if (c == EOF) {
00119                         fatal("Unexpected end of trace file.");
00120                     }
00121                     tmp_req->vaddr |= (c & 0xff) << (8 * i);
00122                 }
00123                 if (type == ITXCode) {
00124                     codeVirtAddr = tmp_req->vaddr + tmp_req->size;
00125                     codeVirtValid = true;
00126                 }
00127                 tmp_req->paddr = 0;
00128                 if (phys_val) {
00129                     c = getc(trace);
00130                     if (c == EOF) {
00131                         fatal("Unexpected end of trace file.");
00132                     }
00133                     // Get the page offset from the virtual address.
00134                     tmp_req->paddr = tmp_req->vaddr & 0xfff;
00135                     tmp_req->paddr |= (c & 0xf0) << 8;
00136                     tmp_req->paddr |= (Addr)(c & 0x0f) << 32;
00137                     for (int i = 2; i < 4; ++i) {
00138                         c = getc(trace);
00139                         if (c == EOF) {
00140                             fatal("Unexpected end of trace file.");
00141                         }
00142                         tmp_req->paddr |= (Addr)(c & 0xff) << (8 * i);
00143                     }
00144                     if (type == ITXCode) {
00145                         if (((tmp_req->paddr & 0xfff) + tmp_req->size)
00146                             & ~0xfff) {
00147                             // Crossing the page boundary, next physical
00148                             // address isn't valid
00149                             codePhysValid = false;
00150                         } else {
00151                             codePhysAddr = tmp_req->paddr + tmp_req->size;
00152                             codePhysValid = true;
00153                         }
00154                     }
00155                     assert(tmp_req->paddr >> 36 == 0);
00156                 } else if (type == ITXCode) {
00157                     codePhysValid = false;
00158                 }
00159                 switch(type) {
00160                   case ITXRead:
00161                     tmp_req->cmd = Read;
00162                     break;
00163                   case ITXWrite:
00164                     tmp_req->cmd = Write;
00165                     break;
00166                   case ITXWriteback:
00167                     tmp_req->cmd = Writeback;
00168                     break;
00169                   case ITXCode:

00172                     break;

00174                     fatal("Unknown ITX type");
00175                 }
00176             }
00177         } else {
00178             // EOF need to return a null request
00179             MemReqPtr null_req;
00180             req = null_req;
00181             return 0;
00182         }
00183     } while (!phys_val);
00184     req = tmp_req;
00185     assert(!req || (req->paddr >> 36) == 0);
