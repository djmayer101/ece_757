00173 void
00174 Pl050::processCommand(uint8_t byte)
00175 {
00176     using namespace Ps2;
00177 
00178     if (ackNext) {
00179         ackNext--;
00180         rxQueue.push_back(Ack);
00181         updateIntStatus();
00182         return;
00183     }
00184 
00185     switch (byte) {
00186       case Ps2Reset:
00187         rxQueue.push_back(Ack);
00188         rxQueue.push_back(SelfTestPass);
00189         break;
00190       case SetResolution:
00191       case SetRate:
00192       case SetStatusLed:
00193       case SetScaling1_1:
00194       case SetScaling1_2:
00195         rxQueue.push_back(Ack);
00196         ackNext = 1;
00197         break;
00198       case ReadId:
00199         rxQueue.push_back(Ack);
00200         if (params()->is_mouse)
00201             rxQueue.push_back(MouseId);
00202         else
00203             rxQueue.push_back(KeyboardId);
00204         break;
00205       case TpReadId:
00206         if (!params()->is_mouse)
00207             break;
00208         // We're not a trackpoint device, this should make the probe go away
00209         rxQueue.push_back(Ack);
00210         rxQueue.push_back(0);
00211         rxQueue.push_back(0);
00212         // fall through
00213       case Disable:
00214       case Enable:
00215       case SetDefaults:
00216         rxQueue.push_back(Ack);
00217         break;
00218       case StatusRequest:
00219         rxQueue.push_back(Ack);
00220         rxQueue.push_back(0);
00221         rxQueue.push_back(2); // default resolution
00222         rxQueue.push_back(100); // default sample rate
00223         break;
00224       case TouchKitId:
00225         ackNext = 2;
00226         rxQueue.push_back(Ack);
00227         rxQueue.push_back(TouchKitId);
00228         rxQueue.push_back(1);
00229         rxQueue.push_back('A');
00230 
00231         driverInitialized = true;
00232         break;
00233       default:
00234         panic("Unknown byte received: %d\n", byte);
00235     }
00236 
00237     updateIntStatus();
00238 }

00274     // If the driver hasn't initialized the device yet, no need to try and send

00280 
00281     // We shouldn't be here unless a vnc server called us in which case

00285     // Convert screen coordinates to touchpad coordinates

00287     uint16_t _y = (2047.0/vnc->videoHeight()) * y;
00288 
00289     rxQueue.push_back(buttons);

00291     rxQueue.push_back(_x & 0x7f);
