00069 void
00070 setupSharedData()
00071 {
00072     int shmfd = shmget('M' << 24 | getuid(), sizeof(SharedData), 0777);
00073     if (shmfd < 0)
00074         fatal("Couldn't get shared memory fd. Is Legion running?");
00075 
00076     shared_data = (SharedData*)shmat(shmfd, NULL, SHM_RND);
00077     if (shared_data == (SharedData*)-1)
00078         fatal("Couldn't allocate shared memory");
00079 
00080     if (shared_data->flags != OWN_M5)
00081         fatal("Shared memory has invalid owner");
00082 
00083     if (shared_data->version != VERSION)
00084         fatal("Shared Data is wrong version! M5: %d Legion: %d", VERSION,
00085               shared_data->version);
00086 
00087     // step legion forward one cycle so we can get register values
00088     shared_data->flags = OWN_LEGION;
00089 }

00096 inline char * genCenteredLabel(int length, char * buffer, const char * label)
00097 {
00098     int labelLength = strlen(label);
00099     assert(labelLength <= length);
00100     int leftPad = (length - labelLength) / 2;
00101     int rightPad = length - leftPad - labelLength;
00102     char format[64];
00103     sprintf(format, "%%%ds%%s%%%ds", leftPad, rightPad);
00104     sprintf(buffer, format, "", label, "");
00105     return buffer;
00106 }
