00047 // create and initialize a LRU/MRU cache structure
00048 LRU::LRU(unsigned _numSets, unsigned _blkSize, unsigned _assoc,
00049          unsigned _hit_latency)
00050     : numSets(_numSets), blkSize(_blkSize), assoc(_assoc),
00051       hitLatency(_hit_latency)
00052 {
00053     // Check parameters
00054     if (blkSize < 4 || !isPowerOf2(blkSize)) {
00055         fatal("Block size must be at least 4 and a power of 2");
00056     }
00057     if (numSets <= 0 || !isPowerOf2(numSets)) {
00058         fatal("# of sets must be non-zero and a power of 2");
00059     }
00060     if (assoc <= 0) {
00061         fatal("associativity must be greater than zero");
00062     }
00063     if (hitLatency <= 0) {
00064         fatal("access latency must be greater than zero");
00065     }
00066 
00067     blkMask = blkSize - 1;
00068     setShift = floorLog2(blkSize);
00069     setMask = numSets - 1;
00070     tagShift = setShift + floorLog2(numSets);
00071     warmedUp = false;
00073     warmupBound = numSets * assoc;
00074 
00075     sets = new CacheSet[numSets];
00076     blks = new BlkType[numSets * assoc];
00077     // allocate data storage in one big chunk
00078     numBlocks = numSets * assoc;
00079     dataBlks = new uint8_t[numBlocks * blkSize];
00080 
00081     unsigned blkIndex = 0;       // index into blks array
00082     for (unsigned i = 0; i < numSets; ++i) {
00083         sets[i].assoc = assoc;
00084 
00085         sets[i].blks = new BlkType*[assoc];
00086 
00087         // link in the data blocks
00088         for (unsigned j = 0; j < assoc; ++j) {
00089             // locate next cache block
00090             BlkType *blk = &blks[blkIndex];
00091             blk->data = &dataBlks[blkSize*blkIndex];
00092             ++blkIndex;
00093 
00094             // invalidate new cache block
00095             blk->invalidate();
00096 
00097             //EGH Fix Me : do we need to initialize blk?
00098 
00099             // Setting the tag to j is just to prevent long chains in the hash
00100             // table; won't matter because the block is invalid
00101             blk->tag = j;
00102             blk->whenReady = 0;
00103             blk->isTouched = false;
00104             blk->size = blkSize;
00105             sets[i].blks[j]=blk;
00106             blk->set = i;
00107         }
00108     }
00109 }

00167     if (!blk->isTouched) {

00175 
00176     // If we're replacing a block that was previously valid update
00177     // stats for it. This can't be done in findBlock() because a
