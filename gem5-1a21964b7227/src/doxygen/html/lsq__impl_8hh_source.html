00057 LSQ<Impl>::LSQ(O3CPU *cpu_ptr, IEW *iew_ptr, DerivO3CPUParams *params)
00058     : cpu(cpu_ptr), iewStage(iew_ptr),
00059       LQEntries(params->LQEntries),
00060       SQEntries(params->SQEntries),
00061       numThreads(params->numThreads),
00062       retryTid(-1)
00063 {
00064     //**********************************************/
00065     //************ Handle SMT Parameters ***********/
00066     //**********************************************/
00067     std::string policy = params->smtLSQPolicy;
00068 
00069     //Convert string to lowercase
00070     std::transform(policy.begin(), policy.end(), policy.begin(),
00071                    (int(*)(int)) tolower);
00072 
00073     //Figure out fetch policy
00074     if (policy == "dynamic") {
00075         lsqPolicy = Dynamic;
00076 
00077         maxLQEntries = LQEntries;
00078         maxSQEntries = SQEntries;
00079 
00080         DPRINTF(LSQ, "LSQ sharing policy set to Dynamic\n");
00081     } else if (policy == "partitioned") {
00082         lsqPolicy = Partitioned;
00083 
00084         //@todo:make work if part_amt doesnt divide evenly.
00085         maxLQEntries = LQEntries / numThreads;
00086         maxSQEntries = SQEntries / numThreads;
00087 
00088         DPRINTF(Fetch, "LSQ sharing policy set to Partitioned: "
00089                 "%i entries per LQ | %i entries per SQ\n",
00090                 maxLQEntries,maxSQEntries);
00091     } else if (policy == "threshold") {
00092         lsqPolicy = Threshold;
00093 
00094         assert(params->smtLSQThreshold > LQEntries);
00095         assert(params->smtLSQThreshold > SQEntries);
00096 
00097         //Divide up by threshold amount
00098         //@todo: Should threads check the max and the total
00099         //amount of the LSQ
00100         maxLQEntries  = params->smtLSQThreshold;
00101         maxSQEntries  = params->smtLSQThreshold;
00102 
00103         DPRINTF(LSQ, "LSQ sharing policy set to Threshold: "
00104                 "%i entries per LQ | %i entries per SQ\n",
00105                 maxLQEntries,maxSQEntries);
00106     } else {
00107         assert(0 && "Invalid LSQ Sharing Policy.Options Are:{Dynamic,"
00108                     "Partitioned, Threshold}");
00109     }
00110 
00111     //Initialize LSQs
00112     for (ThreadID tid = 0; tid < numThreads; tid++) {
00113         thread[tid].init(cpu, iew_ptr, params, this,
00114                          maxLQEntries, maxSQEntries, tid);
00115         thread[tid].setDcachePort(&cpu_ptr->getDataPort());
00116     }
00117 }
