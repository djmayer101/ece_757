00050 int
00051 LWBackEnd<Impl>::wakeDependents(DynInstPtr &inst, bool memory_deps)
00052 {
00053     assert(!inst->isSquashed());
00054     std::vector<DynInstPtr> &dependents = memory_deps ? inst->getMemDeps() :
00055         inst->getDependents();
00056     int num_outputs = dependents.size();
00057 
00058     DPRINTF(BE, "Waking instruction [sn:%lli] dependents in IQ\n", inst->seqNum);
00059 
00060     for (int i = 0; i < num_outputs; i++) {
00061         DynInstPtr dep_inst = dependents[i];
00062         if (!memory_deps) {
00063             dep_inst->markSrcRegReady();
00064         } else {
00065             if (!dep_inst->isSquashed())
00066                 dep_inst->markMemInstReady(inst.get());
00067         }
00068 
00069         DPRINTF(BE, "Marking source reg ready [sn:%lli] in IQ\n", dep_inst->seqNum);
00070 
00071         if (dep_inst->readyToIssue() && dep_inst->isInROB() &&
00072             !dep_inst->isNonSpeculative() && !dep_inst->isStoreConditional() &&
00073             dep_inst->memDepReady() && !dep_inst->isMemBarrier() &&
00074             !dep_inst->isWriteBarrier()) {
00075             DPRINTF(BE, "Adding instruction to exeList [sn:%lli]\n",
00076                     dep_inst->seqNum);
00077             exeList.push(dep_inst);
00078             if (dep_inst->iqItValid) {
00079                 DPRINTF(BE, "Removing instruction from waiting list\n");
00080                 waitingList.erase(dep_inst->iqIt);
00081                 waitingInsts--;
00082                 dep_inst->iqItValid = false;
00083                 assert(waitingInsts >= 0);
00084             }
00085             if (dep_inst->isMemRef()) {
00086                 removeWaitingMemOp(dep_inst);
00087                 DPRINTF(BE, "Issued a waiting mem op [sn:%lli]\n",
00088                         dep_inst->seqNum);
00089             }
00090         }
00091     }
00092     return num_outputs;
00093 }

00519 template <class Impl>
00520 void
00521 LWBackEnd<Impl>::checkInterrupts()
00522 {
00523     if (cpu->checkInterrupts(tc) && !trapSquash && !tcSquash) {
00524         frontEnd->interruptPending = true;
00525         if (robEmpty() && !LSQ.hasStoresToWB()) {
00526             // Will need to squash all instructions currently in flight and have
00527             // the interrupt handler restart at the last non-committed inst.
00528             // Most of that can be handled through the trap() function.  The
00529             // processInterrupts() function really just checks for interrupts
00530             // and then calls trap() if there is an interrupt present.
00531 
00532             // Not sure which thread should be the one to interrupt.  For now
00533             // always do thread 0.
00534             assert(!thread->noSquashFromTC);
00535             thread->noSquashFromTC = true;
00536 
00537             // CPU will handle implementation of the interrupt.
00538             cpu->processInterrupts();
00539 
00540             // Now squash or record that I need to squash this cycle.
00541             commitStatus = TrapPending;
00542 
00543             // Exit state update mode to avoid accidental updating.
00544             thread->noSquashFromTC = false;
00545 
00546             // Generate trap squash event.
00547             generateTrapEvent();
00548 
00549             DPRINTF(BE, "Interrupt detected.\n");
00550         } else {
00551             DPRINTF(BE, "Interrupt must wait for ROB to drain.\n");
00552         }
00553     }
00554 }

00664 template <class Impl>
00665 void
00666 LWBackEnd<Impl>::dispatchInsts()
00667 {
00668     DPRINTF(BE, "Trying to dispatch instructions.\n");
00669 
00670     while (numInsts < numROBEntries &&
00671            numWaitingMemOps < maxOutstandingMemOps) {
00672         // Get instruction from front of time buffer
00673         if (lsqLimits && LSQ.isFull()) {
00674             break;
00675         }
00676 
00677         DynInstPtr inst = frontEnd->getInst();
00678         if (!inst) {
00679             break;
00680         } else if (inst->isSquashed()) {
00681             continue;
00682         }
00683 
00684         ++numInsts;
00685         instList.push_front(inst);
00686 
00687         inst->setInROB();
00688 
00689         DPRINTF(BE, "Dispatching instruction [sn:%lli] PC:%#x\n",
00690                 inst->seqNum, inst->readPC());
00691 
00692         for (int i = 0; i < inst->numDestRegs(); ++i)
00693             renameTable[inst->destRegIdx(i)] = inst;
00694 
00695         if (inst->isMemBarrier() || inst->isWriteBarrier()) {
00696             if (memBarrier) {
00697                 DPRINTF(BE, "Instruction [sn:%lli] is waiting on "
00698                         "barrier [sn:%lli].\n",
00699                         inst->seqNum, memBarrier->seqNum);
00700                 memBarrier->addMemDependent(inst);
