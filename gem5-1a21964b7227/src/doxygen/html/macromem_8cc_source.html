00054 MacroMemOp::MacroMemOp(const char *mnem, ExtMachInst machInst,
00055                        OpClass __opClass, IntRegIndex rn,
00056                        bool index, bool up, bool user, bool writeback,
00057                        bool load, uint32_t reglist) :
00058     PredMacroOp(mnem, machInst, __opClass)
00059 {
00060     uint32_t regs = reglist;
00061     uint32_t ones = number_of_ones(reglist);
00062     // Remember that writeback adds a uop or two and the temp register adds one
00063     numMicroops = ones + (writeback ? (load ? 2 : 1) : 0) + 1;
00064 
00065     // It's technically legal to do a lot of nothing
00066     if (!ones)
00067         numMicroops = 1;
00068 
00069     microOps = new StaticInstPtr[numMicroops];
00070     uint32_t addr = 0;
00071 
00072     if (!up)
00073         addr = (ones << 2) - 4;
00074 
00075     if (!index)
00076         addr += 4;
00077 
00078     StaticInstPtr *uop = microOps;
00079 
00080     // Add 0 to Rn and stick it in ureg0.
00081     // This is equivalent to a move.
00082     *uop = new MicroAddiUop(machInst, INTREG_UREG0, rn, 0);
00083 
00084     unsigned reg = 0;
00085     unsigned regIdx = 0;
00086     bool force_user = user & !bits(reglist, 15);
00087     bool exception_ret = user & bits(reglist, 15);
00088 
00089     for (int i = 0; i < ones; i++) {
00090         // Find the next register.
00091         while (!bits(regs, reg))
00092             reg++;
00093         replaceBits(regs, reg, 0);
00094 
00095         regIdx = reg;
00096         if (force_user) {
00097             regIdx = intRegInMode(MODE_USER, regIdx);
00098         }
00099 
00100         if (load) {
00101             if (writeback && i == ones - 1) {
00102                 // If it's a writeback and this is the last register
00103                 // do the load into a temporary register which we'll move
00104                 // into the final one later
00105                 *++uop = new MicroLdrUop(machInst, INTREG_UREG1, INTREG_UREG0,
00106                         up, addr);
00107             } else {
00108                 // Otherwise just do it normally
00109                 if (reg == INTREG_PC && exception_ret) {
00110                     // This must be the exception return form of ldm.
00111                     *++uop = new MicroLdrRetUop(machInst, regIdx,
00112                                                INTREG_UREG0, up, addr);
00113                 } else {
00114                     *++uop = new MicroLdrUop(machInst, regIdx,
00115                                             INTREG_UREG0, up, addr);
00116                     if (reg == INTREG_PC) {
00117                         (*uop)->setFlag(StaticInst::IsControl);
00118                         if (!(condCode == COND_AL || condCode == COND_UC))
00119                             (*uop)->setFlag(StaticInst::IsCondControl);
00120                         else
00121                             (*uop)->setFlag(StaticInst::IsUncondControl);
00122                         (*uop)->setFlag(StaticInst::IsIndirectControl);
00123                     }
00124                 }
00125             }
00126         } else {
00127             *++uop = new MicroStrUop(machInst, regIdx, INTREG_UREG0, up, addr);
00128         }
00129 
00130         if (up)
00131             addr += 4;
00132         else
00133             addr -= 4;
00134     }
00135 
00136     if (writeback && ones) {
00137         // put the register update after we're done all loading
00138         if (up)
00139             *++uop = new MicroAddiUop(machInst, rn, rn, ones * 4);
00140         else
00141             *++uop = new MicroSubiUop(machInst, rn, rn, ones * 4);
00142 
00143         // If this was a load move the last temporary value into place
00144         // this way we can't take an exception after we update the base
00145         // register.
00146         if (load && reg == INTREG_PC && exception_ret) {
00147             *++uop = new MicroUopRegMovRet(machInst, 0, INTREG_UREG1);
00148             if (!(condCode == COND_AL || condCode == COND_UC))
00149                 (*uop)->setFlag(StaticInst::IsCondControl);
00150             else
00151                 (*uop)->setFlag(StaticInst::IsUncondControl);
00152         } else if (load) {
00153             *++uop = new MicroUopRegMov(machInst, regIdx, INTREG_UREG1);
00154             if (reg == INTREG_PC) {
00155                 (*uop)->setFlag(StaticInst::IsControl);
00156                 (*uop)->setFlag(StaticInst::IsCondControl);
00157                 (*uop)->setFlag(StaticInst::IsIndirectControl);
00158                 // This is created as a RAS POP
00159                 if (rn == INTREG_SP)
00160                     (*uop)->setFlag(StaticInst::IsReturn);
00161 
00162             }
00163         }
00164     }
00165 
00166     (*uop)->setLastMicroop();
00167 
00168     for (StaticInstPtr *curUop = microOps;
00169             !(*curUop)->isLastMicroop(); curUop++) {
00170         MicroOp * uopPtr = dynamic_cast<MicroOp *>(curUop->get());
00171         assert(uopPtr);
00172         uopPtr->setDelayedCommit();
00173     }
00174 }

00833 MacroVFPMemOp::MacroVFPMemOp(const char *mnem, ExtMachInst machInst,
00834                              OpClass __opClass, IntRegIndex rn,
00835                              RegIndex vd, bool single, bool up,
00836                              bool writeback, bool load, uint32_t offset) :
00837     PredMacroOp(mnem, machInst, __opClass)
00838 {
00839     int i = 0;
00840 
00841     // The lowest order bit selects fldmx (set) or fldmd (clear). These seem
00842     // to be functionally identical except that fldmx is deprecated. For now
00843     // we'll assume they're otherwise interchangable.
00844     int count = (single ? offset : (offset / 2));
00845     if (count == 0 || count > NumFloatArchRegs)
00846         warn_once("Bad offset field for VFP load/store multiple.\n");
00847     if (count == 0) {
00848         // Force there to be at least one microop so the macroop makes sense.
00849         writeback = true;
00850     }
00851     if (count > NumFloatArchRegs)
00852         count = NumFloatArchRegs;
00853 
00854     numMicroops = count * (single ? 1 : 2) + (writeback ? 1 : 0);
00855     microOps = new StaticInstPtr[numMicroops];
00856 
00857     int64_t addr = 0;
00858 
00859     if (!up)
00860         addr = 4 * offset;
00861 
00862     bool tempUp = up;
00863     for (int j = 0; j < count; j++) {
00864         if (load) {
00865             if (single) {
00866                 microOps[i++] = new MicroLdrFpUop(machInst, vd++, rn,
00867                                                   tempUp, addr);
00868             } else {
00869                 microOps[i++] = new MicroLdrDBFpUop(machInst, vd++, rn,
00870                                                     tempUp, addr);
00871                 microOps[i++] = new MicroLdrDTFpUop(machInst, vd++, rn, tempUp,
00872                                                     addr + (up ? 4 : -4));
00873             }
00874         } else {
00875             if (single) {
00876                 microOps[i++] = new MicroStrFpUop(machInst, vd++, rn,
00877                                                   tempUp, addr);
00878             } else {
00879                 microOps[i++] = new MicroStrDBFpUop(machInst, vd++, rn,
00880                                                     tempUp, addr);
00881                 microOps[i++] = new MicroStrDTFpUop(machInst, vd++, rn, tempUp,
00882                                                     addr + (up ? 4 : -4));
00883             }
00884         }
00885         if (!tempUp) {
00886             addr -= (single ? 4 : 8);
00887             // The microops don't handle negative displacement, so turn if we
00888             // hit zero, flip polarity and start adding.
00889             if (addr <= 0) {
00890                 tempUp = true;
00891                 addr = -addr;
00892             }
00893         } else {
00894             addr += (single ? 4 : 8);
00895         }
00896     }
00897 
00898     if (writeback) {
00899         if (up) {
00900             microOps[i++] =
00901                 new MicroAddiUop(machInst, rn, rn, 4 * offset);
00902         } else {
00903             microOps[i++] =
00904                 new MicroSubiUop(machInst, rn, rn, 4 * offset);
00905         }
00906     }
00907 
00908     assert(numMicroops == i);
00909     microOps[numMicroops - 1]->setLastMicroop();

00911     for (StaticInstPtr *curUop = microOps;
00912             !(*curUop)->isLastMicroop(); curUop++) {
