00051 static inline unsigned int
00052 number_of_ones(int32_t val)
00053 {
00054     uint32_t ones = 0;
00055     for (int i = 0; i < 32; i++ )
00056     {
00057         if ( val & (1<<i) )
00058             ones++;
00059     }
00060     return ones;
