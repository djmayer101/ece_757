00047 static uint8_t
00048 bcdize(uint8_t val)
00049 {
00050     uint8_t result;
00051     result = val % 10;
00052     result += (val / 10) << 4;

00056 static uint8_t
00057 unbcdize(uint8_t val)
00058 {
00059     uint8_t result;

00070     // Unix is 0-11 for month, data seet says start at 1
00071     mon = time.tm_mon + 1;

00086         mday = bcdize(mday);

00113 void
00114 MC146818::writeData(const uint8_t addr, const uint8_t data)
00115 {
00116     if (addr < RTC_STAT_REGA) {
00117         clock_data[addr] = data;
00118         curTime.tm_sec = unbcdize(sec);
00119         curTime.tm_min = unbcdize(min);
00120         curTime.tm_hour = unbcdize(hour);
00121         curTime.tm_mday = unbcdize(mday);
00122         curTime.tm_mon = unbcdize(mon) - 1;
00123         curTime.tm_year = ((unbcdize(year) + 50) % 100) + 1950;
00124         curTime.tm_wday = unbcdize(wday) - 1;
00125     } else {
00126         switch (addr) {
00127           case RTC_STAT_REGA:
00128             // The "update in progress" bit is read only.
00129             if ((data & ~RTCA_UIP) != (RTCA_32768HZ | RTCA_1024HZ))
00130                 panic("Unimplemented RTC register A value write!\n");
00131             replaceBits(stat_regA, data, 6, 0);
00132             break;
00133           case RTC_STAT_REGB:
00134             if ((data & ~(RTCB_PRDC_IE | RTCB_SQWE)) != RTCB_24HR)
00135                 panic("Write to RTC reg B bits that are not implemented!\n");
00136 
00137             if (data & RTCB_PRDC_IE) {
00138                 if (!event.scheduled())
00139                     event.scheduleIntr();
00140             } else {
00141                 if (event.scheduled())
00142                     deschedule(event);
00143             }
00144             stat_regB = data;
00145             break;
00146           case RTC_STAT_REGC:
00147           case RTC_STAT_REGD:
00148             panic("RTC status registers C and D are not implemented.\n");
00149             break;
00150         }
00151     }
00152 }

00154 uint8_t
00155 MC146818::readData(uint8_t addr)
00156 {
00157     if (addr < RTC_STAT_REGA)
00158         return clock_data[addr];
00159     else {
00160         switch (addr) {
00161           case RTC_STAT_REGA:
00162             // toggle UIP bit for linux
00163             stat_regA ^= RTCA_UIP;
00164             return stat_regA;
00165             break;
00166           case RTC_STAT_REGB:
00167             return stat_regB;
00168             break;
00169           case RTC_STAT_REGC:
00170           case RTC_STAT_REGD:
00171             return 0x00;
00172             break;
00173           default:
00174             panic("Shouldn't be here");
00175         }
00176     }
00177 }
