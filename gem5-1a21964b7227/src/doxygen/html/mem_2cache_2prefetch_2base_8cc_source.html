00216                 delete iter->pkt;
00217                 iter = pf.erase(iter);
00218                 if (iter != pf.begin())
00219                     iter--;
00220             }
00221             if (pf.empty())
00222                 cache->deassertMemSideBusRequest(BaseCache::Request_PF);
00223         }

00233             Addr addr = *addrIter;
00234
