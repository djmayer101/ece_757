00157 void
00158 MemDepUnit<MemDepPred, Impl>::insert(DynInstPtr &inst)
00159 {
00160     ThreadID tid = inst->threadNumber;
00161 
00162     MemDepEntryPtr inst_entry = new MemDepEntry(inst);
00163 
00164     // Add the MemDepEntry to the hash.
00165     memDepHash.insert(
00166         std::pair<InstSeqNum, MemDepEntryPtr>(inst->seqNum, inst_entry));
00167 #ifdef DEBUG
00168     MemDepEntry::memdep_insert++;
00169 #endif
00170 
00171     instList[tid].push_back(inst);
00172 
00173     inst_entry->listIt = --(instList[tid].end());
00174 
00175     // Check any barriers and the dependence predictor for any
00176     // producing memrefs/stores.
00177     InstSeqNum producing_store;
00178     if (inst->isLoad() && loadBarrier) {
00179         DPRINTF(MemDepUnit, "Load barrier [sn:%lli] in flight\n",
00180                 loadBarrierSN);
00181         producing_store = loadBarrierSN;
00182     } else if (inst->isStore() && storeBarrier) {
00183         DPRINTF(MemDepUnit, "Store barrier [sn:%lli] in flight\n",
00184                 storeBarrierSN);
00185         producing_store = storeBarrierSN;
00186     } else {
00187         producing_store = depPred.checkInst(inst->instAddr());
00188     }
00189 
00190     MemDepEntryPtr store_entry = NULL;
00191 
00192     // If there is a producing store, try to find the entry.
00193     if (producing_store != 0) {
00194         DPRINTF(MemDepUnit, "Searching for producer\n");
00195         MemDepHashIt hash_it = memDepHash.find(producing_store);
00196 
00197         if (hash_it != memDepHash.end()) {
00198             store_entry = (*hash_it).second;
00199             DPRINTF(MemDepUnit, "Proucer found\n");
00200         }
00201     }
00202 
00203     // If no store entry, then instruction can issue as soon as the registers
00204     // are ready.
00205     if (!store_entry) {
00206         DPRINTF(MemDepUnit, "No dependency for inst PC "
00207                 "%s [sn:%lli].\n", inst->pcState(), inst->seqNum);
00208 
00209         inst_entry->memDepReady = true;
00210 
00211         if (inst->readyToIssue()) {
00212             inst_entry->regsReady = true;
00213 
00214             moveToReady(inst_entry);
00215         }
00216     } else {
00217         // Otherwise make the instruction dependent on the store/barrier.
00218         DPRINTF(MemDepUnit, "Adding to dependency list; "
00219                 "inst PC %s is dependent on [sn:%lli].\n",
00220                 inst->pcState(), producing_store);
00221 
00222         if (inst->readyToIssue()) {
00223             inst_entry->regsReady = true;
00224         }
00225 
00226         // Clear the bit saying this instruction can issue.
00227         inst->clearCanIssue();
00228 
00229         // Add this instruction to the list of dependents.
00230         store_entry->dependInsts.push_back(inst_entry);
00231 
00232         if (inst->isLoad()) {
00233             ++conflictingLoads;
00234         } else {
00235             ++conflictingStores;
00236         }
00237     }
00238 
00239     if (inst->isStore()) {
00240         DPRINTF(MemDepUnit, "Inserting store PC %s [sn:%lli].\n",
00241                 inst->pcState(), inst->seqNum);
00242 
00243         depPred.insertStore(inst->instAddr(), inst->seqNum, inst->threadNumber);
00244 
00245         ++insertedStores;
00246     } else if (inst->isLoad()) {
00247         ++insertedLoads;
00248     } else {
00249         panic("Unknown type! (most likely a barrier).");
00250     }
00251 }

00435 void
00436 MemDepUnit<MemDepPred, Impl>::wakeDependents(DynInstPtr &inst)
00437 {
00438     // Only stores and barriers have dependents.
00439     if (!inst->isStore() && !inst->isMemBarrier() && !inst->isWriteBarrier()) {
00440         return;
00441     }
00442 
00443     MemDepEntryPtr inst_entry = findInHash(inst);
00444 
00445     for (int i = 0; i < inst_entry->dependInsts.size(); ++i ) {
00446         MemDepEntryPtr woken_inst = inst_entry->dependInsts[i];
00447 
00448         if (!woken_inst->inst) {
00449             // Potentially removed mem dep entries could be on this list
00450             continue;
00451         }
00452 
00453         DPRINTF(MemDepUnit, "Waking up a dependent inst, "
00454                 "[sn:%lli].\n",
00455                 woken_inst->inst->seqNum);
00456 
00457         if (woken_inst->regsReady && !woken_inst->squashed) {
00458             moveToReady(woken_inst);
00459         } else {
00460             woken_inst->memDepReady = true;
00461         }
00462     }

00471 {

00485     --squash_it;

00488

00493                 (*squash_it)->seqNum);

00496               loadBarrier = false;

00499               storeBarrier = false;

00516

00518     depPred.squash(squashed_num, tid);
