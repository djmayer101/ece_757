00158 void
00159 MemTest::completeRequest(PacketPtr pkt)
00160 {
00161     Request *req = pkt->req;
00162 
00163     if (issueDmas) {
00164         dmaOutstanding = false;
00165     }
00166 
00167     DPRINTF(MemTest, "completing %s at address %x (blk %x) %s\n",
00168             pkt->isWrite() ? "write" : "read",
00169             req->getPaddr(), blockAddr(req->getPaddr()),
00170             pkt->isError() ? "error" : "success");
00171 
00172     MemTestSenderState *state =
00173         dynamic_cast<MemTestSenderState *>(pkt->senderState);
00174 
00175     uint8_t *data = state->data;
00176     uint8_t *pkt_data = pkt->getPtr<uint8_t>();
00177 
00178     //Remove the address from the list of outstanding
00179     std::set<unsigned>::iterator removeAddr =
00180         outstandingAddrs.find(req->getPaddr());
00181     assert(removeAddr != outstandingAddrs.end());
00182     outstandingAddrs.erase(removeAddr);
00183 
00184     if (pkt->isError()) {
00185         if (!suppress_func_warnings) {
00186           warn("Functional %s access failed at %#x\n",
00187                pkt->isWrite() ? "write" : "read", req->getPaddr());
00188         }
00189     } else {
00190         if (pkt->isRead()) {
00191             if (memcmp(pkt_data, data, pkt->getSize()) != 0) {
00192                 panic("%s: read of %x (blk %x) @ cycle %d "
00193                       "returns %x, expected %x\n", name(),
00194                       req->getPaddr(), blockAddr(req->getPaddr()), curTick(),
00195                       *pkt_data, *data);
00196             }
00197 
00198             numReads++;
00199             numReadsStat++;
00200 
00201             if (numReads == (uint64_t)nextProgressMessage) {
00202                 ccprintf(cerr, "%s: completed %d read, %d write accesses @%d\n",
00203                          name(), numReads, numWrites, curTick());
00204                 nextProgressMessage += progressInterval;
00205             }
00206 
00207             if (maxLoads != 0 && numReads >= maxLoads)
00208                 exitSimLoop("maximum number of loads reached");
00209         } else {
00210             assert(pkt->isWrite());
00211             funcProxy.writeBlob(req->getPaddr(), pkt_data, req->getSize());
00212             numWrites++;
00213             numWritesStat++;
00214         }
00215     }
00216 
00217     noResponseCycles = 0;
00218     delete state;
00219     delete [] data;
00220     delete pkt->req;
00221     delete pkt;
00222 }

00275     //mem tester
00276     //We can eliminate the lower bits of the offset, and then use the id
00277     //to offset within the blks
00278     offset = blockAddr(offset);

00280     access_size = 0;
00281     dma_access_size = 0;
00282

00284     Addr paddr;
00285

00287         flags.set(Request::UNCACHEABLE);

00290         paddr = ((base) ? baseAddr1 : baseAddr2) + offset;

00297         return;

00299 
00300     bool do_functional = (random() % 100 < percentFunctional) && !uncacheable;
00301     Request *req = new Request();

00303 
00304     if (issueDmas) {

00310         req->setPhys(paddr, 1 << access_size, flags, masterId);
00311         req->setThreadContext(id,0);

00313     assert(req->getSize() == 1);
00314 
00315     if (cmd < percentReads) {
00316         // read

00320         funcProxy.readBlob(req->getPaddr(), result, req->getSize());
00321

00327         PacketPtr pkt = new Packet(req, MemCmd::ReadReq);

00330         pkt->senderState = state;
00331 
00332         if (do_functional) {
00333             assert(pkt->needsResponse());
