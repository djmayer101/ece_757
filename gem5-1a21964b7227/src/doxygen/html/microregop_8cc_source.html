00057             if(findCarry(dataSize*8, _dest, _src1, _src2))

00067                 flags |= AFBit;

00069                 flags ^= AFBit;
00070         }
00071         if(flagMask & (EZFBit | ZFBit) && findZero(dataSize*8, _dest))
00072             flags |= (flagMask & (EZFBit | ZFBit));
00073         if(flagMask & SFBit && findNegative(dataSize*8, _dest))
00074             flags |= SFBit;
00075         if(flagMask & OFBit && findOverflow(dataSize*8, _dest, _src1, _src2))
00076             flags |= OFBit;
