00109 Fault
00110 Interrupts::getInterrupt(ThreadContext * tc)
00111 {
00112     DPRINTF(Interrupt, "Interrupts getInterrupt\n");
00113 
00114     //Check if there are any outstanding interrupts
00115     StatusReg status = tc->readMiscRegNoEffect(MISCREG_STATUS);
00116     // Interrupts must be enabled, error level must be 0 or interrupts
00117     // inhibited, and exception level must be 0 or interrupts inhibited
00118     if ((status.ie == 1) && (status.erl == 0) && (status.exl == 0)) {
00119         // Software interrupts & hardware interrupts are handled in software.
00120         // So if any interrupt that isn't masked is detected, jump to interrupt
00121         // handler
00122         CauseReg cause = tc->readMiscRegNoEffect(MISCREG_CAUSE);
00123         if (status.im && cause.ip) {
00124             DPRINTF(Interrupt, "Interrupt! IM[7:0]=%d IP[7:0]=%d \n",
00125                     (unsigned)status.im, (unsigned)cause.ip);
00126             return new InterruptFault;
00127         }
00128     }
00129 
00130     return NoFault;
00131 }

00149 bool
00150 Interrupts::interruptsPending(ThreadContext *tc) const
00151 {
00152     //if there is a on cpu timer interrupt (i.e. Compare == Count)
00153     //update CauseIP before proceeding to interrupt
00154     if (onCpuTimerInterrupt(tc)) {
00155         DPRINTF(Interrupt, "Interrupts OnCpuTimerINterrupt(tc) == true\n");
00156         //determine timer interrupt IP #
00157         IntCtlReg intCtl = tc->readMiscRegNoEffect(MISCREG_INTCTL);
00158         uint8_t intStatus = getCauseIP(tc);
00159         intStatus |= 1 << intCtl.ipti;
00160         setCauseIP(tc, intStatus);
00161     }
00162 
00163     return (getCauseIP(tc) != 0);
00164 
00165 }
