00061 template <class XC>
00062 inline void
00063 handleLockedSnoop(XC *xc, PacketPtr pkt, Addr cacheBlockMask)
00064 {
00065     if (!xc->readMiscReg(MISCREG_LLFLAG))
00066         return;
00067 
00068     Addr locked_addr = xc->readMiscReg(MISCREG_LLADDR) & cacheBlockMask;
00069     Addr snoop_addr = pkt->getAddr();
00070 
00071     assert((cacheBlockMask & snoop_addr) == snoop_addr);
00072 
00073     if (locked_addr == snoop_addr)
00074         xc->setMiscReg(MISCREG_LLFLAG, false);
00075 }

00089 template <class XC>
00090 inline bool
00091 handleLockedWrite(XC *xc, Request *req)
00092 {
00093     if (req->isUncacheable()) {
00094         // Funky Turbolaser mailbox access...don't update
00095         // result register (see stq_c in decoder.isa)
00096         req->setExtraData(2);
00097     } else {
00098         // standard store conditional
00099         bool lock_flag = xc->readMiscReg(MISCREG_LLFLAG);
00100         Addr lock_addr = xc->readMiscReg(MISCREG_LLADDR);
00101 
00102         if (!lock_flag || (req->getPaddr() & ~0xf) != lock_addr) {
00103             // Lock flag not set or addr mismatch in CPU;
00104             // don't even bother sending to memory system
00105             req->setExtraData(0);
00106             xc->setMiscReg(MISCREG_LLFLAG, false);
00107 
00108             // the rest of this code is not architectural;
00109             // it's just a debugging aid to help detect
00110             // livelock by warning on long sequences of failed
00111             // store conditionals
00112             int stCondFailures = xc->readStCondFailures();
00113             stCondFailures++;
00114             xc->setStCondFailures(stCondFailures);
00115             if (stCondFailures % 100000 == 0) {
00116                 warn("%i: context %d: %d consecutive "
00117                      "store conditional failures\n",
00118                      curTick(), xc->contextId(), stCondFailures);
00119             }
00120 
00121             if (!lock_flag){
00122                 DPRINTF(LLSC, "[tid:%i]: Lock Flag Set, "
00123                               "Store Conditional Failed.\n",
00124                         req->threadId());
00125             } else if ((req->getPaddr() & ~0xf) != lock_addr) {
00126                 DPRINTF(LLSC, "[tid:%i]: Load-Link Address Mismatch, "
00127                               "Store Conditional Failed.\n",
00128                         req->threadId());
00129             }
00130             // store conditional failed already, so don't issue it to mem
00131             return false;
00132         }
00133     }
00134 
00135     return true;
00136 }
