00127 bool
00128 StackTrace::decodeStack(MachInst inst, int &disp)
00129 {
00130     // lda $sp, -disp($sp)
00131     //
00132     // Opcode<31:26> == 0x08
00133     // RA<25:21> == 30
00134     // RB<20:16> == 30
00135     // Disp<15:0>
00136     const MachInst mem_mask = 0xffff0000;
00137     const MachInst lda_pattern = 0x23de0000;
00138     const MachInst lda_disp_mask = 0x0000ffff;
00139 
00140     // subq $sp, disp, $sp
00141     // addq $sp, disp, $sp
00142     //
00143     // Opcode<31:26> == 0x10
00144     // RA<25:21> == 30
00145     // Lit<20:13>
00146     // One<12> = 1
00147     // Func<11:5> == 0x20 (addq)
00148     // Func<11:5> == 0x29 (subq)
00149     // RC<4:0> == 30
00150     const MachInst intop_mask = 0xffe01fff;
00151     const MachInst addq_pattern = 0x43c0141e;
00152     const MachInst subq_pattern = 0x43c0153e;
00153     const MachInst intop_disp_mask = 0x001fe000;
00154     const int intop_disp_shift = 13;
00155 
00156     if ((inst & mem_mask) == lda_pattern)
00157         disp = -sext<16>(inst & lda_disp_mask);
00158     else if ((inst & intop_mask) == addq_pattern)
00159         disp = -int((inst & intop_disp_mask) >> intop_disp_shift);
00160     else if ((inst & intop_mask) == subq_pattern)
00161         disp = int((inst & intop_disp_mask) >> intop_disp_shift);
00162     else
00163         return false;
00164 
00165     return true;
00166 }

00168 bool
00169 StackTrace::decodeSave(MachInst inst, int &reg, int &disp)
00170 {
00171     // lda $stq, disp($sp)
00172     //
00173     // Opcode<31:26> == 0x08
00174     // RA<25:21> == ?
00175     // RB<20:16> == 30
00176     // Disp<15:0>
00177     const MachInst stq_mask = 0xfc1f0000;
00178     const MachInst stq_pattern = 0xb41e0000;
00179     const MachInst stq_disp_mask = 0x0000ffff;
00180     const MachInst reg_mask = 0x03e00000;
00181     const int reg_shift = 21;
00182 
00183     if ((inst & stq_mask) == stq_pattern) {
00184         reg = (inst & reg_mask) >> reg_shift;
00185         disp = sext<16>(inst & stq_disp_mask);
00186     } else {
00187         return false;
00188     }
00189 
00190     return true;
00191 }

00194  * Decode the function prologue for the function we're in, and note

00198 StackTrace::decodePrologue(Addr sp, Addr callpc, Addr func,
00199                            int &size, Addr &ra)

00202     ra = 0;
00203 
00204     for (Addr pc = func; pc < callpc; pc += sizeof(MachInst)) {
00205         MachInst inst;

00207

00210             if (size) {
00211                 return true;
00212             }
