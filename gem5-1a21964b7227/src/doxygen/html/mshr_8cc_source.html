00067 inline void
00068 MSHR::TargetList::add(PacketPtr pkt, Tick readyTime,
00069                       Counter order, Target::Source source, bool markPending)
00070 {
00071     if (source != Target::FromSnoop) {
00072         if (pkt->needsExclusive()) {
00073             needsExclusive = true;
00074         }
00075 
00076         // StoreCondReq is effectively an upgrade if it's in an MSHR
00077         // since it would have been failed already if we didn't have a
00078         // read-only copy
00079         if (pkt->isUpgrade() || pkt->cmd == MemCmd::StoreCondReq) {
00080             hasUpgrade = true;
00081         }
00082     }
00083 
00084     if (markPending) {
00085         // Iterate over the SenderState stack and see if we find
00086         // an MSHR entry. If we do, set the downstreamPending
00087         // flag. Otherwise, do nothing.
00088         MSHR *mshr = pkt->findNextSenderState<MSHR>();
00089         if (mshr != NULL) {
00090             assert(!mshr->downstreamPending);
00091             mshr->downstreamPending = true;
00092         }
00093     }
00094 
00095     push_back(Target(pkt, readyTime, order, source, markPending));
00096 }

00229 MSHR::markInService(PacketPtr pkt)
00230 {
00231     assert(!inService);
00232     if (isForwardNoResponse()) {
00233         // we just forwarded the request packet & don't expect a
00234         // response, so get rid of it

00266 void
00267 MSHR::allocateTarget(PacketPtr pkt, Tick whenReady, Counter _order)
00268 {
00269     // if there's a request already in service for this MSHR, we will
00270     // have to defer the new target until after the response if any of
00271     // the following are true:
00272     // - there are other targets already deferred
00273     // - there's a pending invalidate to be applied after the response
00274     //   comes back (but before this target is processed)
00275     // - this target requires an exclusive block and either we're not
00276     //   getting an exclusive block back or we have already snooped
00277     //   another read request that will downgrade our exclusive block
00278     //   to shared
00279 
00280     // assume we'd never issue a prefetch when we've got an
00281     // outstanding miss
00282     assert(pkt->cmd != MemCmd::HardPFReq);
00283 
00284     if (inService &&
00285         (!deferredTargets->empty() || hasPostInvalidate() ||
00286          (pkt->needsExclusive() &&
00287           (!isPendingDirty() || hasPostDowngrade() || isForward)))) {
00288         // need to put on deferred list
00289         if (hasPostInvalidate())
00290             replaceUpgrade(pkt);
00291         deferredTargets->add(pkt, whenReady, _order, Target::FromCPU, true);
00292     } else {
00293         // No request outstanding, or still OK to append to
00294         // outstanding request: append to regular target list.  Only
00295         // mark pending if current request hasn't been issued yet
00296         // (isn't in service).
00297         targets->add(pkt, whenReady, _order, Target::FromCPU, !inService);
00298     }
00299 
00300     ++ntargets;
00301 }

00303 bool
00304 MSHR::handleSnoop(PacketPtr pkt, Counter _order)
00305 {
00306     if (!inService || (pkt->isExpressSnoop() && downstreamPending)) {
00307         // Request has not been issued yet, or it's been issued
00308         // locally but is buffered unissued at some downstream cache
00309         // which is forwarding us this snoop.  Either way, the packet
00310         // we're snooping logically precedes this MSHR's request, so
00311         // the snoop has no impact on the MSHR, but must be processed
00312         // in the standard way by the cache.  The only exception is
00313         // that if we're an L2+ cache buffering an UpgradeReq from a
00314         // higher-level cache, and the snoop is invalidating, then our
00315         // buffered upgrades must be converted to read exclusives,
00316         // since the upper-level cache no longer has a valid copy.
00317         // That is, even though the upper-level cache got out on its
00318         // local bus first, some other invalidating transaction
00319         // reached the global bus before the upgrade did.
00320         if (pkt->needsExclusive()) {
00321             targets->replaceUpgrades();
00322             deferredTargets->replaceUpgrades();
00323         }
00324 
00325         return false;
00326     }
00327 
00328     // From here on down, the request issued by this MSHR logically
00329     // precedes the request we're snooping.
00330     if (pkt->needsExclusive()) {
00331         // snooped request still precedes the re-request we'll have to
00332         // issue for deferred targets, if any...
00333         deferredTargets->replaceUpgrades();
00334     }
00335 
00336     if (hasPostInvalidate()) {
00337         // a prior snoop has already appended an invalidation, so
00338         // logically we don't have the block anymore; no need for
00339         // further snooping.
00340         return true;
00341     }
00342 
00343     if (isPendingDirty() || pkt->isInvalidate()) {
00344         // We need to save and replay the packet in two cases:
00345         // 1. We're awaiting an exclusive copy, so ownership is pending,
00346         //    and we need to respond after we receive data.
00347         // 2. It's an invalidation (e.g., UpgradeReq), and we need
00348         //    to forward the snoop up the hierarchy after the current
00349         //    transaction completes.
00350         
00351         // Actual target device (typ. a memory) will delete the
00352         // packet on reception, so we need to save a copy here.
00353         PacketPtr cp_pkt = new Packet(pkt, true);
00354         targets->add(cp_pkt, curTick(), _order, Target::FromSnoop,
00355                      downstreamPending && targets->needsExclusive);
00356         ++ntargets;
00357 
00358         if (isPendingDirty()) {
00359             pkt->assertMemInhibit();
00360             pkt->setSupplyExclusive();
00361         }
00362 
00363         if (pkt->needsExclusive()) {
00364             // This transaction will take away our pending copy
00365             postInvalidate = true;
00366         }
00367     }
00368 
00369     if (!pkt->needsExclusive()) {
00370         // This transaction will get a read-shared copy, downgrading
00371         // our copy if we had an exclusive one
00372         postDowngrade = true;
00373         pkt->assertShared();
00374     }
00375 
00376     return true;
00377 }

00405 void
00406 MSHR::handleFill(Packet *pkt, CacheBlk *blk)
00407 {
00408     if (!pkt->sharedAsserted()
00409         && !(hasPostInvalidate() || hasPostDowngrade())
00410         && deferredTargets->needsExclusive) {
00411         // We got an exclusive response, but we have deferred targets
00412         // which are waiting to request an exclusive copy (not because
00413         // of a pending invalidate).  This can happen if the original
00414         // request was for a read-only (non-exclusive) block, but we
00415         // got an exclusive copy anyway because of the E part of the
00416         // MOESI/MESI protocol.  Since we got the exclusive copy
00417         // there's no need to defer the targets, so move them up to
00418         // the regular target list.
00419         assert(!targets->needsExclusive);
00420         targets->needsExclusive = true;
00421         // if any of the deferred targets were upper-level cache
00422         // requests marked downstreamPending, need to clear that
00423         assert(!downstreamPending);  // not pending here anymore
00424         deferredTargets->clearDownstreamPending();
00425         // this clears out deferredTargets too
00426         targets->splice(targets->end(), *deferredTargets);
00427         deferredTargets->resetFlags();
00428     }
00429 }
