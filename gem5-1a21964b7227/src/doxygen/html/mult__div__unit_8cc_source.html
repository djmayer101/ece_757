00116 MultDivUnit::getSlot(DynInstPtr inst)
00117 {
00118     // If MDU already has instruction, return current slot.
00119     int slot_num = findSlot(inst);
00120 
00121     // If we have this instruction's request already then return
00122     if (slot_num != -1 &&         
00123         inst->curSkedEntry->cmd == reqs[slot_num]->cmd)
00124         return slot_num;
00125     
00126     unsigned repeat_rate = 0;
00127 
00129     if (lastOpType == IntMultOp) {
00130         repeat_rate = multRepeatRate;
00131     }
00132 
00134     if (lastOpType == IntDivOp) {
00135         switch (lastDivSize) {
00136           case 8:
00137             repeat_rate = div8RepeatRate;
00138             break;
00139 
00140           case 16:
00141             repeat_rate = div16RepeatRate;
00142             break;
00143 
00144           case 24:
00145             repeat_rate = div24RepeatRate;
00146             break;
00147 
00148           case 32:
00149             repeat_rate = div32RepeatRate;
00150             break;
00151         }
00152     }
00153 
00154     if (lastMDUCycle + repeat_rate > curTick()) {
00155         DPRINTF(InOrderMDU, "MDU not ready to process another inst. until %i, "
00156                 "denying request.\n", lastMDUCycle + repeat_rate);
00157         return -1;
00158     } else {
00159         int rval = Resource::getSlot(inst);
00160         DPRINTF(InOrderMDU, "MDU request should pass: %i.\n",
00161                 rval);
00162 
00163         if (rval != -1) {            
00164             lastMDUCycle = curTick();
00165             lastOpType = inst->opClass();
00166             lastInstName = inst->staticInst->getName();
00167         }
00168       
00169         return rval;
00170     }

00191 MultDivUnit::execute(int slot_num)
00192 {
00193     ResourceRequest* mult_div_req = reqs[slot_num];
00194     DynInstPtr inst = reqs[slot_num]->inst;
00195     if (inst->fault != NoFault) {
00196         DPRINTF(InOrderMDU,
00197                 "[tid:%i]: [sn:%i]: Detected %s fault @ %x. Forwarding to "
00198                 "next stage.\n", inst->readTid(), inst->seqNum, inst->fault->name(),
00199                 inst->pcState());
00200         mult_div_req->done();
00201         return;
00202     }
00203 
00204     DPRINTF(InOrderMDU, "Executing [sn:%i] ...\n", slot_num);
00205 
00206     switch (mult_div_req->cmd)
00207     {
00208       case StartMultDiv:
00209         {
00210             DPRINTF(InOrderMDU, "Start MDU called ...\n");
00211 
00212             OpClass op_class = inst->opClass();
00213             if (op_class == IntMultOp) {
00214                 scheduleEvent(slot_num, multLatency);
00215             } else if (op_class == IntDivOp) {
00216                 int op_size = getDivOpSize(inst);
00217 
00218                 switch (op_size)
00219                 {
00220                   case 8:
00221                     scheduleEvent(slot_num, div8Latency);
00222                     break;
00223 
00224                   case 16:
00225                     scheduleEvent(slot_num, div16Latency);
00226                     break;
00227 
00228                   case 24:
00229                     scheduleEvent(slot_num, div24Latency);
00230                     break;
00231 
00232                   case 32:
00233                     scheduleEvent(slot_num, div32Latency);
00234                     break;
00235                 }
00236 
00237                 lastDivSize = op_size;
00238             }
00239 
00240             // Allow to pass through to next stage while
00241             // event processes
00242             mult_div_req->setProcessing();
00243             mult_div_req->setCompleted();
00244         }
00245         break;
00246         
00247       case MultDiv:
00248         DPRINTF(InOrderMDU, "Execute MDU called ...\n");        
00249         exeMulDiv(slot_num);        
00250         mult_div_req->done();
00251         break;
00252 
00253 
00254       case EndMultDiv:
00255         //@TODO: Why not allow high-latency requests to sleep
00256         //      within stage until event wakes up????
00257         //      Seems wasteful to continually check to see if
00258         //      this is done when we have a event in parallel
00259         //      counting down the time
00260         {
00261             DPRINTF(InOrderMDU, "End MDU called ...\n");    
00262             if (!mult_div_req->isProcessing()) {
00263                 DPRINTF(InOrderMDU, "Mult/Div finished.\n");                    
00264                 mult_div_req->done();            
00265             } else {                
00266                 mult_div_req->setCompleted(false);
00267             }
00268             
00269         }
00270         break;
00271 
00272       default:

00274     }
