00132 void
00133 NetworkTest::tick()
00134 {
00135     if (++noResponseCycles >= 500000) {
00136         cerr << name() << ": deadlocked at cycle " << curTick() << endl;
00137         fatal("");
00138     }
00139 
00140     // make new request based on injection rate
00141     // (injection rate's range depends on precision)
00142     // - generate a random number between 0 and 10^precision
00143     // - send pkt if this number is < injRate*(10^precision)
00144     bool send_this_cycle;
00145     double injRange = pow((double) 10, (double) precision);
00146     unsigned trySending = random() % (int) injRange;
00147     if (trySending < injRate*injRange)
00148         send_this_cycle = true;
00149     else
00150         send_this_cycle = false;
00151 
00152     // always generatePkt unless fixedPkts is enabled
00153     if (send_this_cycle) {
00154         if (fixedPkts) {
00155             if (numPacketsSent < maxPackets) {
00156                 generatePkt();
00157             }
00158         } else {
00159             generatePkt();
00160         }
00161     }
00162 
00163     // Schedule wakeup
00164     if (curTick() >= simCycles)
00165         exitSimLoop("Network Tester completed simCycles");
00166     else {
00167         if (!tickEvent.scheduled())
00168             schedule(tickEvent, clockEdge(Cycles(1)));
00169     }
00170 }

00172 void
00173 NetworkTest::generatePkt()
00174 {
00175     unsigned destination = id;
00176     if (trafficType == 0) { // Uniform Random
00177         destination = random() % numMemories;
00178     } else if (trafficType == 1) { // Tornado
00179         int networkDimension = (int) sqrt(numMemories);
00180         int my_x = id%networkDimension;
00181         int my_y = id/networkDimension;
00182 
00183         int dest_x = my_x + (int) ceil(networkDimension/2) - 1;
00184         dest_x = dest_x%networkDimension;
00185         int dest_y = my_y;
00186 
00187         destination = dest_y*networkDimension + dest_x;
00188     } else if (trafficType == 2) { // Bit Complement
00189         int networkDimension = (int) sqrt(numMemories);
00190         int my_x = id%networkDimension;
00191         int my_y = id/networkDimension;
00192 
00193         int dest_x = networkDimension - my_x - 1;
00194         int dest_y = networkDimension - my_y - 1;
00195 
00196         destination = dest_y*networkDimension + dest_x;
00197     }
00198 
00199     Request *req = new Request();
00200     Request::Flags flags;
00201 
00202     // The source of the packets is a cache.
00203     // The destination of the packets is a directory.
00204     // The destination bits are embedded in the address after byte-offset.
00205     Addr paddr =  destination;
00206     paddr <<= blockSizeBits;
00207     unsigned access_size = 1; // Does not affect Ruby simulation
00208 
00209     // Modeling different coherence msg types over different msg classes.
00210     //
00211     // networktest assumes the Network_test coherence protocol 
00212     // which models three message classes/virtual networks.
00213     // These are: request, forward, response.
00214     // requests and forwards are "control" packets (typically 8 bytes),
00215     // while responses are "data" packets (typically 72 bytes).
00216     //
00217     // Life of a packet from the tester into the network:
00218     // (1) This function generatePkt() generates packets of one of the 
00219     //     following 3 types (randomly) : ReadReq, INST_FETCH, WriteReq
00220     // (2) mem/ruby/system/RubyPort.cc converts these to RubyRequestType_LD,
00221     //     RubyRequestType_IFETCH, RubyRequestType_ST respectively
00222     // (3) mem/ruby/system/Sequencer.cc sends these to the cache controllers
00223     //     in the coherence protocol.
00224     // (4) Network_test-cache.sm tags RubyRequestType:LD,
00225     //     RubyRequestType:IFETCH and RubyRequestType:ST as
00226     //     Request, Forward, and Response events respectively; 
00227     //     and injects them into virtual networks 0, 1 and 2 respectively.
00228     //     It immediately calls back the sequencer.
00229     // (5) The packet traverses the network (simple/garnet) and reaches its
00230     //     destination (Directory), and network stats are updated.
00231     // (6) Network_test-dir.sm simply drops the packet.
00232     // 
00233     MemCmd::Command requestType;
00234 
00235     unsigned randomReqType = random() % 3;
00236     if (randomReqType == 0) {
00237         // generate packet for virtual network 0
00238         requestType = MemCmd::ReadReq;
00239         req->setPhys(paddr, access_size, flags, masterId);
00240     } else if (randomReqType == 1) {
00241         // generate packet for virtual network 1
00242         requestType = MemCmd::ReadReq;
00243         flags.set(Request::INST_FETCH);
00244         req->setVirt(0, 0x0, access_size, flags, 0x0, masterId);
00245         req->setPaddr(paddr);
00246     } else {  // if (randomReqType == 2)
00247         // generate packet for virtual network 2
00248         requestType = MemCmd::WriteReq;
00249         req->setPhys(paddr, access_size, flags, masterId);
00250     }
00251 
00252     req->setThreadContext(id,0);
00253 
00254     //No need to do functional simulation
00255     //We just do timing simulation of the network
00256 
00257     DPRINTF(NetworkTest,
00258             "Generated packet with destination %d, embedded in address %x\n",
00259             destination, req->getPaddr());
00260 
00261     PacketPtr pkt = new Packet(req, requestType);
00262     pkt->dataDynamicArray(new uint8_t[req->getSize()]);
00263     pkt->senderState = NULL;
00264 
00265     sendPkt(pkt);
00266 }
