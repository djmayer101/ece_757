00057 NoncoherentBus::NoncoherentBus(const NoncoherentBusParams *p)
00058     : BaseBus(p), reqLayer(*this, ".reqLayer"),
00059       respLayer(*this, ".respLayer")
00060 {
00061     // create the ports based on the size of the master and slave
00062     // vector ports, and the presence of the default port, the ports
00063     // are enumerated starting from zero
00064     for (int i = 0; i < p->port_master_connection_count; ++i) {
00065         std::string portName = csprintf("%s.master[%d]", name(), i);
00066         MasterPort* bp = new NoncoherentBusMasterPort(portName, *this, i);
00067         masterPorts.push_back(bp);
00068     }
00069 
00070     // see if we have a default slave device connected and if so add
00071     // our corresponding master port
00072     if (p->port_default_connection_count) {
00073         defaultPortID = masterPorts.size();
00074         std::string portName = name() + ".default";
00075         MasterPort* bp = new NoncoherentBusMasterPort(portName, *this,
00076                                                       defaultPortID);
00077         masterPorts.push_back(bp);
00078     }
00079 
00080     // create the slave ports, once again starting at zero
00081     for (int i = 0; i < p->port_slave_connection_count; ++i) {
00082         std::string portName = csprintf("%s.slave[%d]", name(), i);
00083         SlavePort* bp = new NoncoherentBusSlavePort(portName, *this, i);
00084         slavePorts.push_back(bp);
00085     }
00086 
00087     clearPortCache();
00088 }

00091 NoncoherentBus::recvTimingReq(PacketPtr pkt, PortID slave_port_id)
00092 {
00093     // determine the source port based on the id
00094     SlavePort *src_port = slavePorts[slave_port_id];
00095 
00096     // we should never see express snoops on a non-coherent bus
00097     assert(!pkt->isExpressSnoop());
00098 
00099     // test if the bus should be considered occupied for the current
00100     // port
00101     if (!reqLayer.tryTiming(src_port)) {
00102         DPRINTF(NoncoherentBus, "recvTimingReq: src %s %s 0x%x BUSY\n",
00103                 src_port->name(), pkt->cmdString(), pkt->getAddr());
00104         return false;
00105     }
00106 
00107     DPRINTF(NoncoherentBus, "recvTimingReq: src %s %s 0x%x\n",
00108             src_port->name(), pkt->cmdString(), pkt->getAddr());
00109 
00110     // set the source port for routing of the response
00111     pkt->setSrc(slave_port_id);
00112 
00113     calcPacketTiming(pkt);
00114     Tick packetFinishTime = pkt->busLastWordDelay + curTick();
00115 
00116     // since it is a normal request, determine the destination
00117     // based on the address and attempt to send the packet
00118     bool success = masterPorts[findPort(pkt->getAddr())]->sendTimingReq(pkt);
00119 
00120     if (!success)  {
00121         // inhibited packets should never be forced to retry
00122         assert(!pkt->memInhibitAsserted());
00123 
00124         DPRINTF(NoncoherentBus, "recvTimingReq: src %s %s 0x%x RETRY\n",
00125                 src_port->name(), pkt->cmdString(), pkt->getAddr());
00126 
00127         // undo the calculation so we can check for 0 again
00128         pkt->busFirstWordDelay = pkt->busLastWordDelay = 0;
00129 
00130         // occupy until the header is sent
00131         reqLayer.failedTiming(src_port, clockEdge(Cycles(headerCycles)));
00132 
00133         return false;
00134     }
00135 
00136     reqLayer.succeededTiming(packetFinishTime);
00137 
00138     return true;
00139 }
