00748 void
00749 FullO3CPU<Impl>::activateContext(ThreadID tid, Cycles delay)
00750 {
00751     assert(!switchedOut());
00752 
00753     // Needs to set each stage to running as well.
00754     if (delay){
00755         DPRINTF(O3CPU, "[tid:%i]: Scheduling thread context to activate "
00756                 "on cycle %d\n", tid, clockEdge(delay));
00757         scheduleActivateThreadEvent(tid, delay);
00758     } else {
00759         activateThread(tid);
00760     }
00761 
00762     // We don't want to wake the CPU if it is drained. In that case,
00763     // we just want to flag the thread as active and schedule the tick
00764     // event from drainResume() instead.
00765     if (getDrainState() == Drainable::Drained)
00766         return;
00767 
00768     // If we are time 0 or if the last activation time is in the past,
00769     // schedule the next tick and wake up the fetch unit
00770     if (lastActivatedCycle == 0 || lastActivatedCycle < curTick()) {

00774         // deschedule itself.
00775         activityRec.activity();
00776         fetch.wakeFromQuiesce();
00777 
00778         Cycles cycles(curCycle() - lastRunningCycle);
00779         // @todo: This is an oddity that is only here to match the stats
00780         if (cycles != 0)
00781             --cycles;
00782         quiesceCycles += cycles;
00783 
00784         lastActivatedCycle = curTick();

01107 template <class Impl>
01108 unsigned int
01109 FullO3CPU<Impl>::drain(DrainManager *drain_manager)
01110 {
01111     // If the CPU isn't doing anything, then return immediately.
01112     if (switchedOut()) {
01113         setDrainState(Drainable::Drained);
01114         return 0;
01115     }
01116 
01117     DPRINTF(Drain, "Draining...\n");
01118     setDrainState(Drainable::Draining);
01119 
01120     // We only need to signal a drain to the commit stage as this
01121     // initiates squashing controls the draining. Once the commit
01122     // stage commits an instruction where it is safe to stop, it'll
01123     // squash the rest of the instructions in the pipeline and force
01124     // the fetch stage to stall. The pipeline will be drained once all
01125     // in-flight instructions have retired.
01126     commit.drain();
01127 
01128     // Wake the CPU and record activity so everything can drain out if
01129     // the CPU was not able to immediately drain.
01130     if (!isDrained())  {
01131         drainManager = drain_manager;
01132 
01133         wakeCPU();
01134         activityRec.activity();
01135 
01136         DPRINTF(Drain, "CPU not drained\n");
01137 
01138         return 1;
01139     } else {
01140         setDrainState(Drainable::Drained);
01141         DPRINTF(Drain, "CPU is already drained\n");
01142         if (tickEvent.scheduled())
01143             deschedule(tickEvent);
01144 
01145         // Flush out any old data from the time buffers.  In
01146         // particular, there might be some data in flight from the
01147         // fetch stage that isn't visible in any of the CPU buffers we
01148         // test in isDrained().
01149         for (int i = 0; i < timeBuffer.getSize(); ++i) {
01150             timeBuffer.advance();
01151             fetchQueue.advance();
01152             decodeQueue.advance();
01153             renameQueue.advance();
01154             iewQueue.advance();
01155         }
01156 
01157         drainSanityCheck();
01158         return 0;
01159     }
01160 }
