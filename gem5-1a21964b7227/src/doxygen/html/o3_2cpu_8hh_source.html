00250     void
00251     scheduleActivateThreadEvent(ThreadID tid, Cycles delay)
00252     {
00253         // Schedule thread to activate, regardless of its current state.
00254         if (activateThreadEvent[tid].squashed())
00255             reschedule(activateThreadEvent[tid],
00256                        clockEdge(delay));
00257         else if (!activateThreadEvent[tid].scheduled()) {
00258             Tick when = clockEdge(delay);
00259 
00260             // Check if the deallocateEvent is also scheduled, and make
00261             // sure they do not happen at same time causing a sleep that
00262             // is never woken from.
00263             if (deallocateContextEvent[tid].scheduled() &&
00264                 deallocateContextEvent[tid].when() == when) {
00265                 when++;
00266             }
00267 
00268             schedule(activateThreadEvent[tid], when);
00269         }
00270     }
