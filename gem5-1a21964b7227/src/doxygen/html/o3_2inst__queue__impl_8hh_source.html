00378 template <class Impl>
00379 void
00380 InstructionQueue<Impl>::resetState()
00381 {
00382     //Initialize thread IQ counts
00383     for (ThreadID tid = 0; tid <numThreads; tid++) {
00384         count[tid] = 0;
00385         instList[tid].clear();
00386     }
00387 
00388     // Initialize the number of free IQ entries.
00389     freeEntries = numEntries;
00390 
00391     // Note that in actuality, the registers corresponding to the logical
00392     // registers start off as ready.  However this doesn't matter for the
00393     // IQ as the instruction should have been correctly told if those
00394     // registers are ready in rename.  Thus it can all be initialized as
00395     // unready.
00396     for (int i = 0; i < numPhysRegs; ++i) {
00397         regScoreboard[i] = false;
00398     }
00399 
00400     for (ThreadID tid = 0; tid < numThreads; ++tid) {
00401         squashedSeqNum[tid] = 0;
00402     }
00403 
00404     for (int i = 0; i < Num_OpClasses; ++i) {
00405         while (!readyInsts[i].empty())
00406             readyInsts[i].pop();
00407         queueOnList[i] = false;
00408         readyIt[i] = listOrder.end();
00409     }
00410     nonSpecInsts.clear();
00411     listOrder.clear();
00412     deferredMemInsts.clear();
00413 }

00702 template <class Impl>
00703 void
00704 InstructionQueue<Impl>::processFUCompletion(DynInstPtr &inst, int fu_idx)
00705 {
00706     DPRINTF(IQ, "Processing FU completion [sn:%lli]\n", inst->seqNum);
00707     assert(!cpu->switchedOut());
00708     // The CPU could have been sleeping until this op completed (*extremely*
00709     // long latency op).  Wake it if it was.  This may be overkill.
00710     iewStage->wakeCPU();
00711 
00712     if (fu_idx > -1)
00713         fuPool->freeUnitNextCycle(fu_idx);
00714 
00715     // @todo: Ensure that these FU Completions happen at the beginning
00716     // of a cycle, otherwise they could add too many instructions to
00717     // the queue.
00718     issueToExecuteQueue->access(-1)->size++;
00719     instsToExecute.push_back(inst);
00720 }
00721 
00722 // @todo: Figure out a better way to remove the squashed items from the
00723 // lists.  Checking the top item of each list to see if it's squashed
00724 // wastes time and forces jumps.
00725 template <class Impl>
00726 void
00727 InstructionQueue<Impl>::scheduleReadyInsts()
00728 {
00729     DPRINTF(IQ, "Attempting to schedule ready instructions from "
00730             "the IQ.\n");
00731 
00732     IssueStruct *i2e_info = issueToExecuteQueue->access(0);
00733 
00734     DynInstPtr deferred_mem_inst;
00735     int total_deferred_mem_issued = 0;
00736     while (total_deferred_mem_issued < totalWidth &&
00737            (deferred_mem_inst = getDeferredMemInstToExecute()) != 0) {
00738         issueToExecuteQueue->access(0)->size++;
00739         instsToExecute.push_back(deferred_mem_inst);
00740         total_deferred_mem_issued++;
00741     }
00742 
00743     // Have iterator to head of the list
00744     // While I haven't exceeded bandwidth or reached the end of the list,
00745     // Try to get a FU that can do what this op needs.
00746     // If successful, change the oldestInst to the new top of the list, put
00747     // the queue in the proper place in the list.
00748     // Increment the iterator.
00749     // This will avoid trying to schedule a certain op class if there are no
00750     // FUs that handle it.
00751     ListOrderIt order_it = listOrder.begin();
00752     ListOrderIt order_end_it = listOrder.end();
00753     int total_issued = 0;
00754 
00755     while (total_issued < (totalWidth - total_deferred_mem_issued) &&
00756            iewStage->canIssue() &&
00757            order_it != order_end_it) {
00758         OpClass op_class = (*order_it).queueType;
00759 
00760         assert(!readyInsts[op_class].empty());
00761 
00762         DynInstPtr issuing_inst = readyInsts[op_class].top();
00763 
00764         issuing_inst->isFloating() ? fpInstQueueReads++ : intInstQueueReads++;
00765 
00766         assert(issuing_inst->seqNum == (*order_it).oldestInst);
00767 
00768         if (issuing_inst->isSquashed()) {
00769             readyInsts[op_class].pop();
00770 
00771             if (!readyInsts[op_class].empty()) {
00772                 moveToYoungerInst(order_it);
00773             } else {
00774                 readyIt[op_class] = listOrder.end();
00775                 queueOnList[op_class] = false;
00776             }
00777 
00778             listOrder.erase(order_it++);
00779 
00780             ++iqSquashedInstsIssued;
00781 
00782             continue;
00783         }
00784 
00785         int idx = -2;
00786         Cycles op_latency = Cycles(1);
00787         ThreadID tid = issuing_inst->threadNumber;
00788 
00789         if (op_class != No_OpClass) {
00790             idx = fuPool->getUnit(op_class);
00791             issuing_inst->isFloating() ? fpAluAccesses++ : intAluAccesses++;
00792             if (idx > -1) {
00793                 op_latency = fuPool->getOpLatency(op_class);
00794             }
00795         }
00796 
00797         // If we have an instruction that doesn't require a FU, or a
00798         // valid FU, then schedule for execution.
00799         if (idx == -2 || idx != -1) {
00800             if (op_latency == Cycles(1)) {
00801                 i2e_info->size++;
00802                 instsToExecute.push_back(issuing_inst);
00803 
00804                 // Add the FU onto the list of FU's to be freed next
00805                 // cycle if we used one.
00806                 if (idx >= 0)
00807                     fuPool->freeUnitNextCycle(idx);
00808             } else {
00809                 Cycles issue_latency = fuPool->getIssueLatency(op_class);
00810                 // Generate completion event for the FU
00811                 FUCompletion *execution = new FUCompletion(issuing_inst,
00812                                                            idx, this);
00813 
00814                 cpu->schedule(execution,
00815                               cpu->clockEdge(Cycles(op_latency - 1)));
00816 
00817                 // @todo: Enforce that issue_latency == 1 or op_latency
00818                 if (issue_latency > Cycles(1)) {
00819                     // If FU isn't pipelined, then it must be freed
00820                     // upon the execution completing.
00821                     execution->setFreeFU();
00822                 } else {
00823                     // Add the FU onto the list of FU's to be freed next cycle.
00824                     fuPool->freeUnitNextCycle(idx);
00825                 }
00826             }
00827 
00828             DPRINTF(IQ, "Thread %i: Issuing instruction PC %s "
00829                     "[sn:%lli]\n",
00830                     tid, issuing_inst->pcState(),
00831                     issuing_inst->seqNum);
00832 
00833             readyInsts[op_class].pop();
00834 
00835             if (!readyInsts[op_class].empty()) {
00836                 moveToYoungerInst(order_it);
00837             } else {
00838                 readyIt[op_class] = listOrder.end();
00839                 queueOnList[op_class] = false;
00840             }
00841 
00842             issuing_inst->setIssued();
00843             ++total_issued;
00844 
00845 #if TRACING_ON
00846             issuing_inst->issueTick = curTick() - issuing_inst->fetchTick;
00847 #endif
00848 
00849             if (!issuing_inst->isMemRef()) {
00850                 // Memory instructions can not be freed from the IQ until they
00851                 // complete.
00852                 ++freeEntries;
00853                 count[tid]--;
00854                 issuing_inst->clearInIQ();
00855             } else {
00856                 memDepUnit[tid].issue(issuing_inst);
00857             }
00858 
00859             listOrder.erase(order_it++);
00860             statIssuedInstType[tid][op_class]++;
00861             iewStage->incrWb(issuing_inst->seqNum);
00862         } else {
00863             statFuBusy[op_class]++;
00864             fuBusy[tid]++;
00865             ++order_it;
00866         }
00867     }
00868 
00869     numIssuedDist.sample(total_issued);
00870     iqInstsIssued+= total_issued;
00871 
00872     // If we issued any instructions, tell the CPU we had activity.
00873     // @todo If the way deferred memory instructions are handeled due to
00874     // translation changes then the deferredMemInsts condition should be removed
00875     // from the code below.
00876     if (total_issued || total_deferred_mem_issued || deferredMemInsts.size()) {
00877         cpu->activityThisCycle();
00878     } else {
00879         DPRINTF(IQ, "Not able to schedule any instructions.\n");
00880     }
00881 }

00929 template <class Impl>
00930 int
00931 InstructionQueue<Impl>::wakeDependents(DynInstPtr &completed_inst)
00932 {
00933     int dependents = 0;
00934 
00935     // The instruction queue here takes care of both floating and int ops
00936     if (completed_inst->isFloating()) {
00937         fpInstQueueWakeupQccesses++;
00938     } else {
00939         intInstQueueWakeupAccesses++;
00940     }
00941 
00942     DPRINTF(IQ, "Waking dependents of completed instruction.\n");
00943 
00944     assert(!completed_inst->isSquashed());
00945 
00946     // Tell the memory dependence unit to wake any dependents on this
00947     // instruction if it is a memory instruction.  Also complete the memory
00948     // instruction at this point since we know it executed without issues.
00949     // @todo: Might want to rename "completeMemInst" to something that
00950     // indicates that it won't need to be replayed, and call this
00951     // earlier.  Might not be a big deal.
00952     if (completed_inst->isMemRef()) {
00953         memDepUnit[completed_inst->threadNumber].wakeDependents(completed_inst);
00954         completeMemInst(completed_inst);
00955     } else if (completed_inst->isMemBarrier() ||
00956                completed_inst->isWriteBarrier()) {
00957         memDepUnit[completed_inst->threadNumber].completeBarrier(completed_inst);
00958     }
00959 
00960     for (int dest_reg_idx = 0;
00961          dest_reg_idx < completed_inst->numDestRegs();
00962          dest_reg_idx++)
00963     {
00964         PhysRegIndex dest_reg =
00965             completed_inst->renamedDestRegIdx(dest_reg_idx);
00966 
00967         // Special case of uniq or control registers.  They are not
00968         // handled by the IQ and thus have no dependency graph entry.
00969         // @todo Figure out a cleaner way to handle this.
00970         if (dest_reg >= numPhysRegs) {
00971             DPRINTF(IQ, "dest_reg :%d, numPhysRegs: %d\n", dest_reg,
00972                     numPhysRegs);
00973             continue;
00974         }
00975 
00976         DPRINTF(IQ, "Waking any dependents on register %i.\n",
00977                 (int) dest_reg);
00978 
00979         //Go through the dependency chain, marking the registers as
00980         //ready within the waiting instructions.
00981         DynInstPtr dep_inst = dependGraph.pop(dest_reg);
00982 
00983         while (dep_inst) {
00984             DPRINTF(IQ, "Waking up a dependent instruction, [sn:%lli] "
00985                     "PC %s.\n", dep_inst->seqNum, dep_inst->pcState());
00986 
00987             // Might want to give more information to the instruction
00988             // so that it knows which of its source registers is
00989             // ready.  However that would mean that the dependency
00990             // graph entries would need to hold the src_reg_idx.
00991             dep_inst->markSrcRegReady();
00992 
00993             addIfReady(dep_inst);
00994 
00995             dep_inst = dependGraph.pop(dest_reg);
00996 
00997             ++dependents;
00998         }
00999 
01000         // Reset the head node now that all of its dependents have
01001         // been woken up.
01002         assert(dependGraph.empty(dest_reg));
01003         dependGraph.clearInst(dest_reg);
01004 
01005         // Mark the scoreboard as having that register ready.
01006         regScoreboard[dest_reg] = true;
01007     }
01008     return dependents;
01009 }

01123 template <class Impl>
01124 void
01125 InstructionQueue<Impl>::doSquash(ThreadID tid)
01126 {
01127     // Start at the tail.
01128     ListIt squash_it = instList[tid].end();
01129     --squash_it;
01130 
01131     DPRINTF(IQ, "[tid:%i]: Squashing until sequence number %i!\n",
01132             tid, squashedSeqNum[tid]);
01133 
01134     // Squash any instructions younger than the squashed sequence number
01135     // given.
01136     while (squash_it != instList[tid].end() &&
01137            (*squash_it)->seqNum > squashedSeqNum[tid]) {
01138 
01139         DynInstPtr squashed_inst = (*squash_it);
01140         squashed_inst->isFloating() ? fpInstQueueWrites++ : intInstQueueWrites++;
01141 
01142         // Only handle the instruction if it actually is in the IQ and
01143         // hasn't already been squashed in the IQ.
01144         if (squashed_inst->threadNumber != tid ||
01145             squashed_inst->isSquashedInIQ()) {
01146             --squash_it;
01147             continue;
01148         }
01149 
01150         if (!squashed_inst->isIssued() ||
01151             (squashed_inst->isMemRef() &&
01152              !squashed_inst->memOpDone())) {
01153 
01154             DPRINTF(IQ, "[tid:%i]: Instruction [sn:%lli] PC %s squashed.\n",
01155                     tid, squashed_inst->seqNum, squashed_inst->pcState());
01156 
01157             // Remove the instruction from the dependency list.
01158             if (!squashed_inst->isNonSpeculative() &&
01159                 !squashed_inst->isStoreConditional() &&
01160                 !squashed_inst->isMemBarrier() &&
01161                 !squashed_inst->isWriteBarrier()) {
01162 
01163                 for (int src_reg_idx = 0;
01164                      src_reg_idx < squashed_inst->numSrcRegs();
01165                      src_reg_idx++)
01166                 {
01167                     PhysRegIndex src_reg =
01168                         squashed_inst->renamedSrcRegIdx(src_reg_idx);
01169 
01170                     // Only remove it from the dependency graph if it
01171                     // was placed there in the first place.
01172 
01173                     // Instead of doing a linked list traversal, we
01174                     // can just remove these squashed instructions
01175                     // either at issue time, or when the register is
01176                     // overwritten.  The only downside to this is it
01177                     // leaves more room for error.
01178 
01179                     if (!squashed_inst->isReadySrcRegIdx(src_reg_idx) &&
01180                         src_reg < numPhysRegs) {
01181                         dependGraph.remove(src_reg, squashed_inst);
01182                     }
01183 
01184 
01185                     ++iqSquashedOperandsExamined;
01186                 }
01187             } else if (!squashed_inst->isStoreConditional() ||
01188                        !squashed_inst->isCompleted()) {
01189                 NonSpecMapIt ns_inst_it =
01190                     nonSpecInsts.find(squashed_inst->seqNum);
01191 
01192                 if (ns_inst_it == nonSpecInsts.end()) {
01193                     assert(squashed_inst->getFault() != NoFault);
01194                 } else {
01195 
01196                     (*ns_inst_it).second = NULL;
01197 
01198                     nonSpecInsts.erase(ns_inst_it);
01199 
01200                     ++iqSquashedNonSpecRemoved;
01201                 }
01202             }
01203 
01204             // Might want to also clear out the head of the dependency graph.
01205 
01206             // Mark it as squashed within the IQ.
01207             squashed_inst->setSquashedInIQ();
01208 
01209             // @todo: Remove this hack where several statuses are set so the
01210             // inst will flow through the rest of the pipeline.
01211             squashed_inst->setIssued();
01212             squashed_inst->setCanCommit();
01213             squashed_inst->clearInIQ();
01214 
01215             //Update Thread IQ Count
01216             count[squashed_inst->threadNumber]--;
01217 
01218             ++freeEntries;
01219         }
01220 
01221         instList[tid].erase(squash_it--);
01222         ++iqSquashedInstsExamined;
01223     }

01226 template <class Impl>
01227 bool
01228 InstructionQueue<Impl>::addToDependents(DynInstPtr &new_inst)
01229 {
01230     // Loop through the instruction's source registers, adding
01231     // them to the dependency list if they are not ready.
01232     int8_t total_src_regs = new_inst->numSrcRegs();
01233     bool return_val = false;
01234 
01235     for (int src_reg_idx = 0;
01236          src_reg_idx < total_src_regs;
01237          src_reg_idx++)
01238     {
01239         // Only add it to the dependency graph if it's not ready.
01240         if (!new_inst->isReadySrcRegIdx(src_reg_idx)) {
01241             PhysRegIndex src_reg = new_inst->renamedSrcRegIdx(src_reg_idx);
01242 
01243             // Check the IQ's scoreboard to make sure the register
01244             // hasn't become ready while the instruction was in flight
01245             // between stages.  Only if it really isn't ready should
01246             // it be added to the dependency graph.
01247             if (src_reg >= numPhysRegs) {
01248                 continue;
01249             } else if (regScoreboard[src_reg] == false) {
01250                 DPRINTF(IQ, "Instruction PC %s has src reg %i that "
01251                         "is being added to the dependency chain.\n",
01252                         new_inst->pcState(), src_reg);
01253 
01254                 dependGraph.insert(src_reg, new_inst);
01255 
01256                 // Change the return value to indicate that something
01257                 // was added to the dependency graph.
01258                 return_val = true;
01259             } else {
01260                 DPRINTF(IQ, "Instruction PC %s has src reg %i that "
01261                         "became ready before it reached the IQ.\n",
01262                         new_inst->pcState(), src_reg);
01263                 // Mark a register ready within the instruction.
01264                 new_inst->markSrcRegReady(src_reg_idx);
01265             }
01266         }
01267     }
01268 
01269     return return_val;
01270 }
