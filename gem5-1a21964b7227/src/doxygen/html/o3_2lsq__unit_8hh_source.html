00591         if (TheISA::HasUnalignedMemAcc && sreqLow) {
00592             delete sreqLow;
00593             delete sreqHigh;
00594         }

00596                 "Uncachable load [sn:%llx] PC %s\n",
00597                 load_inst->seqNum, load_inst->pcState());

00601     int store_idx = load_inst->sqIdx;
00602 
00603     int store_size = 0;
00604 
00605     DPRINTF(LSQUnit, "Read called, load idx: %i, store idx: %i, "
00606             "storeHead: %i addr: %#x%s\n",
00607             load_idx, store_idx, storeHead, req->getPaddr(),
00608             sreqLow ? " split" : "");

00610     if (req->isLLSC()) {
00611         assert(!sreqLow);
00612         // Disable recording the result temporarily.  Writing to misc
00613         // regs normally updates the result, but this is not the
00614         // desired behavior when handling store conditionals.
00615         load_inst->recordResult(false);
00616         TheISA::handleLockedRead(load_inst.get(), req);
00617         load_inst->recordResult(true);
00618     }

00621         assert(!load_inst->memData);

00641             if (delay2 > delay)

00791         PacketPtr snd_data_pkt = NULL;

00794

00796         state->isLoad = true;
00797         state->idx = load_idx;
00798         state->inst = load_inst;

00804             fst_data_pkt = data_pkt;
00805         } else {

00808             fst_data_pkt = new Packet(sreqLow, command);
00809             snd_data_pkt = new Packet(sreqHigh, command);

00886         if (isLoadBlocked && blockedLoadSeqNum < load_inst->seqNum)

00889         // Record that the load was blocked due to memory.  This
00890         // load will squash all instructions after it, be

00896         return NoFault;
00897     }
00898 
00899     return NoFault;
