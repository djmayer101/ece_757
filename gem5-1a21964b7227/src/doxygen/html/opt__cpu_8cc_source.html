00046 OptCPU::OptCPU(const string &name,
00047                MemTraceReader *_trace,
00048                int block_size,
00049                int cache_size,
00050                int _assoc)
00051     : SimObject(name), tickEvent(this), trace(_trace),
00052       numBlks(cache_size/block_size), assoc(_assoc), numSets(numBlks/assoc),
00053       setMask(numSets - 1)
00054 {
00055     int log_block_size = 0;
00056     int tmp_block_size = block_size;
00057     while (tmp_block_size > 1) {
00058         ++log_block_size;
00059         tmp_block_size = tmp_block_size >> 1;
00060     }
00061     assert(1<<log_block_size == block_size);
00062     MemReqPtr req;
00063     trace->getNextReq(req);
00064     refInfo.resize(numSets);
00065     while (req) {
00066         RefInfo temp;
00067         temp.addr = req->paddr >> log_block_size;
00068         int set = temp.addr & setMask;
00069         refInfo[set].push_back(temp);
00070         trace->getNextReq(req);
00071     }
00072 
00073     // Initialize top level of lookup table.
00074     lookupTable.resize(16);
00075 
00076     // Annotate references with next ref time.
00077     for (int k = 0; k < numSets; ++k) {
00078         for (RefIndex i = refInfo[k].size() - 1; i >= 0; --i) {
00079             Addr addr = refInfo[k][i].addr;
00080             initTable(addr, InfiniteRef);
00081             refInfo[k][i].nextRefTime = lookupValue(addr);
00082             setValue(addr, i);
00083         }
00084     }
00085 
00086     // Reset the lookup table
00087     for (int j = 0; j < 16; ++j) {
00088         if (lookupTable[j].size() == (1<<16)) {
00089             for (int k = 0; k < (1<<16); ++k) {
00090                 if (lookupTable[j][k].size() == (1<<16)) {
00091                     for (int l = 0; l < (1<<16); ++l) {
00092                         lookupTable[j][k][l] = -1;
00093                     }
00094                 }
00095             }
00096         }
00097     }
00098 
00099     tickEvent.schedule(0);
00100 
00101     hits = 0;
00102     misses = 0;
00103 }

00106 OptCPU::processSet(int set)
00107 {
00108     // Initialize cache
00109     int blks_in_cache = 0;
00110     RefIndex i = 0;
00111     cacheHeap.clear();
00112     cacheHeap.resize(assoc);
00113 
00114     while (blks_in_cache < assoc) {
00115         RefIndex cache_index = lookupValue(refInfo[set][i].addr);
00116         if (cache_index == -1) {
00117             // First reference to this block
00118             misses++;
00119             cache_index = blks_in_cache++;
00120             setValue(refInfo[set][i].addr, cache_index);
00121         } else {
00122             hits++;
00123         }
00124         // update cache heap to most recent reference
00125         cacheHeap[cache_index] = i;
00126         if (++i >= refInfo[set].size()) {
00127             return;
00128         }
00129     }
00130     for (int start = assoc/2; start >= 0; --start) {
00131         heapify(set,start);
00132     }
00133     //verifyHeap(set,0);
00134 
00135     for (; i < refInfo[set].size(); ++i) {
00136         RefIndex cache_index = lookupValue(refInfo[set][i].addr);
00137         if (cache_index == -1) {
00138             // miss
00139             misses++;
00140             // replace from cacheHeap[0]
00141             // mark replaced block as absent
00142             setValue(refInfo[set][cacheHeap[0]].addr, -1);
00143             setValue(refInfo[set][i].addr, 0);
00144             cacheHeap[0] = i;
00145             heapify(set, 0);
00146             // Make sure its in the cache
00147             assert(lookupValue(refInfo[set][i].addr) != -1);
00148         } else {
00149             // hit
00150             hits++;
00151             assert(refInfo[set][cacheHeap[cache_index]].addr ==
00152                    refInfo[set][i].addr);
00153             assert(refInfo[set][cacheHeap[cache_index]].nextRefTime == i);
00154             assert(heapLeft(cache_index) >= assoc);
00155 
00156             cacheHeap[cache_index] = i;
00157             processRankIncrease(set, cache_index);
00158             assert(lookupValue(refInfo[set][i].addr) != -1);
00159         }
00160     }
