00174 bool
00175 Packet::checkFunctional(Printable *obj, Addr addr, int size, uint8_t *data)
00176 {
00177     Addr func_start = getAddr();
00178     Addr func_end   = getAddr() + getSize() - 1;
00179     Addr val_start  = addr;
00180     Addr val_end    = val_start + size - 1;
00181 
00182     if (func_start > val_end || val_start > func_end) {
00183         // no intersection
00184         return false;
00185     }
00186 
00187     // check print first since it doesn't require data
00188     if (isPrint()) {
00189         dynamic_cast<PrintReqState*>(senderState)->printObj(obj);
00190         return false;
00191     }
00192 
00193     // if there's no data, there's no need to look further
00194     if (!data) {
00195         return false;
00196     }
00197 
00198     // offset of functional request into supplied value (could be
00199     // negative if partial overlap)
00200     int offset = func_start - val_start;
00201 
00202     if (isRead()) {
00203         if (func_start >= val_start && func_end <= val_end) {
00204             allocate();
00205             memcpy(getPtr<uint8_t>(), data + offset, getSize());
00206             return true;
00207         } else {
00208             // Offsets and sizes to copy in case of partial overlap
00209             int func_offset;
00210             int val_offset;
00211             int overlap_size;
00212 
00213             // calculate offsets and copy sizes for the two byte arrays
00214             if (val_start < func_start && val_end <= func_end) {
00215                 val_offset = func_start - val_start;
00216                 func_offset = 0;
00217                 overlap_size = val_end - func_start;
00218             } else if (val_start >= func_start && val_end > func_end) {
00219                 val_offset = 0;
00220                 func_offset = val_start - func_start;
00221                 overlap_size = func_end - val_start;
00222             } else if (val_start >= func_start && val_end <= func_end) {
00223                 val_offset = 0;
00224                 func_offset = val_start - func_start;
00225                 overlap_size = size;
00226             } else {
00227                 panic("BUG: Missed a case for a partial functional request");
00228             }
00229 
00230             // Figure out how much of the partial overlap should be copied
00231             // into the packet and not overwrite previously found bytes.
00232             if (bytesValidStart == 0 && bytesValidEnd == 0) {
00233                 // No bytes have been copied yet, just set indices
00234                 // to found range
00235                 bytesValidStart = func_offset;
00236                 bytesValidEnd = func_offset + overlap_size;
00237             } else {
00238                 // Some bytes have already been copied. Use bytesValid
00239                 // indices and offset values to figure out how much data
00240                 // to copy and where to copy it to.
00241 
00242                 // Indice overlap conditions to check
00243                 int a = func_offset - bytesValidStart;
00244                 int b = (func_offset + overlap_size) - bytesValidEnd;
00245                 int c = func_offset - bytesValidEnd;
00246                 int d = (func_offset + overlap_size) - bytesValidStart;
00247 
00248                 if (a >= 0 && b <= 0) {
00249                     // bytes already in pkt data array are superset of
00250                     // found bytes, will not copy any bytes
00251                     overlap_size = 0;
00252                 } else if (a < 0 && d >= 0 && b <= 0) {
00253                     // found bytes will move bytesValidStart towards 0
00254                     overlap_size = bytesValidStart - func_offset;
00255                     bytesValidStart = func_offset;
00256                 } else if (b > 0 && c <= 0 && a >= 0) {
00257                     // found bytes will move bytesValidEnd
00258                     // towards end of pkt data array
00259                     overlap_size =
00260                         (func_offset + overlap_size) - bytesValidEnd;
00261                     val_offset += bytesValidEnd - func_offset;
00262                     func_offset = bytesValidEnd;
00263                     bytesValidEnd += overlap_size;
00264                 } else if (a < 0 && b > 0) {
00265                     // Found bytes are superset of copied range. Will move
00266                     // bytesValidStart towards 0 and bytesValidEnd towards
00267                     // end of pkt data array.  Need to break copy into two
00268                     // pieces so as to not overwrite previously found data.
00269 
00270                     // copy the first half
00271                     uint8_t *dest = getPtr<uint8_t>() + func_offset;
00272                     uint8_t *src = data + val_offset;
00273                     memcpy(dest, src, (bytesValidStart - func_offset));
00274 
00275                     // re-calc the offsets and indices to do the copy
00276                     // required for the second half
00277                     val_offset += (bytesValidEnd - func_offset);
00278                     bytesValidStart = func_offset;
00279                     overlap_size =
00280                         (func_offset + overlap_size) - bytesValidEnd;
00281                     func_offset = bytesValidEnd;
00282                     bytesValidEnd += overlap_size;
00283                 } else if ((c > 0 && b > 0)
00284                            || (a < 0 && d < 0)) {
00285                     // region to be copied is discontiguous! Not supported.
00286                     panic("BUG: Discontiguous bytes found"
00287                           "for functional copying!");
00288                 }
00289             }
00290             assert(bytesValidEnd <= getSize());
00291 
00292             // copy partial data into the packet's data array
00293             uint8_t *dest = getPtr<uint8_t>() + func_offset;
00294             uint8_t *src = data + val_offset;
00295             memcpy(dest, src, overlap_size);
00296 
00297             // check if we're done filling the functional access
00298             bool done = (bytesValidStart == 0) && (bytesValidEnd == getSize());
00299             return done;
00300         }
00301     } else if (isWrite()) {
00302         if (offset >= 0) {
00303             memcpy(data + offset, getPtr<uint8_t>(),
00304                    (min(func_end, val_end) - func_start) + 1);
00305         } else {
00306             // val_start > func_start
00307             memcpy(data, getPtr<uint8_t>() - offset,
00308                    (min(func_end, val_end) - val_start) + 1);
00309         }
00310     } else {
00311         panic("Don't know how to handle command %s\n", cmdString());
00312     }
00313 
00314     // keep going with request by default
00315     return false;
00316 }
