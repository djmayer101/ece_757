00091 PacketQueue::schedSendEvent(Tick when)
00092 {
00093     // if we are waiting on a retry, do not schedule a send event, and
00094     // instead rely on retry being called
00095     if (waitingOnRetry) {
00096         assert(!sendEvent.scheduled());
00097         return;
00098     }
00099 
00100     if (!sendEvent.scheduled()) {
00101         em.schedule(&sendEvent, when);
00102     } else if (sendEvent.when() > when) {
00103         em.reschedule(&sendEvent, when);
00104     }

00107 void
00108 PacketQueue::schedSendTiming(PacketPtr pkt, Tick when, bool send_as_snoop)
00109 {
00110     // we can still send a packet before the end of this tick
00111     assert(when >= curTick());
00112 
00113     // express snoops should never be queued
00114     assert(!pkt->isExpressSnoop());
00115 
00116     // add a very basic sanity check on the port to ensure the
00117     // invisible buffer is not growing beyond reasonable limits
00118     if (transmitList.size() > 100) {
00119         panic("Packet queue %s has grown beyond 100 packets\n",
00120               name());
00121     }
00122 
00123     // nothing on the list, or earlier than current front element,
00124     // schedule an event
00125     if (transmitList.empty() || when < transmitList.front().tick) {
00126         // note that currently we ignore a potentially outstanding retry
00127         // and could in theory put a new packet at the head of the
00128         // transmit list before retrying the existing packet
00129         transmitList.push_front(DeferredPacket(when, pkt, send_as_snoop));
00130         schedSendEvent(when);
00131         return;
00132     }
00133 
00134     // list is non-empty and this belongs at the end
00135     if (when >= transmitList.back().tick) {
00136         transmitList.push_back(DeferredPacket(when, pkt, send_as_snoop));
00137         return;
00138     }
00139 
00140     // this belongs in the middle somewhere, insertion sort
00141     DeferredPacketIterator i = transmitList.begin();
00142     ++i; // already checked for insertion at front
00143     while (i != transmitList.end() && when >= i->tick)
00144         ++i;
00145     transmitList.insert(i, DeferredPacket(when, pkt, send_as_snoop));
00146 }

00148 void PacketQueue::trySendTiming()
00149 {
00150     assert(deferredPacketReady());
00151 
00152     // take the next packet off the list here, as we might return to
00153     // ourselves through the sendTiming call below
00154     DeferredPacket dp = transmitList.front();
00155     transmitList.pop_front();
00156 
00157     // use the appropriate implementation of sendTiming based on the
00158     // type of port associated with the queue, and whether the packet
00159     // is to be sent as a snoop or not
00160     waitingOnRetry = !sendTiming(dp.pkt, dp.sendAsSnoop);
00161 
00162     if (waitingOnRetry) {
00163         // put the packet back at the front of the list (packet should
00164         // not have changed since it wasn't accepted)
00165         assert(!sendEvent.scheduled());
00166         transmitList.push_front(dp);
00167     }
00168 }

00170 void
00171 PacketQueue::scheduleSend(Tick time)
00172 {
00173     // the next ready time is either determined by the next deferred packet,
00174     // or in the cache through the MSHR ready time
00175     Tick nextReady = std::min(deferredPacketReadyTime(), time);
00176 
00177     if (nextReady != MaxTick) {
00178         // if the sendTiming caused someone else to call our
00179         // recvTiming we could already have an event scheduled, check
00180         if (!sendEvent.scheduled())
00181             em.schedule(&sendEvent, std::max(nextReady, curTick() + 1));
00182     } else {
00183         // no more to send, so if we're draining, we may be done
00184         if (drainManager && transmitList.empty() && !sendEvent.scheduled()) {
00185             DPRINTF(Drain, "PacketQueue done draining,"
00186                     "processing drain event\n");
00187             drainManager->signalDrainDone();
00188             drainManager = NULL;
00189         }
00190     }
00191 }

00194 PacketQueue::sendDeferredPacket()
