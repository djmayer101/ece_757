00177 Fault
00178 PageTable::translate(RequestPtr req)
00179 {
00180     Addr paddr;
00181     assert(pageAlign(req->getVaddr() + req->getSize() - 1)
00182            == pageAlign(req->getVaddr()));
00183     if (!translate(req->getVaddr(), paddr)) {
00184         return Fault(new GenericPageTableFault(req->getVaddr()));

00186     req->setPaddr(paddr);
00187     if ((paddr & (pageSize - 1)) + req->getSize() > pageSize) {
