00126         dynamic_cast<WalkerSenderState *>(pkt->popSenderState());

00129     delete senderState;

00520 void
00521 Walker::WalkerState::setupWalk(Addr vaddr)
00522 {
00523     VAddr addr = vaddr;
00524     CR3 cr3 = tc->readMiscRegNoEffect(MISCREG_CR3);
00525     // Check if we're in long mode or not
00526     Efer efer = tc->readMiscRegNoEffect(MISCREG_EFER);
00527     dataSize = 8;
00528     Addr topAddr;
00529     if (efer.lma) {
00530         // Do long mode.
00531         state = LongPML4;
00532         topAddr = (cr3.longPdtb << 12) + addr.longl4 * dataSize;
00533         enableNX = efer.nxe;
00534     } else {
00535         // We're in some flavor of legacy mode.
00536         CR4 cr4 = tc->readMiscRegNoEffect(MISCREG_CR4);
00537         if (cr4.pae) {
00538             // Do legacy PAE.
00539             state = PAEPDP;
00540             topAddr = (cr3.paePdtb << 5) + addr.pael3 * dataSize;
00541             enableNX = efer.nxe;
00542         } else {
00543             dataSize = 4;
00544             topAddr = (cr3.pdtb << 12) + addr.norml2 * dataSize;
00545             if (cr4.pse) {
00546                 // Do legacy PSE.
00547                 state = PSEPD;
00548             } else {
00549                 // Do legacy non PSE.
00550                 state = PD;
00551             }
00552             enableNX = false;
00553         }
00554     }
00555 
00556     nextState = Ready;
00557     entry.vaddr = vaddr;
00558 
00559     Request::Flags flags = Request::PHYSICAL;
00560     if (cr3.pcd)
00561         flags.set(Request::UNCACHEABLE);
00562     RequestPtr request = new Request(topAddr, dataSize, flags,
00563                                      walker->masterId);
00564     read = new Packet(request, MemCmd::ReadReq);
00565     read->allocate();
00566 }

00568 bool
00569 Walker::WalkerState::recvPacket(PacketPtr pkt)
00570 {
00571     assert(pkt->isResponse());
00572     assert(inflight);
00573     assert(state == Waiting);
00574     assert(!read);
00575     inflight--;
00576     if (pkt->isRead()) {
00577         // @todo someone should pay for this
00578         pkt->busFirstWordDelay = pkt->busLastWordDelay = 0;
00579 
00580         state = nextState;
00581         nextState = Ready;
00582         PacketPtr write = NULL;
00583         read = pkt;
00584         timingFault = stepWalk(write);
00585         state = Waiting;
00586         assert(timingFault == NoFault || read == NULL);
00587         if (write) {
00588             writes.push_back(write);
00589         }
00590         sendPackets();
00591     } else {
00592         sendPackets();
00593     }
00594     if (inflight == 0 && read == NULL && writes.size() == 0) {
00595         state = Ready;
00596         nextState = Waiting;
00597         if (timingFault == NoFault) {
00598             /*
00599              * Finish the translation. Now that we now the right entry is
00600              * in the TLB, this should work with no memory accesses.
00601              * There could be new faults unrelated to the table walk like
00602              * permissions violations, so we'll need the return value as
00603              * well.
00604              */
00605             bool delayedResponse;
00606             Fault fault = walker->tlb->translate(req, tc, NULL, mode,
00607                                                  delayedResponse, true);
00608             assert(!delayedResponse);
00609             // Let the CPU continue.
00610             translation->finish(fault, req, tc, mode);
00611         } else {
00612             // There was a fault during the walk. Let the CPU know.
00613             translation->finish(timingFault, req, tc, mode);
00614         }
00615         return true;
00616     }
00617 
00618     return false;
00619 }

00621 void
00622 Walker::WalkerState::sendPackets()
00623 {
00624     //If we're already waiting for the port to become available, just return.
00625     if (retrying)
00626         return;
00627 
00628     //Reads always have priority
00629     if (read) {
00630         PacketPtr pkt = read;
00631         read = NULL;
00632         inflight++;
00633         if (!walker->sendTiming(this, pkt)) {
00634             retrying = true;
00635             read = pkt;
00636             inflight--;
00637             return;
00638         }
00639     }
