00087 bool
00088 PCEventQueue::doService(ThreadContext *tc)
00089 {
00090     // This will fail to break on Alpha PALcode addresses, but that is
00091     // a rare use case.
00092     Addr pc = tc->instAddr();
00093     int serviced = 0;
00094     range_t range = equal_range(pc);
00095     for (iterator i = range.first; i != range.second; ++i) {
00096         // Make sure that the pc wasn't changed as the side effect of
00097         // another event.  This for example, prevents two invocations
00098         // of the SkipFuncEvent.  Maybe we should have separate PC
00099         // event queues for each processor?
00100         if (pc != tc->instAddr())
00101             continue;
00102 
00103         DPRINTF(PCEvent, "PC based event serviced at %#x: %s\n",
00104                 (*i)->pc(), (*i)->descr());
00105 
00106         (*i)->process(tc);
00107         ++serviced;
00108     }
00109 
00110     return serviced > 0;
00111 }
