00240             config.latencyTimer = pkt->get<uint8_t>();

00242           /* Do nothing for these read-only registers */

00244           case PCI0_MINIMUM_GRANT:
00245           case PCI0_MAXIMUM_LATENCY:
00246           case PCI_CLASS_CODE:
00247           case PCI_REVISION_ID:
00248             break;
00249           default:
00250             panic("writing to a read only register");
00251         }
00252         DPRINTF(PCIDEV,
00253             "writeConfig: dev %#x func %#x reg %#x 1 bytes: data = %#x\n",
00254             params()->pci_dev, params()->pci_func, offset,
00255             (uint32_t)pkt->get<uint8_t>());
00256         break;
00257       case sizeof(uint16_t):
00258         switch (offset) {
00259           case PCI_COMMAND:
00260             config.command = pkt->get<uint8_t>();
00261             break;
00262           case PCI_STATUS:
00263             config.status = pkt->get<uint8_t>();
00264             break;
00265           case PCI_CACHE_LINE_SIZE:
00266             config.cacheLineSize = pkt->get<uint8_t>();
00267             break;
00268           default:
00269             panic("writing to a read only register");
00270         }
00271         DPRINTF(PCIDEV,
00272             "writeConfig: dev %#x func %#x reg %#x 2 bytes: data = %#x\n",
00273             params()->pci_dev, params()->pci_func, offset,
00274             (uint32_t)pkt->get<uint16_t>());
00275         break;
00276       case sizeof(uint32_t):
00277         switch (offset) {
00278           case PCI0_BASE_ADDR0:
00279           case PCI0_BASE_ADDR1:
00280           case PCI0_BASE_ADDR2:
00281           case PCI0_BASE_ADDR3:
00282           case PCI0_BASE_ADDR4:
00283           case PCI0_BASE_ADDR5:
00284             {
00285                 int barnum = BAR_NUMBER(offset);
00286 
00287                 if (!legacyIO[barnum]) {
00288                     // convert BAR values to host endianness
00289                     uint32_t he_old_bar = letoh(config.baseAddr[barnum]);
00290                     uint32_t he_new_bar = letoh(pkt->get<uint32_t>());
00291 
00292                     uint32_t bar_mask =
00293                         BAR_IO_SPACE(he_old_bar) ? BAR_IO_MASK : BAR_MEM_MASK;
00294 
00295                     // Writing 0xffffffff to a BAR tells the card to set the
00296                     // value of the bar to a bitmask indicating the size of
00297                     // memory it needs
00298                     if (he_new_bar == 0xffffffff) {
00299                         he_new_bar = ~(BARSize[barnum] - 1);
00300                     } else {
00301                         // does it mean something special to write 0 to a BAR?
00302                         he_new_bar &= ~bar_mask;
00303                         if (he_new_bar) {
00304                             BARAddrs[barnum] = BAR_IO_SPACE(he_old_bar) ?
00305                                 platform->calcPciIOAddr(he_new_bar) :
00306                                 platform->calcPciMemAddr(he_new_bar);
00307                             pioPort.sendRangeChange();
00308                         }
00309                     }
00310                     config.baseAddr[barnum] = htole((he_new_bar & ~bar_mask) |
00311                                                     (he_old_bar & bar_mask));
00312                 }
00313             }
00314             break;
00315 
00316           case PCI0_ROM_BASE_ADDR:
00317             if (letoh(pkt->get<uint32_t>()) == 0xfffffffe)
00318                 config.expansionROM = htole((uint32_t)0xffffffff);
00319             else
00320                 config.expansionROM = pkt->get<uint32_t>();
00321             break;
00322 
00323           case PCI_COMMAND:
00324             // This could also clear some of the error bits in the Status
00325             // register. However they should never get set, so lets ignore
00326             // it for now
00327             config.command = pkt->get<uint32_t>();
00328             break;
00329 
00330           default:
00331             DPRINTF(PCIDEV, "Writing to a read only register");
00332         }
00333         DPRINTF(PCIDEV,
00334             "writeConfig: dev %#x func %#x reg %#x 4 bytes: data = %#x\n",
00335             params()->pci_dev, params()->pci_func, offset,
00336             (uint32_t)pkt->get<uint32_t>());
00337         break;
00338       default:
00339         panic("invalid access size(?) for PCI configspace!\n");
00340     }
00341     pkt->makeAtomicResponse();
