00122                 curr_memories.push_back(r->second);
00123             } else {
00124                 vector<AbstractMemory*> single_memory;
00125                 single_memory.push_back(r->second);
00126                 createBackingStore(r->first, single_memory);
00127             }
00128         }
00129     }
00130 
00131     // if there is still interleaved ranges waiting to be merged, go
00132     // ahead and do it

00134         AddrRange merged_range(intlv_ranges);

00142 {

00151     uint8_t* pmem = (uint8_t*) mmap(NULL, range.size(),
00152                                     PROT_READ | PROT_WRITE,
00153                                     map_flags, -1, 0);
00154 
00155     if (pmem == (uint8_t*) MAP_FAILED) {
00156         perror("mmap");
00157         fatal("Could not mmap %d bytes for range %s!\n", range.size(),
00158               range.to_string());
00159     }
00160 
00161     // remember this backing store so we can checkpoint it and unmap
00162     // it appropriately
00163     backingStore.push_back(make_pair(range, pmem));
00164 
00165     // count how many of the memories are to be zero initialized so we
00166     // can see if some but not all have this parameter set
00167     uint32_t init_to_zero = 0;
00168 
00169     // point the memories to their backing store, and if requested,
00170     // initialize the memory range to 0
00171     for (vector<AbstractMemory*>::const_iterator m = _memories.begin();
00172          m != _memories.end(); ++m) {
00173         DPRINTF(BusAddrRanges, "Mapping memory %s to backing store\n",
00174                 (*m)->name());
00175         (*m)->setBackingStore(pmem);
00176 
00177         // if it should be zero, then go and make it so
00178         if ((*m)->initToZero()) {
00179             ++init_to_zero;
00180         }
00181     }
00182 
00183     if (init_to_zero != 0) {
00184         if (init_to_zero != _memories.size())
00185             fatal("Some, but not all memories in range %s are set zero\n",
00186                   range.to_string());
