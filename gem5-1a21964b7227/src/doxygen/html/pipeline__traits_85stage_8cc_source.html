00093     if (!inst->staticInst) {

00098     int stPri = 0;

00103     //
00104     // EX - Stage 2
00105     // ---------------------------------------
00106     for (int idx=0; idx < inst->numSrcRegs(); idx++) {
00107         if (!idx || !inst->isStore())
00108             inst_sched->push(new ScheduleEntry(stNum, stPri++, RegManager, UseDefUnit::ReadSrcReg, idx));
00109     }
00110 
00111     if ( inst->isNonSpeculative() ) {
00112         // skip execution of non speculative insts until later
00113     } else if (inst->isMemRef()) {
00114         inst_sched->push(new ScheduleEntry(stNum, stPri++, AGEN, AGENUnit::GenerateAddr));
00115         if ( inst->isLoad() ) {
00116             inst_sched->push(new ScheduleEntry(stNum, stPri++, DTLB, TLBUnit::DataLookup));
00117             inst_sched->push(new ScheduleEntry(stNum, stPri++, DCache, CacheUnit::InitiateReadData));
00118         }
00119     } else {
00120         inst_sched->push(new ScheduleEntry(stNum, stPri++, ExecUnit, ExecutionUnit::ExecuteInst));
00121     }
00122 
00123     //
00124     // MEM - Stage 3
00125     // ---------------------------------------
00126     stPri = 0; stNum++;
00127     if ( inst->isStore() ) { // for store, need src reg at this point
00128         inst_sched->push(new ScheduleEntry(stNum, stPri++, RegManager, UseDefUnit::ReadSrcReg, 1));
00129     }
00130     if ( inst->isLoad() ) {
00131         inst_sched->push(new ScheduleEntry(stNum, stPri++, DCache, CacheUnit::CompleteReadData));
00132     } else if ( inst->isStore() ) {
00133         inst_sched->push(new ScheduleEntry(stNum, stPri++, DTLB, TLBUnit::DataLookup));
00134         inst_sched->push(new ScheduleEntry(stNum, stPri++, DCache, CacheUnit::InitiateWriteData));
00135     }
00136 
00137     //
