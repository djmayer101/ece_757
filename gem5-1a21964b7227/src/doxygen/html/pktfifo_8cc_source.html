00042

00044     iterator end = fifo.end();

00046         EthPacketPtr &pkt = i->packet;

00048             offset -= pkt->length;

00050         }
00051

00053             panic("invalid fifo");

00055         unsigned size = min(pkt->length - offset, len);
00056         memcpy(data, pkt->data, size);

00061     }
