00456 void
00457 Pl111::fillFifo()
00458 {
00459     while ((dmaPendingNum < maxOutstandingDma) && (maxAddr >= curAddr + dmaSize )) {
00460         // concurrent dma reads need different dma done events
00461         // due to assertion in scheduling state
00462         ++dmaPendingNum;
00463 
00464         assert(!dmaDoneEventFree.empty());
00465         DmaDoneEvent *event(dmaDoneEventFree.back());
00466         dmaDoneEventFree.pop_back();
00467         assert(!event->scheduled());
00468 
00469         // We use a uncachable request here because the requests from the CPU
00470         // will be uncacheable as well. If we have uncacheable and cacheable
00471         // requests in the memory system for the same address it won't be
00472         // pleased
00473         dmaPort.dmaAction(MemCmd::ReadReq, curAddr + startAddr, dmaSize,
00474                           event, curAddr + dmaBuffer,
00475                           0, Request::UNCACHEABLE);
00476         curAddr += dmaSize;
00477     }

00493             lcdRis.underflow = 1;
00494             if (!intEvent.scheduled())
00495                 schedule(intEvent, nextCycle());
00496         }
00497 
00498         assert(!readEvent.scheduled());
00499         if (vnc)
00500             vnc->setDirty();
00501 
00502         DPRINTF(PL111, "-- write out frame buffer into bmp\n");
00503 
00504         if (!pic)
00505             pic = simout.create(csprintf("%s.framebuffer.bmp", sys->name()), true);
00506 
00507         assert(bmp);
00508         assert(pic);
00509         pic->seekp(0);
00510         bmp->write(pic);
00511 
00512         // schedule the next read based on when the last frame started
00513         // and the desired fps (i.e. maxFrameTime), we turn the
00514         // argument into a relative number of cycles in the future
00515         if (lcdControl.lcden)
00516             schedule(readEvent, clockEdge(ticksToCycles(startTime -

00522         return;
