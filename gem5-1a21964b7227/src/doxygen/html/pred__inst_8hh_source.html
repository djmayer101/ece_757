00057 static inline uint32_t
00058 modified_imm(uint8_t ctrlImm, uint8_t dataImm)
00059 {
00060     uint32_t bigData = dataImm;
00061     uint32_t bigCtrl = ctrlImm;
00062     if (bigCtrl < 4) {
00063         switch (bigCtrl) {
00064           case 0:
00065             return bigData;
00066           case 1:
00067             return bigData | (bigData << 16);
00068           case 2:
00069             return (bigData << 8) | (bigData << 24);
00070           case 3:
00071             return (bigData << 0) | (bigData << 8) |
00072                    (bigData << 16) | (bigData << 24);
00073         }
00074     }
00075     bigCtrl = (bigCtrl << 1) | ((bigData >> 7) & 0x1);
00076     bigData |= (1 << 7);
00077     return bigData << (32 - bigCtrl);
00078 }

00080 static inline uint64_t
00081 simd_modified_imm(bool op, uint8_t cmode, uint8_t data, bool &immValid)
00082 {
00083     uint64_t bigData = data;
00084     immValid = true;
00085     switch (cmode) {
00086       case 0x0:
00087       case 0x1:
00088         bigData = (bigData << 0) | (bigData << 32);
00089         break;
00090       case 0x2:
00091       case 0x3:
00092         bigData = (bigData << 8) | (bigData << 40);
00093         break;
00094       case 0x4:
00095       case 0x5:
00096         bigData = (bigData << 16) | (bigData << 48);
00097         break;
00098       case 0x6:
00099       case 0x7:
00100         bigData = (bigData << 24) | (bigData << 56);
00101         break;
00102       case 0x8:
00103       case 0x9:
00104         bigData = (bigData << 0) | (bigData << 16) |
00105                   (bigData << 32) | (bigData << 48);
00106         break;
00107       case 0xa:
00108       case 0xb:
00109         bigData = (bigData << 8) | (bigData << 24) |
00110                   (bigData << 40) | (bigData << 56);
00111         break;
00112       case 0xc:
00113         bigData = (0xffULL << 0) | (bigData << 8) |
00114                   (0xffULL << 32) | (bigData << 40);
00115         break;
00116       case 0xd:
00117         bigData = (0xffffULL << 0) | (bigData << 16) |
00118                   (0xffffULL << 32) | (bigData << 48);
00119         break;
00120       case 0xe:
00121         if (op) {
00122             bigData = 0;
00123             for (int i = 7; i >= 0; i--) {
00124                 if (bits(data, i)) {
00125                     bigData |= (ULL(0xFF) << (i * 8));
00126                 }
00127             }
00128         } else {
00129             bigData = (bigData << 0)  | (bigData << 8)  |
00130                       (bigData << 16) | (bigData << 24) |
00131                       (bigData << 32) | (bigData << 40) |
00132                       (bigData << 48) | (bigData << 56);
00133         }
00134         break;
00135       case 0xf:
00136         if (!op) {
00137             uint64_t bVal = bits(bigData, 6) ? (0x1F) : (0x20);
00138             bigData = (bits(bigData, 5, 0) << 19) |
00139                       (bVal << 25) | (bits(bigData, 7) << 31);
00140             bigData |= (bigData << 32);
00141             break;
00142         }
00143         // Fall through, immediate encoding is invalid.
00144       default:
00145         immValid = false;
00146         break;
00147     }
00148     return bigData;

00151 static inline uint64_t
00152 vfp_modified_imm(uint8_t data, bool wide)
00153 {
00154     uint64_t bigData = data;
00155     uint64_t repData;
00156     if (wide) {
00157         repData = bits(data, 6) ? 0xFF : 0;
00158         bigData = (bits(bigData, 5, 0) << 48) |
00159                   (repData << 54) | (bits(~bigData, 6) << 62) |
00160                   (bits(bigData, 7) << 63);
00161     } else {
00162         repData = bits(data, 6) ? 0x1F : 0;
00163         bigData = (bits(bigData, 5, 0) << 19) |
00164                   (repData << 25) | (bits(~bigData, 6) << 30) |
00165                   (bits(bigData, 7) << 31);
00166     }
