00051         panic("Could not open %s for writing\n", filename);

00053     // Wrap the output file in a zero copy stream, that in turn is
00054     // wrapped in a gzip stream if the filename ends with .gz. The
00055     // latter stream is in turn wrapped in a coded stream

00059         gzipStream = new io::GzipOutputStream(zeroCopyStream);
00060         codedStream = new io::CodedOutputStream(gzipStream);
00061     } else {
00062         codedStream = new io::CodedOutputStream(zeroCopyStream);
00063     }
00064

00066     codedStream->WriteLittleEndian32(magicNumber);
00067 
00068     // Note that each type of stream (packet, instruction etc) should
00069     // add its own header and perform the appropriate checks

00098     if (!fileStream.good())
00099         panic("Could not open %s for reading\n", filename);

00113 void
00114 ProtoInputStream::createStreams()
00115 {
00116     // All streams should be NULL at this point
00117     assert(zeroCopyStream == NULL && gzipStream == NULL &&
00118            codedStream == NULL);
00119 
00120     // Wrap the input file in a zero copy stream, that in turn is
00121     // wrapped in a gzip stream if the filename ends with .gz. The
00122     // latter stream is in turn wrapped in a coded stream
00123     zeroCopyStream = new io::IstreamInputStream(&fileStream);
00124     if (useGzip) {
00125         gzipStream = new io::GzipInputStream(zeroCopyStream);
00126         codedStream = new io::CodedInputStream(gzipStream);
00127     } else {
00128         codedStream = new io::CodedInputStream(zeroCopyStream);
00129     }
00130 
00131     uint32_t magic_check;
00132     if (!codedStream->ReadLittleEndian32(&magic_check) ||
00133         magic_check != magicNumber)
00134         panic("Input file %s is not a valid gem5 proto format.\n",
00135               fileName);

00170 bool
00171 ProtoInputStream::read(Message& msg)
00172 {
00173     // Read a message from the stream by getting the size, using it as
00174     // a limit when parsing the message, then popping the limit again
00175     uint32_t size;
00176     if (codedStream->ReadVarint32(&size)) {
00177         io::CodedInputStream::Limit limit = codedStream->PushLimit(size);
00178         if (msg.ParseFromCodedStream(codedStream)) {
00179             codedStream->PopLimit(limit);
00180             // All went well, the message is parsed and the limit is
00181             // popped again
00182             return true;
00183         } else {
00184             panic("Unable to read message from coded stream %s\n",
00185                   fileName);
00186         }
00187     }
00188 
00189     return false;
00190 }
