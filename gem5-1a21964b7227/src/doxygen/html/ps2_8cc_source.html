00083 {
00084     if (key <= XK_asciitilde) {
00085         uint16_t tmp = keySymToPs2Byte[key];
00086         uint8_t code = tmp & 0xff;
00087         bool shift = tmp >> 8;
00088 
00089         if (down) {
00090             if (!cur_shift && shift) {
00091                 keys.push_back(ShiftKey);
00092                 cur_shift = true;
00093             }
00094             keys.push_back(code);
00095         } else {
00096             if (cur_shift && !shift) {
00097                 keys.push_back(BreakKey);
00098                 keys.push_back(ShiftKey);
00099                 cur_shift = false;
00100             }
00101             keys.push_back(BreakKey);
00102             keys.push_back(code);
00103         }
00104     } else {
00105         if ((key & UpperKeys) == UpperKeys) {
00106             bool extended = false;
00107             switch (key) {
00108               case XK_BackSpace:
00109                 keys.push_back(0x66);
00110                 break;

00112                 keys.push_back(0x0d);
00113                 break;

00115                 keys.push_back(0x5a);
00116                 break;
00117              case XK_Escape:
00118                 keys.push_back(0x76);
00119                 break;
00120              case XK_Delete:
00121                 extended = true;
00122                 keys.push_back(0x71);
00123                 break;
00124              case XK_Home:
00125                 extended = true;
00126                 keys.push_back(0x6c);
00127                 break;

00135                 break;

00137                 extended = true;

00139                 break;

00141                 extended = true;
