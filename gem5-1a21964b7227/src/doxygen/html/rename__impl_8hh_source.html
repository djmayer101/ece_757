00307 void
00308 DefaultRename<Impl>::squash(const InstSeqNum &squash_seq_num, ThreadID tid)
00309 {
00310     DPRINTF(Rename, "[tid:%u]: Squashing instructions.\n",tid);
00311 
00312     // Clear the stall signal if rename was blocked or unblocking before.
00313     // If it still needs to block, the blocking should happen the next
00314     // cycle and there should be space to hold everything due to the squash.
00315     if (renameStatus[tid] == Blocked ||
00316         renameStatus[tid] == Unblocking) {
00317         toDecode->renameUnblock[tid] = 1;
00318 
00319         resumeSerialize = false;
00320         serializeInst[tid] = NULL;
00321     } else if (renameStatus[tid] == SerializeStall) {
00322         if (serializeInst[tid]->seqNum <= squash_seq_num) {
00323             DPRINTF(Rename, "Rename will resume serializing after squash\n");
00324             resumeSerialize = true;
00325             assert(serializeInst[tid]);
00326         } else {
00327             resumeSerialize = false;
00328             toDecode->renameUnblock[tid] = 1;
00329 
00330             serializeInst[tid] = NULL;
00331         }
00332     }
00333 
00334     // Set the status to Squashing.
00335     renameStatus[tid] = Squashing;
00336 
00337     // Squash any instructions from decode.
00338     unsigned squashCount = 0;
00339 
00340     for (int i=0; i<fromDecode->size; i++) {
00341         if (fromDecode->insts[i]->threadNumber == tid &&
00342             fromDecode->insts[i]->seqNum > squash_seq_num) {
00343             fromDecode->insts[i]->setSquashed();
00344             wroteToTimeBuffer = true;
00345             squashCount++;
00346         }
00347 
00348     }
00349 
00350     // Clear the instruction list and skid buffer in case they have any
00351     // insts in them.
00352     insts[tid].clear();
00353 
00354     // Clear the skid buffer in case it has any data in it.
00355     skidBuffer[tid].clear();
00356 
00357     doSquash(squash_seq_num, tid);
00358 }

00421 template<class Impl>
00422 void
00423 DefaultRename<Impl>::rename(bool &status_change, ThreadID tid)
00424 {
00425     // If status is Running or idle,
00426     //     call renameInsts()
00427     // If status is Unblocking,
00428     //     buffer any instructions coming from decode
00429     //     continue trying to empty skid buffer
00430     //     check if stall conditions have passed
00431 
00432     if (renameStatus[tid] == Blocked) {
00433         ++renameBlockCycles;
00434     } else if (renameStatus[tid] == Squashing) {
00435         ++renameSquashCycles;
00436     } else if (renameStatus[tid] == SerializeStall) {
00437         ++renameSerializeStallCycles;
00438         // If we are currently in SerializeStall and resumeSerialize
00439         // was set, then that means that we are resuming serializing
00440         // this cycle.  Tell the previous stages to block.
00441         if (resumeSerialize) {
00442             resumeSerialize = false;
00443             block(tid);
00444             toDecode->renameUnblock[tid] = false;
00445         }
00446     } else if (renameStatus[tid] == Unblocking) {
00447         if (resumeUnblocking) {
00448             block(tid);
00449             resumeUnblocking = false;
00450             toDecode->renameUnblock[tid] = false;
00451         }
00452     }
00453 
00454     if (renameStatus[tid] == Running ||
00455         renameStatus[tid] == Idle) {
00456         DPRINTF(Rename, "[tid:%u]: Not blocked, so attempting to run "
00457                 "stage.\n", tid);
00458 
00459         renameInsts(tid);
00460     } else if (renameStatus[tid] == Unblocking) {
00461         renameInsts(tid);
00462 
00463         if (validInsts()) {
00464             // Add the current inputs to the skid buffer so they can be
00465             // reprocessed when this stage unblocks.
00466             skidInsert(tid);
00467         }
00468 
00469         // If we switched over to blocking, then there's a potential for
00470         // an overall status change.
00471         status_change = unblock(tid) || status_change || blockThisCycle;
00472     }
00473 }

00655 
00656         renameSrcRegs(inst, inst->threadNumber);
00657 
00658         renameDestRegs(inst, inst->threadNumber);
00659 
00660         ++renamed_insts;
00661 
00662 
00663         // Put instruction in rename queue.
00664         toIEW->insts[toIEWIndex] = inst;
00665         ++(toIEW->size);
00666 
00667         // Increment which instruction we're on.
00668         ++toIEWIndex;
00669 
00670         // Decrement how many instructions are available.
00671         --insts_available;
00672     }
00673 
00674     instsInProgress[tid] += renamed_insts;
00675     renameRenamedInsts += renamed_insts;
00676 
00677     // If we wrote to the time buffer, record this.
00678     if (toIEWIndex) {
00679         wroteToTimeBuffer = true;
00680     }
00681 
00682     // Check if there's any instructions left that haven't yet been renamed.
00683     // If so then block.
00684     if (insts_available) {
00685         blockThisCycle = true;
00686     }
00687 
00688     if (blockThisCycle) {
00689         block(tid);
00690         toDecode->renameUnblock[tid] = false;
00691     }
00692 }

00694 template<class Impl>
00695 void
00696 DefaultRename<Impl>::skidInsert(ThreadID tid)
00697 {
00698     DynInstPtr inst = NULL;
00699 
00700     while (!insts[tid].empty()) {
00701         inst = insts[tid].front();
00702 
00703         insts[tid].pop_front();
00704 
00705         assert(tid == inst->threadNumber);
00706 
00707         DPRINTF(Rename, "[tid:%u]: Inserting [sn:%lli] PC: %s into Rename "
00708                 "skidBuffer\n", tid, inst->seqNum, inst->pcState());
00709 
00710         ++renameSkidInsts;
00711 
00712         skidBuffer[tid].push_back(inst);
00713     }
00714 
00715     if (skidBuffer[tid].size() > skidBufferMax)
00716     {
00717         typename InstQueue::iterator it;
00718         warn("Skidbuffer contents:\n");
00719         for(it = skidBuffer[tid].begin(); it != skidBuffer[tid].end(); it++)
00720         {
00721             warn("[tid:%u]: %s [sn:%i].\n", tid,
00722                     (*it)->staticInst->disassemble(inst->instAddr()),
00723                     (*it)->seqNum);
00724         }
00725         panic("Skidbuffer Exceeded Max Size");
00726     }
00727 }

00762 template<class Impl>
00763 void
00764 DefaultRename<Impl>::updateStatus()
00765 {
00766     bool any_unblocking = false;
00767 
00768     list<ThreadID>::iterator threads = activeThreads->begin();
00769     list<ThreadID>::iterator end = activeThreads->end();
00770 
00771     while (threads != end) {
00772         ThreadID tid = *threads++;
00773 
00774         if (renameStatus[tid] == Unblocking) {
00775             any_unblocking = true;
00776             break;
00777         }
00778     }
00779 
00780     // Rename will have activity if it's unblocking.
00781     if (any_unblocking) {
00782         if (_status == Inactive) {
00783             _status = Active;
00784

00786 
00787             cpu->activateStage(O3CPU::RenameIdx);
00788         }
00789     } else {
00790         // If it's not unblocking, then rename will not have any internal
00791         // activity.  Switch it to inactive.
00792         if (_status == Active) {
00793             _status = Inactive;
00794             DPRINTF(Activity, "Deactivating stage.\n");
00795 
00796             cpu->deactivateStage(O3CPU::RenameIdx);
00797         }
00798     }

00801 template <class Impl>
00802 bool
00803 DefaultRename<Impl>::block(ThreadID tid)
00804 {
00805     DPRINTF(Rename, "[tid:%u]: Blocking.\n", tid);
00806 
00807     // Add the current inputs onto the skid buffer, so they can be
00808     // reprocessed when this stage unblocks.
00809     skidInsert(tid);
00810 
00811     // Only signal backwards to block if the previous stages do not think
00812     // rename is already blocked.
00813     if (renameStatus[tid] != Blocked) {
00814         // If resumeUnblocking is set, we unblocked during the squash,
00815         // but now we're have unblocking status. We need to tell earlier
00816         // stages to block.
00817         if (resumeUnblocking || renameStatus[tid] != Unblocking) {
00818             toDecode->renameBlock[tid] = true;
00819             toDecode->renameUnblock[tid] = false;
00820             wroteToTimeBuffer = true;
00821         }
00822 
00823         // Rename can not go from SerializeStall to Blocked, otherwise
00824         // it would not know to complete the serialize stall.
00825         if (renameStatus[tid] != SerializeStall) {
00826             // Set status to Blocked.
00827             renameStatus[tid] = Blocked;
00828             return true;
00829         }
00830     }
00831 
00832     return false;
00833 }

00835 template <class Impl>
00836 bool
00837 DefaultRename<Impl>::unblock(ThreadID tid)
00838 {
00839     DPRINTF(Rename, "[tid:%u]: Trying to unblock.\n", tid);
00840 
00841     // Rename is done unblocking if the skid buffer is empty.
00842     if (skidBuffer[tid].empty() && renameStatus[tid] != SerializeStall) {
00843 
00844         DPRINTF(Rename, "[tid:%u]: Done unblocking.\n", tid);
00845 
00846         toDecode->renameUnblock[tid] = true;
00847         wroteToTimeBuffer = true;
00848 
00849         renameStatus[tid] = Running;
00850         return true;
00851     }
00852 
00853     return false;
00854 }

00856 template <class Impl>
00857 void
00858 DefaultRename<Impl>::doSquash(const InstSeqNum &squashed_seq_num, ThreadID tid)
00859 {
00860     typename std::list<RenameHistory>::iterator hb_it =
00861         historyBuffer[tid].begin();
00862 
00863     // After a syscall squashes everything, the history buffer may be empty
00864     // but the ROB may still be squashing instructions.
00865     if (historyBuffer[tid].empty()) {
00866         return;
00867     }
00868 
00869     // Go through the most recent instructions, undoing the mappings
00870     // they did and freeing up the registers.
00871     while (!historyBuffer[tid].empty() &&
00872            (*hb_it).instSeqNum > squashed_seq_num) {
00873         assert(hb_it != historyBuffer[tid].end());
00874 
00875         DPRINTF(Rename, "[tid:%u]: Removing history entry with sequence "
00876                 "number %i.\n", tid, (*hb_it).instSeqNum);
00877 
00878         // Tell the rename map to set the architected register to the
00879         // previous physical register that it was renamed to.
00880         renameMap[tid]->setEntry(hb_it->archReg, hb_it->prevPhysReg);
00881 
00882         // Put the renamed physical register back on the free list.
00883         freeList->addReg(hb_it->newPhysReg);
00884 
00885         // Be sure to mark its register as ready if it's a misc register.
00886         if (hb_it->newPhysReg >= maxPhysicalRegs) {
00887             scoreboard->setReg(hb_it->newPhysReg);
00888         }
00889 
00890         historyBuffer[tid].erase(hb_it++);
00891 
00892         ++renameUndoneMaps;
00893     }
00894 }

00896 template<class Impl>
00897 void
00898 DefaultRename<Impl>::removeFromHistory(InstSeqNum inst_seq_num, ThreadID tid)
00899 {
00900     DPRINTF(Rename, "[tid:%u]: Removing a committed instruction from the "
00901             "history buffer %u (size=%i), until [sn:%lli].\n",
00902             tid, tid, historyBuffer[tid].size(), inst_seq_num);
00903 
00904     typename std::list<RenameHistory>::iterator hb_it =
00905         historyBuffer[tid].end();
00906 
00907     --hb_it;
00908 
00909     if (historyBuffer[tid].empty()) {
00910         DPRINTF(Rename, "[tid:%u]: History buffer is empty.\n", tid);
00911         return;
00912     } else if (hb_it->instSeqNum > inst_seq_num) {
00913         DPRINTF(Rename, "[tid:%u]: Old sequence number encountered.  Ensure "
00914                 "that a syscall happened recently.\n", tid);
00915         return;
00916     }
00917 
00918     // Commit all the renames up until (and including) the committed sequence
00919     // number. Some or even all of the committed instructions may not have
00920     // rename histories if they did not have destination registers that were
00921     // renamed.
00922     while (!historyBuffer[tid].empty() &&
00923            hb_it != historyBuffer[tid].end() &&
00924            (*hb_it).instSeqNum <= inst_seq_num) {
00925 
00926         DPRINTF(Rename, "[tid:%u]: Freeing up older rename of reg %i, "
00927                 "[sn:%lli].\n",
00928                 tid, (*hb_it).prevPhysReg, (*hb_it).instSeqNum);
00929 
00930         freeList->addReg((*hb_it).prevPhysReg);
00931         ++renameCommittedMaps;
00932 
00933         historyBuffer[tid].erase(hb_it--);
00934     }
00935 }

00937 template <class Impl>
00938 inline void
00939 DefaultRename<Impl>::renameSrcRegs(DynInstPtr &inst, ThreadID tid)
00940 {
00941     assert(renameMap[tid] != 0);
00942 
00943     unsigned num_src_regs = inst->numSrcRegs();
00944 
00945     // Get the architectual register numbers from the source and
00946     // destination operands, and redirect them to the right register.
00947     // Will need to mark dependencies though.
00948     for (int src_idx = 0; src_idx < num_src_regs; src_idx++) {
00949         RegIndex src_reg = inst->srcRegIdx(src_idx);
00950         RegIndex flat_src_reg = src_reg;
00951         if (src_reg < TheISA::FP_Base_DepTag) {
00952             flat_src_reg = inst->tcBase()->flattenIntIndex(src_reg);
00953             DPRINTF(Rename, "Flattening index %d to %d.\n",
00954                     (int)src_reg, (int)flat_src_reg);
00955         } else if (src_reg < TheISA::Ctrl_Base_DepTag) {
00956             src_reg = src_reg - TheISA::FP_Base_DepTag;
00957             flat_src_reg = inst->tcBase()->flattenFloatIndex(src_reg);
00958             DPRINTF(Rename, "Flattening index %d to %d.\n",
00959                     (int)src_reg, (int)flat_src_reg);
00960             flat_src_reg += TheISA::NumIntRegs;
00961         } else if (src_reg < TheISA::Max_DepTag) {
00962             flat_src_reg = src_reg - TheISA::Ctrl_Base_DepTag +
00963                            TheISA::NumFloatRegs + TheISA::NumIntRegs;
00964             DPRINTF(Rename, "Adjusting reg index from %d to %d.\n",
00965                     src_reg, flat_src_reg);
00966         } else {
00967             panic("Reg index is out of bound: %d.", src_reg);
00968         }
00969 
00970         // Look up the source registers to get the phys. register they've
00971         // been renamed to, and set the sources to those registers.
00972         PhysRegIndex renamed_reg = renameMap[tid]->lookup(flat_src_reg);
00973 
00974         DPRINTF(Rename, "[tid:%u]: Looking up arch reg %i, got "
00975                 "physical reg %i.\n", tid, (int)flat_src_reg,
00976                 (int)renamed_reg);
00977 
00978         inst->renameSrcReg(src_idx, renamed_reg);
00979 
00980         // See if the register is ready or not.
00981         if (scoreboard->getReg(renamed_reg) == true) {
00982             DPRINTF(Rename, "[tid:%u]: Register %d is ready.\n",
00983                     tid, renamed_reg);
00984 
00985             inst->markSrcRegReady(src_idx);
00986         } else {
00987             DPRINTF(Rename, "[tid:%u]: Register %d is not ready.\n",
00988                     tid, renamed_reg);
00989         }
00990 
00991         ++renameRenameLookups;
00992         inst->isFloating() ? fpRenameLookups++ : intRenameLookups++;
00993     }
00994 }

00996 template <class Impl>
00997 inline void
00998 DefaultRename<Impl>::renameDestRegs(DynInstPtr &inst, ThreadID tid)
00999 {
01000     typename RenameMap::RenameInfo rename_result;
01001 
01002     unsigned num_dest_regs = inst->numDestRegs();
01003 
01004     // Rename the destination registers.
01005     for (int dest_idx = 0; dest_idx < num_dest_regs; dest_idx++) {
01006         RegIndex dest_reg = inst->destRegIdx(dest_idx);
01007         RegIndex flat_dest_reg = dest_reg;
01008         if (dest_reg < TheISA::FP_Base_DepTag) {
01009             // Integer registers are flattened.
01010             flat_dest_reg = inst->tcBase()->flattenIntIndex(dest_reg);
01011             DPRINTF(Rename, "Flattening index %d to %d.\n",
01012                     (int)dest_reg, (int)flat_dest_reg);
01013         } else if (dest_reg < TheISA::Ctrl_Base_DepTag) {
01014             dest_reg = dest_reg - TheISA::FP_Base_DepTag;
01015             flat_dest_reg = inst->tcBase()->flattenFloatIndex(dest_reg);
01016             DPRINTF(Rename, "Flattening index %d to %d.\n",
01017                     (int)dest_reg, (int)flat_dest_reg);
01018             flat_dest_reg += TheISA::NumIntRegs;
01019         } else if (dest_reg < TheISA::Max_DepTag) {
01020             // Floating point and Miscellaneous registers need their indexes
01021             // adjusted to account for the expanded number of flattened int regs.
01022             flat_dest_reg = dest_reg - TheISA::Ctrl_Base_DepTag +
01023                             TheISA::NumIntRegs + TheISA::NumFloatRegs;
01024             DPRINTF(Rename, "Adjusting reg index from %d to %d.\n",
01025                     dest_reg, flat_dest_reg);
01026         } else {
01027             panic("Reg index is out of bound: %d.", dest_reg);
01028         }
01029 
01030         inst->flattenDestReg(dest_idx, flat_dest_reg);
01031 
01032         // Get the physical register that the destination will be
01033         // renamed to.
01034         rename_result = renameMap[tid]->rename(flat_dest_reg);
01035 
01036         //Mark Scoreboard entry as not ready
01037         if (dest_reg < TheISA::Ctrl_Base_DepTag)
01038             scoreboard->unsetReg(rename_result.first);
01039 
01040         DPRINTF(Rename, "[tid:%u]: Renaming arch reg %i to physical "
01041                 "reg %i.\n", tid, (int)flat_dest_reg,
01042                 (int)rename_result.first);
01043 
01044         // Record the rename information so that a history can be kept.
01045         RenameHistory hb_entry(inst->seqNum, flat_dest_reg,
01046                                rename_result.first,
01047                                rename_result.second);
01048 
01049         historyBuffer[tid].push_front(hb_entry);
01050 
01051         DPRINTF(Rename, "[tid:%u]: Adding instruction to history buffer "
01052                 "(size=%i), [sn:%lli].\n",tid,
01053                 historyBuffer[tid].size(),
01054                 (*historyBuffer[tid].begin()).instSeqNum);
01055 
01056         // Tell the instruction to rename the appropriate destination
01057         // register (dest_idx) to the new physical register
01058         // (rename_result.first), and record the previous physical
01059         // register that the same logical register was renamed to
01060         // (rename_result.second).
01061         inst->renameDestReg(dest_idx,
01062                             rename_result.first,
01063                             rename_result.second);
01064 
01065         ++renameRenamedOperands;
01066     }
01067 }

01167                (!emptyROB[tid] || instsInProgress[tid])) {

01169                 "empty.\n",

01203 template <class Impl>
01204 bool
01205 DefaultRename<Impl>::checkSignalsAndUpdate(ThreadID tid)
01206 {
01207     // Check if there's a squash signal, squash if there is
01208     // Check stall signals, block if necessary.
01209     // If status was blocked
01210     //     check if stall conditions have passed
01211     //         if so then go to unblocking
01212     // If status was Squashing
01213     //     check if squashing is not high.  Switch to running this cycle.
01214     // If status was serialize stall
01215     //     check if ROB is empty and no insts are in flight to the ROB
01216 
01217     readFreeEntries(tid);
01218     readStallSignals(tid);
01219 
01220     if (fromCommit->commitInfo[tid].squash) {
01221         DPRINTF(Rename, "[tid:%u]: Squashing instructions due to squash from "
01222                 "commit.\n", tid);
01223 
01224         squash(fromCommit->commitInfo[tid].doneSeqNum, tid);
01225 
01226         return true;
01227     }
01228 
01229     if (fromCommit->commitInfo[tid].robSquashing) {
01230         DPRINTF(Rename, "[tid:%u]: ROB is still squashing.\n", tid);
01231 
01232         renameStatus[tid] = Squashing;
01233 
01234         return true;
01235     }
01236 
01237     if (checkStall(tid)) {
01238         return block(tid);
01239     }
01240 
01241     if (renameStatus[tid] == Blocked) {
01242         DPRINTF(Rename, "[tid:%u]: Done blocking, switching to unblocking.\n",
01243                 tid);
01244 
01245         renameStatus[tid] = Unblocking;
01246 
01247         unblock(tid);
01248 
01249         return true;
01250     }
01251 
01252     if (renameStatus[tid] == Squashing) {
01253         // Switch status to running if rename isn't being told to block or
01254         // squash this cycle.
01255         if (resumeSerialize) {
01256             DPRINTF(Rename, "[tid:%u]: Done squashing, switching to serialize.\n",
01257                     tid);
01258 
01259             renameStatus[tid] = SerializeStall;
01260             return true;
01261         } else if (resumeUnblocking) {
01262             DPRINTF(Rename, "[tid:%u]: Done squashing, switching to unblocking.\n",
01263                     tid);
01264             renameStatus[tid] = Unblocking;
01265             return true;
01266         } else {
01267             DPRINTF(Rename, "[tid:%u]: Done squashing, switching to running.\n",
01268                     tid);
01269 
01270             renameStatus[tid] = Running;
01271             return false;
01272         }
01273     }
01274 
01275     if (renameStatus[tid] == SerializeStall) {
01276         // Stall ends once the ROB is free.
01277         DPRINTF(Rename, "[tid:%u]: Done with serialize stall, switching to "
01278                 "unblocking.\n", tid);
01279 
01280         DynInstPtr serial_inst = serializeInst[tid];
01281 
01282         renameStatus[tid] = Unblocking;
01283 
01284         unblock(tid);
01285 
01286         DPRINTF(Rename, "[tid:%u]: Processing instruction [%lli] with "
01287                 "PC %s.\n", tid, serial_inst->seqNum, serial_inst->pcState());
01288 
01289         // Put instruction into queue here.
01290         serial_inst->clearSerializeBefore();
01291 
01292         if (!skidBuffer[tid].empty()) {
01293             skidBuffer[tid].push_front(serial_inst);
01294         } else {
01295             insts[tid].push_front(serial_inst);
01296         }
01297 
01298         DPRINTF(Rename, "[tid:%u]: Instruction must be processed by rename."
