00105         for (PhysRegIndex index = numLogicalIntRegs;

00113         PhysRegIndex temp_ireg = ireg_idx;

00137 SimpleRenameMap::RenameInfo
00138 SimpleRenameMap::rename(RegIndex arch_reg)
00139 {
00140     PhysRegIndex renamed_reg;
00141     PhysRegIndex prev_reg;
00142 
00143     if (arch_reg < numLogicalIntRegs) {
00144 
00145         // Record the current physical register that is renamed to the
00146         // requested architected register.
00147         prev_reg = intRenameMap[arch_reg].physical_reg;
00148 
00149         // If it's not referencing the zero register, then rename the
00150         // register.
00151         if (arch_reg != intZeroReg) {
00152             renamed_reg = freeList->getIntReg();
00153 
00154             intRenameMap[arch_reg].physical_reg = renamed_reg;
00155 
00156             assert(renamed_reg >= 0 && renamed_reg < numPhysicalIntRegs);
00157 
00158         } else {
00159             // Otherwise return the zero register so nothing bad happens.
00160             renamed_reg = intZeroReg;
00161             prev_reg = intZeroReg;
00162         }
00163     } else if (arch_reg < numLogicalRegs) {
00164         // Record the current physical register that is renamed to the
00165         // requested architected register.
00166         prev_reg = floatRenameMap[arch_reg].physical_reg;
00167 
00168         // If it's not referencing the zero register, then rename the
00169         // register.
00170 #if THE_ISA == ALPHA_ISA
00171         if (arch_reg != floatZeroReg) {
00172 #endif
00173             renamed_reg = freeList->getFloatReg();
00174 
00175             floatRenameMap[arch_reg].physical_reg = renamed_reg;
00176 
00177             assert(renamed_reg < numPhysicalRegs &&
00178                    renamed_reg >= numPhysicalIntRegs);
00179 #if THE_ISA == ALPHA_ISA
00180         } else {
00181             // Otherwise return the zero register so nothing bad happens.
00182             renamed_reg = floatZeroReg;
00183         }
00184 #endif
00185     } else {
00186         // Subtract off the base offset for miscellaneous registers.
00187         arch_reg = arch_reg - numLogicalRegs;
00188 
00189         DPRINTF(Rename, "Renamed misc reg %d\n", arch_reg);
00190 
00191         // No renaming happens to the misc. registers.  They are
00192         // simply the registers that come after all the physical
00193         // registers; thus take the base architected register and add
00194         // the physical registers to it.
00195         renamed_reg = arch_reg + numPhysicalRegs;
00196 
00197         // Set the previous register to the same register; mainly it must be
00198         // known that the prev reg was outside the range of normal registers
00199         // so the free list can avoid adding it.
00200         prev_reg = renamed_reg;
00201     }
00202 
00203     DPRINTF(Rename, "Renamed reg %d to physical reg %d old mapping was %d\n",
00204             arch_reg, renamed_reg, prev_reg);
00205 
00206     return RenameInfo(renamed_reg, prev_reg);
00207 }

00209 PhysRegIndex
00210 SimpleRenameMap::lookup(RegIndex arch_reg)
00211 {
00212     if (arch_reg < numLogicalIntRegs) {
00213         return intRenameMap[arch_reg].physical_reg;
00214     } else if (arch_reg < numLogicalRegs) {
00215         return floatRenameMap[arch_reg].physical_reg;
00216     } else {
00217         // Subtract off the misc registers offset.
00218         arch_reg = arch_reg - numLogicalRegs;
00219 
00220         // Misc. regs don't rename, so simply add the base arch reg to
00221         // the number of physical registers.
00222         return numPhysicalRegs + arch_reg;
00223     }
00224 }

00226 void
00227 SimpleRenameMap::setEntry(RegIndex arch_reg, PhysRegIndex renamed_reg)
00228 {
00229     // In this implementation the miscellaneous registers do not
00230     // actually rename, so this function does not allow you to try to
00231     // change their mappings.
00232     if (arch_reg < numLogicalIntRegs) {
00233         DPRINTF(Rename, "Rename Map: Integer register %i being set to %i.\n",
00234                 (int)arch_reg, renamed_reg);
00235 
00236         intRenameMap[arch_reg].physical_reg = renamed_reg;
00237     } else if (arch_reg < numLogicalIntRegs + numLogicalFloatRegs) {
00238         DPRINTF(Rename, "Rename Map: Float register %i being set to %i.\n",
00239                 (int)arch_reg - numLogicalIntRegs, renamed_reg);
00240 
00241         floatRenameMap[arch_reg].physical_reg = renamed_reg;
00242     }
