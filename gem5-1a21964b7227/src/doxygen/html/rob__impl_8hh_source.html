00065     std::transform(policy.begin(), policy.end(), policy.begin(),
00066                    (int(*)(int)) tolower);
00067 
00068     //Figure out rob policy
00069     if (policy == "dynamic") {
00070         robPolicy = Dynamic;
00071 
00072         //Set Max Entries to Total ROB Capacity
00073         for (ThreadID tid = 0; tid < numThreads; tid++) {
00074             maxEntries[tid] = numEntries;

00076 
00077     } else if (policy == "partitioned") {
00078         robPolicy = Partitioned;
00079         DPRINTF(Fetch, "ROB sharing policy set to Partitioned\n");
00080 
00081         //@todo:make work if part_amt doesnt divide evenly.
00082         int part_amt = numEntries / numThreads;
00083 
00084         //Divide ROB up evenly
00085         for (ThreadID tid = 0; tid < numThreads; tid++) {
00086             maxEntries[tid] = part_amt;
00087         }
00088 
00089     } else if (policy == "threshold") {
00090         robPolicy = Threshold;
00091         DPRINTF(Fetch, "ROB sharing policy set to Threshold\n");
00092 
00093         int threshold =  _smtROBThreshold;;
00094 
00095         //Divide up by threshold amount
00096         for (ThreadID tid = 0; tid < numThreads; tid++) {
00097             maxEntries[tid] = threshold;
00098         }
00099     } else {
00100         assert(0 && "Invalid ROB Sharing Policy.Options Are:{Dynamic,"
00101                     "Partitioned, Threshold}");

00328 ROB<Impl>::doSquash(ThreadID tid)
00329 {
00330     robWrites++;
00331     DPRINTF(ROB, "[tid:%u]: Squashing instructions until [sn:%i].\n",
00332             tid, squashedSeqNum[tid]);
00333 
00334     assert(squashIt[tid] != instList[tid].end());
00335 
00336     if ((*squashIt[tid])->seqNum < squashedSeqNum[tid]) {
00337         DPRINTF(ROB, "[tid:%u]: Done squashing instructions.\n",
00338                 tid);
00339 
00340         squashIt[tid] = instList[tid].end();
00341 
00342         doneSquashing[tid] = true;
00343         return;
00344     }
00345 
00346     bool robTailUpdate = false;
00347 
00348     for (int numSquashed = 0;
00349          numSquashed < squashWidth &&
00350          squashIt[tid] != instList[tid].end() &&
00351          (*squashIt[tid])->seqNum > squashedSeqNum[tid];
00352          ++numSquashed)
00353     {
00354         DPRINTF(ROB, "[tid:%u]: Squashing instruction PC %s, seq num %i.\n",
00355                 (*squashIt[tid])->threadNumber,
00356                 (*squashIt[tid])->pcState(),
00357                 (*squashIt[tid])->seqNum);
00358 
00359         // Mark the instruction as squashed, and ready to commit so that
00360         // it can drain out of the pipeline.
00361         (*squashIt[tid])->setSquashed();
00362 
00363         (*squashIt[tid])->setCanCommit();
00364 
00365 
00366         if (squashIt[tid] == instList[tid].begin()) {
00367             DPRINTF(ROB, "Reached head of instruction list while "
00368                     "squashing.\n");
00369 
00370             squashIt[tid] = instList[tid].end();
00371 
00372             doneSquashing[tid] = true;
00373 
00374             return;
00375         }
00376 
00377         InstIt tail_thread = instList[tid].end();
00378         tail_thread--;
00379 
00380         if ((*squashIt[tid]) == (*tail_thread))
00381             robTailUpdate = true;
00382 
00383         squashIt[tid]--;
00384     }
00385 
00386 
00387     // Check if ROB is done squashing.
00388     if ((*squashIt[tid])->seqNum <= squashedSeqNum[tid]) {
00389         DPRINTF(ROB, "[tid:%u]: Done squashing instructions.\n",
00390                 tid);
00391 
00392         squashIt[tid] = instList[tid].end();
00393 
00394         doneSquashing[tid] = true;
00395     }
00396 
00397     if (robTailUpdate) {
00398         updateTail();
00399     }
