00126 PL031::write(PacketPtr pkt)
00127 {
00128     assert(pkt->getAddr() >= pioAddr && pkt->getAddr() < pioAddr + pioSize);
00129     assert(pkt->getSize() == 4);
00130     Addr daddr = pkt->getAddr() - pioAddr;

00133 
00134     switch (daddr) {
00135       case DataReg:
00136         break;
00137       case MatchReg:
00138         matchVal = pkt->get<uint32_t>();
00139         resyncMatch();
00140         break;
00141       case LoadReg:
00142         lastWrittenTick = curTick();
00143         timeVal = pkt->get<uint32_t>();
00144         loadVal = timeVal;
00145         resyncMatch();
00146         break;
00147       case ControlReg:
00148         break; // Can't stop when started
00149       case IntMask:
00150         maskInt = pkt->get<uint32_t>();
00151         break;
00152       case IntClear:
00153         if (pkt->get<uint32_t>()) {
00154             rawInt = false;
00155             pendingInt = false;
00156         }
00157         break;
00158       default:
00159         if (AmbaDev::readId(pkt, ambaId, pioAddr))
00160             break;
00161         panic("Tried to read PL031 at offset %#x that doesn't exist\n", daddr);
00162         break;
00163     }
00164 
00165     pkt->makeAtomicResponse();

00169 void
00170 PL031::resyncMatch()
00171 {
00172     DPRINTF(Timer, "Setting up new match event match=%d time=%d\n", matchVal,
00173             timeVal);
00174 
00175     uint32_t seconds_until = matchVal - timeVal;
00176     Tick ticks_until = SimClock::Int::s * seconds_until;
00177 
00178     if (matchEvent.scheduled()) {
00179         DPRINTF(Timer, "-- Event was already schedule, de-scheduling\n");
00180         deschedule(matchEvent);
00181     }
00182     schedule(matchEvent, curTick() + ticks_until);
00183     DPRINTF(Timer, "-- Scheduling new event for: %d\n", curTick() + ticks_until);
