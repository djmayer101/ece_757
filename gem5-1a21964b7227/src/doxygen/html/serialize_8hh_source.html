00183 //
00184 // A SerializableBuilder serves as an evaluation context for a set of
00185 // parameters that describe a specific instance of a Serializable.  This
00186 // evaluation context corresponds to a section in the .ini file (as
00187 // with the base ParamContext) plus an optional node in the
00188 // configuration hierarchy (the configNode member) for resolving
00189 // Serializable references.  SerializableBuilder is an abstract superclass;
00190 // derived classes specialize the class for particular subclasses of
00191 // Serializable (e.g., BaseCache).
00192 //
00193 // For typical usage, see the definition of
00194 // SerializableClass::createObject().
00195 //
00196 class SerializableBuilder
00197 {
00198   public:
00199 
00200     SerializableBuilder() {}
00201 
00202     virtual ~SerializableBuilder() {}
00203 
00204     // Create the actual Serializable corresponding to the parameter
00205     // values in this context.  This function is overridden in derived
00206     // classes to call a specific constructor for a particular
00207     // subclass of Serializable.
00208     virtual Serializable *create() = 0;
00209 };

00211 //
00212 // An instance of SerializableClass corresponds to a class derived from
00213 // Serializable.  The SerializableClass instance serves to bind the string
00214 // name (found in the config file) to a function that creates an
00215 // instance of the appropriate derived class.
00216 //
00217 // This would be much cleaner in Smalltalk or Objective-C, where types
00218 // are first-class objects themselves.
00219 //
00220 class SerializableClass
00221 {
00222   public:
00223 
00224     // Type CreateFunc is a pointer to a function that creates a new
00225     // simulation object builder based on a .ini-file parameter
00226     // section (specified by the first string argument), a unique name
00227     // for the object (specified by the second string argument), and
00228     // an optional config hierarchy node (specified by the third
00229     // argument).  A pointer to the new SerializableBuilder is returned.
00230     typedef Serializable *(*CreateFunc)(Checkpoint *cp,
00231                                         const std::string &section);
00232 
00233     static std::map<std::string,CreateFunc> *classMap;
00234 
00235     // Constructor.  For example:
00236     //
00237     // SerializableClass baseCacheSerializableClass("BaseCacheSerializable",
00238     //                         newBaseCacheSerializableBuilder);
00239     //
00240     SerializableClass(const std::string &className, CreateFunc createFunc);
00241 
00242     // create Serializable given name of class and pointer to
00243     // configuration hierarchy node
00244     static Serializable *createObject(Checkpoint *cp,
00245                                       const std::string &section);
