00205     int id;

00215         if (threadContexts.size() <= assigned)

00221         fatal("Cannot have two CPUs with the same id (%d)\n", id);

00232         if (rgdb_wait != -1 && rgdb_wait == id)
00233             gdbl->accept();

00235         if (remoteGDB.size() <= id) {
00236             remoteGDB.resize(id + 1);
00237         }
00238

00263             TheISA::startupCPU(threadContexts[i], i);
00264         // Moved from the constructor to here since it relies on the

00421 MasterID
00422 System::getMasterId(std::string master_name)
00423 {
00424     // strip off system name if the string starts with it
00425     if (startswith(master_name, name()))
00426         master_name = master_name.erase(0, name().size() + 1);
00427 
00428     // CPUs in switch_cpus ask for ids again after switching
00429     for (int i = 0; i < masterIds.size(); i++) {
00430         if (masterIds[i] == master_name) {
00431             return i;
00432         }
00433     }
00434 
00435     // Verify that the statistics haven't been enabled yet
00436     // Otherwise objects will have sized their stat buckets and
00437     // they will be too small
00438 
00439     if (Stats::enabled())
00440         fatal("Can't request a masterId after regStats(). \
00441                 You must do so in init().\n");
00442 
00443     masterIds.push_back(master_name);
00444 
00445     return masterIds.size() - 1;
00446 }
