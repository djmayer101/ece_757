00048 //
00049 // handle termination event
00050 //
00051 void
00052 SimLoopExitEvent::process()
00053 {
00054     // if this got scheduled on a different queue (e.g. the committed
00055     // instruction queue) then make a corresponding event on the main
00056     // queue.
00057     if (!isFlagSet(IsMainQueue)) {
00058         exitSimLoop(cause, code);
00059         setFlags(AutoDelete);
00060     }
00061 
00062     // otherwise do nothing... the IsExitEvent flag takes care of
00063     // exiting the simulation loop and returning this object to Python
00064 
00065     // but if you are doing this on intervals, don't forget to make another
00066     if (repeat) {
00067         assert(isFlagSet(IsMainQueue));
00068         mainEventQueue.schedule(this, curTick() + repeat);
00069     }
00070 }
