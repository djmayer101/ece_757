00094 bool
00095 SimpleMemory::recvTimingReq(PacketPtr pkt)
00096 {
00099     for (int x = 0; x < pendingDelete.size(); x++)
00100         delete pendingDelete[x];
00101     pendingDelete.clear();
00102 
00103     if (pkt->memInhibitAsserted()) {
00104         // snooper will supply based on copy of packet
00105         // still target's responsibility to delete packet
00106         pendingDelete.push_back(pkt);
00107         return true;
00108     }
00109 
00110     // we should never get a new request after committing to retry the
00111     // current one, the bus violates the rule as it simply sends a
00112     // retry to the next one waiting on the retry list, so simply
00113     // ignore it
00114     if (retryReq)
00115         return false;
00116 
00117     // if we are busy with a read or write, remember that we have to
00118     // retry
00119     if (isBusy) {
00120         retryReq = true;
00121         return false;
00122     }
00123 
00124     // @todo someone should pay for this
00125     pkt->busFirstWordDelay = pkt->busLastWordDelay = 0;
00126 
00127     // update the release time according to the bandwidth limit, and
00128     // do so with respect to the time it takes to finish this request
00129     // rather than long term as it is the short term data rate that is
00130     // limited for any real memory
00131 
00132     // only look at reads and writes when determining if we are busy,
00133     // and for how long, as it is not clear what to regulate for the
00134     // other types of commands
00135     if (pkt->isRead() || pkt->isWrite()) {
00136         // calculate an appropriate tick to release to not exceed
00137         // the bandwidth limit
00138         Tick duration = pkt->getSize() * bandwidth;
00139 
00140         // only consider ourselves busy if there is any need to wait
00141         // to avoid extra events being scheduled for (infinitely) fast
00142         // memories
00143         if (duration != 0) {
00144             schedule(releaseEvent, curTick() + duration);
00145             isBusy = true;
00146         }
00147     }
00148 
00149     // go ahead and deal with the packet and put the response in the
00150     // queue if there is one
00151     bool needsResponse = pkt->needsResponse();
00152     Tick latency = doAtomicAccess(pkt);
00153     // turn packet around to go back to requester if response expected
00154     if (needsResponse) {
00155         // doAtomicAccess() should already have turned packet into
00156         // atomic response
00157         assert(pkt->isResponse());
00158         port.schedTimingResp(pkt, curTick() + latency);
00159     } else {
00160         pendingDelete.push_back(pkt);
00161     }
00162 
00163     return true;
00164 }
