00037 #define __SINIC_VAL32(NAME, OFFSET, WIDTH) \
00038         static const uint32_t NAME##_width = WIDTH; \
00039         static const uint32_t NAME##_offset = OFFSET; \
00040         static const uint32_t NAME##_mask = (1 << WIDTH) - 1; \
00041         static const uint32_t NAME = ((1 << WIDTH) - 1) << OFFSET; \
00042         static inline uint32_t get_##NAME(uint32_t reg) \
00043         { return (reg & NAME) >> OFFSET; } \
00044         static inline uint32_t set_##NAME(uint32_t reg, uint32_t val) \
00045         { return (reg & ~NAME) | ((val << OFFSET) & NAME); }
