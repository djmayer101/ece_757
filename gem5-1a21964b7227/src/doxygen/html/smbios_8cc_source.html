00064 template <class T>
00065 uint64_t
00066 composeBitVector(T vec)
00067 {
00068     uint64_t val = 0;
00069     typename T::iterator vecIt;
00070     for (vecIt = vec.begin(); vecIt != vec.end(); vecIt++) {
00071         val |= (1 << (*vecIt));
00072     }
00073     return val;
00074 }
