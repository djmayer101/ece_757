00065     moreBytes(const PCState &pc, Addr fetchPC, MachInst inst)
00066     {
00067         emi = inst;
00068         // The I bit, bit 13, is used to figure out where the ASI
00069         // should come from. Use that in the ExtMachInst. This is
00070         // slightly redundant, but it removes the need to put a condition
00071         // into all the execute functions
00072         if (inst & (1 << 13)) {
00073             emi |= (static_cast<ExtMachInst>(
00074                         asi << (sizeof(MachInst) * 8)));
00075         } else {
00076             emi |= (static_cast<ExtMachInst>(bits(inst, 12, 5))
00077                     << (sizeof(MachInst) * 8));
00078         }
00079         instDone = true;
00080     }
