00078 void
00079 ISA::reloadRegMap()
00080 {
00081     installGlobals(gl, CurrentGlobalsOffset);
00082     installWindow(cwp, CurrentWindowOffset);
00083     // Microcode registers.
00084     for (int i = 0; i < NumMicroIntRegs; i++)
00085         intRegMap[MicroIntOffset + i] = i + TotalGlobals + NWindows * 16;
00086     installGlobals(gl, NextGlobalsOffset);
00087     installWindow(cwp - 1, NextWindowOffset);
00088     installGlobals(gl, PreviousGlobalsOffset);
00089     installWindow(cwp + 1, PreviousWindowOffset);
00090 }

00093 ISA::installWindow(int cwp, int offset)
00094 {
00095     assert(offset >= 0 && offset + NumWindowedRegs <= NumIntRegs);
00096     RegIndex *mapChunk = intRegMap + offset;
00097     for (int i = 0; i < NumWindowedRegs; i++)
00098         mapChunk[i] = TotalGlobals +
00099             ((i - cwp * RegsPerWindow + TotalWindowed) % (TotalWindowed));
