00040 // The caller uses %o0-%05 for the first 6 arguments even if their floating
00041 // point. Double precision floating point values take two registers/args.
00042 // Quads, structs, and unions are passed as pointers. All arguments beyond
00043 // the sixth are passed on the stack past the 16 word window save area,
00044 // space for the struct/union return pointer, and space reserved for the
00045 // first 6 arguments which the caller may use but doesn't have to.
00046 uint64_t
00047 getArgument(ThreadContext *tc, int &number, uint16_t size, bool fp)
00048 {
00049     if (!FullSystem) {
00050         panic("getArgument() only implemented for full system\n");
00051         M5_DUMMY_RETURN
00052     }
00053 
00054     const int NumArgumentRegs = 6;
00055     if (number < NumArgumentRegs) {
00056         return tc->readIntReg(8 + number);
00057     } else {
00058         Addr sp = tc->readIntReg(StackPointerReg);
00059         FSTranslatingPortProxy &vp = tc->getVirtProxy();
00060         uint64_t arg = vp.read<uint64_t>(sp + 92 +
00061                             (number-NumArgumentRegs) * sizeof(uint64_t));
00062         return arg;
00063     }
00064 }

00202 void
00203 copyRegs(ThreadContext *src, ThreadContext *dest)
00204 {
00205     // First loop through the integer registers.
00206     int old_gl = src->readMiscRegNoEffect(MISCREG_GL);
00207     int old_cwp = src->readMiscRegNoEffect(MISCREG_CWP);
00208     // Globals
00209     for (int x = 0; x < MaxGL; ++x) {
00210         src->setMiscReg(MISCREG_GL, x);
00211         dest->setMiscReg(MISCREG_GL, x);
00212         // Skip %g0 which is always zero.
00213         for (int y = 1; y < 8; y++)
00214             dest->setIntReg(y, src->readIntReg(y));
00215     }
00216     // Locals and ins. Outs are all also ins.
00217     for (int x = 0; x < NWindows; ++x) {
00218          src->setMiscReg(MISCREG_CWP, x);
00219          dest->setMiscReg(MISCREG_CWP, x);
00220          for (int y = 16; y < 32; y++)
00221              dest->setIntReg(y, src->readIntReg(y));
00222     }
00223     // Microcode reg and pseudo int regs (misc regs in the integer regfile).
00224     for (int y = NumIntArchRegs; y < NumIntArchRegs + NumMicroIntRegs; ++y)
00225         dest->setIntReg(y, src->readIntReg(y));
00226 
00227     // Restore src's GL, CWP
00228     src->setMiscReg(MISCREG_GL, old_gl);
00229     src->setMiscReg(MISCREG_CWP, old_cwp);
00230 
00231 
00232     // Then loop through the floating point registers.
00233     for (int i = 0; i < SparcISA::NumFloatArchRegs; ++i) {
00234         dest->setFloatRegBits(i, src->readFloatRegBits(i));
00235     }
00236 
00237     // Copy misc. registers
00238     copyMiscRegs(src, dest);
00239 
00240     // Lastly copy PC/NPC
00241     dest->pcState(src->pcState());
00242 }
