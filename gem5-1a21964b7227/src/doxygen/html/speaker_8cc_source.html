00055 Tick
00056 X86ISA::Speaker::write(PacketPtr pkt)
00057 {
00058     assert(pkt->getAddr() == pioAddr);
00059     assert(pkt->getSize() == 1);
00060     SpeakerControl val = pkt->get<uint8_t>();
00061     controlVal.gate = val.gate;
00062     //Change the gate value in the timer.
00063     if (!val.gate)
00064         warn("The gate bit of the pc speaker isn't implemented and "
00065                 "is always on.\n");
00066     //This would control whether the timer output is hooked up to a physical
00067     //speaker. Since M5 can't make noise, it's value doesn't actually do
00068     //anything.
00069     controlVal.speaker = val.speaker;
00070     DPRINTF(PcSpeaker, "Writing to speaker device: gate %s, speaker %s.\n",
00071             controlVal.gate ? "on" : "off", controlVal.speaker ? "on" : "off");
00072     pkt->makeAtomicResponse();
00073     return latency;
00074 }
