00255 void
00256 periodicStatDump(Tick period)
00257 {
00258     /*
00259      * If the period is set to 0, then we do not want to dump periodically,
00260      * thus we deschedule the event. Else, if the period is not 0, but the event
00261      * has already been scheduled, we need to get rid of the old event before we
00262      * create a new one, as the old event will no longer be moved forward in the
00263      * event that we resume from a checkpoint.
00264      */
00265     if (dumpEvent != NULL && (period == 0 || dumpEvent->scheduled())) {
00266         // Event should AutoDelete, so we do not need to free it.
00267         mainEventQueue.deschedule(dumpEvent);
00268     }
00269 
00270     /*
00271      * If the period is not 0, we schedule the event. If this is called with a
00272      * period that is less than the current tick, then we shift the first dump
00273      * by curTick. This ensures that we do not schedule the event is the past.
00274      */
00275     if (period != 0) {
00276         // Schedule the event
00277         if (period >= curTick()) {
00278             schedStatEvent(true, true, (Tick)period, (Tick)period);
00279         } else {
00280             schedStatEvent(true, true, (Tick)period + curTick(), (Tick)period);
00281         }
00282     }
00283 }

00285 void
00286 updateEvents()
00287 {
00288     /*
00289      * If the dumpEvent has been scheduled, but is scheduled in the past, then
00290      * we need to shift the event to be at a valid point in time. Therefore, we
00291      * shift the event by curTick.
00292      */
00293     if (dumpEvent != NULL &&
00294         (dumpEvent->scheduled() && dumpEvent->when() < curTick())) {
00295         // shift by curTick() and reschedule
00296         Tick _when = dumpEvent->when();
00297         mainEventQueue.reschedule(dumpEvent, _when + curTick());
00298     }
00299 }
