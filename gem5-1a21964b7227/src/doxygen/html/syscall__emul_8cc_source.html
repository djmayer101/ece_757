00153 brkFunc(SyscallDesc *desc, int num, LiveProcess *p, ThreadContext *tc)
00154 {
00155     // change brk addr to first arg
00156     int index = 0;
00157     Addr new_brk = p->getSyscallArg(tc, index);
00158 
00159     // in Linux at least, brk(0) returns the current break value
00160     // (note that the syscall and the glibc function have different behavior)
00161     if (new_brk == 0)
00162         return p->brk_point;
00163 
00164     if (new_brk > p->brk_point) {
00165         // might need to allocate some new pages
00166         for (ChunkGenerator gen(p->brk_point, new_brk - p->brk_point,
00167                                 VMPageSize); !gen.done(); gen.next()) {
00168             if (!p->pTable->translate(gen.addr()))
00169                 p->allocateMem(roundDown(gen.addr(), VMPageSize), VMPageSize);
00170 
00171             // if the address is already there, zero it out
00172             else {
00173                 uint8_t zero  = 0;
00174                 SETranslatingPortProxy &tp = tc->getMemProxy();
00175 
00176                 // split non-page aligned accesses
00177                 Addr next_page = roundUp(gen.addr(), VMPageSize);
00178                 uint32_t size_needed = next_page - gen.addr();
00179                 tp.memsetBlob(gen.addr(), zero, size_needed);
00180                 if (gen.addr() + VMPageSize > next_page &&
00181                     next_page < new_brk &&
00182                     p->pTable->translate(next_page))
00183                 {
00184                     size_needed = VMPageSize - size_needed;
00185                     tp.memsetBlob(next_page, zero, size_needed);
00186                 }
00187             }
00188         }
00189     }
00190 
00191     p->brk_point = new_brk;
00192     DPRINTF(SyscallVerbose, "Break Point changed to: %#X\n", p->brk_point);
00193     return p->brk_point;

00773 cloneFunc(SyscallDesc *desc, int callnum, LiveProcess *process,
00774            ThreadContext *tc)
00775 {
00776     int index = 0;
00777     IntReg flags = process->getSyscallArg(tc, index);
00778     IntReg newStack = process->getSyscallArg(tc, index);
00779 
00780     DPRINTF(SyscallVerbose, "In sys_clone:\n");
00781     DPRINTF(SyscallVerbose, " Flags=%llx\n", flags);
00782     DPRINTF(SyscallVerbose, " Child stack=%llx\n", newStack);
00783 
00784 
00785     if (flags != 0x10f00) {
00786         warn("This sys_clone implementation assumes flags "
00787              "CLONE_VM|CLONE_FS|CLONE_FILES|CLONE_SIGHAND|CLONE_THREAD "
00788              "(0x10f00), and may not work correctly with given flags "
00789              "0x%llx\n", flags);
00790     }
00791 
00792     ThreadContext* ctc; // child thread context
00793     if ( ( ctc = process->findFreeContext() ) != NULL ) {
00794         DPRINTF(SyscallVerbose, " Found unallocated thread context\n");
00795 
00796         ctc->clearArchRegs();
00797 
00798         // Arch-specific cloning code
00799         #if THE_ISA == ALPHA_ISA or THE_ISA == X86_ISA
00800             // Cloning the misc. regs for these archs is enough
00801             TheISA::copyMiscRegs(tc, ctc);
00802         #elif THE_ISA == SPARC_ISA
00803             TheISA::copyRegs(tc, ctc);
00804 
00805             // TODO: Explain what this code actually does :-)
00806             ctc->setIntReg(NumIntArchRegs + 6, 0);
00807             ctc->setIntReg(NumIntArchRegs + 4, 0);
00808             ctc->setIntReg(NumIntArchRegs + 3, NWindows - 2);
00809             ctc->setIntReg(NumIntArchRegs + 5, NWindows);
00810             ctc->setMiscReg(MISCREG_CWP, 0);
00811             ctc->setIntReg(NumIntArchRegs + 7, 0);
00812             ctc->setMiscRegNoEffect(MISCREG_TL, 0);
00813             ctc->setMiscReg(MISCREG_ASI, ASI_PRIMARY);
00814 
00815             for (int y = 8; y < 32; y++)
00816                 ctc->setIntReg(y, tc->readIntReg(y));
00817         #elif THE_ISA == ARM_ISA
00818             TheISA::copyRegs(tc, ctc);
00819         #else
00820             fatal("sys_clone is not implemented for this ISA\n");
00821         #endif
00822 
00823         // Set up stack register
00824         ctc->setIntReg(TheISA::StackPointerReg, newStack);
00825 
00826         // Set up syscall return values in parent and child
00827         ctc->setIntReg(ReturnValueReg, 0); // return value, child
00828 
00829         // Alpha needs SyscallSuccessReg=0 in child
00830         #if THE_ISA == ALPHA_ISA
00831             ctc->setIntReg(TheISA::SyscallSuccessReg, 0);
00832         #endif
00833 
00834         // In SPARC/Linux, clone returns 0 on pseudo-return register if
00835         // parent, non-zero if child
00836         #if THE_ISA == SPARC_ISA
00837             tc->setIntReg(TheISA::SyscallPseudoReturnReg, 0);
00838             ctc->setIntReg(TheISA::SyscallPseudoReturnReg, 1);
00839         #endif
00840 
00841         ctc->pcState(tc->nextInstAddr());
00842 
00843         ctc->activate();
00844 
00845         // Should return nonzero child TID in parent's syscall return register,
00846         // but for our pthread library any non-zero value will work
00847         return 1;
00848     } else {
00849         fatal("Called sys_clone, but no unallocated thread contexts found!\n");
00850         return 0;
00851     }
00852 }
