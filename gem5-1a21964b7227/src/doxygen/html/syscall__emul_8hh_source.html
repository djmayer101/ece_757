00112     {
00113         bufPtr = new uint8_t[size];
00114         // clear out buffer: in case we only partially populate this,
00115         // and then do a copyOut(), we want to make sure we don't
00116         // introduce any random junk into the simulated address space
00117         memset(bufPtr, 0, size);
00118     }
00119 
00120     virtual ~BaseBufferArg() { delete [] bufPtr; }

00122     //
00123     // copy data into simulator space (read from target memory)
00124     //
00125     virtual bool copyIn(SETranslatingPortProxy &memproxy)
00126     {
00127         memproxy.readBlob(addr, bufPtr, size);
00128         return true;    // no EFAULT detection for now
00129     }
00130 
00131     //
00132     // copy data out of simulator space (write to target memory)
00133     //
00134     virtual bool copyOut(SETranslatingPortProxy &memproxy)
00135     {
00136         memproxy.writeBlob(addr, bufPtr, size);
00137         return true;    // no EFAULT detection for now
00138     }
00139

00141     Addr addr;
00142     int size;
00143     uint8_t *bufPtr;

00345 template <class OS>
00346 SyscallReturn
00347 futexFunc(SyscallDesc *desc, int callnum, LiveProcess *process,
00348           ThreadContext *tc)
00349 {
00350     int index_uaddr = 0;
00351     int index_op = 1;
00352     int index_val = 2;
00353     int index_timeout = 3;
00354 
00355     uint64_t uaddr = process->getSyscallArg(tc, index_uaddr);
00356     int op = process->getSyscallArg(tc, index_op);
00357     int val = process->getSyscallArg(tc, index_val);
00358     uint64_t timeout = process->getSyscallArg(tc, index_timeout);
00359 
00360     std::map<uint64_t, std::list<ThreadContext *> * >
00361         &futex_map = tc->getSystemPtr()->futexMap;
00362 
00363     DPRINTF(SyscallVerbose, "In sys_futex: Address=%llx, op=%d, val=%d\n",
00364             uaddr, op, val);
00365 
00366     op &= ~OS::TGT_FUTEX_PRIVATE_FLAG;
00367 
00368     if (op == OS::TGT_FUTEX_WAIT) {
00369         if (timeout != 0) {
00370             warn("sys_futex: FUTEX_WAIT with non-null timeout unimplemented;"
00371                  "we'll wait indefinitely");
00372         }
00373 
00374         uint8_t *buf = new uint8_t[sizeof(int)];
00375         tc->getMemProxy().readBlob((Addr)uaddr, buf, (int)sizeof(int));
00376         int mem_val = *((int *)buf);
00377         delete buf;
00378 
00379         if(val != mem_val) {
00380             DPRINTF(SyscallVerbose, "sys_futex: FUTEX_WAKE, read: %d, "
00381                                     "expected: %d\n", mem_val, val);
00382             return -OS::TGT_EWOULDBLOCK;
00383         }
00384 
00385         // Queue the thread context
00386         std::list<ThreadContext *> * tcWaitList;
00387         if (futex_map.count(uaddr)) {
00388             tcWaitList = futex_map.find(uaddr)->second;
00389         } else {
00390             tcWaitList = new std::list<ThreadContext *>();
00391             futex_map.insert(std::pair< uint64_t,
00392                             std::list<ThreadContext *> * >(uaddr, tcWaitList));
00393         }
00394         tcWaitList->push_back(tc);
00395         DPRINTF(SyscallVerbose, "sys_futex: FUTEX_WAIT, suspending calling "
00396                                 "thread context\n");
00397         tc->suspend();
00398         return 0;
00399     } else if (op == OS::TGT_FUTEX_WAKE){
00400         int wokenUp = 0;
00401         std::list<ThreadContext *> * tcWaitList;
00402         if (futex_map.count(uaddr)) {
00403             tcWaitList = futex_map.find(uaddr)->second;
00404             while (tcWaitList->size() > 0 && wokenUp < val) {
00405                 tcWaitList->front()->activate();
00406                 tcWaitList->pop_front();
00407                 wokenUp++;
00408             }
00409             if(tcWaitList->empty()) {
00410                 futex_map.erase(uaddr);
00411                 delete tcWaitList;
00412             }
00413         }
00414         DPRINTF(SyscallVerbose, "sys_futex: FUTEX_WAKE, activated %d waiting "
00415                                 "thread contexts\n", wokenUp);
00416         return wokenUp;
00417     } else {
00418         warn("sys_futex: op %d is not implemented, just returning...", op);
00419         return 0;
00420     }
00421 
00422 }

00607          ThreadContext *tc)
00608 {

00610 
00611     int index = 0;
00612     if (!tc->getMemProxy().tryReadString(path,
00613                 process->getSyscallArg(tc, index)))
00614         return -EFAULT;
00615 
00616     if (path == "/dev/sysdev0") {
00617         // This is a memory-mapped high-resolution timer device on Alpha.
00618         // We don't support it, so just punt.
00619         warn("Ignoring open(%s, ...)\n", path);
00620         return -ENOENT;
00621     }
00622 
00623     int tgtFlags = process->getSyscallArg(tc, index);
00624     int mode = process->getSyscallArg(tc, index);
00625     int hostFlags = 0;
00626 
00627     // translate open flags
00628     for (int i = 0; i < OS::NUM_OPEN_FLAGS; i++) {
00629         if (tgtFlags & OS::openFlagTable[i].tgtFlag) {
00630             tgtFlags &= ~OS::openFlagTable[i].tgtFlag;
00631             hostFlags |= OS::openFlagTable[i].hostFlag;
00632         }
00633     }
00634 
00635     // any target flags left?
00636     if (tgtFlags != 0)
00637         warn("Syscall: open: cannot decode flags 0x%x", tgtFlags);
00638 
00639 #ifdef __CYGWIN32__
00640     hostFlags |= O_BINARY;
00641 #endif
00642 
00643     // Adjust path for current working directory
00644     path = process->fullPath(path);
00645 
00646     DPRINTF(SyscallVerbose, "opening file %s\n", path.c_str());
00647 
00648     int fd;
00649     int local_errno;
00650     if (startswith(path, "/proc/") || startswith(path, "/system/") ||
00651         startswith(path, "/platform/") || startswith(path, "/sys/")) {
00652         // It's a proc/sys entry and requires special handling
00653         fd = OS::openSpecialFile(path, process, tc);
00654         local_errno = ENOENT;
00655      } else {
00656         // open the file

00658         local_errno = errno;
00659      }
00660

00781                 process->mmap_end += new_length;
00782                 warn("returning %08p as start\n", start);

01081 mmapFunc(SyscallDesc *desc, int num, LiveProcess *p, ThreadContext *tc)
01082 {
01083     int index = 0;
01084     Addr start = p->getSyscallArg(tc, index);
01085     uint64_t length = p->getSyscallArg(tc, index);
01086     index++; // int prot = p->getSyscallArg(tc, index);
01087     int flags = p->getSyscallArg(tc, index);
01088     int tgt_fd = p->getSyscallArg(tc, index);
01089     // int offset = p->getSyscallArg(tc, index);
01090 
01091     if (length > 0x100000000ULL)
01092         warn("mmap length argument %#x is unreasonably large.\n", length);
01093 
01094     if (!(flags & OS::TGT_MAP_ANONYMOUS)) {
01095         Process::FdMap *fd_map = p->sim_fd_obj(tgt_fd);
01096         if (!fd_map || fd_map->fd < 0) {
01097             warn("mmap failing: target fd %d is not valid\n", tgt_fd);
01098             return -EBADF;
01099         }
01100 
01101         if (fd_map->filename != "/dev/zero") {
01102             // This is very likely broken, but leave a warning here
01103             // (rather than panic) in case /dev/zero is known by
01104             // another name on some platform
01105             warn("allowing mmap of file %s; mmap not supported on files"
01106                  " other than /dev/zero\n", fd_map->filename);
01107         }
01108     }
01109 
01110     if ((start  % TheISA::VMPageSize) != 0 ||
01111         (length % TheISA::VMPageSize) != 0) {
01112         warn("mmap failing: arguments not page-aligned: "
01113              "start 0x%x length 0x%x",
01114              start, length);
01115         return -EINVAL;
01116     }
01117 
01118     // are we ok with clobbering existing mappings?  only set this to
01119     // true if the user has been warned.
01120     bool clobber = false;
01121 
01122     // try to use the caller-provided address if there is one
01123     bool use_provided_address = (start != 0);
01124 
01125     if (use_provided_address) {
01126         // check to see if the desired address is already in use
01127         if (!p->pTable->isUnmapped(start, length)) {
01128             // there are existing mappings in the desired range
01129             // whether we clobber them or not depends on whether the caller
01130             // specified MAP_FIXED
01131             if (flags & OS::TGT_MAP_FIXED) {
01132                 // MAP_FIXED specified: clobber existing mappings
01133                 warn("mmap: MAP_FIXED at 0x%x overwrites existing mappings\n",
01134                      start);
01135                 clobber = true;
01136             } else {
01137                 // MAP_FIXED not specified: ignore suggested start address
01138                 warn("mmap: ignoring suggested map address 0x%x\n", start);
01139                 use_provided_address = false;
01140             }
01141         }
01142     }
01143 
01144     if (!use_provided_address) {
01145         // no address provided, or provided address unusable:
01146         // pick next address from our "mmap region"
01147         if (OS::mmapGrowsDown()) {
01148             start = p->mmap_end - length;
01149             p->mmap_end = start;
01150         } else {
01151             start = p->mmap_end;
01152             p->mmap_end += length;
01153         }
01154     }
01155
