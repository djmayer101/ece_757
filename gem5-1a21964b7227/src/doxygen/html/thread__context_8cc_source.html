00052 void
00053 ThreadContext::compare(ThreadContext *one, ThreadContext *two)
00054 {
00055     DPRINTF(Context, "Comparing thread contexts\n");
00056 
00057     // First loop through the integer registers.
00058     for (int i = 0; i < TheISA::NumIntRegs; ++i) {
00059         TheISA::IntReg t1 = one->readIntReg(i);
00060         TheISA::IntReg t2 = two->readIntReg(i);
00061         if (t1 != t2)
00062             panic("Int reg idx %d doesn't match, one: %#x, two: %#x",
00063                   i, t1, t2);
00064     }
00065 
00066     // Then loop through the floating point registers.
00067     for (int i = 0; i < TheISA::NumFloatRegs; ++i) {
00068         TheISA::FloatRegBits t1 = one->readFloatRegBits(i);
00069         TheISA::FloatRegBits t2 = two->readFloatRegBits(i);
00070         if (t1 != t2)
00071             panic("Float reg idx %d doesn't match, one: %#x, two: %#x",
00072                   i, t1, t2);
00073     }
00074     for (int i = 0; i < TheISA::NumMiscRegs; ++i) {
00075         TheISA::MiscReg t1 = one->readMiscRegNoEffect(i);
00076         TheISA::MiscReg t2 = two->readMiscRegNoEffect(i);
00077         if (t1 != t2)
00078             panic("Misc reg idx %d doesn't match, one: %#x, two: %#x",
00079                   i, t1, t2);
00080     }
00081 
00082     if (!(one->pcState() == two->pcState()))
00083         panic("PC state doesn't match.");
00084     int id1 = one->cpuId();
00085     int id2 = two->cpuId();
00086     if (id1 != id2)
00087         panic("CPU ids don't match, one: %d, two: %d", id1, id2);
00088 
00089     id1 = one->contextId();
00090     id2 = two->contextId();
00091     if (id1 != id2)
00092         panic("Context ids don't match, one: %d, two: %d", id1, id2);
00093 
00094 
00095 }
