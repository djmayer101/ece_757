00172 void
00173 CpuLocalTimer::Timer::write(PacketPtr pkt, Addr daddr)
00174 {
00175     DPRINTF(Timer, "Writing to CpuLocalTimer at offset: %#x\n", daddr);
00176     bool old_enable;
00177     bool old_wd_mode;
00178     uint32_t old_val;
00179 
00180     switch (daddr) {
00181       case TimerLoadReg:
00182         // Writing to this register also resets the counter register and
00183         // starts decrementing if the counter is enabled.
00184         timerLoadValue = pkt->get<uint32_t>();
00185         restartTimerCounter(timerLoadValue);
00186         break;
00187       case TimerCounterReg:
00188         // Can be written, doesn't start counting unless the timer is enabled
00189         restartTimerCounter(pkt->get<uint32_t>());
00190         break;
00191       case TimerControlReg:
00192         old_enable = timerControl.enable;
00193         timerControl = pkt->get<uint32_t>();
00194         if ((old_enable == 0) && timerControl.enable)
00195             restartTimerCounter(timerLoadValue);
00196         break;
00197       case TimerIntStatusReg:
00198         rawIntTimer = false;
00199         if (pendingIntTimer) {
00200             pendingIntTimer = false;
00201             DPRINTF(Timer, "Clearing interrupt\n");
00202         }
00203         break;
00204       case WatchdogLoadReg:
00205         watchdogLoadValue = pkt->get<uint32_t>();
00206         restartWatchdogCounter(watchdogLoadValue);
00207         break;
00208       case WatchdogCounterReg:
00209         // Can't be written when in watchdog mode, but can in timer mode
00210         if (!watchdogControl.watchdogMode) {
00211             restartWatchdogCounter(pkt->get<uint32_t>());
00212         }
00213         break;
00214       case WatchdogControlReg:
00215         old_enable = watchdogControl.enable;
00216         old_wd_mode = watchdogControl.watchdogMode;
00217         watchdogControl = pkt->get<uint32_t>();
00218         if ((old_enable == 0) && watchdogControl.enable)
00219             restartWatchdogCounter(watchdogLoadValue);
00220         // cannot disable watchdog using control register
00221         if ((old_wd_mode == 1) && watchdogControl.watchdogMode == 0)
00222             watchdogControl.watchdogMode = 1;
00223         break;
00224       case WatchdogIntStatusReg:
00225         rawIntWatchdog = false;
00226         if (pendingIntWatchdog) {
00227             pendingIntWatchdog = false;
00228             DPRINTF(Timer, "Clearing watchdog interrupt\n");
00229         }
00230         break;
00231       case WatchdogResetStatusReg:
00232         rawResetWatchdog = false;
00233         DPRINTF(Timer, "Clearing watchdog reset flag\n");
00234         break;
00235       case WatchdogDisableReg:
00236         old_val = watchdogDisableReg;
00237         watchdogDisableReg = pkt->get<uint32_t>();
00238         // if this sequence is observed, turn off watchdog mode
00239         if (old_val == 0x12345678 && watchdogDisableReg == 0x87654321)
00240             watchdogControl.watchdogMode = 0;
00241         break;
00242       default:
00243         panic("Tried to write CpuLocalTimer timer at offset %#x\n", daddr);
00244         break;
00245     }
00246 }
