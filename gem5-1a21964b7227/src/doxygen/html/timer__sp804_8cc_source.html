00140 Sp804::Timer::write(PacketPtr pkt, Addr daddr)
00141 {
00142     DPRINTF(Timer, "Writing %#x to Timer at offset: %#x\n", pkt->get<uint32_t>(), daddr);
00143     switch (daddr) {
00144       case LoadReg:
00145         loadValue = pkt->get<uint32_t>();
00146         restartCounter(loadValue);
00147         break;
00148       case CurrentReg:
00149         // Spec says this value can't be written, but linux writes it anyway
00150         break;
00151       case ControlReg:
00152         bool old_enable;
00153         old_enable = control.timerEnable;
00154         control = pkt->get<uint32_t>();
00155         if ((old_enable == 0) && control.timerEnable)
00156             restartCounter(loadValue);
00157         break;
00158       case IntClear:
00159         rawInt = false;
00160         if (pendingInt) {
00161             pendingInt = false;
00162             DPRINTF(Timer, "Clearing interrupt\n");
00163             parent->gic->clearInt(intNum);
00164         }
00165         break;
00166       case BGLoad:
00167         loadValue = pkt->get<uint32_t>();
00168         break;
00169       default:
00170         panic("Tried to write SP804 timer at offset %#x\n", daddr);
00171         break;
00172     }

00176 Sp804::Timer::restartCounter(uint32_t val)
00177 {
00178     DPRINTF(Timer, "Resetting counter with value %#x\n", val);
00179     if (!control.timerEnable)
00180         return;
00181 
00182     Tick time = clock * power(16, control.timerPrescale);
00183     if (control.timerSize)
00184         time *= val;

00186         time *= bits(val,15,0);

00189         DPRINTF(Timer, "-- Event was already schedule, de-scheduling\n");

00193     DPRINTF(Timer, "-- Scheduling new event for: %d\n", curTick() + time);
