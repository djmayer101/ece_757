00610 void

00612 {
00613     if (_status == Faulting)
00614         return;
00615 
00616     if (fault != NoFault) {
00617         advancePC(fault);
00618         DPRINTF(SimpleCPU, "Fault occured, scheduling fetch event\n");
00619         reschedule(fetchEvent, nextCycle(), true);
00620         _status = Faulting;
00621         return;
00622     }
00623 
00624 
00625     if (!stayAtPC)
00626         advancePC(fault);
00627 
00628     if (tryCompleteDrain())

00631     if (_status == BaseSimpleCPU::Running) {
00632         // kick off fetch of next instruction... callback from icache
00633         // response will cause that instruction to be executed,
00634         // keeping the CPU running.

00742 void
00743 TimingSimpleCPU::completeDataAccess(PacketPtr pkt)
00744 {
00745     // received a response from the dcache: complete the load or store
00746     // instruction
00747     assert(!pkt->isError());
00748     assert(_status == DcacheWaitResponse || _status == DTBWaitResponse ||
00749            pkt->req->getFlags().isSet(Request::NO_ACCESS));
00750 
00751     numCycles += curCycle() - previousCycle;
00752     previousCycle = curCycle();
00753 
00754     if (pkt->senderState) {
00755         SplitFragmentSenderState * send_state =
00756             dynamic_cast<SplitFragmentSenderState *>(pkt->senderState);
00757         assert(send_state);
00758         delete pkt->req;
00759         delete pkt;
00760         PacketPtr big_pkt = send_state->bigPkt;
00761         delete send_state;
00762         
00763         SplitMainSenderState * main_send_state =
00764             dynamic_cast<SplitMainSenderState *>(big_pkt->senderState);
00765         assert(main_send_state);
00766         // Record the fact that this packet is no longer outstanding.
00767         assert(main_send_state->outstanding != 0);
00768         main_send_state->outstanding--;
00769 
00770         if (main_send_state->outstanding) {
00771             return;
00772         } else {
00773             delete main_send_state;
00774             big_pkt->senderState = NULL;
00775             pkt = big_pkt;
00776         }
00777     }
00778 
00779     _status = BaseSimpleCPU::Running;
00780 
00781     Fault fault = curStaticInst->completeAcc(pkt, this, traceData);
00782 
00783     // keep an instruction count
00784     if (fault == NoFault)
00785         countInst();
00786     else if (traceData) {
00787         // If there was a fault, we shouldn't trace this instruction.
00788         delete traceData;
00789         traceData = NULL;
00790     }
00791 
00792     // the locked flag may be cleared on the response packet, so check
00793     // pkt->req and not pkt to see if it was a load-locked
00794     if (pkt->isRead() && pkt->req->isLLSC()) {
00795         TheISA::handleLockedRead(thread, pkt->req);
00796     }
00797 
00798     delete pkt->req;
00799     delete pkt;
00800 
00801     postExecute();
00802 
00803     advanceInst(fault);
00804 }

00808 {
00809     // delay processing of returned data until next CPU clock edge
00810     Tick next_tick = cpu->nextCycle();
00811

00819             // faster than a CPU we could get two responses before
00820             // next_tick expires

00823             return false;
00824         }
00825     }
00826

00836 void
00837 TimingSimpleCPU::DcachePort::recvRetry()
00838 {
00839     // we shouldn't get a retry unless we have a packet that we're
00840     // waiting to transmit
00841     assert(cpu->dcache_pkt != NULL);
00842     assert(cpu->_status == DcacheRetry);
00843     PacketPtr tmp = cpu->dcache_pkt;
00844     if (tmp->senderState) {
00845         // This is a packet from a split access.
00846         SplitFragmentSenderState * send_state =
00847             dynamic_cast<SplitFragmentSenderState *>(tmp->senderState);
00848         assert(send_state);
00849         PacketPtr big_pkt = send_state->bigPkt;
00850         
00851         SplitMainSenderState * main_send_state =
00852             dynamic_cast<SplitMainSenderState *>(big_pkt->senderState);
00853         assert(main_send_state);
00854 
00855         if (sendTimingReq(tmp)) {
00856             // If we were able to send without retrying, record that fact
00857             // and try sending the other fragment.
00858             send_state->clearFromParent();
00859             int other_index = main_send_state->getPendingFragment();
00860             if (other_index > 0) {
00861                 tmp = main_send_state->fragments[other_index];
00862                 cpu->dcache_pkt = tmp;
00863                 if ((big_pkt->isRead() && cpu->handleReadPacket(tmp)) ||
00864                         (big_pkt->isWrite() && cpu->handleWritePacket())) {
00865                     main_send_state->fragments[other_index] = NULL;
00866                 }
00867             } else {
00868                 cpu->_status = DcacheWaitResponse;
00869                 // memory system takes ownership of packet
00870                 cpu->dcache_pkt = NULL;
00871             }
00872         }
00873     } else if (sendTimingReq(tmp)) {
00874         cpu->_status = DcacheWaitResponse;
00875         // memory system takes ownership of packet
00876         cpu->dcache_pkt = NULL;
00877     }
00878 }
