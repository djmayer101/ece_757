00054       globalHistoryBits(params->globalHistoryBits),
00055       choicePredictorSize(params->choicePredictorSize),
00056       choiceCtrBits(params->choiceCtrBits),
00057       instShiftAmt(params->instShiftAmt)
00058 {
00059     localPredictorSize = ULL(1) << localHistoryBits;
00060 
00061     //Set up the array of counters for the local predictor
00062     localCtrs.resize(localPredictorSize);
00063 
00064     for (int i = 0; i < localPredictorSize; ++i)
00065         localCtrs[i].setBits(localCtrBits);
00066 
00067     localPredictorMask = mask(localHistoryBits);
00068 
00069     if (!isPowerOf2(localHistoryTableSize)) {
00070         fatal("Invalid local history table size!\n");
00071     }
00072 
00073     //Setup the history table for the local table
00074     localHistoryTable.resize(localHistoryTableSize);
00075 
00076     for (int i = 0; i < localHistoryTableSize; ++i)
00077         localHistoryTable[i] = 0;
00078 
00079     if (!isPowerOf2(globalPredictorSize)) {
00080         fatal("Invalid global predictor size!\n");
00081     }
00082 
00083     //Setup the array of counters for the global predictor
00084     globalCtrs.resize(globalPredictorSize);
00085 
00086     for (int i = 0; i < globalPredictorSize; ++i)
00087         globalCtrs[i].setBits(globalCtrBits);
00088 
00089     //Clear the global history
00090     globalHistory = 0;
00091     // Set up the global history mask
00092     // this is equivalent to mask(log2(globalPredictorSize)
00093     globalHistoryMask = globalPredictorSize - 1;
00094 
00095     if (!isPowerOf2(choicePredictorSize)) {
00096         fatal("Invalid choice predictor size!\n");
00097     }
00098 
00099     // Set up choiceHistoryMask
00100     // this is equivalent to mask(log2(choicePredictorSize)
00101     choiceHistoryMask = choicePredictorSize - 1;
00102 
00103     //Setup the array of counters for the choice predictor
00104     choiceCtrs.resize(choicePredictorSize);
00105 
00106     for (int i = 0; i < choicePredictorSize; ++i)
00107         choiceCtrs[i].setBits(choiceCtrBits);
00108 
00109     //Set up historyRegisterMask
00110     historyRegisterMask = mask(globalHistoryBits);
00111 
00112     //Check that predictors don't use more bits than they have available
00113     if (globalHistoryMask > historyRegisterMask) {
00114         fatal("Global predictor too large for global history bits!\n");
00115     }
00116     if (choiceHistoryMask > historyRegisterMask) {
00117         fatal("Choice predictor too large for global history bits!\n");
00118     }
00119 
00120     if (globalHistoryMask < historyRegisterMask &&
00121         choiceHistoryMask < historyRegisterMask) {
00122         inform("More global history bits than required by predictors\n");
00123     }
00124 
00125     // Set thresholds for the three predictors' counters
00126     // This is equivalent to (2^(Ctr))/2 - 1
00127     localThreshold  = (ULL(1) << (localCtrBits  - 1)) - 1;
00128     globalThreshold = (ULL(1) << (globalCtrBits - 1)) - 1;
00129     choiceThreshold = (ULL(1) << (choiceCtrBits - 1)) - 1;

00291             // Update the choice predictor to tell it which one was correct if
00292             // there was a prediction.
00293             if (history->localPredTaken != history->globalPredTaken) {
00294                  // If the local prediction matches the actual outcome,

00300                      choiceCtrs[choice_predictor_idx].decrement();

00302                      choiceCtrs[choice_predictor_idx].increment();

00305              }
00306

00318              } else {

00323              }
00324         }
