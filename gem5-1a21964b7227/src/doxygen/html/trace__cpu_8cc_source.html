00061 void
00062 TraceCPU::tick()
00063 {
00064     assert(outstandingRequests >= 0);
00065     assert(outstandingRequests < 1000);
00066     int instReqs = 0;
00067     int dataReqs = 0;
00068 
00069     while (nextReq && curTick() >= nextCycle) {
00070         assert(nextReq->thread_num < 4 && "Not enough threads");
00071         if (nextReq->isInstFetch() && icacheInterface) {
00072             if (icacheInterface->isBlocked())
00073                 break;
00074 
00075             nextReq->time = curTick();
00076             if (nextReq->cmd == Squash) {
00077                 icacheInterface->squash(nextReq->asid);
00078             } else {
00079                 ++instReqs;
00080                 if (icacheInterface->doEvents()) {
00081                     nextReq->completionEvent =
00082                         new TraceCompleteEvent(nextReq, this);
00083                     icacheInterface->access(nextReq);
00084                 } else {
00085                     icacheInterface->access(nextReq);
00086                     completeRequest(nextReq);
00087                 }
00088             }
00089         } else {
00090             if (dcacheInterface->isBlocked())
00091                 break;
00092 
00093             ++dataReqs;
00094             nextReq->time = curTick();
00095             if (dcacheInterface->doEvents()) {
00096                 nextReq->completionEvent =
00097                     new TraceCompleteEvent(nextReq, this);
00098                 dcacheInterface->access(nextReq);
00099             } else {
00100                 dcacheInterface->access(nextReq);
00101                 completeRequest(nextReq);
00102             }
00103 
00104         }
00105         nextCycle = dataTrace->getNextReq(nextReq);
00106     }
00107 
00108     if (!nextReq) {
00109         // No more requests to send. Finish trailing events and exit.
00110         if (mainEventQueue.empty()) {
00111             exitSimLoop("end of memory trace reached");
00112         } else {
00113             tickEvent.schedule(mainEventQueue.nextEventTime() + ticks(1));
00114         }
00115     } else {
00116         tickEvent.schedule(max(curTick() + ticks(1), nextCycle));
00117     }
00118 }
