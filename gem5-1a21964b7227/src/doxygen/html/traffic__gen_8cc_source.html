00323 
00324     // determine next state

00412 Tick

00414 {
00415     // If we have reached the end of the address space, reset the
00416     // address to the start of the range
00417     if (nextAddr + blocksize > endAddr) {
00418         DPRINTF(TrafficGen, "Wrapping address to the start of "

00421     }
00422 
00423     // Check to see if we have reached the data limit. If dataLimit is
00424     // zero we do not have a data limit and therefore we will keep

00427         DPRINTF(TrafficGen, "Data limit for LinearGen reached.\n");
00428         // there are no more requests, therefore return MaxTick
00429         return MaxTick;
00430     } else {
00431         // return the time when the next request should take place
00432         return curTick() + random_mt.random<Tick>(minPeriod, maxPeriod);
