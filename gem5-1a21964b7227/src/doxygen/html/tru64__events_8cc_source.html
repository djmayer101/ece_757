00052 BadAddrEvent::process(ThreadContext *tc)
00053 {
00054     // The following gross hack is the equivalent function to the
00055     // annotation for vmunix::badaddr in:
00056     // simos/simulation/apps/tcl/osf/tlaser.tcl
00057 
00058     uint64_t a0 = tc->readIntReg(16);
00059 
00060     AddrRangeIter iter;
00061     bool found = false;
00062 
00063     MasterPort &dataPort = tc->getCpuPtr()->getDataPort();
00064 
00065     // get the address ranges of the connected slave port
00066     AddrRangeList resp = dataPort.getAddrRanges();
00067     for (iter = resp.begin(); iter != resp.end(); iter++) {
00068         if (iter->contains(K0Seg2Phys(a0) & PAddrImplMask))
00069             found = true;
00070     }
00071 
00072     if (!IsK0Seg(a0) || found ) {
00073 
00074         DPRINTF(BADADDR, "badaddr arg=%#x bad\n", a0);
00075         tc->setIntReg(ReturnValueReg, 0x1);
00076         SkipFuncEvent::process(tc);
00077     } else {
00078         DPRINTF(BADADDR, "badaddr arg=%#x good\n", a0);
00079     }
