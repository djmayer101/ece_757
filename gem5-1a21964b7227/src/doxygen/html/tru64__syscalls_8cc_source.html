00427 const char *
00428 SystemCalls<Tru64>::name(int num)
00429 {
00430     if (num >= Number)
00431         return 0;
00432     else if (num >= StandardNumber)
00433         return mach_strings[num - StandardNumber];
00434     else if (num >= 0)
00435         return standard_strings[num];
00436     else if (num > -MachNumber)
00437         return mach_strings[-num];
00438     else
00439         return 0;
00440 }
