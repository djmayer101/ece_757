00092 TsunamiIO::read(PacketPtr pkt)
00093 {
00094     assert(pkt->getAddr() >= pioAddr && pkt->getAddr() < pioAddr + pioSize);
00095 
00096     Addr daddr = pkt->getAddr() - pioAddr;
00097 
00098     DPRINTF(Tsunami, "io read  va=%#x size=%d IOPorrt=%#x\n", pkt->getAddr(),
00099             pkt->getSize(), daddr);
00100 
00101     pkt->allocate();
00102 
00103     if (pkt->getSize() == sizeof(uint8_t)) {
00104         switch(daddr) {
00105           // PIC1 mask read
00106           case TSDEV_PIC1_MASK:
00107             pkt->set(~mask1);
00108             break;
00109           case TSDEV_PIC2_MASK:
00110             pkt->set(~mask2);
00111             break;
00112           case TSDEV_PIC1_ISR:
00113               // !!! If this is modified 64bit case needs to be too
00114               // Pal code has to do a 64 bit physical read because there is
00115               // no load physical byte instruction
00116               pkt->set(picr);
00117               break;
00118           case TSDEV_PIC2_ISR:
00119               // PIC2 not implemnted... just return 0
00120               pkt->set(0x00);
00121               break;
00122           case TSDEV_TMR0_DATA:
00123             pkt->set(pitimer.readCounter(0));
00124             break;
00125           case TSDEV_TMR1_DATA:
00126             pkt->set(pitimer.readCounter(1));
00127             break;
00128           case TSDEV_TMR2_DATA:
00129             pkt->set(pitimer.readCounter(2));
00130             break;
00131           case TSDEV_RTC_DATA:
00132             pkt->set(rtc.readData(rtcAddr));
00133             break;
00134           case TSDEV_CTRL_PORTB:
00135             if (pitimer.outputHigh(2))
00136                 pkt->set(PORTB_SPKR_HIGH);
00137             else
00138                 pkt->set(0x00);
00139             break;
00140           default:
00141             panic("I/O Read - va%#x size %d\n", pkt->getAddr(), pkt->getSize());
00142         }
00143     } else if (pkt->getSize() == sizeof(uint64_t)) {
00144         if (daddr == TSDEV_PIC1_ISR)
00145             pkt->set<uint64_t>(picr);
00146         else
00147            panic("I/O Read - invalid addr - va %#x size %d\n",
00148                    pkt->getAddr(), pkt->getSize());
00149     } else {
00150        panic("I/O Read - invalid size - va %#x size %d\n", pkt->getAddr(), pkt->getSize());
00151     }
00152     pkt->makeAtomicResponse();
00153     return pioDelay;
00154 }

00156 Tick
00157 TsunamiIO::write(PacketPtr pkt)
00158 {
00159     assert(pkt->getAddr() >= pioAddr && pkt->getAddr() < pioAddr + pioSize);
00160     Addr daddr = pkt->getAddr() - pioAddr;
00161 
00162     DPRINTF(Tsunami, "io write - va=%#x size=%d IOPort=%#x Data=%#x\n",
00163             pkt->getAddr(), pkt->getSize(), pkt->getAddr() & 0xfff, (uint32_t)pkt->get<uint8_t>());
00164 
00165     assert(pkt->getSize() == sizeof(uint8_t));
00166 
00167     switch(daddr) {
00168       case TSDEV_PIC1_MASK:
00169         mask1 = ~(pkt->get<uint8_t>());
00170         if ((picr & mask1) && !picInterrupting) {
00171             picInterrupting = true;
00172             tsunami->cchip->postDRIR(55);
00173             DPRINTF(Tsunami, "posting pic interrupt to cchip\n");
00174         }
00175         if ((!(picr & mask1)) && picInterrupting) {
00176             picInterrupting = false;
00177             tsunami->cchip->clearDRIR(55);
00178             DPRINTF(Tsunami, "clearing pic interrupt\n");
00179         }
00180         break;
00181       case TSDEV_PIC2_MASK:
00182         mask2 = pkt->get<uint8_t>();
00183         //PIC2 Not implemented to interrupt
00184         break;
00185       case TSDEV_PIC1_ACK:
00186         // clear the interrupt on the PIC
00187         picr &= ~(1 << (pkt->get<uint8_t>() & 0xF));
00188         if (!(picr & mask1))
00189             tsunami->cchip->clearDRIR(55);
00190         break;
00191       case TSDEV_DMA1_MODE:
00192         mode1 = pkt->get<uint8_t>();
00193         break;
00194       case TSDEV_DMA2_MODE:
00195         mode2 = pkt->get<uint8_t>();
00196         break;
00197       case TSDEV_TMR0_DATA:
00198         pitimer.writeCounter(0, pkt->get<uint8_t>());
00199         break;
00200       case TSDEV_TMR1_DATA:
00201         pitimer.writeCounter(1, pkt->get<uint8_t>());
00202         break;
00203       case TSDEV_TMR2_DATA:
00204         pitimer.writeCounter(2, pkt->get<uint8_t>());
00205         break;
00206       case TSDEV_TMR_CTRL:
00207         pitimer.writeControl(pkt->get<uint8_t>());
00208         break;
00209       case TSDEV_RTC_ADDR:
00210         rtcAddr = pkt->get<uint8_t>();
00211         break;
00212       case TSDEV_RTC_DATA:
00213         rtc.writeData(rtcAddr, pkt->get<uint8_t>());
00214         break;
00215       case TSDEV_KBD:
00216       case TSDEV_DMA1_CMND:
00217       case TSDEV_DMA2_CMND:
00218       case TSDEV_DMA1_MMASK:
00219       case TSDEV_DMA2_MMASK:
00220       case TSDEV_PIC2_ACK:
00221       case TSDEV_DMA1_RESET:
00222       case TSDEV_DMA2_RESET:
00223       case TSDEV_DMA1_MASK:
00224       case TSDEV_DMA2_MASK:
00225       case TSDEV_CTRL_PORTB:
00226         break;
00227       default:
00228         panic("I/O Write - va%#x size %d data %#x\n", pkt->getAddr(), pkt->getSize(), pkt->get<uint8_t>());
00229     }
00230 
00231     pkt->makeAtomicResponse();
00232     return pioDelay;
00233 }
