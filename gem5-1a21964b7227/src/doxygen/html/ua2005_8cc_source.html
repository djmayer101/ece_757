00045 void
00046 ISA::checkSoftInt(ThreadContext *tc)
00047 {
00048     BaseCPU *cpu = tc->getCpuPtr();
00049 
00050     // If PIL < 14, copy over the tm and sm bits
00051     if (pil < 14 && softint & 0x10000)
00052         cpu->postInterrupt(IT_SOFT_INT, 16);
00053     else
00054         cpu->clearInterrupt(IT_SOFT_INT, 16);
00055     if (pil < 14 && softint & 0x1)
00056         cpu->postInterrupt(IT_SOFT_INT, 0);
00057     else
00058         cpu->clearInterrupt(IT_SOFT_INT, 0);
00059 
00060     // Copy over any of the other bits that are set
00061     for (int bit = 15; bit > 0; --bit) {
00062         if (1 << bit & softint && bit > pil)
00063             cpu->postInterrupt(IT_SOFT_INT, bit);
00064         else
00065             cpu->clearInterrupt(IT_SOFT_INT, bit);
00066     }
00067 }

00094 
00095     int64_t time;
00096     switch (miscReg) {
00097         /* Full system only ASRs */
00098       case MISCREG_SOFTINT:
00099         setMiscRegNoEffect(miscReg, val);;
00100         checkSoftInt(tc);
00101         break;
00102       case MISCREG_SOFTINT_CLR:
00103         return setMiscReg(MISCREG_SOFTINT, ~val & softint, tc);
00104       case MISCREG_SOFTINT_SET:
00105         return setMiscReg(MISCREG_SOFTINT, val | softint, tc);
00106 
00107       case MISCREG_TICK_CMPR:
00108         if (tickCompare == NULL)
00109             tickCompare = new TickCompareEvent(this, tc);
00110         setMiscRegNoEffect(miscReg, val);
00111         if ((tick_cmpr & ~mask(63)) && tickCompare->scheduled())
00112             cpu->deschedule(tickCompare);
00113         time = (tick_cmpr & mask(63)) - (tick & mask(63));
00114         if (!(tick_cmpr & ~mask(63)) && time > 0) {
00115             if (tickCompare->scheduled())
00116                 cpu->deschedule(tickCompare);
00117             cpu->schedule(tickCompare, cpu->clockEdge(Cycles(time)));
00118         }
00119         panic("writing to TICK compare register %#X\n", val);
00120         break;
00121 
00122       case MISCREG_STICK_CMPR:
00123         if (sTickCompare == NULL)
00124             sTickCompare = new STickCompareEvent(this, tc);
00125         setMiscRegNoEffect(miscReg, val);
00126         if ((stick_cmpr & ~mask(63)) && sTickCompare->scheduled())
00127             cpu->deschedule(sTickCompare);
00128         time = ((int64_t)(stick_cmpr & mask(63)) - (int64_t)stick) -
00129             cpu->instCount();

00131             if (sTickCompare->scheduled())
00132                 cpu->deschedule(sTickCompare);
00133             cpu->schedule(sTickCompare, cpu->clockEdge(Cycles(time)));
00134         }

00136         break;
00137 
00138       case MISCREG_PSTATE:
00139         setMiscRegNoEffect(miscReg, val);
00140 
00141       case MISCREG_PIL:
00142         setMiscRegNoEffect(miscReg, val);
00143         checkSoftInt(tc);
00144         break;
00145 
00146       case MISCREG_HVER:
00147         panic("Shouldn't be writing HVER\n");
00148 
00149       case MISCREG_HINTP:
00150         setMiscRegNoEffect(miscReg, val);
00151         if (hintp)
00152             cpu->postInterrupt(IT_HINTP, 0);
00153         else
00154             cpu->clearInterrupt(IT_HINTP, 0);
00155         break;
00156 
00157       case MISCREG_HTBA:
00158         // clear lower 7 bits on writes.
00159         setMiscRegNoEffect(miscReg, val & ULL(~0x7FFF));

00161 
00162       case MISCREG_QUEUE_CPU_MONDO_HEAD:
00163       case MISCREG_QUEUE_CPU_MONDO_TAIL:

00184             cpu->clearInterrupt(IT_RES_ERROR, 0);

00187       case MISCREG_QUEUE_NRES_ERROR_TAIL:

00189         // This one doesn't have an interrupt to report to the guest OS

00191 
00192       case MISCREG_HSTICK_CMPR:
00193         if (hSTickCompare == NULL)
00194             hSTickCompare = new HSTickCompareEvent(this, tc);

00207 
00208       case MISCREG_HPSTATE:
00209         {
00210             HPSTATE newVal = val;
00211             newVal.id = 1;
00212             // T1000 spec says impl. dependent val must always be 1
00213             setMiscRegNoEffect(miscReg, newVal);

00216                 cpu->postInterrupt(IT_TRAP_LEVEL_ZERO, 0);
00217             else
00218                 cpu->clearInterrupt(IT_TRAP_LEVEL_ZERO, 0);
00219             break;
00220         }
00221       case MISCREG_HTSTATE:
00222         setMiscRegNoEffect(miscReg, val);
00223         break;
00224 
00225       case MISCREG_STRAND_STS_REG:
00226         if (bits(val,2,2))
00227             panic("No support for setting spec_en bit\n");
00228         setMiscRegNoEffect(miscReg, bits(val,0,0));
00229         if (!bits(val,0,0)) {
00230             DPRINTF(Quiesce, "Cpu executed quiescing instruction\n");
00231             // Time to go to sleep
00232             tc->suspend();
00233             if (FullSystem && tc->getKernelStats())
00234                 tc->getKernelStats()->quiesce();
00235         }
00236         break;
00237 
00238       default:
00239         panic("Invalid write to FS misc register %s\n",
00240               getMiscRegName(miscReg));
00241     }

00260       case MISCREG_TICK_CMPR:
00261       case MISCREG_STICK_CMPR:
00262       case MISCREG_PIL:
00263       case MISCREG_HPSTATE:
00264       case MISCREG_HINTP:
00265       case MISCREG_HTSTATE:
00266       case MISCREG_HSTICK_CMPR:
00267         return readMiscRegNoEffect(miscReg) ;
00268 
00269       case MISCREG_HTBA:
00270         return readMiscRegNoEffect(miscReg) & ULL(~0x7FFF);
00271       case MISCREG_HVER:
00272         // XXX set to match Legion
00273         return ULL(0x3e) << 48 |
00274                ULL(0x23) << 32 |
00275                ULL(0x20) << 24 |
00276                    // MaxGL << 16 | XXX For some reason legion doesn't set GL
00277                    MaxTL << 8  |
00278            (NWindows -1) << 0;
00279 
00280       case MISCREG_STRAND_STS_REG:
00281         System *sys;
00282         int x;
00283         sys = tc->getSystemPtr();
00284 
00285         temp = readMiscRegNoEffect(miscReg) & (STS::active | STS::speculative);
00286         // Check that the CPU array is fully populated
00287         // (by calling getNumCPus())
00288         assert(sys->numContexts() > tc->contextId());
00289 
00290         temp |= tc->contextId()  << STS::shft_id;
00291 
00292         for (x = tc->contextId() & ~3; x < sys->threadContexts.size(); x++) {
00293             switch (sys->threadContexts[x]->status()) {
00294               case ThreadContext::Active:
00295                 temp |= STS::st_run << (STS::shft_fsm0 -
00296                         ((x & 0x3) * (STS::shft_fsm0-STS::shft_fsm1)));
00297                 break;
00298               case ThreadContext::Suspended:
00299                 // should this be idle?
00300                 temp |= STS::st_idle << (STS::shft_fsm0 -
00301                         ((x & 0x3) * (STS::shft_fsm0-STS::shft_fsm1)));
00302                 break;
00303               case ThreadContext::Halted:
00304                 temp |= STS::st_halt << (STS::shft_fsm0 -
00305                         ((x & 0x3) * (STS::shft_fsm0-STS::shft_fsm1)));
00306                 break;
00307               default:
00308                 panic("What state are we in?!\n");
00309             } // switch
00310         } // for
00311 
00312         return temp;
00313       default:
00314         panic("Invalid read to FS misc register\n");
00315     }

00324 void
00325 ISA::processSTickCompare(ThreadContext *tc)
00326 {
00327     BaseCPU *cpu = tc->getCpuPtr();
00328 
00329     // since our microcode instructions take two cycles we need to check if
00330     // we're actually at the correct cycle or we need to wait a little while
00331     // more
00332     int delay;
00333     delay = ((int64_t)(stick_cmpr & mask(63)) - (int64_t)stick) -
00334         cpu->instCount();
00335     assert(delay >= 0 && "stick compare missed interrupt cycle");
00336 
00337     if (delay == 0 || tc->status() == ThreadContext::Suspended) {
00338         DPRINTF(Timer, "STick compare cycle reached at %#x\n",
00339                 (stick_cmpr & mask(63)));
00340         if (!(tc->readMiscRegNoEffect(MISCREG_STICK_CMPR) & (ULL(1) << 63))) {
00341             setMiscReg(MISCREG_SOFTINT, softint | (ULL(1) << 16), tc);
00342         }
00343     } else {
00344         cpu->schedule(sTickCompare, cpu->clockEdge(Cycles(delay)));
00345     }
00346 }
