00079 /* The linux serial driver (8250.c about line 1182) loops reading from
00080  * the device until the device reports it has no more data to
00081  * read. After a maximum of 255 iterations the code prints "serial8250
00082  * too much work for irq X," and breaks out of the loop. Since the
00083  * simulated system is so much slower than the actual system, if a
00084  * user is typing on the keyboard it is very easy for them to provide
00085  * input at a fast enough rate to not allow the loop to exit and thus
00086  * the error to be printed. This magic number provides a delay between
00087  * the time the UART receives a character to send to the simulated
00088  * system and the time it actually notifies the system it has a
00089  * character to send to alleviate this problem. --Ali
00090  */
00091 void
00092 Uart8250::IntrEvent::scheduleIntr()
00093 {
00094     static const Tick interval = 225 * SimClock::Int::ns;
00095     DPRINTF(Uart, "Scheduling IER interrupt for %#x, at cycle %lld\n", intrBit,
00096             curTick() + interval);
00097     if (!scheduled())
00098         uart->schedule(this, curTick() + interval);
00099     else
00100         uart->reschedule(this, curTick() + interval);
00101 }
