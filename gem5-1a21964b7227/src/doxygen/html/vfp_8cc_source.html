00332 uint16_t
00333 vcvtFpSFpH(FPSCR &fpscr, bool flush, bool defaultNan,
00334            uint32_t rMode, bool ahp, float op)
00335 {
00336     uint32_t opBits = fpToBits(op);
00337     // Extract the operand.
00338     bool neg = bits(opBits, 31);
00339     uint32_t exponent = bits(opBits, 30, 23);
00340     uint32_t oldMantissa = bits(opBits, 22, 0);
00341     uint32_t mantissa = oldMantissa >> (23 - 10);
00342     // Do the conversion.
00343     uint32_t extra = oldMantissa & mask(23 - 10);
00344     if (exponent == 0xff) {
00345         if (oldMantissa != 0) {
00346             // Nans.
00347             if (bits(mantissa, 9) == 0) {
00348                 // Signalling nan.
00349                 fpscr.ioc = 1;
00350             }
00351             if (ahp) {
00352                 mantissa = 0;
00353                 exponent = 0;
00354                 fpscr.ioc = 1;
00355             } else if (defaultNan) {
00356                 mantissa = (1 << 9);
00357                 exponent = 0x1f;
00358                 neg = false;
00359             } else {
00360                 exponent = 0x1f;
00361                 mantissa |= (1 << 9);
00362             }
00363         } else {
00364             // Infinities.
00365             exponent = 0x1F;
00366             if (ahp) {
00367                 fpscr.ioc = 1;
00368                 mantissa = 0x3ff;
00369             } else {
00370                 mantissa = 0;
00371             }
00372         }
00373     } else if (exponent == 0 && oldMantissa == 0) {
00374         // Zero, don't need to do anything.
00375     } else {
00376         // Normalized or denormalized numbers.
00377 
00378         bool inexact = (extra != 0);
00379 
00380         if (exponent == 0) {
00381             // Denormalized.
00382 
00383             // If flush to zero is on, this shouldn't happen.
00384             assert(!flush);
00385 
00386             // Check for underflow
00387             if (inexact || fpscr.ufe)
00388                 fpscr.ufc = 1;
00389 
00390             // Handle rounding.
00391             unsigned mode = rMode;
00392             if ((mode == VfpRoundUpward && !neg && extra) ||
00393                 (mode == VfpRoundDown && neg && extra) ||
00394                 (mode == VfpRoundNearest &&
00395                  (extra > (1 << 9) ||
00396                   (extra == (1 << 9) && bits(mantissa, 0))))) {
00397                 mantissa++;
00398             }
00399 
00400             // See if the number became normalized after rounding.
00401             if (mantissa == (1 << 10)) {
00402                 mantissa = 0;
00403                 exponent = 1;
00404             }
00405         } else {
00406             // Normalized.
00407 
00408             // We need to track the dropped bits differently since
00409             // more can be dropped by denormalizing.
00410             bool topOne = bits(extra, 12);
00411             bool restZeros = bits(extra, 11, 0) == 0;
00412 
00413             if (exponent <= (127 - 15)) {
00414                 // The result is too small. Denormalize.
00415                 mantissa |= (1 << 10);
00416                 while (mantissa && exponent <= (127 - 15)) {
00417                     restZeros = restZeros && !topOne;
00418                     topOne = bits(mantissa, 0);
00419                     mantissa = mantissa >> 1;
00420                     exponent++;
00421                 }
00422                 if (topOne || !restZeros)
00423                     inexact = true;
00424                 exponent = 0;
00425             } else {
00426                 // Change bias.
00427                 exponent -= (127 - 15);
00428             }
00429 
00430             if (exponent == 0 && (inexact || fpscr.ufe)) {
00431                 // Underflow
00432                 fpscr.ufc = 1;
00433             }
00434 
00435             // Handle rounding.
00436             unsigned mode = rMode;
00437             bool nonZero = topOne || !restZeros;
00438             if ((mode == VfpRoundUpward && !neg && nonZero) ||
00439                 (mode == VfpRoundDown && neg && nonZero) ||
00440                 (mode == VfpRoundNearest && topOne &&
00441                  (!restZeros || bits(mantissa, 0)))) {
00442                 mantissa++;
00443             }
00444 
00445             // See if we rounded up and need to bump the exponent.
00446             if (mantissa == (1 << 10)) {
00447                 mantissa = 0;
00448                 exponent++;
00449             }
00450 
00451             // Deal with overflow
00452             if (ahp) {
00453                 if (exponent >= 0x20) {
00454                     exponent = 0x1f;
00455                     mantissa = 0x3ff;
00456                     fpscr.ioc = 1;
00457                     // Supress inexact exception.
00458                     inexact = false;
00459                 }
00460             } else {
00461                 if (exponent >= 0x1f) {
00462                     if ((mode == VfpRoundNearest) ||
00463                         (mode == VfpRoundUpward && !neg) ||
00464                         (mode == VfpRoundDown && neg)) {
00465                         // Overflow to infinity.
00466                         exponent = 0x1f;
00467                         mantissa = 0;
00468                     } else {
00469                         // Overflow to max normal.
00470                         exponent = 0x1e;
00471                         mantissa = 0x3ff;
00472                     }
00473                     fpscr.ofc = 1;
00474                     inexact = true;
00475                 }
00476             }
00477         }
00478 
00479         if (inexact) {
00480             fpscr.ixc = 1;
00481         }
00482     }
00483     // Reassemble and install the result.
00484     uint32_t result = bits(mantissa, 9, 0);
00485     replaceBits(result, 14, 10, exponent);
00486     if (neg)
00487         result |= (1 << 15);
00488     return result;
00489 }
