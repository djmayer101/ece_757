00115 template <class fpType>
00116 static inline bool
00117 flushToZero(fpType &op)
00118 {
00119     fpType junk = 0.0;
00120     if (std::fpclassify(op) == FP_SUBNORMAL) {
00121         uint64_t bitMask = ULL(0x1) << (sizeof(fpType) * 8 - 1);
00122         op = bitsToFp(fpToBits(op) & bitMask, junk);
00123         return true;
00124     }
00125     return false;
00126 }

00361 static inline float
00362 fpRSqrtsS(float a, float b)
00363 {
00364     int fpClassA = std::fpclassify(a);
00365     int fpClassB = std::fpclassify(b);
00366     float aXb;
00367     int fpClassAxB;
00368 
00369     if ((fpClassA == FP_ZERO && fpClassB == FP_INFINITE) ||
00370         (fpClassA == FP_INFINITE && fpClassB == FP_ZERO)) {
00371         return 1.5;
00372     }
00373     aXb = a*b;
00374     fpClassAxB = std::fpclassify(aXb);
00375     if(fpClassAxB == FP_SUBNORMAL) {
00376        feraiseexcept(FeUnderflow);
00377        return 1.5;
00378     }
00379     return (3.0 - (a * b)) / 2.0;
