00070 void
00071 VncInput::setFrameBufferParams(VideoConvert::Mode mode, uint16_t width,
00072     uint16_t height)
00073 {
00074     DPRINTF(VNC, "Updating video params: mode: %d width: %d height: %d\n", mode,
00075             width, height);
00076 
00077     if (mode != videoMode || width != videoWidth() || height != videoHeight()) {
00078         videoMode = mode;
00079         _videoWidth = width;
00080         _videoHeight = height;
00081 
00082         if (vc)
00083             delete vc;
00084 
00085         vc = new VideoConvert(mode, VideoConvert::rgb8888, videoWidth(),
00086                 videoHeight());
00087 
00088         if (captureEnabled) {
00089             // create bitmap of the frame with new attributes
00090             if (captureBitmap)
00091                 delete captureBitmap;
00092 
00093             assert(fbPtr);
00094             captureBitmap = new Bitmap(videoMode, width, height, fbPtr);
00095             assert(captureBitmap);
00096         }
00097     }
00098 }

00100 void
00101 VncInput::captureFrameBuffer()
00102 {
00103     assert(captureBitmap);
00104 
00105     // skip identical frames
00106     uint64_t new_hash = captureBitmap->getHash();
00107     if (captureLastHash == new_hash)
00108         return;
00109     captureLastHash = new_hash;
00110 
00111     // get the filename for the current frame
00112     char frameFilenameBuffer[64];
00113     snprintf(frameFilenameBuffer, 64, "fb.%06d.%lld.bmp.gz",
00114             captureCurrentFrame, static_cast<long long int>(curTick()));
00115     const string frameFilename(frameFilenameBuffer);
00116 
00117     // create the compressed framebuffer file
00118     ostream *fb_out = simout.create(captureOutputDirectory + frameFilename,
00119                     true);
00120     captureBitmap->write(fb_out);
00121     simout.close(fb_out);
00122 
00123     ++captureCurrentFrame;
00124 }
