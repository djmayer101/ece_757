00042 void
00043 ISA::updateHandyM5Reg(Efer efer, CR0 cr0,
00044                       SegAttr csAttr, SegAttr ssAttr, RFLAGS rflags,
00045                       ThreadContext *tc)
00046 {
00047     HandyM5Reg m5reg = 0;
00048     if (efer.lma) {
00049         m5reg.mode = LongMode;
00050         if (csAttr.longMode)
00051             m5reg.submode = SixtyFourBitMode;
00052         else
00053             m5reg.submode = CompatabilityMode;
00054     } else {
00055         m5reg.mode = LegacyMode;
00056         if (cr0.pe) {
00057             if (rflags.vm)
00058                 m5reg.submode = Virtual8086Mode;
00059             else
00060                 m5reg.submode = ProtectedMode;
00061         } else {
00062             m5reg.submode = RealMode;
00063         }
00064     }
00065     m5reg.cpl = csAttr.dpl;
00066     m5reg.paging = cr0.pg;
00067     m5reg.prot = cr0.pe;
00068 
00069     // Compute the default and alternate operand size.
00070     if (m5reg.submode == SixtyFourBitMode || csAttr.defaultSize) {
00071         m5reg.defOp = 2;
00072         m5reg.altOp = 1;
00073     } else {
00074         m5reg.defOp = 1;
00075         m5reg.altOp = 2;
00076     }
00077 
00078     // Compute the default and alternate address size.
00079     if (m5reg.submode == SixtyFourBitMode) {
00080         m5reg.defAddr = 3;
00081         m5reg.altAddr = 2;
00082     } else if (csAttr.defaultSize) {
00083         m5reg.defAddr = 2;
00084         m5reg.altAddr = 1;
00085     } else {
00086         m5reg.defAddr = 1;
00087         m5reg.altAddr = 2;
00088     }
00089 
00090     // Compute the stack size
00091     if (m5reg.submode == SixtyFourBitMode) {
00092         m5reg.stack = 3;
00093     } else if (ssAttr.defaultSize) {
00094         m5reg.stack = 2;
00095     } else {
00096         m5reg.stack = 1;
00097     }
00098 
00099     regVal[MISCREG_M5_REG] = m5reg;
00100     if (tc)
00101         tc->getDecoderPtr()->setM5Reg(m5reg);
00102 }

00141 MiscReg
00142 ISA::readMiscReg(int miscReg, ThreadContext * tc)
00143 {
00144     if (miscReg == MISCREG_TSC) {
00145         return regVal[MISCREG_TSC] + tc->getCpuPtr()->curCycle();
00146     }
00147 
00148     if (miscReg == MISCREG_FSW) {
00149         MiscReg fsw = regVal[MISCREG_FSW];
00150         MiscReg top = regVal[MISCREG_X87_TOP];
00151         return (fsw & (~(7ULL << 11))) + (top << 11);
00152     }
00153 
00154     return readMiscRegNoEffect(miscReg);
00155 }

00174     MiscReg newVal = val;
00175     switch(miscReg)
00176     {
00177       case MISCREG_CR0:
00178         {
00179             CR0 toggled = regVal[miscReg] ^ val;
00180             CR0 newCR0 = val;
00181             Efer efer = regVal[MISCREG_EFER];
00182             if (toggled.pg && efer.lme) {
00183                 if (newCR0.pg) {
00184                     //Turning on long mode
00185                     efer.lma = 1;
00186                     regVal[MISCREG_EFER] = efer;
00187                 } else {
00188                     //Turning off long mode
00189                     efer.lma = 0;
00190                     regVal[MISCREG_EFER] = efer;
00191                 }
00192             }
00193             if (toggled.pg) {
00194                 tc->getITBPtr()->flushAll();
00195                 tc->getDTBPtr()->flushAll();
00196             }
00197             //This must always be 1.
00198             newCR0.et = 1;
00199             newVal = newCR0;
00200             updateHandyM5Reg(regVal[MISCREG_EFER],
00201                              newCR0,
00202                              regVal[MISCREG_CS_ATTR],
00203                              regVal[MISCREG_SS_ATTR],
00204                              regVal[MISCREG_RFLAGS],
00205                              tc);
00206         }
00207         break;
00208       case MISCREG_CR2:
00209         break;
00210       case MISCREG_CR3:
00211         tc->getITBPtr()->flushNonGlobal();
00212         tc->getDTBPtr()->flushNonGlobal();
00213         break;
00214       case MISCREG_CR4:
00215         {
00216             CR4 toggled = regVal[miscReg] ^ val;
00217             if (toggled.pae || toggled.pse || toggled.pge) {
00218                 tc->getITBPtr()->flushAll();
00219                 tc->getDTBPtr()->flushAll();
00220             }
00221         }
00222         break;
00223       case MISCREG_CR8:
00224         break;
00225       case MISCREG_CS_ATTR:
00226         {
00227             SegAttr toggled = regVal[miscReg] ^ val;
00228             SegAttr newCSAttr = val;
00229             if (toggled.longMode) {
00230                 if (newCSAttr.longMode) {
00231                     regVal[MISCREG_ES_EFF_BASE] = 0;
00232                     regVal[MISCREG_CS_EFF_BASE] = 0;
00233                     regVal[MISCREG_SS_EFF_BASE] = 0;
00234                     regVal[MISCREG_DS_EFF_BASE] = 0;
00235                 } else {
00236                     regVal[MISCREG_ES_EFF_BASE] = regVal[MISCREG_ES_BASE];
00237                     regVal[MISCREG_CS_EFF_BASE] = regVal[MISCREG_CS_BASE];
00238                     regVal[MISCREG_SS_EFF_BASE] = regVal[MISCREG_SS_BASE];
00239                     regVal[MISCREG_DS_EFF_BASE] = regVal[MISCREG_DS_BASE];
00240                 }
00241             }
00242             updateHandyM5Reg(regVal[MISCREG_EFER],
00243                              regVal[MISCREG_CR0],
00244                              newCSAttr,
00245                              regVal[MISCREG_SS_ATTR],
00246                              regVal[MISCREG_RFLAGS],
00247                              tc);
00248         }
00249         break;
00250       case MISCREG_SS_ATTR:
00251         updateHandyM5Reg(regVal[MISCREG_EFER],
00252                          regVal[MISCREG_CR0],
00253                          regVal[MISCREG_CS_ATTR],
00254                          val,
00255                          regVal[MISCREG_RFLAGS],
00256                          tc);
00257         break;
00258       // These segments always actually use their bases, or in other words
00259       // their effective bases must stay equal to their actual bases.
00260       case MISCREG_FS_BASE:
00261       case MISCREG_GS_BASE:
00262       case MISCREG_HS_BASE:
00263       case MISCREG_TSL_BASE:
00264       case MISCREG_TSG_BASE:
00265       case MISCREG_TR_BASE:
00266       case MISCREG_IDTR_BASE:

00268         break;
00269       // These segments ignore their bases in 64 bit mode.
00270       // their effective bases must stay equal to their actual bases.
00271       case MISCREG_ES_BASE:
00272       case MISCREG_CS_BASE:
00273       case MISCREG_SS_BASE:
00274       case MISCREG_DS_BASE:
00275         {
00276             Efer efer = regVal[MISCREG_EFER];
00277             SegAttr csAttr = regVal[MISCREG_CS_ATTR];
00278             if (!efer.lma || !csAttr.longMode) // Check for non 64 bit mode.
00279                 regVal[MISCREG_SEG_EFF_BASE(miscReg -
00280                         MISCREG_SEG_BASE_BASE)] = val;
00281         }
00282         break;
00283       case MISCREG_TSC:
00284         regVal[MISCREG_TSC] = val - tc->getCpuPtr()->curCycle();
00285         return;
00286       case MISCREG_DR0:
00287       case MISCREG_DR1:
00288       case MISCREG_DR2:
00289       case MISCREG_DR3:
00290         /* These should eventually set up breakpoints. */
00291         break;
00292       case MISCREG_DR4:
00293         miscReg = MISCREG_DR6;
00294         /* Fall through to have the same effects as DR6. */
00295       case MISCREG_DR6:
00296         {
00297             DR6 dr6 = regVal[MISCREG_DR6];
00298             DR6 newDR6 = val;
00299             dr6.b0 = newDR6.b0;
00300             dr6.b1 = newDR6.b1;
00301             dr6.b2 = newDR6.b2;
00302             dr6.b3 = newDR6.b3;

00307         }
00308         break;
00309       case MISCREG_DR5:
00310         miscReg = MISCREG_DR7;
00311         /* Fall through to have the same effects as DR7. */
00312       case MISCREG_DR7:
00313         {
00314             DR7 dr7 = regVal[MISCREG_DR7];

00316             dr7.l0 = newDR7.l0;
00317             dr7.g0 = newDR7.g0;
00318             if (dr7.l0 || dr7.g0) {

00320             } else {
00321                 /* Disable breakpoint 0. */
00322             }
00323             dr7.l1 = newDR7.l1;
00324             dr7.g1 = newDR7.g1;
00325             if (dr7.l1 || dr7.g1) {
00326                 panic("Debug register breakpoints not implemented.\n");
00327             } else {
00328                 /* Disable breakpoint 1. */
00329             }
00330             dr7.l2 = newDR7.l2;
00331             dr7.g2 = newDR7.g2;
00332             if (dr7.l2 || dr7.g2) {
00333                 panic("Debug register breakpoints not implemented.\n");
00334             } else {
00335                 /* Disable breakpoint 2. */
00336             }
00337             dr7.l3 = newDR7.l3;
00338             dr7.g3 = newDR7.g3;
00339             if (dr7.l3 || dr7.g3) {
00340                 panic("Debug register breakpoints not implemented.\n");
00341             } else {
00342                 /* Disable breakpoint 3. */
00343             }
00344             dr7.gd = newDR7.gd;
00345             dr7.rw0 = newDR7.rw0;
00346             dr7.len0 = newDR7.len0;
00347             dr7.rw1 = newDR7.rw1;
00348             dr7.len1 = newDR7.len1;
00349             dr7.rw2 = newDR7.rw2;
00350             dr7.len2 = newDR7.len2;
00351             dr7.rw3 = newDR7.rw3;
00352             dr7.len3 = newDR7.len3;
00353         }
00354         break;
00355       case MISCREG_M5_REG:
00356         // Writing anything to the m5reg with side effects makes it update

00358         // value written is discarded.
00359         updateHandyM5Reg(regVal[MISCREG_EFER],

00361                          regVal[MISCREG_CS_ATTR],
00362                          regVal[MISCREG_SS_ATTR],
00363                          regVal[MISCREG_RFLAGS],
00364                          tc);
00365         return;
00366       default:
00367         break;
00368     }
00369     setMiscRegNoEffect(miscReg, newVal);
