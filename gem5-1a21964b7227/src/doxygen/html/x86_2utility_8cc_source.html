00185 void startupCPU(ThreadContext *tc, int cpuId)
00186 {
00187     if (cpuId == 0 || !FullSystem) {
00188         tc->activate(Cycles(0));
00189     } else {
00190         // This is an application processor (AP). It should be initialized to
00191         // look like only the BIOS POST has run on it and put then put it into
00192         // a halted state.

00197 void
00198 copyMiscRegs(ThreadContext *src, ThreadContext *dest)
00199 {
00200     // This function assumes no side effects other than TLB invalidation
00201     // need to be considered while copying state. That will likely not be
00202     // true in the future.
00203     for (int i = 0; i < NUM_MISCREGS; ++i) {
00204         if ( ( i != MISCREG_CR1 &&
00205              !(i > MISCREG_CR4 && i < MISCREG_CR8) &&
00206              !(i > MISCREG_CR8 && i <= MISCREG_CR15) ) == false) {
00207              continue;
00208         }
00209         dest->setMiscRegNoEffect(i, src->readMiscRegNoEffect(i));
00210     }
00211 
00212     dest->getITBPtr()->flushAll();
00213     dest->getDTBPtr()->flushAll();
00214 }
